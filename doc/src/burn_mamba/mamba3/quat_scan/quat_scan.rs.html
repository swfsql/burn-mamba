1//! # Quaternion cumulative-product scan with a custom, memory-efficient backward
2//!
3//! The forward is the same Hillis–Steele parallel scan as
4//! [`crate::mamba3::rotation::quat_cumprod`], but routed through the
5//! [`Mamba3QuatScanBackendExt`] trait so that `Autodiff` backends substitute a
6//! custom backward that recomputes the scan instead of retaining its
7//! intermediates (see [`super::backward`]). Plain backends use the trait's
8//! default body, which runs the scan on `B`'s primitives.
9//!
10//! The default body runs under a generic backend `B`, where the high-level
11//! [`Tensor`] (pinned to `Dispatch`) is unavailable, so the quaternion algebra
12//! goes through the rank-tagged [`F`] primitive wrapper, held in a
13//! struct-of-arrays [`Quat`] (the four components as separate tensors) so the
14//! Hamilton product is narrow/cat-free on the hot path. The same [`Quat`] helper
15//! and [`quat_prefix_product_soa`] are reused by the recompute backward.
16
17#![allow(non_snake_case)]
18

25// ---------------------------------------------------------------------------
26// Primitive quaternion algebra — struct-of-arrays (SoA)
27// ---------------------------------------------------------------------------
28//
29// The scan's workhorse is the Hamilton product, run ~10× in the forward prefix
30// product and again in the backward. Holding a quaternion as one packed
31// `[…, 4]` tensor forces every product to `narrow` out the four components and
32// `cat` them back — fusion-breaking ops (strided reads + a concat kernel) on the
33// hottest path. Instead, [`Quat`] keeps the four components `(w, x, y, z)` as
34// separate `[batch, sequence, nheads, blocks]` tensors threaded through the
35// whole scan; the product is then pure (fusible) element-wise arithmetic with no
36// narrow/cat. Packing to/from the `[…, 4]` layout happens once, at the
37// boundaries ([`Quat::from_rank5`] / [`Quat::pack`]). This is also the layout a
38// future custom kernel would use (w,x,y,z in registers).
39
40/// A quaternion field in struct-of-arrays form: the four components
41/// `(w, x, y, z)` as separate rank-4 `[batch, sequence, nheads, blocks]` tensors
42/// (a leading seq-length of `1` broadcasts, e.g. the carry).
43pub(crate) struct Quat<B: Backend> {
44    /// Real part `w`.
45    pub w: F<B, 4>,
46    /// Imaginary `x`.
47    pub x: F<B, 4>,
48    /// Imaginary `y`.
49    pub y: F<B, 4>,
50    /// Imaginary `z`.
51    pub z: F<B, 4>,
52}
53
54impl<B: Backend> Clone for Quat<B> {
55    fn clone(&self) -> Self {
56        Quat {
57            w: self.w.clone(),
58            x: self.x.clone(),
59            y: self.y.clone(),
60            z: self.z.clone(),
61        }
62    }
63}
64
65impl<B: Backend> Quat<B> {
66    /// Unpack a packed `[batch, sequence, nheads, blocks, 4]` tensor into SoA
67    /// components (the only `narrow`s in the scan — done once at entry).
68    pub fn from_rank5(q_bshj4: F<B, 5>) -> Self {
69        let w = q_bshj4.clone().narrow(4, 0, 1).squeeze_dim::<4>(4);
70        let x = q_bshj4.clone().narrow(4, 1, 1).squeeze_dim::<4>(4);
71        let y = q_bshj4.clone().narrow(4, 2, 1).squeeze_dim::<4>(4);
72        let z = q_bshj4.narrow(4, 3, 1).squeeze_dim::<4>(4);
73        Quat { w, x, y, z }
74    }
75
76    /// Pack SoA components back into a `[batch, sequence, nheads, blocks, 4]`
77    /// tensor (the only `cat` in the scan — done once at exit).
78    pub fn pack(self) -> F<B, 5> {
79        F::cat(
80            vec![
81                self.w.unsqueeze_dim::<5>(4),
82                self.x.unsqueeze_dim::<5>(4),
83                self.y.unsqueeze_dim::<5>(4),
84                self.z.unsqueeze_dim::<5>(4),
85            ],
86            4,
87        )
88    }
89
90    /// Identity quaternion `(1, 0, 0, 0)` of the given component shape.
91    pub fn identity(shape: [usize; 4], device: &Device<B>, dtype: FloatDType) -> Self {
92        let zero = F::<B, 4>::zeros(shape, device, dtype);
93        Quat {
94            w: F::<B, 4>::full(shape, 1.0, device, dtype),
95            x: zero.clone(),
96            y: zero.clone(),
97            z: zero,
98        }
99    }
100
101    /// Hamilton product `self ⊗ other` (pure element-wise; no narrow/cat).
102    /// Broadcasts over the leading dims, so a `[b,1,h,j]` carry multiplies a
103    /// `[b,s,h,j]` sequence.
104    pub fn mul(self, other: Quat<B>) -> Quat<B> {
105        let Quat {
106            w: aw,
107            x: ax,
108            y: ay,
109            z: az,
110        } = self;
111        let Quat {
112            w: bw,
113            x: bx,
114            y: by,
115            z: bz,
116        } = other;
117        Quat {
118            w: aw.clone() * bw.clone()
119                - ax.clone() * bx.clone()
120                - ay.clone() * by.clone()
121                - az.clone() * bz.clone(),
122            x: aw.clone() * bx.clone() + ax.clone() * bw.clone() + ay.clone() * bz.clone()
123                - az.clone() * by.clone(),
124            y: aw.clone() * by.clone() - ax.clone() * bz.clone()
125                + ay.clone() * bw.clone()
126                + az.clone() * bx.clone(),
127            z: aw * bz + ax * by - ay * bx + az * bw,
128        }
129    }
130
131    /// Quaternion conjugate `(w, −x, −y, −z)`. For a unit quaternion `q* = q⁻¹`.
132    pub fn conj(self) -> Quat<B> {
133        Quat {
134            w: self.w,
135            x: -self.x,
136            y: -self.y,
137            z: -self.z,
138        }
139    }
140
141    /// Prepend `ident` along the sequence axis and re-narrow to `sequence` — the
142    /// Hillis–Steile shift `shifted[t] = self[t-offset]` (identity before the
143    /// start), with `offset = ident`'s sequence length.
144    pub fn shift_prepend(self, ident: Quat<B>, sequence: usize) -> Quat<B> {
145        let shift =
146            |head: F<B, 4>, tail: F<B, 4>| F::cat(vec![head, tail], 1).narrow(1, 0, sequence);
147        Quat {
148            w: shift(ident.w, self.w),
149            x: shift(ident.x, self.x),
150            y: shift(ident.y, self.y),
151            z: shift(ident.z, self.z),
152        }
153    }
154
155    /// Inclusive **suffix**-sum along the sequence axis (`out[t] = Σ_{s≥t} in[s]`),
156    /// per component — `flip → cumsum → flip`. Used by the backward's `S[t]`.
157    pub fn reverse_cumsum_seq(self) -> Quat<B> {
158        let rc = |t: F<B, 4>| t.flip(&[1]).cumsum(1).flip(&[1]);
159        Quat {
160            w: rc(self.w),
161            x: rc(self.x),
162            y: rc(self.y),
163            z: rc(self.z),
164        }
165    }
166}
167
168/// Pure prefix product `P[t] = qₜ ⊗ qₜ₋₁ ⊗ ⋯ ⊗ q₀` (no carry) along the sequence
169/// axis, via Hillis–Steele doubling — `O(log seq)` dependency depth, all on SoA
170/// [`Quat`] components (no per-step narrow/cat).
171///
172/// The carry is folded in by the caller (one extra [`Quat::mul`]); keeping `P`
173/// separate is what the recompute backward needs (`G[t] = P[t] ⊗ S[t]`).
174pub(crate) fn quat_prefix_product_soa<B: Backend>(q: Quat<B>) -> Quat<B> {
175    let [batch, sequence, nheads, blocks] = q.w.dims();
176    let device = q.w.device();
177    let dtype = q.w.dtype();
178
179    let mut a = q;
180    let mut offset = 1usize;
181    while offset < sequence {
182        let ident = Quat::<B>::identity([batch, offset, nheads, blocks], &device, dtype);
183        let shifted = a.clone().shift_prepend(ident, sequence);
184        a = a.mul(shifted); // recent (a) ⊗ older (shifted)
185        offset *= 2;
186    }
187    a
188}

190// ---------------------------------------------------------------------------
191// Backend extension trait (default body = the primitive scan)
192// ---------------------------------------------------------------------------
193
194/// Extends the backend with the quaternion cumulative-product scan.
195///
196/// The default body runs the Hillis–Steele scan on primitive tensors. The
197/// `Autodiff` wrapper overrides it with a memory-efficient custom backward (in
198/// [`super::backward`]) that recomputes the scan instead of saving its
199/// intermediates.
200#[backend_extension(
201    Cpu:  cfg(feature = "backend-cpu"),
202    Cuda: cfg(feature = "backend-cuda"),
203    Rocm:  cfg(feature = "backend-rocm"),
204    Metal:  cfg(feature = "backend-metal"),
205    Vulkan:  cfg(feature = "backend-vulkan"),
206    Wgpu:  cfg(feature = "backend-wgpu"),
207    WebGpu:  cfg(feature = "backend-webgpu"),
208    Flex:  cfg(feature = "backend-flex"),
209    NdArray:  cfg(feature = "backend-ndarray"),
210    LibTorch:  cfg(any(feature = "backend-tch-cpu", feature = "backend-tch-gpu")),
211    Autodiff:  cfg(feature = "autodiff"),
212)]
213pub trait Mamba3QuatScanBackendExt: Backend {
214    /// Cumulative quaternion product `cum[t] = qₜ ⊗ ⋯ ⊗ q₀ ⊗ init` along the
215    /// sequence axis (newest on the left), returning only `cum`.
216    ///
217    /// The caller derives `final_carry = cum[:, −1]` (a thin autodiff slice) in
218    /// high-level land — see [`quat_cumprod_recalculated`].
219    ///
220    /// # Shapes
221    /// - `q_bshj4`   : `[batch, sequence, nheads, blocks, 4]` — per-step **unit**
222    ///   quaternions.
223    /// - `init_bhj4` : `[batch, nheads, blocks, 4]` — the cross-chunk carry
224    ///   (identity `(1,0,0,0)` for a fresh start).
225    /// - returns `cum` : `[batch, sequence, nheads, blocks, 4]`.
226    fn quat_cumprod(q_bshj4: FloatTensor<Self>, init_bhj4: FloatTensor<Self>) -> FloatTensor<Self> {
227        let q = Quat::from_rank5(F::<Self, 5>::new(q_bshj4));
228        // Carry as a 1-long sequence [batch, 1, nheads, blocks, 4] → SoA, broadcasts over seq.
229        let init = Quat::from_rank5(F::<Self, 4>::new(init_bhj4).unsqueeze_dim::<5>(1));
230
231        let p = quat_prefix_product_soa::<Self>(q);
232        let cum = p.mul(init); // cum[t] = Pₜ ⊗ init
233        cum.pack().inner()
234    }
235}

239crate::impl_ssd_backend_ext_for_burn_backends!(Mamba3QuatScanBackendExt);

241// ---------------------------------------------------------------------------
242// High-level wrapper (Dispatch-pinned `Tensor`)
243// ---------------------------------------------------------------------------
244
245/// Cumulative quaternion product with a custom, memory-efficient backward — the
246/// drop-in for [`crate::mamba3::rotation::quat_cumprod`] used by the
247/// Quaternion4D `forward`.
248///
249/// Routes the scan through [`Mamba3QuatScanBackendExt`] (so `Autodiff` gets the
250/// recompute backward), then takes `final_carry = cum[:, −1]` as a thin autodiff
251/// slice — its gradient folds back into `cum`'s gradient before the custom node
252/// runs, so the node needs only the single `cum` output.
253///
254/// Mathematically identical to `quat_cumprod` (asserted on values and gradients
255/// by the tests); only the backward's memory profile differs.
256///
257/// # Shapes
258/// - `q_bshj4` : `[batch, sequence, nheads, blocks, 4]` — per-step **unit**
259///   quaternions (the recompute backward's gradient identities assume unit norm,
260///   which holds for [`quat_from_scaled_axis`](crate::mamba3::rotation::quat_from_scaled_axis)
261///   outputs and products thereof).
262/// - `init`    : optional carry `[batch, nheads, blocks, 4]` (identity if `None`).
263/// - returns `(cum, final_carry)` — `[batch, sequence, nheads, blocks, 4]` and
264///   `[batch, nheads, blocks, 4]`.
265pub fn quat_cumprod_recalculated(
266    q_bshj4: Tensor<5>,
267    init: Option<Tensor<4>>,
268) -> (Tensor<5>, Tensor<4>) {
269    let [batch, sequence, nheads, blocks, _four] = q_bshj4.dims();
270    let device = q_bshj4.device();
271
272    let init_bhj4 = init.unwrap_or_else(|| {
273        let w = Tensor::<4>::ones([batch, nheads, blocks, 1], &device);
274        let xyz = Tensor::<4>::zeros([batch, nheads, blocks, 3], &device);
275        Tensor::cat(vec![w, xyz], 3)
276    });
277
278    let cum_bshj4 =
279        Tensor::<5>::from_primitive(<Dispatch as Mamba3QuatScanBackendExt>::quat_cumprod(
280            q_bshj4.into_primitive(),
281            init_bhj4.into_primitive(),
282        ));
283
284    let final_carry_bhj4 = cum_bshj4
285        .clone()
286        .narrow(1, sequence - 1, 1)
287        .squeeze_dim::<4>(1);
288    (cum_bshj4, final_carry_bhj4)
289}
