1//! # Quaternion cumulative-product scan with a custom, memory-efficient backward
2//!
3//! The forward is the same Hillis–Steele parallel scan as
4//! [`crate::mamba3::rotation::quat_cumprod`], but routed through the
5//! [`Mamba3QuatScanBackendExt`] trait so that `Autodiff` backends substitute a
6//! custom backward that recomputes the scan instead of retaining its
7//! intermediates (see [`super::backward`]). Plain backends use the trait's
8//! default body, which runs the scan on `B`'s primitives.
9//!
10//! The default body runs under a generic backend `B`, where the high-level
11//! [`Tensor`] (pinned to `Dispatch`) is unavailable, so the quaternion algebra
12//! goes through the rank-tagged [`F`] primitive wrapper. The same primitive
13//! helpers ([`fquat_mul`], [`fquat_conj`], [`fquat_prefix_product`]) are reused
14//! by the recompute backward.
15
16#![allow(non_snake_case)]
17

24// ---------------------------------------------------------------------------
25// Primitive quaternion algebra (rank-5: [batch, sequence, nheads, blocks, 4])
26// ---------------------------------------------------------------------------
27
28/// Hamilton product `a ⊗ b` on the trailing `(w, x, y, z)` axis (axis 4), the
29/// primitive-`F` analogue of [`crate::mamba3::rotation::quat_mul`]. Broadcasts
30/// over the leading dims, so a `[b,1,h,j,4]` carry multiplies a `[b,s,h,j,4]`
31/// sequence.
32pub(crate) fn fquat_mul<B: Backend>(a: F<B, 5>, b: F<B, 5>) -> F<B, 5> {
33    let aw = a.clone().narrow(4, 0, 1);
34    let ax = a.clone().narrow(4, 1, 1);
35    let ay = a.clone().narrow(4, 2, 1);
36    let az = a.narrow(4, 3, 1);
37    let bw = b.clone().narrow(4, 0, 1);
38    let bx = b.clone().narrow(4, 1, 1);
39    let by = b.clone().narrow(4, 2, 1);
40    let bz = b.narrow(4, 3, 1);
41
42    let w = aw.clone() * bw.clone() - ax.clone() * bx.clone() - ay.clone() * by.clone()
43        - az.clone() * bz.clone();
44    let x = aw.clone() * bx.clone() + ax.clone() * bw.clone() + ay.clone() * bz.clone()
45        - az.clone() * by.clone();
46    let y = aw.clone() * by.clone() - ax.clone() * bz.clone() + ay.clone() * bw.clone()
47        + az.clone() * bx.clone();
48    let z = aw * bz + ax * by - ay * bx + az * bw;
49
50    F::cat(vec![w, x, y, z], 4)
51}

53/// Quaternion conjugate `q* = (w, −x, −y, −z)` on axis 4 (primitive-`F` analogue
54/// of [`crate::mamba3::rotation::quat_conj`]). For a unit quaternion `q* = q⁻¹`.
55pub(crate) fn fquat_conj<B: Backend>(q: F<B, 5>) -> F<B, 5> {
56    let w = q.clone().narrow(4, 0, 1);
57    let xyz = q.narrow(4, 1, 3);
58    F::cat(vec![w, -xyz], 4)
59}

61/// Pure prefix product `P[t] = qₜ ⊗ qₜ₋₁ ⊗ ⋯ ⊗ q₀` (no carry) along the sequence
62/// axis (axis 1), via the same Hillis–Steele doubling as
63/// [`crate::mamba3::rotation::quat_cumprod`] — `O(log seq)` dependency depth.
64///
65/// The carry is folded in by the caller (one extra [`fquat_mul`]); keeping `P`
66/// separate is what the recompute backward needs (`G[t] = P[t] ⊗ S[t]`).
67pub(crate) fn fquat_prefix_product<B: Backend>(q_bshj4: F<B, 5>) -> F<B, 5> {
68    let [batch, sequence, nheads, blocks, _four] = q_bshj4.dims();
69    let device = q_bshj4.device();
70    let dtype = q_bshj4.dtype();
71
72    let mut a = q_bshj4;
73    let mut offset = 1usize;
74    while offset < sequence {
75        // identity quaternion (1,0,0,0) for the first `offset` (shifted-past-start) slots.
76        let w = F::<B, 5>::full([batch, offset, nheads, blocks, 1], 1.0, &device, dtype);
77        let xyz = F::<B, 5>::zeros([batch, offset, nheads, blocks, 3], &device, dtype);
78        let ident = F::cat(vec![w, xyz], 4);
79        let shifted = F::cat(vec![ident, a.clone()], 1).narrow(1, 0, sequence);
80        a = fquat_mul::<B>(a, shifted); // recent (a) ⊗ older (shifted)
81        offset *= 2;
82    }
83    a
84}

86// ---------------------------------------------------------------------------
87// Backend extension trait (default body = the primitive scan)
88// ---------------------------------------------------------------------------

90/// Extends the backend with the quaternion cumulative-product scan.
91///
92/// The default body runs the Hillis–Steele scan on primitive tensors. The
93/// `Autodiff` wrapper overrides it with a memory-efficient custom backward (in
94/// [`super::backward`]) that recomputes the scan instead of saving its
95/// intermediates.
96#[backend_extension(
97    Cpu:  cfg(feature = "backend-cpu"),
98    Cuda: cfg(feature = "backend-cuda"),
99    Rocm:  cfg(feature = "backend-rocm"),
100    Metal:  cfg(feature = "backend-metal"),
101    Vulkan:  cfg(feature = "backend-vulkan"),
102    Wgpu:  cfg(feature = "backend-wgpu"),
103    WebGpu:  cfg(feature = "backend-webgpu"),
104    Flex:  cfg(feature = "backend-flex"),
105    NdArray:  cfg(feature = "backend-ndarray"),
106    LibTorch:  cfg(any(feature = "backend-tch-cpu", feature = "backend-tch-gpu")),
107    Autodiff:  cfg(feature = "autodiff"),
108)]
109pub trait Mamba3QuatScanBackendExt: Backend {
110    /// Cumulative quaternion product `cum[t] = qₜ ⊗ ⋯ ⊗ q₀ ⊗ init` along the
111    /// sequence axis (newest on the left), returning only `cum`.
112    ///
113    /// The caller derives `final_carry = cum[:, −1]` (a thin autodiff slice) in
114    /// high-level land — see [`quat_cumprod_recalculated`].
115    ///
116    /// # Shapes
117    /// - `q_bshj4`   : `[batch, sequence, nheads, blocks, 4]` — per-step **unit**
118    ///   quaternions.
119    /// - `init_bhj4` : `[batch, nheads, blocks, 4]` — the cross-chunk carry
120    ///   (identity `(1,0,0,0)` for a fresh start).
121    /// - returns `cum` : `[batch, sequence, nheads, blocks, 4]`.
122    fn quat_cumprod(
123        q_bshj4: FloatTensor<Self>,
124        init_bhj4: FloatTensor<Self>,
125    ) -> FloatTensor<Self> {
126        let q = F::<Self, 5>::new(q_bshj4);
127        let init = F::<Self, 4>::new(init_bhj4);
128
129        let p_bshj4 = fquat_prefix_product::<Self>(q);
130        let init5_b1hj4 = init.unsqueeze_dim::<5>(1); // [batch, 1, nheads, blocks, 4]
131        let cum_bshj4 = fquat_mul::<Self>(p_bshj4, init5_b1hj4); // broadcasts over seq
132        cum_bshj4.inner()
133    }
134}

137// backward lives in `super::backward` as a separate `Autodiff<B>` impl.
138crate::impl_ssd_backend_ext_for_burn_backends!(Mamba3QuatScanBackendExt);

140// ---------------------------------------------------------------------------
141// High-level wrapper (Dispatch-pinned `Tensor`)
142// ---------------------------------------------------------------------------
143
144/// Cumulative quaternion product with a custom, memory-efficient backward — the
145/// drop-in for [`crate::mamba3::rotation::quat_cumprod`] used by the
146/// Quaternion4D `forward`.
147///
148/// Routes the scan through [`Mamba3QuatScanBackendExt`] (so `Autodiff` gets the
149/// recompute backward), then takes `final_carry = cum[:, −1]` as a thin autodiff
150/// slice — its gradient folds back into `cum`'s gradient before the custom node
151/// runs, so the node needs only the single `cum` output.
152///
153/// Mathematically identical to `quat_cumprod` (asserted on values and gradients
154/// by the tests); only the backward's memory profile differs.
155///
156/// # Shapes
157/// - `q_bshj4` : `[batch, sequence, nheads, blocks, 4]` — per-step **unit**
158///   quaternions (the recompute backward's gradient identities assume unit norm,
159///   which holds for [`quat_from_scaled_axis`](crate::mamba3::rotation::quat_from_scaled_axis)
160///   outputs and products thereof).
161/// - `init`    : optional carry `[batch, nheads, blocks, 4]` (identity if `None`).
162/// - returns `(cum, final_carry)` — `[batch, sequence, nheads, blocks, 4]` and
163///   `[batch, nheads, blocks, 4]`.
164pub fn quat_cumprod_recalculated(
165    q_bshj4: Tensor<5>,
166    init: Option<Tensor<4>>,
167) -> (Tensor<5>, Tensor<4>) {
168    let [batch, sequence, nheads, blocks, _four] = q_bshj4.dims();
169    let device = q_bshj4.device();
170
171    let init_bhj4 = init.unwrap_or_else(|| {
172        let w = Tensor::<4>::ones([batch, nheads, blocks, 1], &device);
173        let xyz = Tensor::<4>::zeros([batch, nheads, blocks, 3], &device);
174        Tensor::cat(vec![w, xyz], 3)
175    });
176
177    let cum_bshj4 = Tensor::<5>::from_primitive(<Dispatch as Mamba3QuatScanBackendExt>::quat_cumprod(
178        q_bshj4.into_primitive(),
179        init_bhj4.into_primitive(),
180    ));
181
182    let final_carry_bhj4 = cum_bshj4
183        .clone()
184        .narrow(1, sequence - 1, 1)
185        .squeeze_dim::<4>(1);
186    (cum_bshj4, final_carry_bhj4)
187}
