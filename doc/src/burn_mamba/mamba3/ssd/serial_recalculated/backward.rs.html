16impl<B: Backend + Mamba3BackendExt, C: CheckpointStrategy> Mamba3BackendExt for Autodiff<B, C> {
17    /// Memory-efficient combined forward+backward for the Mamba-3 MIMO SSD.
18    ///
19    /// The two output tensors (`y_bnlmhp`, `final_state_bhpr`) are flattened
20    /// and concatenated into a single 1-dimensional tracked tensor so a single
21    /// `Backward<B, 5>` node covers both. The caller receives split+reshaped
22    /// slices of that combined tensor; burn's autodiff accumulates their
23    /// upstream gradients back into one gradient vector before invoking
24    /// `backward`.
25    fn ssd_serial_recalculated(
26        v_bnlmhp: FloatTensor<Self>,
27        da_bnlh: FloatTensor<Self>,
28        b_bnlmhr: FloatTensor<Self>,
29        c_bnlmhr: FloatTensor<Self>,
30        initial_state_bhpr: FloatTensor<Self>,
31    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
32        // ── Backward node definition ─────────────────────────────────────────
33        #[derive(Debug)]
34        struct CombinedKernelsBackward;
35
36        #[derive(Clone, Debug)]
37        struct State<B: Backend> {
38            // Saved forward inputs
39            v_bnlmhp: FloatTensor<B>,
40            da_bnlh: FloatTensor<B>,
41            b_bnlmhr: FloatTensor<B>,
42            c_bnlmhr: FloatTensor<B>,
43            initial_state_bhpr: FloatTensor<B>,
44            // Flat lengths for splitting the combined upstream gradient
45            flat_len_y_BNLMHP: usize,
46            flat_len_final_state_BHPR: usize,
47            // Shapes needed to reconstruct tensors at the right ranks
48            shape_v_bnlmhp: [usize; 6],
49            shape_da_bnlh: [usize; 4],
50            shape_b_bnlmhr: [usize; 6],
51            shape_c_bnlmhr: [usize; 6],
52            shape_initial_state_bhpr: [usize; 4],
53            shape_y_bnlmhp: [usize; 6],
54            shape_final_state_bhpr: [usize; 4],
55        }
56
57        impl<B: Backend + Mamba3BackendExt> Backward<B, 5> for CombinedKernelsBackward {
58            type State = State<B>;
59
60            fn backward(
61                self,
62                ops: Ops<Self::State, 5>,
63                grads: &mut Gradients,
64                _checkpointer: &mut Checkpointer,
65            ) {
66                let [
67                    node_v_bnlmhp,
68                    node_da_bnlh,
69                    node_b_bnlmhr,
70                    node_c_bnlmhr,
71                    node_initial_state_bhpr,
72                ] = ops.parents;
73
74                // Upstream gradient of the combined 1-dimensional output.
75                let d_combined: Tensor<B, 1> =
76                    Tensor::from_primitive(TensorPrimitive::Float(grads.consume::<B>(&ops.node)));
77
78                let State {
79                    v_bnlmhp,
80                    da_bnlh,
81                    b_bnlmhr,
82                    c_bnlmhr,
83                    initial_state_bhpr,
84                    flat_len_y_BNLMHP,
85                    flat_len_final_state_BHPR,
86                    shape_v_bnlmhp,
87                    shape_da_bnlh,
88                    shape_b_bnlmhr,
89                    shape_c_bnlmhr,
90                    shape_initial_state_bhpr,
91                    shape_y_bnlmhp,
92                    shape_final_state_bhpr,
93                } = ops.state;
94
95                // ── Reconstruct saved tensors ───────────────────────────
96                use super::serial_recalculated::mk;
97                let v_bnlmhp = mk::<_, 6>(v_bnlmhp).reshape(shape_v_bnlmhp);
98                let da_bnlh = mk::<_, 4>(da_bnlh).reshape(shape_da_bnlh);
99                let b_bnlmhr = mk::<_, 6>(b_bnlmhr).reshape(shape_b_bnlmhr);
100                let c_bnlmhr = mk::<_, 6>(c_bnlmhr).reshape(shape_c_bnlmhr);
101                let initial_state_bhpr =
102                    mk::<_, 4>(initial_state_bhpr).reshape(shape_initial_state_bhpr);
103
104                // ── Split combined gradient vector ──────────────────────
105                let flat_d_y_BNLMHP = d_combined.clone().narrow(0, 0, flat_len_y_BNLMHP);
106                let flat_d_final_state_BHPR =
107                    d_combined.narrow(0, flat_len_y_BNLMHP, flat_len_final_state_BHPR);
108                let d_y_bnlmhp: Tensor<B, 6> = flat_d_y_BNLMHP.reshape(shape_y_bnlmhp);
109                let d_final_state_bhpr: Tensor<B, 4> =
110                    flat_d_final_state_BHPR.reshape(shape_final_state_bhpr);
111
112                // ── Core gradient computation ───────────────────────────
113                let CombinedGrads {
114                    d_v_bnlmhp,
115                    d_da_bnlh,
116                    d_b_bnlmhr,
117                    d_c_bnlmhr,
118                    d_initial_state_bhpr,
119                    ..
120                } = combined_backward::combined_backward(
121                    d_y_bnlmhp,
122                    d_final_state_bhpr,
123                    v_bnlmhp,
124                    da_bnlh,
125                    b_bnlmhr,
126                    c_bnlmhr,
127                    initial_state_bhpr,
128                );
129
130                // ── Register gradients with autodiff ────────────────────
131                if let Some(n) = node_v_bnlmhp {
132                    grads.register::<B>(n.id, d_v_bnlmhp.into_primitive().tensor());
133                }
134                if let Some(n) = node_da_bnlh {
135                    grads.register::<B>(n.id, d_da_bnlh.into_primitive().tensor());
136                }
137                if let Some(n) = node_b_bnlmhr {
138                    grads.register::<B>(n.id, d_b_bnlmhr.into_primitive().tensor());
139                }
140                if let Some(n) = node_c_bnlmhr {
141                    grads.register::<B>(n.id, d_c_bnlmhr.into_primitive().tensor());
142                }
143                if let Some(n) = node_initial_state_bhpr {
144                    grads.register::<B>(n.id, d_initial_state_bhpr.into_primitive().tensor());
145                }
146            }
147        }
148
149        // ── Shape extraction (via the AutodiffTensor wrappers) ─────────────
150        use burn::tensor::TensorMetadata;
151        let [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim] =
152            v_bnlmhp.primitive.shape().dims();
153        let [.., state_rank] = b_bnlmhr.primitive.shape().dims::<6>();
154
155        let flat_len_y_BNLMHP = batch * nchunks * chunk_len * mimo_rank * nheads * per_head_dim;
156        let flat_len_final_state_BHPR = batch * nheads * per_head_dim * state_rank;
157
158        let shape_v_bnlmhp: [usize; 6] =
159            [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim];
160        let shape_da_bnlh: [usize; 4] = [batch, nchunks, chunk_len, nheads];
161        let shape_b_bnlmhr: [usize; 6] = [batch, nchunks, chunk_len, mimo_rank, nheads, state_rank];
162        let shape_c_bnlmhr: [usize; 6] = [batch, nchunks, chunk_len, mimo_rank, nheads, state_rank];
163        let shape_initial_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
164        let shape_y_bnlmhp: [usize; 6] =
165            [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim];
166        let shape_final_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
167
168        // ── Register backward / run forward ───────────────────────────────
169        match CombinedKernelsBackward
170            .prepare::<C>([
171                v_bnlmhp.node.clone(),
172                da_bnlh.node.clone(),
173                b_bnlmhr.node.clone(),
174                c_bnlmhr.node.clone(),
175                initial_state_bhpr.node.clone(),
176            ])
177            .compute_bound()
178            .stateful()
179        {
180            OpsKind::Tracked(prep) => {
181                let (prim_y_bnlmhp, prim_final_state_bhpr) = B::ssd_serial_recalculated(
182                    v_bnlmhp.primitive.clone(),
183                    da_bnlh.primitive.clone(),
184                    b_bnlmhr.primitive.clone(),
185                    c_bnlmhr.primitive.clone(),
186                    initial_state_bhpr.primitive.clone(),
187                );
188
189                // Flatten + cat both outputs into one tracked 1-dimensional tensor so a
190                // single backward node covers both.
191                let flat_y_BNLMHP: Tensor<B, 1> =
192                    Tensor::<B, 6>::from_primitive(TensorPrimitive::Float(prim_y_bnlmhp))
193                        .reshape([flat_len_y_BNLMHP]);
194                let flat_final_state_BHPR: Tensor<B, 1> =
195                    Tensor::<B, 4>::from_primitive(TensorPrimitive::Float(prim_final_state_bhpr))
196                        .reshape([flat_len_final_state_BHPR]);
197                let combined: Tensor<B, 1> =
198                    Tensor::cat(vec![flat_y_BNLMHP, flat_final_state_BHPR], 0);
199
200                let state = State {
201                    v_bnlmhp: v_bnlmhp.primitive.clone(),
202                    da_bnlh: da_bnlh.primitive.clone(),
203                    b_bnlmhr: b_bnlmhr.primitive.clone(),
204                    c_bnlmhr: c_bnlmhr.primitive.clone(),
205                    initial_state_bhpr: initial_state_bhpr.primitive.clone(),
206                    flat_len_y_BNLMHP,
207                    flat_len_final_state_BHPR,
208                    shape_v_bnlmhp,
209                    shape_da_bnlh,
210                    shape_b_bnlmhr,
211                    shape_c_bnlmhr,
212                    shape_initial_state_bhpr,
213                    shape_y_bnlmhp,
214                    shape_final_state_bhpr,
215                };
216                let tracked_combined: FloatTensor<Autodiff<B, C>> =
217                    prep.finish(state, combined.into_primitive().tensor());
218
219                // Split the tracked combined tensor back into the two outputs.
220                // The narrow/reshape ops are thin autodiff pass-throughs whose
221                // backwards accumulate into the combined gradient vector that
222                // `backward` above consumes.
223                let tracked_combined: Tensor<Autodiff<B, C>, 1> =
224                    Tensor::from_primitive(TensorPrimitive::Float(tracked_combined));
225
226                let tracked_y_bnlmhp: Tensor<Autodiff<B, C>, 6> = tracked_combined
227                    .clone()
228                    .narrow(0, 0, flat_len_y_BNLMHP)
229                    .reshape(shape_y_bnlmhp);
230                let tracked_final_state_bhpr: Tensor<Autodiff<B, C>, 4> = tracked_combined
231                    .narrow(0, flat_len_y_BNLMHP, flat_len_final_state_BHPR)
232                    .reshape(shape_final_state_bhpr);
233
234                (
235                    tracked_y_bnlmhp.into_primitive().tensor(),
236                    tracked_final_state_bhpr.into_primitive().tensor(),
237                )
238            }
239
240            OpsKind::UnTracked(prep) => {
241                // No gradient tracking — just run the bare forward.
242                let (prim_y_bnlmhp, prim_final_state_bhpr) = B::ssd_serial_recalculated(
243                    v_bnlmhp.primitive,
244                    da_bnlh.primitive,
245                    b_bnlmhr.primitive,
246                    c_bnlmhr.primitive,
247                    initial_state_bhpr.primitive,
248                );
249
250                let flat_y_BNLMHP: Tensor<B, 1> =
251                    Tensor::<B, 6>::from_primitive(TensorPrimitive::Float(prim_y_bnlmhp))
252                        .reshape([flat_len_y_BNLMHP]);
253                let flat_final_state_BHPR: Tensor<B, 1> =
254                    Tensor::<B, 4>::from_primitive(TensorPrimitive::Float(prim_final_state_bhpr))
255                        .reshape([flat_len_final_state_BHPR]);
256                let combined: Tensor<B, 1> =
257                    Tensor::cat(vec![flat_y_BNLMHP, flat_final_state_BHPR], 0);
258
259                let tracked_combined: FloatTensor<Autodiff<B, C>> =
260                    prep.finish(combined.into_primitive().tensor());
261
262                let tracked_combined: Tensor<Autodiff<B, C>, 1> =
263                    Tensor::from_primitive(TensorPrimitive::Float(tracked_combined));
264                let tracked_y_bnlmhp: Tensor<Autodiff<B, C>, 6> = tracked_combined
265                    .clone()
266                    .narrow(0, 0, flat_len_y_BNLMHP)
267                    .reshape(shape_y_bnlmhp);
268                let tracked_final_state_bhpr: Tensor<Autodiff<B, C>, 4> = tracked_combined
269                    .narrow(0, flat_len_y_BNLMHP, flat_len_final_state_BHPR)
270                    .reshape(shape_final_state_bhpr);
271
272                (
273                    tracked_y_bnlmhp.into_primitive().tensor(),
274                    tracked_final_state_bhpr.into_primitive().tensor(),
275                )
276            }
277        }
278    }
