6impl<B: Backend> Mamba3<B> {
7    /// MIMO-first (Hybrid) Serial SSD.
8    ///
9    /// Implements K1-K5 with a sequential loop (K4) for the inter-chunk scan instead
10    /// of the quadratic segsum approach in [`ssd_minimal`](Self::ssd_minimal).
11    /// This is more memory-efficient for long sequences with many chunks.
12    ///
13    /// SISO (mimo_rank=1) is the special case where the fused length equals the chunk length.
14    ///
15    /// # Returns
16    /// - `y_bnlmhp`: `[batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]`
17    /// - `final_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
18    pub fn ssd_serial(input: super::Mamba3SsdInput<B>) -> (Tensor<B, 6>, Tensor<B, 4>) {
19        let [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim] = input.v_bnlmhp.dims();
20        let [.., state_rank] = input.b_bnlmhr.dims();
21
22        assert!(
23            input.init_state_hpr.is_none(),
24            "init_state_hpr is not yet supported in ssd_serial; use ssd_minimal instead"
25        );
26        assert!(nchunks > 0, "sequence length must be at least 1");
27
28        // ── K1: da_cumsum from da_bnlh ────────────────────────────────────────
29        let (da_cumsum_bhnl, da_chunk_end_bhn) = k1_ssd_chunk_cumsum(input.da_bnlh.clone());
30        assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
31        assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
32
33        // ── K2: CB matrix on fused tensors ────────────────────────────────────
34        let cb_bnhLMLM: Tensor<B, 5> = k2_ssd_bmm(input.c_bnlmhr.clone(), input.b_bnlmhr.clone());
35        assert_eq!(
36            [
37                batch,
38                nchunks,
39                nheads,
40                chunk_len * mimo_rank,
41                chunk_len * mimo_rank
42            ],
43            cb_bnhLMLM.dims()
44        );
45
46        // ── K3: intra-chunk state ─────────────────────────────────────────────
47        let intra_chunk_state_bnhpr: Tensor<B, 5> = k3_ssd_chunk_state(
48            input.v_bnlmhp.clone(),
49            input.b_bnlmhr.clone(),
50            da_cumsum_bhnl.clone(),
51        );
52        assert_eq!(
53            [batch, nchunks, nheads, per_head_dim, state_rank],
54            intra_chunk_state_bnhpr.dims()
55        );
56
57        // ── K4: state passing (sequential loop) ───────────────────────────────
58        let (chunk_input_state_bnhpr, final_state_bhpr): (Tensor<B, 5>, Tensor<B, 4>) =
59            k4_ssd_state_passing(
60                intra_chunk_state_bnhpr,
61                da_chunk_end_bhn,
62                input.initial_state_bhpr,
63            );
64        assert_eq!(
65            [batch, nchunks, nheads, per_head_dim, state_rank],
66            chunk_input_state_bnhpr.dims()
67        );
68        assert_eq!(
69            [batch, nheads, per_head_dim, state_rank],
70            final_state_bhpr.dims()
71        );
72
73        // ── K5: MIMO chunk scan ───────────────────────────────────────────────
74        let y_bnlmhp: Tensor<B, 6> = k5_ssd_chunk_scan(
75            da_cumsum_bhnl,
76            input.v_bnlmhp,
77            input.c_bnlmhr,
78            cb_bnhLMLM,
79            chunk_input_state_bnhpr,
80        );
81
82        (y_bnlmhp, final_state_bhpr)
83    }
84}

86// ---------------------------------------------------------------------------
87// K1 — chunk cumulative log-decay
88// ---------------------------------------------------------------------------
89
90/// Compute the intra-chunk cumulative log-decay and per-chunk decay totals.
91///
92/// # Arguments
93/// - `da_bnlh`: pre-combined `Δ·A`, shape `[batch, nchunks, chunk_len, nheads]`
94///
95/// # Returns
96/// - `da_cumsum_bhnl`: `[batch, nheads, nchunks, chunk_len]` — intra-chunk prefix sums
97/// - `da_chunk_end_bhn`: `[batch, nheads, nchunks]` — last prefix sum per chunk (total decay)
98pub fn k1_ssd_chunk_cumsum<B: Backend>(da_bnlh: Tensor<B, 4>) -> (Tensor<B, 4>, Tensor<B, 3>) {
99    let [batch, nchunks, chunk_len, nheads] = da_bnlh.dims();
100    // Permute to [batch, nheads, nchunks, chunk_len] for the cumsum along the last dim
101    let da_bhnl = da_bnlh.permute([0, 3, 1, 2]);
102    let da_cumsum_bhnl = da_bhnl.cumsum(3);
103    assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
104
105    let da_chunk_end_bhn: Tensor<B, 3> = da_cumsum_bhnl
106        .clone()
107        .slice(s![.., .., .., -1]) // da_cumsum_end_bhn1
108        .squeeze_dim(3); // da_cumsum_end_bhn
109    assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
110
111    (da_cumsum_bhnl, da_chunk_end_bhn)
112}

114// ---------------------------------------------------------------------------
115// K2 — CB block matrix (C @ B^T on fused MIMO tensors)
116// ---------------------------------------------------------------------------
117
118/// Compute the intra-chunk CB matrix on fused (mimo_rank-into-chunk_len) tensors.
119///
120/// # Arguments
121/// - `c_bnlmhr`: `[batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]`
122/// - `b_bnlmhr`: `[batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]`
123///
124/// # Returns
125/// - `cb_bnhLMLM`: `[batch, nchunks, nheads, chunk_len*mimo_rank, chunk_len*mimo_rank]`
126pub fn k2_ssd_bmm<B: Backend>(c_bnlmhr: Tensor<B, 6>, b_bnlmhr: Tensor<B, 6>) -> Tensor<B, 5> {
127    let [batch, nchunks, chunk_len, mimo_rank, nheads, state_rank] = c_bnlmhr.dims();
128
129    // Fuse R into chunk_len
130    let c_bnLMhr = c_bnlmhr.reshape([batch, nchunks, chunk_len * mimo_rank, nheads, state_rank]);
131    let b_bnLMhr = b_bnlmhr.reshape([batch, nchunks, chunk_len * mimo_rank, nheads, state_rank]);
132
133    let c_bnhLMr = c_bnLMhr.permute([0, 1, 3, 2, 4]);
134    let b_bnhrLM = b_bnLMhr.permute([0, 1, 3, 4, 2]);
135    let cb_bnhLMLM: Tensor<B, 5> = c_bnhLMr.matmul(b_bnhrLM);
136    assert_eq!(
137        [
138            batch,
139            nchunks,
140            nheads,
141            chunk_len * mimo_rank,
142            chunk_len * mimo_rank
143        ],
144        cb_bnhLMLM.dims()
145    );
146    cb_bnhLMLM
147}

149// ---------------------------------------------------------------------------
150// K3 — intra-chunk state (chunk-end state assuming zero initial state)
151// ---------------------------------------------------------------------------
152
153/// Compute the SSM state at the end of each chunk, assuming zero initial hidden state.
154///
155/// Uses the pre-scaled V tensor — no `dt·B` scaling is performed here.
156///
157/// # Arguments
158/// - `v_bnlmhp`: `[batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]` — pre-scaled V
159/// - `b_bnlmhr`: `[batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]`
160/// - `da_cumsum_bhnl`: `[batch, nheads, nchunks, chunk_len]`
161///
162/// # Returns
163/// - `intra_chunk_state_bnhpr`: `[batch, nchunks, nheads, per_head_dim, state_rank]`
164pub fn k3_ssd_chunk_state<B: Backend>(
165    v_bnlmhp: Tensor<B, 6>,
166    b_bnlmhr: Tensor<B, 6>,
167    da_cumsum_bhnl: Tensor<B, 4>,
168) -> Tensor<B, 5> {
169    let [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim] = v_bnlmhp.dims();
170    let [.., state_rank] = b_bnlmhr.dims();
171
172    // Fuse mimo_rank into chunk_len
173    let v_bnLMhp = v_bnlmhp.reshape([batch, nchunks, chunk_len * mimo_rank, nheads, per_head_dim]);
174    let b_bnLMhr = b_bnlmhr.reshape([batch, nchunks, chunk_len * mimo_rank, nheads, state_rank]);
175
176    // Decay from each fused position to end of chunk
177    let a_cumsum_last_bhn1 = da_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
178    // Expand base cumsum to fused
179    let a_cumsum_bhnLM = da_cumsum_bhnl
180        .unsqueeze_dim::<5>(4) // da_cumsum_bhnl1
181        .expand([batch, nheads, nchunks, chunk_len, mimo_rank]) // da_cumsum_bhnlm
182        .reshape([batch, nheads, nchunks, chunk_len * mimo_rank]); // da_cumsum_bhnLM
183    let decay_bhnLM = (a_cumsum_last_bhn1 - a_cumsum_bhnLM).exp();
184
185    // decay * V
186    let decay_bnLMh1 = decay_bhnLM
187        .permute([0, 2, 3, 1]) // decay_bnLMh
188        .unsqueeze_dim(4); // decay_bnLMh1
189    let decayed_v_bnLMhp = decay_bnLMh1 * v_bnLMhp;
190
191    // state = decayed_V^T @ B
192    let decayed_v_bnhpLM = decayed_v_bnLMhp.permute([0, 1, 3, 4, 2]);
193    let b_bnhLMr = b_bnLMhr.permute([0, 1, 3, 2, 4]);
194    let intra_chunk_state_bnhpr: Tensor<B, 5> = decayed_v_bnhpLM.matmul(b_bnhLMr);
195    assert_eq!(
196        [batch, nchunks, nheads, per_head_dim, state_rank],
197        intra_chunk_state_bnhpr.dims()
198    );
199    intra_chunk_state_bnhpr
200}

202// ---------------------------------------------------------------------------
203// K4 — inter-chunk state scan (sequential loop)
204// ---------------------------------------------------------------------------
205
206/// Propagate hidden state across chunk boundaries using a sequential scan.
207///
208/// This kernel is independent of MIMO rank — it operates on the `[nheads, per_head_dim, state_rank]` state
209/// which is already aggregated over ranks.
210///
211/// # Arguments
212/// - `intra_chunk_state_bnhpr`: `[batch, nchunks, nheads, per_head_dim, state_rank]`
213/// - `da_chunk_end_bhn`: `[batch, nheads, nchunks]` — total log-decay per chunk
214/// - `initial_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
215///
216/// # Returns
217/// - `chunk_input_state_bnhpr`: `[batch, nchunks, nheads, per_head_dim, state_rank]`
218/// - `final_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
219pub fn k4_ssd_state_passing<B: Backend>(
220    intra_chunk_state_bnhpr: Tensor<B, 5>,
221    da_chunk_end_bhn: Tensor<B, 3>,
222    initial_state_bhpr: Tensor<B, 4>,
223) -> (Tensor<B, 5>, Tensor<B, 4>) {
224    let [batch, nchunks, nheads, per_head_dim, state_rank] = intra_chunk_state_bnhpr.dims();
225
226    let mut running_state_bhpr = initial_state_bhpr;
227    assert_eq!(
228        [batch, nheads, per_head_dim, state_rank],
229        running_state_bhpr.dims()
230    );
231
232    let mut chunk_input_state_vec_bhpr = Vec::with_capacity(nchunks + 1);
233    chunk_input_state_vec_bhpr.push(running_state_bhpr.clone());
234
235    for i_chunk in 0..nchunks {
236        let intra_state_bhpr: Tensor<B, 4> = intra_chunk_state_bnhpr
237            .clone()
238            .slice(s![.., i_chunk, .., .., ..]) // intra_chunk_state_b1hpr
239            .squeeze_dim(1); // intra_state_bhpr
240
241        let decay_bhpr = da_chunk_end_bhn
242            .clone()
243            .slice(s![.., .., i_chunk]) // da_chunk_end_bh1
244            .unsqueeze_dim::<4>(3) // da_chunk_end_bh
245            .exp()
246            .expand([batch, nheads, per_head_dim, state_rank]); // decay_bhpr
247
248        // SSM recurrence: h[n] = decay * h[n-1] + s[n]
249        running_state_bhpr = decay_bhpr * running_state_bhpr + intra_state_bhpr;
250        chunk_input_state_vec_bhpr.push(running_state_bhpr.clone());
251    }
252
253    let final_state_bhpr = chunk_input_state_vec_bhpr.pop().unwrap();
254    assert_eq!(
255        [batch, nheads, per_head_dim, state_rank],
256        final_state_bhpr.dims()
257    );
258
259    let chunk_input_state_bnhpr = Tensor::stack(chunk_input_state_vec_bhpr, 1);
260    assert_eq!(
261        [batch, nchunks, nheads, per_head_dim, state_rank],
262        chunk_input_state_bnhpr.dims()
263    );
264
265    (chunk_input_state_bnhpr, final_state_bhpr)
266}

268// ---------------------------------------------------------------------------
269// K5 — MIMO chunk scan (Y_diag + Y_off)
270// ---------------------------------------------------------------------------
271
272/// Compute the chunk output by combining the intra-chunk (diagonal) and
273/// inter-chunk (off-diagonal) contributions.
274///
275/// The MIMO causal mask uses interleaved time-step ordering:
276/// `L_mimo[i,j] = exp(cumA[i//m] - cumA[j//m])` if `i//m >= j//m`, else 0.
277///
278/// No D skip is applied — the caller handles it.
279///
280/// # Arguments
281/// - `da_cumsum_bhnl`: `[batch, nheads, nchunks, chunk_len]` — base (not fused)
282/// - `v_bnlmhp`: `[batch, nchunks, chunk_len, R, nheads, per_head_dim]`
283/// - `c_bnlmhr`: `[batch, nchunks, chunk_len, R, nheads, state_rank]`
284/// - `cb_bnhLMLM`: `[batch, nchunks, nheads, L, L]` from K2
285/// - `chunk_input_state_bnhpr`: `[batch, nchunks, nheads, per_head_dim, state_rank]`
286///
287/// # Returns
288/// - `y_bnlmhp`: `[batch, nchunks, chunk_len, R, nheads, per_head_dim]`
289pub fn k5_ssd_chunk_scan<B: Backend>(
290    da_cumsum_bhnl: Tensor<B, 4>,
291    v_bnlmhp: Tensor<B, 6>,
292    c_bnlmhr: Tensor<B, 6>,
293    cb_bnhLMLM: Tensor<B, 5>,
294    chunk_input_state_bnhpr: Tensor<B, 5>,
295) -> Tensor<B, 6> {
296    let [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim] = v_bnlmhp.dims();
297    let [.., state_rank] = c_bnlmhr.dims();
298    let device = v_bnlmhp.device();
299
300    // Fuse mimo_rank into chunk_len
301    let v_bnLMhp = v_bnlmhp.reshape([batch, nchunks, chunk_len * mimo_rank, nheads, per_head_dim]);
302    let c_bnLMhr = c_bnlmhr.reshape([batch, nchunks, chunk_len * mimo_rank, nheads, state_rank]);
303
304    // Expand base da_cumsum to fused length: [b, nheads, n, l] → [b, nheads, n, L]
305    let da_cumsum_bhnLM = da_cumsum_bhnl
306        .unsqueeze_dim::<5>(4) // da_cumsum_bhnl1
307        .expand([batch, nheads, nchunks, chunk_len, mimo_rank]) // da_cumsum_bhnlm
308        .reshape([batch, nheads, nchunks, chunk_len * mimo_rank]); // da_cumsum_bhnLM
309
310    // ── BLUE (Y_off): exp(cumA[i]) · C[i] · h[n-1] ─────────────────────
311    let exp_da_bnhLMp = da_cumsum_bhnLM
312        .clone()
313        .exp()
314        .permute([0, 2, 1, 3]) // exp_da_bnhLM
315        .unsqueeze_dim::<5>(4) // // exp_da_bnhLM1
316        .expand([batch, nchunks, nheads, chunk_len * mimo_rank, per_head_dim]); // exp_da_bnhLMp
317
318    let c_bnhLMr = c_bnLMhr.permute([0, 1, 3, 2, 4]);
319    let chunk_input_state_bnhrp = chunk_input_state_bnhpr.permute([0, 1, 2, 4, 3]);
320    let ch_bnhLMp = c_bnhLMr.matmul(chunk_input_state_bnhrp);
321    let blue_bnhLMp = ch_bnhLMp * exp_da_bnhLMp;
322
323    // ── ORANGE (Y_diag): MIMO causal decay matrix · CB @ V ────────────────────
324    //
325    // MIMO pairwise decay: diff[i,j] = cumA[i] - cumA[j]
326    //                                = cumA_base[i//m] - cumA_base[j//m]
327    let da_cumsum_bnhLM = da_cumsum_bhnLM.permute([0, 2, 1, 3]);
328    let target_da_cumsum_bnhLMLM = da_cumsum_bnhLM
329        .clone()
330        .unsqueeze_dim::<5>(4) // da_cumsum_bnhLM1
331        .expand([
332            batch,
333            nchunks,
334            nheads,
335            chunk_len * mimo_rank,
336            chunk_len * mimo_rank,
337        ]);
338    let source_da_cumsum_bnhLMLM = da_cumsum_bnhLM
339        .unsqueeze_dim::<5>(3) // da_cumsum_bnh1LM
340        .expand([
341            batch,
342            nchunks,
343            nheads,
344            chunk_len * mimo_rank,
345            chunk_len * mimo_rank,
346        ]);
347    let diff_da_cumsum_bnhLMLM = target_da_cumsum_bnhLMLM - source_da_cumsum_bnhLMLM;
348
349    // MIMO causal neg-inf mask: −∞ where j//m > i//m (source strictly ahead of target in time).
350    // Build as interleaved expansion of the standard 2-dimensional upper-triangle mask.
351    let neg_inf_base_bnhll: Tensor<B, 5> =
352        Tensor::<B, 2>::full([chunk_len, chunk_len], f32::NEG_INFINITY, &device)
353            .triu(1) // [chunk_len, chunk_len]: -inf above diagonal
354            .unsqueeze_dims::<5>(&[0, 1, 2]) // neg_inf_base_111ll
355            .expand([batch, nchunks, nheads, chunk_len, chunk_len]); // neg_inf_base_bnhll
356    // Interleave-expand
357    let neg_inf_bnhLMLM: Tensor<B, 5> = neg_inf_base_bnhll
358        .unsqueeze_dim::<6>(4) // neg_inf_base_bnhl1l
359        .expand([batch, nchunks, nheads, chunk_len, mimo_rank, chunk_len]) // neg_inf_base_bnhlml
360        .reshape([batch, nchunks, nheads, chunk_len * mimo_rank, chunk_len]) // neg_inf_base_bnhLMl
361        .unsqueeze_dim::<6>(5) // neg_inf_base_bnhLMl1
362        .expand([
363            batch,
364            nchunks,
365            nheads,
366            chunk_len * mimo_rank,
367            chunk_len,
368            mimo_rank,
369        ]) // neg_inf_base_bnhLMlm
370        .reshape([
371            batch,
372            nchunks,
373            nheads,
374            chunk_len * mimo_rank,
375            chunk_len * mimo_rank,
376        ]); // neg_inf_bnhLMLM
377
378    let decay_bnhLMLM = (diff_da_cumsum_bnhLMLM + neg_inf_bnhLMLM).exp();
379
380    let v_bnhLMp = v_bnLMhp.permute([0, 1, 3, 2, 4]);
381    let orange_bnhLMp = (cb_bnhLMLM * decay_bnhLMLM).matmul(v_bnhLMp);
382
383    // ── Combine and reshape ────────────────────────────────────────────────────
384    let y_bnlmhp = (blue_bnhLMp + orange_bnhLMp)
385        .permute([0, 1, 3, 2, 4]) // y_bnLMhp
386        .reshape([batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]); // y_bnlmhp
387
388    y_bnlmhp
389}
