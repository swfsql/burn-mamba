10use burn::prelude::*;
11use burn::tensor::{TensorPrimitive, ops::FloatTensor};
12use ssd::serial_recalculated::{
13    Mamba3SingleSsdBackendExt,
14    combined_backward::{self, CombinedSingleSsdGrads},

17impl<B: Backend + Mamba3SingleSsdBackendExt, C: CheckpointStrategy> Mamba3SingleSsdBackendExt
18    for Autodiff<B, C>
19{
20    /// Memory-efficient combined forward+backward for the Mamba-3 MIMO
21    /// Single-SSD.
22    ///
23    /// The two outputs (`y_bnlmhp`, `final_state_bhpr`) are flattened and
24    /// concatenated into a single 1-D tracked tensor so one `Backward<B, 7>`
25    /// node covers both. The seven differentiable inputs are `v, da, b, c,
26    /// gamma, scale, initial_state`.
27    fn single_ssd_serial_recalculated(
28        v_bnlmhp: FloatTensor<Self>,
29        da_bnlh: FloatTensor<Self>,
30        b_bnlmhr: FloatTensor<Self>,
31        c_bnlmhr: FloatTensor<Self>,
32        gamma_bnlh: FloatTensor<Self>,
33        scale_bnlh: FloatTensor<Self>,
34        initial_state_bhpr: FloatTensor<Self>,
35    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
36        #[derive(Debug)]
37        struct CombinedKernelsBackward;
38
39        #[derive(Clone, Debug)]
40        struct State<B: Backend> {
41            v_bnlmhp: FloatTensor<B>,
42            da_bnlh: FloatTensor<B>,
43            b_bnlmhr: FloatTensor<B>,
44            c_bnlmhr: FloatTensor<B>,
45            gamma_bnlh: FloatTensor<B>,
46            scale_bnlh: FloatTensor<B>,
47            initial_state_bhpr: FloatTensor<B>,
48            flat_len_y_BNLMHP: usize,
49            flat_len_final_state_BHPR: usize,
50            shape_v_bnlmhp: [usize; 6],
51            shape_da_bnlh: [usize; 4],
52            shape_b_bnlmhr: [usize; 6],
53            shape_c_bnlmhr: [usize; 6],
54            shape_gamma_bnlh: [usize; 4],
55            shape_scale_bnlh: [usize; 4],
56            shape_initial_state_bhpr: [usize; 4],
57            shape_y_bnlmhp: [usize; 6],
58            shape_final_state_bhpr: [usize; 4],
59        }
60
61        impl<B: Backend + Mamba3SingleSsdBackendExt> Backward<B, 7> for CombinedKernelsBackward {
62            type State = State<B>;
63
64            fn backward(
65                self,
66                ops: Ops<Self::State, 7>,
67                grads: &mut Gradients,
68                _checkpointer: &mut Checkpointer,
69            ) {
70                let [
71                    node_v_bnlmhp,
72                    node_da_bnlh,
73                    node_b_bnlmhr,
74                    node_c_bnlmhr,
75                    node_gamma_bnlh,
76                    node_scale_bnlh,
77                    node_initial_state_bhpr,
78                ] = ops.parents;
79
80                let d_combined: Tensor<B, 1> =
81                    Tensor::from_primitive(TensorPrimitive::Float(grads.consume::<B>(&ops.node)));
82
83                let State {
84                    v_bnlmhp,
85                    da_bnlh,
86                    b_bnlmhr,
87                    c_bnlmhr,
88                    gamma_bnlh,
89                    scale_bnlh,
90                    initial_state_bhpr,
91                    flat_len_y_BNLMHP,
92                    flat_len_final_state_BHPR,
93                    shape_v_bnlmhp,
94                    shape_da_bnlh,
95                    shape_b_bnlmhr,
96                    shape_c_bnlmhr,
97                    shape_gamma_bnlh,
98                    shape_scale_bnlh,
99                    shape_initial_state_bhpr,
100                    shape_y_bnlmhp,
101                    shape_final_state_bhpr,
102                } = ops.state;
103
104                use crate::utils::primitive::mk;
105                let v_bnlmhp = mk::<_, 6>(v_bnlmhp).reshape(shape_v_bnlmhp);
106                let da_bnlh = mk::<_, 4>(da_bnlh).reshape(shape_da_bnlh);
107                let b_bnlmhr = mk::<_, 6>(b_bnlmhr).reshape(shape_b_bnlmhr);
108                let c_bnlmhr = mk::<_, 6>(c_bnlmhr).reshape(shape_c_bnlmhr);
109                let gamma_bnlh = mk::<_, 4>(gamma_bnlh).reshape(shape_gamma_bnlh);
110                let scale_bnlh = mk::<_, 4>(scale_bnlh).reshape(shape_scale_bnlh);
111                let initial_state_bhpr =
112                    mk::<_, 4>(initial_state_bhpr).reshape(shape_initial_state_bhpr);
113
114                let (d_y_bnlmhp, d_final_state_bhpr) = crate::utils::combined_grad::unflatten_pair(
115                    d_combined,
116                    flat_len_y_BNLMHP,
117                    flat_len_final_state_BHPR,
118                    shape_y_bnlmhp,
119                    shape_final_state_bhpr,
120                );
121
122                let CombinedSingleSsdGrads {
123                    d_v_bnlmhp,
124                    d_da_bnlh,
125                    d_b_bnlmhr,
126                    d_c_bnlmhr,
127                    d_gamma_bnlh,
128                    d_scale_bnlh,
129                    d_initial_state_bhpr,
130                } = combined_backward::combined_backward(
131                    d_y_bnlmhp,
132                    d_final_state_bhpr,
133                    v_bnlmhp,
134                    da_bnlh,
135                    b_bnlmhr,
136                    c_bnlmhr,
137                    gamma_bnlh,
138                    scale_bnlh,
139                    initial_state_bhpr,
140                );
141
142                if let Some(n) = node_v_bnlmhp {
143                    grads.register::<B>(n.id, d_v_bnlmhp.into_primitive().tensor());
144                }
145                if let Some(n) = node_da_bnlh {
146                    grads.register::<B>(n.id, d_da_bnlh.into_primitive().tensor());
147                }
148                if let Some(n) = node_b_bnlmhr {
149                    grads.register::<B>(n.id, d_b_bnlmhr.into_primitive().tensor());
150                }
151                if let Some(n) = node_c_bnlmhr {
152                    grads.register::<B>(n.id, d_c_bnlmhr.into_primitive().tensor());
153                }
154                if let Some(n) = node_gamma_bnlh {
155                    grads.register::<B>(n.id, d_gamma_bnlh.into_primitive().tensor());
156                }
157                if let Some(n) = node_scale_bnlh {
158                    grads.register::<B>(n.id, d_scale_bnlh.into_primitive().tensor());
159                }
160                if let Some(n) = node_initial_state_bhpr {
161                    grads.register::<B>(n.id, d_initial_state_bhpr.into_primitive().tensor());
162                }
163            }
164        }
165
166        use burn::tensor::TensorMetadata;
167        let [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim] =
168            v_bnlmhp.primitive.shape().dims();
169        let [.., state_rank] = b_bnlmhr.primitive.shape().dims::<6>();
170
171        let flat_len_y_BNLMHP = batch * nchunks * chunk_len * mimo_rank * nheads * per_head_dim;
172        let flat_len_final_state_BHPR = batch * nheads * per_head_dim * state_rank;
173
174        let shape_v_bnlmhp: [usize; 6] =
175            [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim];
176        let shape_da_bnlh: [usize; 4] = [batch, nchunks, chunk_len, nheads];
177        let shape_b_bnlmhr: [usize; 6] = [batch, nchunks, chunk_len, mimo_rank, nheads, state_rank];
178        let shape_c_bnlmhr: [usize; 6] = [batch, nchunks, chunk_len, mimo_rank, nheads, state_rank];
179        let shape_gamma_bnlh: [usize; 4] = [batch, nchunks, chunk_len, nheads];
180        let shape_scale_bnlh: [usize; 4] = [batch, nchunks, chunk_len, nheads];
181        let shape_initial_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
182        let shape_y_bnlmhp: [usize; 6] =
183            [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim];
184        let shape_final_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
185
186        match CombinedKernelsBackward
187            .prepare::<C>([
188                v_bnlmhp.node.clone(),
189                da_bnlh.node.clone(),
190                b_bnlmhr.node.clone(),
191                c_bnlmhr.node.clone(),
192                gamma_bnlh.node.clone(),
193                scale_bnlh.node.clone(),
194                initial_state_bhpr.node.clone(),
195            ])
196            .compute_bound()
197            .stateful()
198        {
199            OpsKind::Tracked(prep) => {
200                let (prim_y_bnlmhp, prim_final_state_bhpr) = B::single_ssd_serial_recalculated(
201                    v_bnlmhp.primitive.clone(),
202                    da_bnlh.primitive.clone(),
203                    b_bnlmhr.primitive.clone(),
204                    c_bnlmhr.primitive.clone(),
205                    gamma_bnlh.primitive.clone(),
206                    scale_bnlh.primitive.clone(),
207                    initial_state_bhpr.primitive.clone(),
208                );
209
210                let (combined, _, _) = crate::utils::combined_grad::flatten_pair(
211                    crate::utils::primitive::mk::<B, 6>(prim_y_bnlmhp),
212                    crate::utils::primitive::mk::<B, 4>(prim_final_state_bhpr),
213                );
214
215                let state = State {
216                    v_bnlmhp: v_bnlmhp.primitive.clone(),
217                    da_bnlh: da_bnlh.primitive.clone(),
218                    b_bnlmhr: b_bnlmhr.primitive.clone(),
219                    c_bnlmhr: c_bnlmhr.primitive.clone(),
220                    gamma_bnlh: gamma_bnlh.primitive.clone(),
221                    scale_bnlh: scale_bnlh.primitive.clone(),
222                    initial_state_bhpr: initial_state_bhpr.primitive.clone(),
223                    flat_len_y_BNLMHP,
224                    flat_len_final_state_BHPR,
225                    shape_v_bnlmhp,
226                    shape_da_bnlh,
227                    shape_b_bnlmhr,
228                    shape_c_bnlmhr,
229                    shape_gamma_bnlh,
230                    shape_scale_bnlh,
231                    shape_initial_state_bhpr,
232                    shape_y_bnlmhp,
233                    shape_final_state_bhpr,
234                };
235                let tracked_combined: FloatTensor<Autodiff<B, C>> =
236                    prep.finish(state, combined.into_primitive().tensor());
237
238                let (tracked_y_bnlmhp, tracked_final_state_bhpr) =
239                    crate::utils::combined_grad::unflatten_pair::<Autodiff<B, C>, 6, 4>(
240                        crate::utils::primitive::mk(tracked_combined),
241                        flat_len_y_BNLMHP,
242                        flat_len_final_state_BHPR,
243                        shape_y_bnlmhp,
244                        shape_final_state_bhpr,
245                    );
246
247                (
248                    tracked_y_bnlmhp.into_primitive().tensor(),
249                    tracked_final_state_bhpr.into_primitive().tensor(),
250                )
251            }
252
253            OpsKind::UnTracked(prep) => {
254                let (prim_y_bnlmhp, prim_final_state_bhpr) = B::single_ssd_serial_recalculated(
255                    v_bnlmhp.primitive,
256                    da_bnlh.primitive,
257                    b_bnlmhr.primitive,
258                    c_bnlmhr.primitive,
259                    gamma_bnlh.primitive,
260                    scale_bnlh.primitive,
261                    initial_state_bhpr.primitive,
262                );
263
264                let (combined, _, _) = crate::utils::combined_grad::flatten_pair(
265                    crate::utils::primitive::mk::<B, 6>(prim_y_bnlmhp),
266                    crate::utils::primitive::mk::<B, 4>(prim_final_state_bhpr),
267                );
268
269                let tracked_combined: FloatTensor<Autodiff<B, C>> =
270                    prep.finish(combined.into_primitive().tensor());
271
272                let (tracked_y_bnlmhp, tracked_final_state_bhpr) =
273                    crate::utils::combined_grad::unflatten_pair::<Autodiff<B, C>, 6, 4>(
274                        crate::utils::primitive::mk(tracked_combined),
275                        flat_len_y_BNLMHP,
276                        flat_len_final_state_BHPR,
277                        shape_y_bnlmhp,
278                        shape_final_state_bhpr,
279                    );
280
281                (
282                    tracked_y_bnlmhp.into_primitive().tensor(),
283                    tracked_final_state_bhpr.into_primitive().tensor(),
284                )
285            }
286        }
287    }
