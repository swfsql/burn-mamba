1//! # Serial SSD with a custom, memory-efficient backward (Mamba-3 single-SSD)
2//!
3//! The `SerialRecalculated` path for the single-SSD pathway.  The forward is the
4//! same serial scan as [`super::super::serial`], routed through the
5//! [`Mamba3SingleSsdBackendExt`] trait so `Autodiff` backends substitute a
6//! custom backward that recomputes per-chunk intermediates rather than storing
7//! them (see [`super::backward`] / [`super::combined_backward`]).  Unlike the
8//! double-SSD form, the kernels here apply the trapezoid `gamma`/`scale` and the
9//! boundary-β seed internally, so the backward also returns `d_gamma`/`d_scale`.
10
11#![allow(non_snake_case)]
12
13use crate::mamba3::single_ssd::prelude::*;
14use crate::mamba3::single_ssd::ssd;
15use crate::utils::primitive::mk;
16use burn::prelude::*;
17use burn::tensor::{Tensor, TensorPrimitive, ops::FloatTensor};
18use ssd::serial;

20impl<B: Backend + Mamba3SingleSsdBackendExt> Mamba3SingleSsdInput<B> {
21    /// MIMO-first single-ssd form Serial SSD with recalculated backward.
22    ///
23    /// Delegates the full K1–K5 (single-ssd) computation to
24    /// [`Mamba3SingleSsdBackendExt::single_ssd_serial_recalculated`], which can provide
25    /// a memory-efficient custom backward for supported backends (the Autodiff
26    /// wrapper) and falls back to the standard K1–K5 forward on others.
27    ///
28    /// # Returns
29    /// - `y_bnlmhp`:         `[batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]`
30    /// - `final_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
31    pub fn single_ssd_serial_recalculated(self) -> (Tensor<B, 6>, Tensor<B, 4>) {
32        let input = self;
33        input.sanity();
34        assert!(
35            input.init_state_hpr.is_none(),
36            "init_state_hpr not yet implemented for single_ssd_serial_recalculated"
37        );
38
39        let (y_bnlmhp, final_state_bhpr) =
40            <B as Mamba3SingleSsdBackendExt>::single_ssd_serial_recalculated(
41                input.v_bnlmhp.into_primitive().tensor(),
42                input.da_bnlh.into_primitive().tensor(),
43                input.b_bnlmhr.into_primitive().tensor(),
44                input.c_bnlmhr.into_primitive().tensor(),
45                input.gamma_bnlh.into_primitive().tensor(),
46                input.scale_bnlh.into_primitive().tensor(),
47                input.initial_state_bhpr.into_primitive().tensor(),
48            );
49        let y_bnlmhp = Tensor::from_primitive(TensorPrimitive::Float(y_bnlmhp));
50        let final_state_bhpr = Tensor::from_primitive(TensorPrimitive::Float(final_state_bhpr));
51        (y_bnlmhp, final_state_bhpr)
52    }
53}

55/// Extends the backend for the memory-efficient single-ssd form serial SSD.
56///
57/// The default implementation runs K1–K5 using standard tensor operations,
58/// reusing K1/K2/K4 from [`crate::mamba3::double_ssd::ssd::serial`] (mode-agnostic) and the
59/// single-ssd form K5 from [`crate::mamba3::single_ssd::ssd::serial`]. Backends that
60/// support a custom memory-efficient backward (the Autodiff wrapper) override
61/// this to recompute forward intermediates during backward instead of saving
62/// them.
63pub trait Mamba3SingleSsdBackendExt: burn::tensor::backend::Backend {
64    /// Memory-efficient MIMO single-ssd form serial SSD.
65    ///
66    /// # Arguments
67    /// - `v_bnlmhp`:           `[batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]`
68    /// - `da_bnlh`:            `[batch, nchunks, chunk_len, nheads]` — pre-combined Δ·A
69    /// - `b_bnlmhr`:           `[batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]`
70    /// - `c_bnlmhr`:           `[batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]`
71    /// - `gamma_bnlh`:         `[batch, nchunks, chunk_len, nheads]` — `γₜ = λₜ Δₜ`
72    /// - `scale_bnlh`:         `[batch, nchunks, chunk_len, nheads]` — `scaleₜ = γₜ + (1−λₜ₊₁)Δₜ₊₁`
73    /// - `initial_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
74    ///
75    /// # Returns
76    /// - `y_bnlmhp`:         `[batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]`
77    /// - `final_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
78    fn single_ssd_serial_recalculated(
79        v_bnlmhp: FloatTensor<Self>,
80        da_bnlh: FloatTensor<Self>,
81        b_bnlmhr: FloatTensor<Self>,
82        c_bnlmhr: FloatTensor<Self>,
83        gamma_bnlh: FloatTensor<Self>,
84        scale_bnlh: FloatTensor<Self>,
85        initial_state_bhpr: FloatTensor<Self>,
86    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
87        // Default impl: replicate the single-ssd form K1–K5 (see `single_ssd_serial`).
88        let v_bnlmhp: Tensor<Self, 6> = mk(v_bnlmhp);
89        let da_bnlh: Tensor<Self, 4> = mk(da_bnlh);
90        let b_bnlmhr: Tensor<Self, 6> = mk(b_bnlmhr);
91        let c_bnlmhr: Tensor<Self, 6> = mk(c_bnlmhr);
92        let gamma_bnlh: Tensor<Self, 4> = mk(gamma_bnlh);
93        let scale_bnlh: Tensor<Self, 4> = mk(scale_bnlh);
94        let initial_state_bhpr: Tensor<Self, 4> = mk(initial_state_bhpr);
95
96        // K1 — chunk cumulative decay.
97        let (da_cumsum_bhnl, da_chunk_end_bhn) = serial::k1_ssd_chunk_cumsum(da_bnlh);
98        // K2 — CB matrix on unscaled B/C.
99        let cb_bnhLMLM = serial::k2_ssd_bmm(c_bnlmhr.clone(), b_bnlmhr.clone());
100        // K3 — chunk state on K_scaled = scaleₜ · B.
101        let scale_bnlh11 = scale_bnlh.clone().unsqueeze_dims::<6>(&[3, 5]);
102        let k_scaled_bnlmhr = b_bnlmhr.clone() * scale_bnlh11;
103        let intra_chunk_state_bnhpr =
104            serial::k3_ssd_chunk_state(v_bnlmhp.clone(), k_scaled_bnlmhr, da_cumsum_bhnl.clone());
105        // K4 — sequential state passing.
106        let (chunk_input_state_bnhpr, final_state_bhpr) = serial::k4_ssd_state_passing(
107            intra_chunk_state_bnhpr,
108            da_chunk_end_bhn,
109            initial_state_bhpr,
110        );
111        // K5 — single-ssd form chunk scan.
112        let y_bnlmhp = serial::k5_single_ssd_chunk_scan(
113            da_cumsum_bhnl,
114            v_bnlmhp,
115            c_bnlmhr,
116            b_bnlmhr,
117            cb_bnhLMLM,
118            gamma_bnlh,
119            scale_bnlh,
120            chunk_input_state_bnhpr,
121        );
122
123        let y_bnlmhp = y_bnlmhp.into_primitive().tensor();
124        let final_state_bhpr = final_state_bhpr.into_primitive().tensor();
125        (y_bnlmhp, final_state_bhpr)
126    }
127}
