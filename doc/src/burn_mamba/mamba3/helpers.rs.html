1//! Shared helpers used by both [`Mamba3::forward`](super::mamba3::Mamba3::forward)
2//! and [`Mamba3::step`](super::mamba3::Mamba3::step). They isolate three blocks
3//! that previously appeared in both methods at different ranks:
4//!
5//! 1. Trapezoidal discretisation: `dt`, `α`, `β`, `γ`, `da`.
6//! 2. QK-norm + GQA expansion + per-(head, mimo-rank) bias on B / C.
7//! 3. MIMO `V` construction: broadcast-multiply `x` by `mimo_x_hmp`.
8//!
9//! Each helper is generic over the rank `D` of the data tensors so a single
10//! definition serves both the sequence-aware (`forward`) and single-token
11//! (`step`) code paths.
12

19/// Output of [`trapezoidal_coefficients`].
20///
21/// All tensors share the rank `D` of the inputs.
22pub struct TrapezoidCoeffs<const D: usize> {
23    /// `Δₜ = softplus(dd_dt + dt_bias)`, clamped.
24    pub dt: Tensor<D>,
25    /// `Δₜ · Aₜ` (negative; the log-decay).
26    pub da: Tensor<D>,
27    /// `αₜ = exp(Δₜ · Aₜ) ∈ (0, 1]` — decay.
28    pub alpha: Tensor<D>,
29    /// `βₜ = (1 − λₜ) · Δₜ · αₜ` — left-endpoint weight.
30    pub beta: Tensor<D>,
31    /// `γₜ = λₜ · Δₜ` — right-endpoint weight.
32    pub gamma: Tensor<D>,
33}

35/// Compute the trapezoidal discretisation coefficients from the raw
36/// (data-dependent) projections. See the top-of-`mamba3.rs` docs for the
37/// formulas.
38///
39/// All four data tensors share rank `D` and have `nheads` as the last dim.
40/// `dt_bias_h` is broadcast to match.
41pub fn trapezoidal_coefficients<const D: usize>(
42    dd_dt: Tensor<D>,
43    dd_a_raw: Tensor<D>,
44    lambda_raw: Tensor<D>,
45    dt_bias_h: Tensor<1>,
46    dt_limit: (f64, f64),
47    a_floor: f64,
48) -> TrapezoidCoeffs<D> {
49    // Broadcast dt_bias_h [nheads] → [1, ..., 1, nheads] so the addition aligns
50    // on the last dim regardless of leading shape.
51    let dt_bias_broadcast = dt_bias_h.unsqueeze::<D>();
52    let dt = softplus(dd_dt + dt_bias_broadcast).clamp(dt_limit.0, dt_limit.1);
53    let a = -softplus(dd_a_raw).clamp(f64::NEG_INFINITY, -a_floor);
54    let da = dt.clone() * a;
55    let lambda = burn::tensor::activation::sigmoid(lambda_raw);
56    let alpha = da.clone().exp();
57    let beta = (-lambda.clone() + 1.0) * dt.clone() * alpha.clone();
58    let gamma = lambda * dt.clone();
59    TrapezoidCoeffs {
60        dt,
61        da,
62        alpha,
63        beta,
64        gamma,
65    }
66}

68/// QK-Norm → GQA-expand groups→heads → add per-(head, mimo-rank) bias.
69///
70/// The input is the raw B/C projection already reshaped to expose the group
71/// dim, with last dim = `state_rank`. The output replaces the group dim with
72/// the head dim, leaving the last dim untouched.
73///
74/// `DP1 = D + 1` (required by [`gqa_expand_to_heads`]'s intermediate rank).
75pub fn qk_norm_expand_bias<const D: usize, const DP1: usize>(
76    raw_mgr: Tensor<D>,
77    norm: &RmsNorm,
78    bias_hmr: Tensor<3>,
79    group_dim: usize,
80    nheads: usize,
81) -> Tensor<D> {
82    // RmsNorm operates on the last dim only, so the leading shape passes through.
83    let normed = norm.forward(raw_mgr);
84    let expanded = gqa_expand_to_heads::<D, DP1>(normed, group_dim, nheads);
85    // Broadcast bias [nheads, mimo_rank, state_rank] → [1, ..., 1, mimo_rank, nheads, state_rank].
86    let bias = bias_hmr.permute([1, 0, 2]).unsqueeze::<D>();
87    expanded + bias
88}

90/// Build the MIMO value tensor `v = x ⊙ mimo_x` with broadcasting.
91///
92/// Inserts a `mimo_rank` axis at `insert_dim`. When `mimo_x_hmp` is `None`
93/// (SISO), the inserted axis has size 1 and `x` is passed through; otherwise
94/// broadcasting fills the inserted axis to size `mimo_rank`.
95///
96/// `DP1 = D + 1`.
97pub fn build_v_with_mimo<const D: usize, const DP1: usize>(
98    x: Tensor<D>,
99    mimo_x_hmp: Option<&Tensor<3>>,
100    insert_dim: usize,
101) -> Tensor<DP1> {
102    let x_with_rank_axis = x.unsqueeze_dim::<DP1>(insert_dim);
103    match mimo_x_hmp {
104        None => x_with_rank_axis,
105        Some(mimo_x_hmp) => {
106            // mimo_x_hmp [nheads, mimo_rank, per_head_dim] → permute to
107            // [mimo_rank, nheads, per_head_dim] → unsqueeze leading 1s. The
108            // result broadcasts against `x_with_rank_axis` over (batch, seq, …).
109            let mimo_x_broadcast = mimo_x_hmp.clone().permute([1, 0, 2]).unsqueeze::<DP1>();
110            x_with_rank_axis * mimo_x_broadcast
111        }
112    }
113}
