1//! Strategies for merging the straight (→) and reverse (←) passes of a
2//! bidirectional Mamba-3 layer pair into a single `[batch, sequence, d_model]`
3//! output.
4
5use burn::backend::Backend;
6use burn::nn::{Linear, LinearConfig};
7use burn::prelude::*;
8
9/// A zero-parameter placeholder used where the `Module` derive expects a value
10/// (the `Mean` merge carries no weights).
11#[derive(Module, Debug)]
12pub struct NoOp;
13
14/// How the two directions of a bidirectional pair are combined.
15#[allow(clippy::large_enum_variant)]
16#[derive(Module, Debug)]
17pub enum OutputMerge {
18    /// Element-wise average of the two directions (no parameters).
19    Mean(NoOp),
20    /// Concatenate along the feature axis and project back down with a learnable
21    /// `[2 · d_model, d_model]` linear layer.
22    CatLinear(Linear),
23}
24
25impl OutputMerge {
26    /// Merge the two directional outputs.
27    ///
28    /// # Shapes
29    ///   - Input straight `[batch, sequence, d_model]`
30    ///   - Input reverse `[batch, sequence, d_model]`
31    ///   - Output `[batch, sequence, d_model]`
32    pub fn forward(&self, straight: Tensor<3>, reverse: Tensor<3>) -> Tensor<3> {
33        let [batch_size, sequence_len, d_model] = straight.dims();
34        assert_eq!(straight.dims(), reverse.dims());
35        match self {
36            OutputMerge::Mean(_noop) => (straight + reverse) * 0.5,
37            OutputMerge::CatLinear(proj) => {
38                let cat = Tensor::cat([straight, reverse].to_vec(), 2);
39                assert_eq!([batch_size, sequence_len, 2 * d_model], cat.dims());
40                let merged = proj.forward(cat);
41                assert_eq!([batch_size, sequence_len, d_model], merged.dims());
42                merged
43            }
44        }
45    }
46}
47
48/// Configuration / factory for [`OutputMerge`].
49#[derive(Config, Debug)]
50pub enum OutputMergeConfig {
51    /// Build an [`OutputMerge::Mean`].
52    Mean,
53    /// Build an [`OutputMerge::CatLinear`].
54    CatLinear,
55}
56
57impl OutputMergeConfig {
58    /// A vector of `n_real_layers / 2` [`Self::Mean`] configs (one per pair).
59    pub fn mean(n_real_layers: usize) -> Vec<Self> {
60        vec![Self::Mean; n_real_layers / 2]
61    }
62    /// A vector of `n_real_layers / 2` [`Self::CatLinear`] configs (one per pair).
63    pub fn cat_linear(n_real_layers: usize) -> Vec<Self> {
64        vec![Self::CatLinear; n_real_layers / 2]
65    }
66
67    /// Allocate the merge module on `device` for the given `d_model`.
68    pub fn init(&self, d_model: usize, device: &Device) -> OutputMerge {
69        match self {
70            OutputMergeConfig::Mean => OutputMerge::Mean(NoOp),
71            OutputMergeConfig::CatLinear => {
72                let cat_linear = LinearConfig::new(d_model * 2, d_model).init(device);
73                OutputMerge::CatLinear(cat_linear)
74            }
75        }
76    }
77}
