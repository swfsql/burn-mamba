1//! # Mamba-3 Inference Caches
2//!
3//! During autoregressive (token-by-token) generation, three pieces of state
4//! must be preserved between calls:
5//!
6//! 1. **SSM hidden state** — `hₜ ∈ ℝ^{per_head_dim×state_rank}` per head, compressed context.
7//! 2. **Previous K state** — `Bₜ₋₁` per rank `[batch, mimo_rank, nheads, state_rank]`,
8//!    needed for the β term of the (double-ssd) trapezoidal recurrence.
9//! 3. **Previous V state** — `xₜ₋₁` per head `[batch, nheads, per_head_dim]`,
10//!    paired with k_state to reconstruct β Bₜ₋₁ ⊗ xₜ₋₁.
11//! 4. **Cumulative RoPE angle** — the accumulated rotation angle up to position
12//!    `t`, needed to correctly continue data-dependent rotary embeddings.
13//!
14//! Note: Mamba-3 has **no conv cache** (the short 1-dimensional convolution present in
15//! Mamba-3 is removed; its role is absorbed by the trapezoidal discretization
16//! and the learnable B/C biases).
17

24// ---------------------------------------------------------------------------
25// Mamba3DoubleSsdCaches  (one cache entry per layer)
26// ---------------------------------------------------------------------------
27
28/// A collection of per-layer caches for a complete Mamba-3 network.
29#[derive(Module, Debug)]
30pub struct Mamba3DoubleSsdCaches {
31    /// Per-layer caches.  Length equals the number of virtual layers.
32    pub caches: Vec<Mamba3DoubleSsdCache>,
33}
34
35/// Configuration / factory for [`Mamba3DoubleSsdCaches`].
36#[derive(Config, Debug)]
37pub struct Mamba3DoubleSsdCachesConfig {
38    /// Number of cache slots (= number of virtual layers).
39    pub n_real_caches: usize,
40
41    /// Shared configuration that determines the shape of each cache.
42    pub cache: Mamba3DoubleSsdCacheConfig,
43}
44
45impl Mamba3DoubleSsdCachesConfig {
46    /// Convenience constructor from a block config.
47    pub fn new_from_block_config(
48        n_real_caches: usize,
49        batch: usize,
50        block_config: Mamba3Config,
51    ) -> Self {
52        Self {
53            n_real_caches,
54            cache: Mamba3DoubleSsdCacheConfig::new_from_block_config(batch, block_config),
55        }
56    }
57
58    /// Allocate all cache tensors (zero-initialised) on `device`.
59    pub fn init(&self, device: &Device) -> Mamba3DoubleSsdCaches {
60        let caches = (0..self.n_real_caches)
61            .map(|_| self.cache.clone().init(device))
62            .collect();
63        Mamba3DoubleSsdCaches { caches }
64    }
65}

67// ---------------------------------------------------------------------------
68// Mamba3DoubleSsdCache  (state for a single layer)
69// ---------------------------------------------------------------------------
70
71/// The mutable state carried between decoding steps for a **single** Mamba-3 layer.
72///
73/// All tensors are updated at every call to [`crate::mamba3::mamba3::Mamba3::step`].
74#[derive(Module, Debug)]
75pub struct Mamba3DoubleSsdCache {
76    /// **SSM hidden state** `hₜ`.
77    ///
78    /// Updated via the (double-ssd) trapezoidal recurrence:
79    /// `hₜ = αₜ hₜ₋₁ + βₜ (sumₘ Kₜ₋₁[m] ⊗ (Vₜ₋₁ * mimo_x[m])) + γₜ (sumₘ Bₜ[m] ⊗ (xₜ * mimo_x[m]))`
80    ///
81    /// Shape: `[batch, nheads, per_head_dim, state_rank]`
82    pub ssm_bhpr: Tensor<4>,
83
84    /// **Previous token's B per mimo rank** = `Bₜ₋₁[m]`.
85    ///
86    /// Used to reconstruct the β term: `β * sum_r Bₜ₋₁[m] ⊗ (xₜ₋₁ * mimo_x[m])`.
87    /// For SISO (mimo_rank=1) this is shape `[batch, 1, nheads, state_rank]`.
88    ///
89    /// Shape: `[batch, mimo_rank, nheads, state_rank]`
90    pub k_state_bmhr: Tensor<4>,
91
92    /// **Previous token's x** = `xₜ₋₁`.
93    ///
94    /// Combined with `k_state_bmhr` and `mimo_x` to produce the β term.
95    ///
96    /// Shape: `[batch, nheads, per_head_dim]`
97    pub v_state_bhp: Tensor<3>,
98
99    /// **Cumulative data-dependent rotation** up to the current position
100    /// ([`RotationState`]): the abelian RoPE angle for
101    /// [`Complex2D`](crate::mamba3::rotation::RotationKind::Complex2D) (each step
102    /// `cum_angleₜ = cum_angleₜ₋₁ + Δₜ · tanh(θₜ) · π`), or the cumulative unit
103    /// quaternion for [`Quaternion4D`](crate::mamba3::rotation::RotationKind::Quaternion4D).
104    ///
105    /// Starts at the identity for fresh sequences; continued across calls for
106    /// streaming.
107    pub rotation: RotationState,
108}
109
110impl Mamba3DoubleSsdCache {
111    /// Run the [`NaN`/`Inf` guards](crate::utils::sanity) on every cached tensor.
112    pub fn sanity(&self) {
113        san(&self.ssm_bhpr);
114        san(&self.k_state_bmhr);
115        san(&self.v_state_bhp);
116        self.rotation.sanity();
117    }
118}

120/// Configuration / factory for a single [`Mamba3DoubleSsdCache`].
121#[derive(Config, Debug)]
122pub struct Mamba3DoubleSsdCacheConfig {
123    /// Batch size.
124    pub batch: usize,
125
126    /// State rank.
127    #[config(default = 128)]
128    pub state_rank: usize,
129
130    /// Head dimension per_head_dim.
131    #[config(default = 64)]
132    pub per_head_dim: usize,
133
134    /// Number of SSM heads.
135    pub nheads: usize,
136
137    /// MIMO rank.  1 = SISO.
138    #[config(default = 1)]
139    pub mimo_rank: usize,
140
141    /// Number of RoPE angle pairs = `rope_dim / 2` = `(state_rank * rope_fraction) / 2`
142    /// (rounded down to even via `Mamba3Config::rope_dim`).
143    pub num_rope_angles: usize,
144
145    /// Which positional rotation the block uses ([`RotationKind`]); selects the
146    /// accumulator variant — [`RotationState::Quaternion`] for
147    /// [`RotationKind::Quaternion4D`], else [`RotationState::Angle`].
148    #[config(default = "crate::mamba3::rotation::RotationKind::Complex2D")]
149    pub rotation: RotationKind,
150
151    /// Number of quaternion blocks (`rope_dim / 4`); only used for
152    /// [`RotationKind::Quaternion4D`].
153    #[config(default = 1)]
154    pub num_quat_blocks: usize,
155}
156
157impl Mamba3DoubleSsdCacheConfig {
158    /// Derive cache shapes from a Mamba-3 block configuration plus a batch size.
159    pub fn new_from_block_config(batch: usize, block_config: Mamba3Config) -> Self {
160        Self {
161            batch,
162            state_rank: block_config.state_rank,
163            per_head_dim: block_config.per_head_dim,
164            nheads: block_config.nheads(),
165            mimo_rank: block_config.mimo_rank,
166            num_rope_angles: block_config.num_rope_angles(),
167            rotation: block_config.rotation,
168            num_quat_blocks: block_config.num_quat_blocks(),
169        }
170    }
171
172    /// Allocate zero/identity-initialised cache tensors on `device`.
173    pub fn init(&self, device: &Device) -> Mamba3DoubleSsdCache {
174        let ssm_bhpr = Tensor::zeros(
175            [self.batch, self.nheads, self.per_head_dim, self.state_rank],
176            device,
177        );
178        let k_state_bmhr = Tensor::zeros(
179            [self.batch, self.mimo_rank, self.nheads, self.state_rank],
180            device,
181        );
182        let v_state_bhp = Tensor::zeros([self.batch, self.nheads, self.per_head_dim], device);
183        let rotation = match self.rotation {
184            RotationKind::Quaternion4D => RotationState::identity_quaternion(
185                self.batch,
186                self.nheads,
187                self.num_quat_blocks,
188                device,
189            ),
190            RotationKind::Complex2D => {
191                RotationState::zeros_angle(self.batch, self.nheads, self.num_rope_angles, device)
192            }
193        };
194        Mamba3DoubleSsdCache {
195            ssm_bhpr,
196            k_state_bmhr,
197            v_state_bhp,
198            rotation,
199        }
200    }
