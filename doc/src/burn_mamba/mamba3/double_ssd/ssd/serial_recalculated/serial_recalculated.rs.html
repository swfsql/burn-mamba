10impl<B: Backend + Mamba3DoubleSsdBackendExt> Mamba3DoubleSsdInput<B> {
11    /// MIMO-first Serial SSD with recalculated backward.
12    ///
13    /// Delegates the full K1-K5 computation to [`Mamba3DoubleSsdBackendExt::double_ssd_serial_recalculated`]
14    /// which can provide a memory-efficient custom backward for supported backends.
15    ///
16    /// Falls back to the standard K1-K5 serial computation on unsupported backends.
17    ///
18    /// # Returns
19    /// - `y_bnlmhp`:         `[batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]`
20    /// - `final_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
21    pub fn double_ssd_serial_recalculated(self) -> (Tensor<B, 6>, Tensor<B, 4>) {
22        let input = self;
23        assert!(
24            input.init_state_hpr.is_none(),
25            "init_state_hpr not yet implemented for ssd_serial_recalculated"
26        );
27
28        let (y_bnlmhp, final_state_bhpr) =
29            <B as Mamba3DoubleSsdBackendExt>::double_ssd_serial_recalculated(
30                input.v_bnlmhp.into_primitive().tensor(),
31                input.da_bnlh.into_primitive().tensor(),
32                input.b_bnlmhr.into_primitive().tensor(),
33                input.c_bnlmhr.into_primitive().tensor(),
34                input.initial_state_bhpr.into_primitive().tensor(),
35            );
36        let y_bnlmhp = Tensor::from_primitive(TensorPrimitive::Float(y_bnlmhp));
37        let final_state_bhpr = Tensor::from_primitive(TensorPrimitive::Float(final_state_bhpr));
38        (y_bnlmhp, final_state_bhpr)
39    }
40}

42/// Extends the backend for the memory-efficient serial recalculated SSD.
43///
44/// The default implementation runs K1-K5 using standard tensor operations.
45/// Backends that support a custom memory-efficient backward (specifically the
46/// Autodiff wrapper) override this to recompute forward intermediates during
47/// the backward pass instead of saving them.
48pub trait Mamba3DoubleSsdBackendExt: burn::tensor::backend::Backend {
49    /// Memory-efficient MIMO serial SSD.
50    ///
51    /// # Arguments
52    /// - `v_bnlmhp`:           `[batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]`
53    /// - `da_bnlh`:            `[batch, nchunks, chunk_len, nheads]` — pre-combined Δ·A
54    /// - `b_bnlmhr`:           `[batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]`
55    /// - `c_bnlmhr`:           `[batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]`
56    /// - `initial_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
57    ///
58    /// # Returns
59    /// - `y_bnlmhp`:         `[batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]`
60    /// - `final_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
61    fn double_ssd_serial_recalculated(
62        v_bnlmhp: FloatTensor<Self>,
63        da_bnlh: FloatTensor<Self>,
64        b_bnlmhr: FloatTensor<Self>,
65        c_bnlmhr: FloatTensor<Self>,
66        initial_state_bhpr: FloatTensor<Self>,
67    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
68        // Default impl: replicate Mamba3::ssd_serial (K1-K5).
69
70        let v_bnlmhp: Tensor<Self, 6> = mk(v_bnlmhp);
71        let da_bnlh: Tensor<Self, 4> = mk(da_bnlh);
72        let b_bnlmhr: Tensor<Self, 6> = mk(b_bnlmhr);
73        let c_bnlmhr: Tensor<Self, 6> = mk(c_bnlmhr);
74        let initial_state_bhpr: Tensor<Self, 4> = mk(initial_state_bhpr);
75
76        let (da_cumsum_bhnl, da_chunk_end_bhn) = serial::k1_ssd_chunk_cumsum(da_bnlh);
77        let cb_bnhLMLM = serial::k2_ssd_bmm(c_bnlmhr.clone(), b_bnlmhr.clone());
78        let intra_chunk_state_bnhpr =
79            serial::k3_ssd_chunk_state(v_bnlmhp.clone(), b_bnlmhr, da_cumsum_bhnl.clone());
80        let (chunk_input_state_bnhpr, final_state_bhpr) = serial::k4_ssd_state_passing(
81            intra_chunk_state_bnhpr,
82            da_chunk_end_bhn,
83            initial_state_bhpr,
84        );
85        let y_bnlmhp = serial::k5_ssd_chunk_scan(
86            da_cumsum_bhnl,
87            v_bnlmhp,
88            c_bnlmhr,
89            cb_bnhLMLM,
90            chunk_input_state_bnhpr,
91        );
92
93        let y_bnlmhp = y_bnlmhp.into_primitive().tensor();
94        let final_state_bhpr = final_state_bhpr.into_primitive().tensor();
95        (y_bnlmhp, final_state_bhpr)
96    }
97}
