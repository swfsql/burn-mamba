1//! # Custom autodiff node for the Mamba-3 double-SSD recompute backward
2//!
3//! Implements [`Mamba3DoubleSsdBackendExt`](crate::mamba3::double_ssd::ssd::Mamba3DoubleSsdBackendExt)
4//! for `Autodiff<B>` via a single Burn [`Backward`] node.  The forward keeps only
5//! its leaf inputs; backprop replays the serial kernels and the gradient math in
6//! [`super::combined_backward`], so the large intermediates are never retained.
7//! The two outputs (`y`, `final_state`) are flattened into one tracked tensor
8//! (via [`crate::utils::combined_grad`]) so one node covers both.
9
10#![allow(non_snake_case)]
11
12use crate::mamba3::double_ssd::ssd;
13use crate::utils::fprim::F;
14use burn::backend::autodiff::{
15    Autodiff,
16    checkpoint::{base::Checkpointer, strategy::CheckpointStrategy},
17    grads::Gradients,
18    ops::{Backward, Ops, OpsKind},
19};
20use burn::backend::tensor::FloatTensor;
21use burn::backend::{Backend, BackendTypes};
22use burn::prelude::*;
23use ssd::serial_recalculated::{
24    Mamba3DoubleSsdBackendExt,
25    combined_backward::{self, CombinedGrads},
26};
27
28impl<B: Backend + Mamba3DoubleSsdBackendExt, C: CheckpointStrategy> Mamba3DoubleSsdBackendExt
29    for Autodiff<B, C>
30{
31    /// Memory-efficient combined forward+backward for the Mamba-3 MIMO SSD.
32    ///
33    /// The two output tensors (`y_bnlmhp`, `final_state_bhpr`) are flattened
34    /// and concatenated into a single 1-dimensional tracked tensor so a single
35    /// `Backward<B, 5>` node covers both. The caller receives split+reshaped
36    /// slices of that combined tensor; burn's autodiff accumulates their
37    /// upstream gradients back into one gradient vector before invoking
38    /// `backward`.
39    fn double_ssd_serial_recalculated(
40        v_bnlmhp: FloatTensor<Self>,
41        da_bnlh: FloatTensor<Self>,
42        b_bnlmhr: FloatTensor<Self>,
43        c_bnlmhr: FloatTensor<Self>,
44        initial_state_bhpr: FloatTensor<Self>,
45    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
46        // ── Backward node definition ─────────────────────────────────────────
47        #[derive(Debug)]
48        struct CombinedKernelsBackward;
49
50        #[derive(Clone, Debug)]
51        struct State<B: Backend> {
52            // Saved forward inputs
53            v_bnlmhp: <B as BackendTypes>::FloatTensorPrimitive,
54            da_bnlh: <B as BackendTypes>::FloatTensorPrimitive,
55            b_bnlmhr: <B as BackendTypes>::FloatTensorPrimitive,
56            c_bnlmhr: <B as BackendTypes>::FloatTensorPrimitive,
57            initial_state_bhpr: <B as BackendTypes>::FloatTensorPrimitive,
58            // Flat lengths for splitting the combined upstream gradient
59            flat_len_y_BNLMHP: usize,
60            flat_len_final_state_BHPR: usize,
61            // Shapes needed to reconstruct tensors at the right ranks
62            shape_v_bnlmhp: [usize; 6],
63            shape_da_bnlh: [usize; 4],
64            shape_b_bnlmhr: [usize; 6],
65            shape_c_bnlmhr: [usize; 6],
66            shape_initial_state_bhpr: [usize; 4],
67            shape_y_bnlmhp: [usize; 6],
68            shape_final_state_bhpr: [usize; 4],
69        }
70
71        impl<B: Backend + Mamba3DoubleSsdBackendExt> Backward<B, 5> for CombinedKernelsBackward {
72            type State = State<B>;
73
74            fn backward(
75                self,
76                ops: Ops<Self::State, 5>,
77                grads: &mut Gradients,
78                _checkpointer: &mut Checkpointer,
79            ) {
80                let [
81                    node_v_bnlmhp,
82                    node_da_bnlh,
83                    node_b_bnlmhr,
84                    node_c_bnlmhr,
85                    node_initial_state_bhpr,
86                ] = ops.parents;
87
88                // Upstream gradient of the combined 1-dimensional output.
89                let d_combined: <B as BackendTypes>::FloatTensorPrimitive =
90                    grads.consume::<B>(&ops.node);
91
92                let State {
93                    v_bnlmhp,
94                    da_bnlh,
95                    b_bnlmhr,
96                    c_bnlmhr,
97                    initial_state_bhpr,
98                    flat_len_y_BNLMHP,
99                    flat_len_final_state_BHPR,
100                    shape_v_bnlmhp,
101                    shape_da_bnlh,
102                    shape_b_bnlmhr,
103                    shape_c_bnlmhr,
104                    shape_initial_state_bhpr,
105                    shape_y_bnlmhp,
106                    shape_final_state_bhpr,
107                } = ops.state;
108
109                // ── Reconstruct saved tensors as rank-tagged primitives ──
110                let v_bnlmhp = F::<B, 6>::new(v_bnlmhp).reshape(shape_v_bnlmhp);
111                let da_bnlh = F::<B, 4>::new(da_bnlh).reshape(shape_da_bnlh);
112                let b_bnlmhr = F::<B, 6>::new(b_bnlmhr).reshape(shape_b_bnlmhr);
113                let c_bnlmhr = F::<B, 6>::new(c_bnlmhr).reshape(shape_c_bnlmhr);
114                let initial_state_bhpr =
115                    F::<B, 4>::new(initial_state_bhpr).reshape(shape_initial_state_bhpr);
116
117                // ── Split combined gradient vector ──────────────────────
118                let (d_y_bnlmhp, d_final_state_bhpr) =
119                    crate::utils::combined_grad::unflatten_pair::<B, 6, 4>(
120                        d_combined,
121                        flat_len_y_BNLMHP,
122                        flat_len_final_state_BHPR,
123                        shape_y_bnlmhp,
124                        shape_final_state_bhpr,
125                    );
126
127                // ── Core gradient computation ───────────────────────────
128                let CombinedGrads {
129                    d_v_bnlmhp,
130                    d_da_bnlh,
131                    d_b_bnlmhr,
132                    d_c_bnlmhr,
133                    d_initial_state_bhpr,
134                    ..
135                } = combined_backward::combined_backward(
136                    F::<B, 6>::new(d_y_bnlmhp),
137                    F::<B, 4>::new(d_final_state_bhpr),
138                    v_bnlmhp,
139                    da_bnlh,
140                    b_bnlmhr,
141                    c_bnlmhr,
142                    initial_state_bhpr,
143                );
144
145                // ── Register gradients with autodiff ────────────────────
146                if let Some(n) = node_v_bnlmhp {
147                    grads.register::<B>(n.id, d_v_bnlmhp.inner());
148                }
149                if let Some(n) = node_da_bnlh {
150                    grads.register::<B>(n.id, d_da_bnlh.inner());
151                }
152                if let Some(n) = node_b_bnlmhr {
153                    grads.register::<B>(n.id, d_b_bnlmhr.inner());
154                }
155                if let Some(n) = node_c_bnlmhr {
156                    grads.register::<B>(n.id, d_c_bnlmhr.inner());
157                }
158                if let Some(n) = node_initial_state_bhpr {
159                    grads.register::<B>(n.id, d_initial_state_bhpr.inner());
160                }
161            }
162        }
163
164        // ── Shape extraction (via the AutodiffTensor wrappers) ─────────────
165        use burn::backend::TensorMetadata;
166        let [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim] =
167            v_bnlmhp.primitive.shape().dims();
168        let [.., state_rank] = b_bnlmhr.primitive.shape().dims::<6>();
169
170        let flat_len_y_BNLMHP = batch * nchunks * chunk_len * mimo_rank * nheads * per_head_dim;
171        let flat_len_final_state_BHPR = batch * nheads * per_head_dim * state_rank;
172
173        let shape_v_bnlmhp: [usize; 6] =
174            [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim];
175        let shape_da_bnlh: [usize; 4] = [batch, nchunks, chunk_len, nheads];
176        let shape_b_bnlmhr: [usize; 6] = [batch, nchunks, chunk_len, mimo_rank, nheads, state_rank];
177        let shape_c_bnlmhr: [usize; 6] = [batch, nchunks, chunk_len, mimo_rank, nheads, state_rank];
178        let shape_initial_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
179        let shape_y_bnlmhp: [usize; 6] =
180            [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim];
181        let shape_final_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
182
183        // ── Register backward / run forward ───────────────────────────────
184        match CombinedKernelsBackward
185            .prepare::<C>([
186                v_bnlmhp.node.clone(),
187                da_bnlh.node.clone(),
188                b_bnlmhr.node.clone(),
189                c_bnlmhr.node.clone(),
190                initial_state_bhpr.node.clone(),
191            ])
192            .compute_bound()
193            .stateful()
194        {
195            OpsKind::Tracked(prep) => {
196                let (prim_y_bnlmhp, prim_final_state_bhpr) = B::double_ssd_serial_recalculated(
197                    v_bnlmhp.primitive.clone(),
198                    da_bnlh.primitive.clone(),
199                    b_bnlmhr.primitive.clone(),
200                    c_bnlmhr.primitive.clone(),
201                    initial_state_bhpr.primitive.clone(),
202                );
203
204                // prep.finish takes a single tensor, so pack both outputs into a
205                // single 1-D tensor; one Backward node then covers both.
206                let (prim_combined, _, _) =
207                    crate::utils::combined_grad::flatten_pair::<B>(prim_y_bnlmhp, prim_final_state_bhpr);
208
209                let state = State {
210                    v_bnlmhp: v_bnlmhp.primitive.clone(),
211                    da_bnlh: da_bnlh.primitive.clone(),
212                    b_bnlmhr: b_bnlmhr.primitive.clone(),
213                    c_bnlmhr: c_bnlmhr.primitive.clone(),
214                    initial_state_bhpr: initial_state_bhpr.primitive.clone(),
215                    flat_len_y_BNLMHP,
216                    flat_len_final_state_BHPR,
217                    shape_v_bnlmhp,
218                    shape_da_bnlh,
219                    shape_b_bnlmhr,
220                    shape_c_bnlmhr,
221                    shape_initial_state_bhpr,
222                    shape_y_bnlmhp,
223                    shape_final_state_bhpr,
224                };
225                let tracked_combined: FloatTensor<Autodiff<B, C>> = prep.finish(state, prim_combined);
226
227                // Split the tracked combined tensor back into the two outputs.
228                // The narrow/reshape ops are thin autodiff pass-throughs whose
229                // backwards accumulate into the combined gradient vector that
230                // `backward` above consumes.
231                let (tracked_y_bnlmhp, tracked_final_state_bhpr) =
232                    crate::utils::combined_grad::autodiff_unflatten_pair::<B, C, 6, 4>(
233                        tracked_combined,
234                        flat_len_y_BNLMHP,
235                        flat_len_final_state_BHPR,
236                        shape_y_bnlmhp,
237                        shape_final_state_bhpr,
238                    );
239
240                (tracked_y_bnlmhp, tracked_final_state_bhpr)
241            }
242
243            OpsKind::UnTracked(prep) => {
244                // No gradient tracking — just run the bare forward.
245                let (prim_y_bnlmhp, prim_final_state_bhpr) = B::double_ssd_serial_recalculated(
246                    v_bnlmhp.primitive,
247                    da_bnlh.primitive,
248                    b_bnlmhr.primitive,
249                    c_bnlmhr.primitive,
250                    initial_state_bhpr.primitive,
251                );
252
253                let (prim_combined, _, _) =
254                    crate::utils::combined_grad::flatten_pair::<B>(prim_y_bnlmhp, prim_final_state_bhpr);
255
256                let tracked_combined: FloatTensor<Autodiff<B, C>> = prep.finish(prim_combined);
257
258                let (tracked_y_bnlmhp, tracked_final_state_bhpr) =
259                    crate::utils::combined_grad::autodiff_unflatten_pair::<B, C, 6, 4>(
260                        tracked_combined,
261                        flat_len_y_BNLMHP,
262                        flat_len_final_state_BHPR,
263                        shape_y_bnlmhp,
264                        shape_final_state_bhpr,
265                    );
266
267                (tracked_y_bnlmhp, tracked_final_state_bhpr)
268            }
269        }
270    }
