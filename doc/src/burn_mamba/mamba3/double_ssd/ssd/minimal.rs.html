1//! ## The Chunkwise MIMO-SSD Algorithm (Minimal/Segsum variant)
2//!
3//! During training (and prefill), a naive sequential recurrence cannot
4//! exploit GPU tensor cores.  The **chunkwise SSD algorithm** achieves this
5//! by splitting the sequence into chunks of length chunk_len and decomposing the
6//! computation into four steps.
7//!
8//! ```text
9//!   Step 1  (intra-chunk, MIMO quadratic form)  →  Y_diag   [batch, nchunks, chunk_len*mimo_rank, nheads, per_head_dim]
10//!   Step 2  (input → chunk state)               →  state    [batch, nchunks, nheads, per_head_dim, state_rank]
11//!   Step 3  (inter-chunk state scan)            →  state    [batch, nchunks, nheads, per_head_dim, state_rank], final_state
12//!   Step 4  (chunk state → output)              →  Y_off    [batch, nchunks, chunk_len*mimo_rank, nheads, per_head_dim]
13//!
14//!   Y = Y_diag + Y_off   →  reshape to [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]
15//! ```
16//!
17//! The MIMO causal mask `LM_mimo[i,j] = exp(cumA[i//m] - cumA[j//m])` for `i//m >= j//m`
18//! allows all mimo_ranks ranks at the same time step to attend to each other while
19//! maintaining causal ordering across time steps.
20
21use crate::mamba3::double_ssd::prelude::*;
22use crate::utils::segsum::segsum;
23use burn::backend::Backend;
24use burn::prelude::*;
25
26impl Mamba3DoubleSsdInput {
27    /// MIMO-first chunkwise SSD — minimal/segsum variant.
28    ///
29    /// Implements the four-step decomposition for the MIMO (double-ssd) trapezoidal recurrence.
30    /// SISO (mimo_rank=1) is the degenerate case where the fused length equals the chunk length.
31    ///
32    /// No D skip is applied here — the caller handles it.
33    ///
34    /// # Shapes
35    /// - input: see [`Mamba3DoubleSsdInput`]
36    /// - output.0 `y_bnlrhp`:       `[batch, nchunks, chunk_len, R, nheads, per_head_dim]`
37    /// - output.1 `final_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
38    #[allow(non_snake_case)]
39    pub fn double_ssd_minimal(self) -> (Tensor<6>, Tensor<4>) {
40        let input = self;
41        let [batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim] = input.v_bnlmhp.dims();
42        let [.., state_rank] = input.b_bnlmhr.dims();
43        // note: L above denotes the chunk_len
44        let device = &input.v_bnlmhp.device();
45
46        assert!(nchunks >= 1, "sequence must be non-empty");
47        assert!(chunk_len > 0, "chunk_len must be positive");
48
49        // ── Fuse mimo_rank into chunk_len ────────────────────────────────────────
50        let b_bnLMhr =
51            input
52                .b_bnlmhr
53                .reshape([batch, nchunks, chunk_len * mimo_rank, nheads, state_rank]);
54        let c_bnLMhr =
55            input
56                .c_bnlmhr
57                .reshape([batch, nchunks, chunk_len * mimo_rank, nheads, state_rank]);
58        let v_bnLMhp =
59            input
60                .v_bnlmhp
61                .reshape([batch, nchunks, chunk_len * mimo_rank, nheads, per_head_dim]);
62
63        // Base per-time-step cumulative log-decay
64        let a_bhnl = input.da_bnlh.clone().permute([0, 3, 1, 2]);
65        let a_cumsum_bhnl = a_bhnl.clone().cumsum(3);
66
67        // =============================================================
68        // STEP 1: Intra-chunk outputs (Y_diag)
69        //
70        // Y_diag[m] = (L_mimo[m] ∘ C[m] B[m]ᵀ) · V[m]
71        // note: L above does not denote the chunk_len, but L in the Mamba-3 paper.
72        //
73        // MIMO mask: L_mimo[i,j] = exp(cumA[i//m] - cumA[j//m]) if i//m >= j//m, else 0
74        // =============================================================
75        let y_diag_bnLMhp = {
76            // CB = C @ B^T: contract over state_rank
77            let c_bnhLMr = c_bnLMhr.clone().permute([0, 1, 3, 2, 4]);
78            let b_bnhrLM = b_bnLMhr.clone().permute([0, 1, 3, 4, 2]);
79            // [batch, nchunks, nheads, chunk_len*mimo_rank, chunk_len*mimo_rank]
80            let cb_bnhLMLM = c_bnhLMr.matmul(b_bnhrLM);
81
82            // Build MIMO causal mask from segsum on base dimension, then interleave-expand.
83            // l_base_bhnll[i,j] = exp(cumA[i] - cumA[j]) if i >= j, else 0
84            let l_base_bhnll = segsum::<4, 5>(a_bhnl.clone()).exp();
85
86            // Interleave-expand
87            // L_mimo[i, j] = L_base[i//m, j//m]  (same decay for all ranks at a given time)
88            let l_mimo_bhnLMLM = l_base_bhnll
89                // row interleaving: insert mimo_rank copies of each l-row
90                .unsqueeze_dim::<6>(4) // l_base_bhnl1l
91                .expand([batch, nheads, nchunks, chunk_len, mimo_rank, chunk_len]) // l_base_bhnlml
92                .reshape([batch, nheads, nchunks, chunk_len * mimo_rank, chunk_len]) // l_base_bhnLMl
93                // col interleaving: insert mimo_rank copies of each l-col
94                .unsqueeze_dim::<6>(5) // l_base_bhnLMl1
95                .expand([
96                    batch,
97                    nheads,
98                    nchunks,
99                    chunk_len * mimo_rank,
100                    chunk_len,
101                    mimo_rank,
102                ]) // l_base_bhnLMlm
103                .reshape([
104                    batch,
105                    nheads,
106                    nchunks,
107                    chunk_len * mimo_rank,
108                    chunk_len * mimo_rank,
109                ]); // l_base_bhnLMLM
110
111            // Apply mask: (CB ∘ L_mimo) · V
112            let cb_bnLMhLM = cb_bnhLMLM.permute([0, 1, 3, 2, 4]);
113            let l_bnLMhLM = l_mimo_bhnLMLM.permute([0, 2, 3, 1, 4]);
114            let masked_cb_bnhLMLM = (cb_bnLMhLM * l_bnLMhLM).permute([0, 1, 3, 2, 4]);
115
116            let v_bnhLMp = v_bnLMhp.clone().permute([0, 1, 3, 2, 4]);
117            let y_diag_bnhLMp = masked_cb_bnhLMLM.matmul(v_bnhLMp);
118
119            y_diag_bnhLMp.permute([0, 1, 3, 2, 4])
120        };
121
122        // =============================================================
123        // STEP 2: Chunk state (input → state, zero initial state)
124        //
125        // s[n] = Σ_{t,r} exp(cumA[n,-1] - cumA[n,t]) · V[n,t*m+r] · B[n,t*m+r]ᵀ
126        //      (outer product over per_head_dim and state_rank)
127        // =============================================================
128        let state_bnhpr = {
129            // Decay from each fused position to end of chunk:
130            let a_cumsum_last_bhn1 = a_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
131            // Expand base cumsum to fused length (each time repeated mimo_rank times):
132            // [b, nheads, n, l] → [b, nheads, n, l, R] → [b, nheads, n, L]
133            let a_cumsum_bhnLM = a_cumsum_bhnl
134                .clone()
135                .unsqueeze_dim::<5>(4) // a_cumsum_bhnl1
136                .expand([batch, nheads, nchunks, chunk_len, mimo_rank]) // a_cumsum_bhnlm
137                .reshape([batch, nheads, nchunks, chunk_len * mimo_rank]); // a_cumsum_bhnLM
138            let decay_bhnLM = (a_cumsum_last_bhn1 - a_cumsum_bhnLM).exp();
139
140            // Multiply decay into V
141            let decay_bnLMh1 = decay_bhnLM
142                .permute([0, 2, 3, 1]) // decay_bnLMh
143                .unsqueeze_dim(4); // decay_bnLMh1
144            let decayed_v_bnLMhp = decay_bnLMh1 * v_bnLMhp.clone();
145
146            // state = decayed_V^T @ B
147            let decayed_v_bnhpLM = decayed_v_bnLMhp.permute([0, 1, 3, 4, 2]);
148            let b_bnhLMr = b_bnLMhr.permute([0, 1, 3, 2, 4]);
149            decayed_v_bnhpLM.matmul(b_bnhLMr) // state_bnhpr
150        };
151
152        // =============================================================
153        // STEP 3: Inter-chunk state scan (state passing via segsum)
154        //
155        // h[n] = Ā_chunk[n] · h[n-1] + s[n]
156        // =============================================================
157        let (state_bnhpr, final_state_bhpr) = {
158            let initial_state_b1hpr = input.initial_state_bhpr.unsqueeze_dim(1);
159            let initial_state_b1hpr = if let Some(init_hpr) = input.init_state_hpr {
160                let init_b1hpr = init_hpr.unsqueeze_dim::<4>(0).expand([
161                    batch,
162                    1,
163                    nheads,
164                    per_head_dim,
165                    state_rank,
166                ]);
167                initial_state_b1hpr + init_b1hpr
168            } else {
169                initial_state_b1hpr
170            };
171
172            // Prepend initial state: [batch, 1+nchunks, nheads, per_head_dim, state_rank]
173            let state_bNhpr = Tensor::cat(vec![initial_state_b1hpr, state_bnhpr], 1);
174
175            // Per-chunk cumulative decay (last position of each chunk)
176            let a_cumsum_last_bhn: Tensor<3> = a_cumsum_bhnl
177                .clone()
178                .slice(s![.., .., .., -1]) // a_cumsum_last_bhn1
179                .squeeze_dim(3); // a_cumsum_last_bhn
180            // Prepend zero for the initial state (no decay before chunk 0):
181            let a_chunk_pad_bhN = Tensor::cat(
182                vec![Tensor::zeros([batch, nheads, 1], device), a_cumsum_last_bhn],
183                2,
184            ); // [batch, nheads, 1+nchunks]
185
186            // Inter-chunk decay matrix via segsum: [batch, nheads, 1+nchunks, 1+nchunks]
187            let decay_chunk_bhNN = segsum::<3, 4>(a_chunk_pad_bhN).exp();
188
189            // Flatten (per_head_dim, state_rank) for matmul
190            let flat = per_head_dim * state_rank;
191            let state_bhNPR = state_bNhpr
192                .clone()
193                .permute([0, 2, 1, 3, 4]) // state_bhNpr
194                .reshape([batch, nheads, 1 + nchunks, flat]); // [batch, nheads, 1+nchunks, per_head_dim*state_rank]
195
196            let new_state_bhNPR = decay_chunk_bhNN.matmul(state_bhNPR);
197            let new_state_bhNpr =
198                new_state_bhNPR.reshape([batch, nheads, 1 + nchunks, per_head_dim, state_rank]);
199
200            // Split: chunk input states [0..n], final state [n]
201            let new_state_bnhpr = new_state_bhNpr
202                .clone()
203                .slice(s![.., .., 0..nchunks, .., ..]) // new_state_bhnpr
204                .permute([0, 2, 1, 3, 4]); // new_state_bnhpr
205            let last_state_bhpr: Tensor<4> = new_state_bhNpr
206                .slice(s![.., .., nchunks, .., ..]) // new_state_bh1pr
207                .squeeze_dim(2); // last_state_bhpr
208
209            (new_state_bnhpr, last_state_bhpr)
210        };
211
212        // =============================================================
213        // STEP 4: State-to-output (Y_off)
214        //
215        // Y_off[n, t*m+r] = C[t*m+r]ᵀ · exp(cumA[t]) · h[n-1]
216        // =============================================================
217        let y_off_bnLMhp = {
218            // Expand base cumsum to fused, then exp:
219            let state_decay_bhnLM = a_cumsum_bhnl
220                .clone()
221                .unsqueeze_dim::<5>(4) // a_cumsum_bhnl1
222                .expand([batch, nheads, nchunks, chunk_len, mimo_rank]) // a_cumsum_bhnlm
223                .reshape([batch, nheads, nchunks, chunk_len * mimo_rank]) // a_cumsum_bhnLM
224                .exp();
225
226            // C
227            let c_bnhLMr = c_bnLMhr.permute([0, 1, 3, 2, 4]);
228            let state_bnhrp = state_bnhpr.permute([0, 1, 2, 4, 3]);
229            let ch_bnhLMp = c_bnhLMr.matmul(state_bnhrp);
230
231            // Multiply by intra-chunk decay
232            let decay_bnhLM1 = state_decay_bhnLM
233                .permute([0, 2, 1, 3]) // state_decay_bnhLM
234                .unsqueeze_dim(4); // state_decay_bnhLM1
235            let y_off_bnhLMp = ch_bnhLMp * decay_bnhLM1;
236            y_off_bnhLMp.permute([0, 1, 3, 2, 4]) // y_off_bnLMhp
237        };
238
239        // ── Combine and reshape ───────────────────────────────────────────────
240        let y_bnLMhp = y_diag_bnLMhp + y_off_bnLMhp;
241        let y_bnlmhp =
242            y_bnLMhp.reshape([batch, nchunks, chunk_len, mimo_rank, nheads, per_head_dim]);
243
244        (y_bnlmhp, final_state_bhpr)
245    }
