1//! # Mamba-3 — Double-Pass SSD Forward
2//!
3//! This module provides the [`Mamba3::forward_double_ssd`] method:
4//! The burn-mamba implementation of the [`VikramLex/mamba3-minimal`](https://github.com/VikramLex/mamba3-minimal) decomposition:
5//!
6//! ```text
7//!   hₜ = αₜ hₜ₋₁ + βₜ Bₜ₋₁ ⊗ xₜ₋₁ + γₜ Bₜ ⊗ xₜ      (original double-ssd trapezoidal)
8//!
9//!   forward:    h = SSD(γ-scaled V, B)   +   SSD(β-scaled V_shifted, B_shifted)
10//! ```
11//!
12//! This is simple to derive and to verify (everything reuses the standard SSD)
13//! but increases the intra-chunk and chunk-state memory during training.
14//!
15//! See also: [`crate::mamba3::mamba3`] and [`crate::mamba3::single_ssd::single_ssd`].
16
17use crate::mamba3::double_ssd::prelude::*;
18use crate::mamba3::helpers;
19use crate::mamba3::prelude::*;
20use crate::mamba3::rotation::{RotationState, rotate_bc_forward, rotate_bc_step};
21use crate::modules::Silu;

25// ---------------------------------------------------------------------------
26// Mamba3::forward  (chunkwise double-SSD — training / prefill)
27// ---------------------------------------------------------------------------
28
29impl Mamba3 {
30    /// Process a full input sequence using the (double-ssd) trapezoidal algorithm.
31    ///
32    /// For SISO (mimo_rank=1), this is the standard double-SSD decomposition.
33    /// For MIMO (mimo_rank>1), B/C have mimo_rank parallel rank channels.
34    /// The hidden state is shared across mimo ranks; each mimo rank contributes independently.
35    ///
36    /// # Shapes
37    /// - `input_bsm` : `[batch, sequence, d_model]`
38    /// - output      : `[batch, sequence, d_model]`
39    #[allow(non_snake_case)]
40    pub fn forward_double_ssd(
41        &self,
42        input_bsm: Tensor<3>,
43        cache: Option<Mamba3DoubleSsdCache>,
44        ssd_path: &Mamba3SsdPath,
45    ) -> (Tensor<3>, Mamba3DoubleSsdCache) {
46        let [batch, sequence, _d_model] = input_bsm.dims();
47        let d_inner = self.d_inner();
48        let nheads = self.nheads();
49        let ngroups = self.ngroups;
50        let per_head_dim = self.per_head_dim();
51        let state_rank = self.state_rank;
52        let num_rope_angles = self.num_rope_angles;
53        let mimo_rank = self.mimo_rank;
54        let device = input_bsm.device();
55
56        assert!(sequence > 0, "sequence length must be at least 1");
57        assert_eq!(nheads % ngroups, 0);
58        san(&input_bsm);
59
60        // ── Initialise cache if not provided ──────────────────────────────────
61        let mut cache = cache.unwrap_or_else(|| {
62            let ssm_bhpr = Tensor::zeros([batch, nheads, per_head_dim, state_rank], &device);
63            let k_state_bmhr = Tensor::zeros([batch, mimo_rank, nheads, state_rank], &device);
64            let v_state_bhp = Tensor::zeros([batch, nheads, per_head_dim], &device);
65            let rotation = match self.rotation {
66                RotationKind::Quaternion4D => {
67                    RotationState::identity_quaternion(batch, nheads, self.num_quat_blocks, &device)
68                }
69                RotationKind::Complex2D => {
70                    RotationState::zeros_angle(batch, nheads, num_rope_angles, &device)
71                }
72            };
73            Mamba3DoubleSsdCache {
74                ssm_bhpr,
75                k_state_bmhr,
76                v_state_bhp,
77                rotation,
78            }
79        });
80
81        // ── Step 1: In-projection ─────────────────────────────────────────────
82        let proj_bsd = self.in_proj.forward(input_bsm);
83        let bc_size = ngroups * state_rank * mimo_rank;
84
85        // [batch, sequence, *] split along channel dim.
86        // b_raw_bsMGR / c_raw_bsMGR have channel size `mimo_rank * ngroups * state_rank`.
87        #[rustfmt::skip]
88        let [
89                z_bsi, x_bsi,
90                b_raw_bsMGR, c_raw_bsMGR,
91                dd_dt_bsh, dd_A_raw_bsh, lambda_raw_bsh,
92                rot_bsa
93        ] = crate::modules::split_into(
94            proj_bsd,
95            [
96                d_inner, d_inner,
97                bc_size, bc_size,
98                nheads, nheads, nheads,
99                self.num_rotation_channels,
100            ],
101            2,
102        );
103
104        san(&z_bsi);
105        san(&x_bsi);
106        san(&dd_dt_bsh);
107
108        // ── Step 2: Discretisation + trapezoidal coefficients ─────────────────
109        let helpers::TrapezoidCoeffs {
110            dt: dt_bsh,
111            da: da_bsh,
112            alpha: _alpha_bsh,
113            beta: beta_bsh,
114            gamma: gamma_bsh,
115        } = helpers::trapezoidal_coefficients(
116            dd_dt_bsh,
117            dd_A_raw_bsh,
118            lambda_raw_bsh,
119            self.dt_bias_h.val(),
120            self.dt_limit,
121            self.a_floor,
122        );
123
124        san(&dt_bsh);
125        san(&da_bsh);
126        san(&beta_bsh);
127        san(&gamma_bsh);
128
129        // ── Step 3: Reshape x ─────────────────────────────────────────────────
130        let x_bshp = x_bsi.reshape([batch, sequence, nheads, per_head_dim]);
131
132        // ── Step 4: QK-Norm on B and C  ───────────────────────────────────────
133        // QK-Norm over state_rank, then expand ngroups→nheads, then add per-(head,
134        // mimo-rank) bias [nheads, mimo_rank, state_rank]. Group dim is axis 3 of
135        // `_bsmgr` (D = 5).
136        let b_bsmhr = helpers::qk_norm_expand_bias::<5, 6>(
137            b_raw_bsMGR.reshape([batch, sequence, mimo_rank, ngroups, state_rank]),
138            &self.b_norm,
139            self.b_bias_hmr.val(),
140            3,
141            nheads,
142        );
143        let c_bsmhr = helpers::qk_norm_expand_bias::<5, 6>(
144            c_raw_bsMGR.reshape([batch, sequence, mimo_rank, ngroups, state_rank]),
145            &self.c_norm,
146            self.c_bias_hmr.val(),
147            3,
148            nheads,
149        );
150        assert_eq!(
151            [batch, sequence, mimo_rank, nheads, state_rank],
152            b_bsmhr.dims()
153        );
154        assert_eq!(
155            [batch, sequence, mimo_rank, nheads, state_rank],
156            c_bsmhr.dims()
157        );
158
159        // ── Step 5: Data-dependent positional rotation of B and C ─────────────
160        // Complex2D: abelian RoPE (cumulative angle). Quaternion4D: cumulative
161        // unit quaternion. The new cache accumulator is returned for Step (cache
162        // update) below. See [`rotate_bc_forward`].
163        let (b_bsmhr, c_bsmhr, new_rotation) = rotate_bc_forward(
164            rot_bsa,
165            dt_bsh.clone(),
166            cache.rotation.clone(),
167            b_bsmhr,
168            c_bsmhr,
169            self.rotation_kind(),
170            self.rope_dim,
171        );
172        san(&b_bsmhr);
173        san(&c_bsmhr);
174
175        // ── Step 6: Build shifted inputs for β term ───────────────────────────
176        //
177        // "Shift-Before-Chunking": prepend the cached xₜ₋₁ / Bₜ₋₁ at the
178        // sequence level (before SSD chunking) so the β term at t=0 sees the
179        // prior token from a continued cache. For a fresh (zero) cache this is
180        // equivalent to zero-padding.
181        let x_prev_first_b1hp = cache.v_state_bhp.clone().unsqueeze_dim::<4>(1);
182        let x_prev_bshp = if sequence == 1 {
183            x_prev_first_b1hp
184        } else {
185            Tensor::cat(
186                vec![x_prev_first_b1hp, x_bshp.clone().narrow(1, 0, sequence - 1)],
187                1,
188            )
189        };
190        let b_prev_first_b1mhr = cache.k_state_bmhr.clone().unsqueeze_dim::<5>(1);
191        let b_prev_bsmhr = if sequence == 1 {
192            b_prev_first_b1mhr
193        } else {
194            Tensor::cat(
195                vec![
196                    b_prev_first_b1mhr,
197                    b_bsmhr.clone().narrow(1, 0, sequence - 1),
198                ],
199                1,
200            )
201        };
202
203        // ── Step 7: Scale inputs by trapezoidal coefficients ──────────────────
204        // gamma and beta are per-head scalars, broadcast over mimo_rank and per_head_dim:
205        let gamma_bsh1 = gamma_bsh.unsqueeze_dim::<4>(3);
206        let beta_bsh1 = beta_bsh.unsqueeze_dim::<4>(3);
207        let x_gamma_bshp = x_bshp.clone() * gamma_bsh1; // γₜ · xₜ
208        let x_beta_bshp = x_prev_bshp * beta_bsh1; // βₜ · xₜ₋₁
209
210        // ── Save last-token B for cache ───────────────────────────────────────
211        let b_last_bmhr = b_bsmhr
212            .clone()
213            .narrow(1, sequence - 1, 1)
214            .reshape([batch, mimo_rank, nheads, state_rank]);
215
216        // ── Step 8: Pad sequence to multiple of chunk_len ─────────────────────
217        let chunk_len = ssd_path.chunk_len_or_optimal(state_rank, per_head_dim);
218        let sequence_padded = sequence.next_multiple_of(chunk_len);
219        let pad = sequence_padded - sequence;
220
221        #[rustfmt::skip]
222        let (x_gamma_bShp, x_beta_bShp, da_bSh, b_bSmhr, b_prev_bSmhr, c_bSmhr) = if pad == 0 {
223            (x_gamma_bshp, x_beta_bshp, da_bsh, b_bsmhr, b_prev_bsmhr, c_bsmhr)
224        } else {
225            let pad_bShp = Tensor::zeros([batch, pad, nheads, per_head_dim], &device);
226            let pad_bSh = Tensor::zeros([batch, pad, nheads], &device);
227            let pad_bSmhr = Tensor::zeros([batch, pad, mimo_rank, nheads, state_rank], &device);
228            (
229                Tensor::cat(vec![x_gamma_bshp, pad_bShp.clone()], 1),
230                Tensor::cat(vec![x_beta_bshp, pad_bShp], 1),
231                Tensor::cat(vec![da_bsh, pad_bSh], 1),
232                Tensor::cat(vec![b_bsmhr, pad_bSmhr.clone()], 1),
233                Tensor::cat(vec![b_prev_bsmhr, pad_bSmhr.clone()], 1),
234                Tensor::cat(vec![c_bsmhr, pad_bSmhr], 1),
235            )
236        };
237
238        // ── Reshape into chunks ───────────────────────────────────────────────
239        let nchunks = sequence_padded / chunk_len;
240        let x_gamma_bnlhp = x_gamma_bShp.reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
241        let x_beta_bnlhp = x_beta_bShp.reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
242        let da_bnlh = da_bSh.reshape([batch, nchunks, chunk_len, nheads]);
243        let b_bnlmhr = b_bSmhr.reshape([batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]);
244        let b_prev_bnlmhr =
245            b_prev_bSmhr.reshape([batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]);
246        let c_bnlmhr = c_bSmhr.reshape([batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]);
247
248        // ── Step 9: Double MIMO-SSD calls ────────────────────────────────────────
249        // Build V tensors — insert the mimo_rank axis at position 3 of `_bnlhp`.
250        let mimo_x_hmp = self.mimo_x_hmp.as_ref().map(|p| p.val());
251        let v_gamma_bnlmhp =
252            helpers::build_v_with_mimo::<5, 6>(x_gamma_bnlhp.clone(), mimo_x_hmp.as_ref(), 3);
253        let v_beta_bnlmhp =
254            helpers::build_v_with_mimo::<5, 6>(x_beta_bnlhp, mimo_x_hmp.as_ref(), 3);
255
256        let input_gamma = Mamba3DoubleSsdInput {
257            v_bnlmhp: v_gamma_bnlmhp,
258            da_bnlh: da_bnlh.clone(),
259            b_bnlmhr: b_bnlmhr.clone(),
260            c_bnlmhr: c_bnlmhr.clone(),
261            initial_state_bhpr: cache.ssm_bhpr,
262            init_state_hpr: self.init_state_hpr.as_ref().map(|s| s.val()),
263        };
264        let (y_gamma_bnlmhp, final_state_gamma_bhpr) = input_gamma.run(ssd_path);
265
266        let input_beta = Mamba3DoubleSsdInput {
267            v_bnlmhp: v_beta_bnlmhp,
268            da_bnlh,
269            b_bnlmhr: b_prev_bnlmhr,
270            c_bnlmhr,
271            initial_state_bhpr: Tensor::zeros([batch, nheads, per_head_dim, state_rank], &device),
272            init_state_hpr: None,
273        };
274        let (y_beta_bnlmhp, final_state_beta_bhpr) = input_beta.run(ssd_path);
275
276        let y_bnlmhp = y_gamma_bnlmhp + y_beta_bnlmhp;
277        let final_state_bhpr = final_state_gamma_bhpr + final_state_beta_bhpr;
278
279        san(&y_bnlmhp);
280        san(&final_state_bhpr);
281
282        cache.ssm_bhpr = final_state_bhpr;
283
284        // ── Step 10: Unpad ────────────────────────────────────────────────────
285        let y_bSmhp = y_bnlmhp.reshape([batch, sequence_padded, mimo_rank, nheads, per_head_dim]);
286        let y_bsmhp = if pad == 0 {
287            y_bSmhp
288        } else {
289            y_bSmhp.narrow(1, 0, sequence)
290        };
291
292        // ── Step 11: D skip + gate + aggregate ranks ──────────────────────────
293        // D skip uses raw x * mimo_x_hmp (not gamma-scaled)
294        // Insert the mimo_rank axis at position 2 of `_bshp`.
295        let v_raw_bsmhp =
296            helpers::build_v_with_mimo::<4, 5>(x_bshp.clone(), mimo_x_hmp.as_ref(), 2);
297
298        let d_111h1 = self.d_h.val().unsqueeze_dims::<5>(&[0, 1, 2, 4]);
299        let y_bsmhp = y_bsmhp + d_111h1 * v_raw_bsmhp.clone();
300
301        // ── Gate (or gated norm) and rank aggregation ─────────────────────────
302        // When `out_norm` is set, the SiLU gate is replaced by a per-head
303        // gated RMSNorm: `RmsNormGated(y, z) = norm(y) * silu(z)`.
304        let y_bsi = if mimo_rank > 1 {
305            let mimo_z_hmp = self.mimo_z_hmp.as_ref().map(|p| p.val()).unwrap();
306            let mimo_o_hmp = self.mimo_o_hmp.as_ref().map(|p| p.val()).unwrap();
307
308            let z_bshp = z_bsi
309                .clone()
310                .reshape([batch, sequence, nheads, per_head_dim]);
311            let z_bsmhp = {
312                let z_bsmhp = z_bshp
313                    .unsqueeze_dim::<5>(2) // z_bs1hp
314                    .expand([batch, sequence, mimo_rank, nheads, per_head_dim]); // z_bsmhp
315                let mimo_z_bsmhp = mimo_z_hmp
316                    .permute([1, 0, 2]) // mimo_z_mhp
317                    .unsqueeze_dims::<5>(&[0, 1]) // mimo_z_11mhp
318                    .expand([batch, sequence, mimo_rank, nheads, per_head_dim]); // mimo_z_bsmhp
319                z_bsmhp * mimo_z_bsmhp
320            };
321
322            // gate or gated norm:
323            //   without out_norm: y_r * silu(z_r)
324            //   with    out_norm: norm(y_r) * silu(z_r)  (norm over per_head_dim)
325            let y_combined_bsmhp = match &self.out_norm {
326                Some(norm) => norm.forward(y_bsmhp, z_bsmhp),
327                None => y_bsmhp * Silu::new().forward(z_bsmhp),
328            };
329
330            // Down-project with mimoₒ_hmp: out = sumₘ mimoₒ_hmp[h, r, p] * yᵣ
331            let mimo_o_bsmhp = mimo_o_hmp
332                .permute([1, 0, 2]) // mimo_o_mhp
333                .unsqueeze_dims::<5>(&[0, 1]) // mimo_o_11mhp
334                .expand([batch, sequence, mimo_rank, nheads, per_head_dim]); // mimo_o_bsmhp
335            // sum over mimo rank dim
336            let y_bshp: Tensor<4> = (y_combined_bsmhp * mimo_o_bsmhp)
337                .sum_dim(2) // y_bs1hp
338                .squeeze_dim(2); // y_bshp
339            y_bshp.reshape([batch, sequence, d_inner])
340        } else {
341            // SISO: squeeze rank dim, apply gate (or gated norm) over per_head_dim.
342            let y_bshp: Tensor<4> = y_bsmhp.squeeze_dim(2); // mimo_rank == 1
343            let z_bshp = z_bsi.reshape([batch, sequence, nheads, per_head_dim]);
344            let y_combined_bshp = match &self.out_norm {
345                Some(norm) => norm.forward(y_bshp, z_bshp),
346                None => y_bshp * Silu::new().forward(z_bshp),
347            };
348            y_combined_bshp.reshape([batch, sequence, d_inner])
349        };
350        san(&y_bsi);
351
352        // ── Out-projection ────────────────────────────────────────────────────
353        let out_bsm = self.out_proj.forward(y_bsi);
354        san(&out_bsm);
355
356        // ── Update remaining cache fields ─────────────────────────────────────
357        // k_state = B at last token
358        cache.k_state_bmhr = b_last_bmhr;
359
360        // v_state = x at last token
361        cache.v_state_bhp = x_bshp
362            .narrow(1, sequence - 1, 1) // x_b1hp
363            .squeeze_dim::<3>(1); // x_bhp
364
365        // Cumulative rotation at the last token (angle wrapped to [−π, π], or
366        // the cumulative quaternion), to continue a longer sequence.
367        cache.rotation = new_rotation;
368
369        (out_bsm, cache)
370    }
371}

377mod step {
378    use super::*;
379
380    /// One token's in-projection unpacked into the step-shaped pieces shared by
381    /// [`Mamba3::step_double_ssd`] and the constant-input shortcuts
382    /// ([`Mamba3::step_n_approx`] / [`Mamba3::step_infinite`]): the gate/value
383    /// streams, the **pre-rotation** QK-normed B/C, the raw rotation channels,
384    /// and the trapezoid coefficients.
385    pub(crate) struct StepProjection {
386        /// Gate stream `[batch, d_inner]`.
387        pub z_bi: Tensor<2>,
388        /// Value stream `[batch, nheads, per_head_dim]`.
389        pub x_bhp: Tensor<3>,
390        /// QK-normed, GQA-expanded, biased B — **before** the positional rotation.
391        pub b_bmhr: Tensor<4>,
392        /// QK-normed, GQA-expanded, biased C — **before** the positional rotation.
393        pub c_bmhr: Tensor<4>,
394        /// Raw rotation channels `[batch, num_rotation_channels]`.
395        pub rot_ba: Tensor<2>,
396        /// `Δ` `[batch, nheads]`.
397        pub dt_bh: Tensor<2>,
398        /// `Δ·A` (negative; the log-decay) `[batch, nheads]`.
399        pub da_bh: Tensor<2>,
400        /// `α = exp(Δ·A)` `[batch, nheads]`.
401        pub alpha_bh: Tensor<2>,
402        /// `β = (1−λ)·Δ·α` `[batch, nheads]`.
403        pub beta_bh: Tensor<2>,
404        /// `γ = λ·Δ` `[batch, nheads]`.
405        pub gamma_bh: Tensor<2>,
406    }
407
408    impl Mamba3 {
409        /// In-projection → split → trapezoid coefficients → QK-norm for a
410        /// single token, **stopping before** the positional rotation (which
411        /// needs the cache's cumulative rotation).
412        #[allow(non_snake_case)]
413        pub(crate) fn step_project(&self, input_bd: Tensor<2>) -> StepProjection {
414            let [batch, _d_model] = input_bd.dims();
415            let d_inner = self.d_inner();
416            let nheads = self.nheads();
417            let ngroups = self.ngroups;
418            let per_head_dim = self.per_head_dim();
419            let state_rank = self.state_rank;
420            let mimo_rank = self.mimo_rank;
421
422            assert_eq!(nheads % ngroups, 0);
423            san(&input_bd);
424
425            // ── In-projection ─────────────────────────────────────────────────
426            let proj_bd = self.in_proj.forward(input_bd);
427            san(&proj_bd);
428            let bc_size = ngroups * state_rank * mimo_rank;
429            // [batch, *] split along channel dim.
430            // b_raw_bMGR / c_raw_bMGR have channel size `mimo_rank * ngroups * state_rank`.
431            #[rustfmt::skip]
432            let [
433                    z_bi, x_bi,
434                    b_raw_bMGR, c_raw_bMGR,
435                    dd_dt_bh, dd_a_raw_bh, lambda_raw_bh,
436                    rot_ba,
437            ] = crate::modules::split_into(
438                proj_bd,
439                [
440                    d_inner, d_inner,
441                    bc_size, bc_size,
442                    nheads, nheads, nheads,
443                    self.num_rotation_channels,
444                ],
445                1,
446            );
447
448            // ── Reshape x ─────────────────────────────────────────────────────
449            let x_bhp = x_bi.reshape([batch, nheads, per_head_dim]);
450
451            // ── Discretisation + trapezoidal coefficients ─────────────────────
452            let helpers::TrapezoidCoeffs {
453                dt: dt_bh,
454                da: da_bh,
455                alpha: alpha_bh,
456                beta: beta_bh,
457                gamma: gamma_bh,
458            } = helpers::trapezoidal_coefficients(
459                dd_dt_bh,
460                dd_a_raw_bh,
461                lambda_raw_bh,
462                self.dt_bias_h.val(),
463                self.dt_limit,
464                self.a_floor,
465            );
466            san(&dt_bh);
467            san(&alpha_bh);
468            san(&beta_bh);
469            san(&gamma_bh);
470
471            // ── QK-Norm on B and C ────────────────────────────────────────────
472            // Group dim is axis 2 of `_bmgr` (D = 4).
473            let b_bmhr = helpers::qk_norm_expand_bias::<4, 5>(
474                b_raw_bMGR.reshape([batch, mimo_rank, ngroups, state_rank]),
475                &self.b_norm,
476                self.b_bias_hmr.val(),
477                2,
478                nheads,
479            );
480            let c_bmhr = helpers::qk_norm_expand_bias::<4, 5>(
481                c_raw_bMGR.reshape([batch, mimo_rank, ngroups, state_rank]),
482                &self.c_norm,
483                self.c_bias_hmr.val(),
484                2,
485                nheads,
486            );
487            assert_eq!([batch, mimo_rank, nheads, state_rank], b_bmhr.dims());
488            san(&b_bmhr);
489            san(&c_bmhr);
490
491            StepProjection {
492                z_bi,
493                x_bhp,
494                b_bmhr,
495                c_bmhr,
496                rot_ba,
497                dt_bh,
498                da_bh,
499                alpha_bh,
500                beta_bh,
501                gamma_bh,
502            }
503        }
504
505        /// State→output contraction:
506        /// `out[b, m, h, p] = Σᵣ C[b, m, h, r] · state[b, h, p, r]`
507        /// (`einsum('bhpr,bmhr->bmhp', state, C)`).
508        pub(crate) fn step_readout(state_bhpr: Tensor<4>, c_bmhr: Tensor<4>) -> Tensor<4> {
509            let c_bhrm = c_bmhr.permute([0, 2, 3, 1]);
510            let out_bhpm = state_bhpr.matmul(c_bhrm);
511            out_bhpm.permute([0, 3, 1, 2])
512        }
513
514        /// Shared block tail: `D` skip, gate (or gated RMSNorm), MIMO rank
515        /// aggregation, and the output projection.
516        ///
517        /// `out_m_bmhp` is the raw SSM readout (see [`Mamba3::step_readout`]);
518        /// `x_vals_bmhp` the MIMO-expanded values; `z_bi` the gate stream.
519        pub(crate) fn step_finish(
520            &self,
521            out_m_bmhp: Tensor<4>,
522            x_vals_bmhp: Tensor<4>,
523            z_bi: Tensor<2>,
524        ) -> Tensor<2> {
525            let [batch, mimo_rank, nheads, per_head_dim] = x_vals_bmhp.dims();
526            let d_inner = self.d_inner();
527
528            // D skip
529            let d_bmhp = self
530                .d_h
531                .val()
532                .unsqueeze_dims::<4>(&[0, 1, 3]) // d_11h1
533                .expand([batch, mimo_rank, nheads, per_head_dim]); // d_bmhp
534            let out_m_bmhp = out_m_bmhp + d_bmhp * x_vals_bmhp;
535            san(&out_m_bmhp);
536
537            // ── Gate (or gated norm) and rank aggregation ─────────────────────
538            // When `out_norm` is set, the SiLU gate is replaced by a per-head
539            // gated RMSNorm: `RmsNormGated(y, z) = norm(y) * silu(z)`.
540            let z_bhp = z_bi.reshape([batch, nheads, per_head_dim]);
541            let y_bi = if mimo_rank > 1 {
542                let mimo_z_hmp = self.mimo_z_hmp.as_ref().map(|p| p.val()).unwrap();
543                let mimo_o_hmp = self.mimo_o_hmp.as_ref().map(|p| p.val()).unwrap();
544
545                // zₘ = z * mimo_z_hmp[m]
546                let z_bmhp = z_bhp
547                    .unsqueeze_dim::<4>(1) // z_b1hp
548                    .expand([batch, mimo_rank, nheads, per_head_dim]); // z_bmhp
549                // mimo_z_hmp
550                let mimo_z_bmhp = mimo_z_hmp
551                    .permute([1, 0, 2]) // mimo_z_mhp
552                    .unsqueeze_dim::<4>(0) // mimo_z_1mhp
553                    .expand([batch, mimo_rank, nheads, per_head_dim]); // mimo_z_bmhp
554                let z_bmhp = z_bmhp * mimo_z_bmhp;
555                san(&z_bmhp);
556
557                // Per-rank gate or gated norm.
558                let combined_bmhp = match &self.out_norm {
559                    Some(norm) => norm.forward(out_m_bmhp, z_bmhp),
560                    None => out_m_bmhp * Silu::new().forward(z_bmhp),
561                };
562                san(&combined_bmhp);
563
564                // Project down: out = sumₘ mimo_o_hmp[m] * combined_bmhp[m]
565                let mimo_o_bmhp = mimo_o_hmp
566                    .permute([1, 0, 2]) // mimo_o_mhp
567                    .unsqueeze_dim::<4>(0) // mimo_o_1mhp
568                    .expand([batch, mimo_rank, nheads, per_head_dim]); // mimo_o_bmhp
569                let out_bhp: Tensor<3> = (combined_bmhp * mimo_o_bmhp)
570                    .sum_dim(1) // out_b1hp
571                    .squeeze_dim(1); // out_bhp
572                san(&out_bhp);
573                out_bhp.reshape([batch, d_inner]) // y_bi
574            } else {
575                // SISO: squeeze rank dim, gate (or gated norm) over per_head_dim.
576                let y_bhp: Tensor<3> = out_m_bmhp.squeeze_dim(1);
577                let combined = match &self.out_norm {
578                    Some(norm) => norm.forward(y_bhp, z_bhp),
579                    None => y_bhp * Silu::new().forward(z_bhp),
580                };
581                san(&combined);
582                combined.reshape([batch, d_inner])
583            };
584
585            // ── Out-projection ────────────────────────────────────────────────
586            let out_bm = self.out_proj.forward(y_bi);
587            san(&out_bm);
588            out_bm
589        }
590
591        /// Process a **single token** using the pure recurrent form.
592        ///
593        /// For SISO (mimo_rank=1):
594        /// ```text
595        ///   hₜ = αₜ hₜ₋₁ + βₜ Bₜ₋₁ ⊗ xₜ₋₁ + γₜ Bₜ ⊗ xₜ
596        ///   yₜ = Cₜᵀ hₜ + D xₜ
597        /// ```
598        ///
599        /// For MIMO (mimo_rank>1):
600        /// ```text
601        ///   hₜ = αₜ hₜ₋₁ + Σₘ βₜ Bₜ₋₁[m] ⊗ (xₜ₋₁ ⊙ mimo_x_hmp[m]) + Σₘ γₜ Bₜ[m] ⊗ (xₜ ⊙ mimo_x_hmp[m])
602        ///   yₜ[r] = Cₜ[r]ᵀ hₜ + D xₜ ⊙ mimo_x_hmp[r]
603        ///   outₜ = Σₘ mimo_o_hmp[m] ⊙ silu(zₜ ⊙ mimo_z_hmp[m]) ⊙ yₜ[m]
604        /// ```
605        ///
606        /// # Shapes
607        /// - `input_bd` : `[batch, d_model]`
608        /// - output     : `[batch, d_model]`
609        #[allow(non_snake_case)]
610        pub fn step_double_ssd(
611            &self,
612            input_bd: Tensor<2>,
613            cache: Option<Mamba3DoubleSsdCache>,
614        ) -> (Tensor<2>, Mamba3DoubleSsdCache) {
615            let [batch, _d_model] = input_bd.dims();
616            let nheads = self.nheads();
617            let per_head_dim = self.per_head_dim();
618            let state_rank = self.state_rank;
619            let num_rope_angles = self.num_rope_angles;
620            let mimo_rank = self.mimo_rank;
621            let device = &input_bd.device();
622            let ssm_shape = [batch, nheads, per_head_dim, state_rank];
623
624            let mut cache = cache.unwrap_or_else(|| {
625                let ssm_bhpr = Tensor::zeros(ssm_shape, device);
626                let k_state_bmhr = Tensor::zeros([batch, mimo_rank, nheads, state_rank], device);
627                let v_state_bhp = Tensor::zeros([batch, nheads, per_head_dim], device);
628                let rotation = match self.rotation {
629                    RotationKind::Quaternion4D => RotationState::identity_quaternion(
630                        batch,
631                        nheads,
632                        self.num_quat_blocks,
633                        device,
634                    ),
635                    RotationKind::Complex2D => {
636                        RotationState::zeros_angle(batch, nheads, num_rope_angles, device)
637                    }
638                };
639                Mamba3DoubleSsdCache {
640                    ssm_bhpr,
641                    k_state_bmhr,
642                    v_state_bhp,
643                    rotation,
644                }
645            });
646
647            // ── In-projection → coefficients → QK-norm ────────────────────────
648            let StepProjection {
649                z_bi,
650                x_bhp,
651                b_bmhr,
652                c_bmhr,
653                rot_ba,
654                dt_bh,
655                da_bh: _,
656                alpha_bh,
657                beta_bh,
658                gamma_bh,
659            } = self.step_project(input_bd);
660
661            // ── Update cumulative rotation, rotate B and C ─────────────────────
662            // Complex2D: abelian RoPE angle. Quaternion4D: cumulative quaternion.
663            // See [`rotate_bc_step`].
664            let (b_bmhr, c_bmhr, new_rotation) = rotate_bc_step(
665                rot_ba,
666                dt_bh.clone(),
667                cache.rotation.clone(),
668                b_bmhr,
669                c_bmhr,
670                self.rotation_kind(),
671                self.rope_dim,
672            );
673            san(&b_bmhr);
674            san(&c_bmhr);
675            new_rotation.sanity();
676
677            // ── Build MIMO value tensors ───────────────────────────────────────
678            // Insert the mimo_rank axis at position 1 of `_bhp`.
679            let mimo_x_hmp = self.mimo_x_hmp.as_ref().map(|p| p.val());
680            let x_vals_bmhp =
681                helpers::build_v_with_mimo::<3, 4>(x_bhp.clone(), mimo_x_hmp.as_ref(), 1);
682            san(&x_vals_bmhp);
683            let xs_vals_bmhp = helpers::build_v_with_mimo::<3, 4>(
684                cache.v_state_bhp.clone(),
685                mimo_x_hmp.as_ref(),
686                1,
687            );
688            san(&xs_vals_bmhp);
689
690            // ── SSM state update ───────────────────────────────────────────────
691            // new_state[b, h, p, r] = alpha * state
692            //   + sumₘ gamma * x_vals[m] ⊗ B_cur[m]
693            //   + sumₘ beta  * xs_vals[m] ⊗ B_state[m]
694            //
695            // For the outer product sum:
696            //   xBt[b, h, p, r] = sumₘ coeff[m, h, p] * B[m, h, n]
697            //   = einsum('bmhp,bmhr->bhpr', coeff*x_vals, B)
698            //   = matmul over m: [b, h, p, m] @ [b, h, m, r]
699            // x_vals_bmhp * gamma_b1h1
700            // Need gamma as [b, 1, h, 1] to broadcast over m and p:
701            let gamma_b1h1 = gamma_bh.clone().unsqueeze_dims::<4>(&[1, 3]);
702            let beta_b1h1 = beta_bh.clone().unsqueeze_dims::<4>(&[1, 3]);
703
704            let x_gamma_bmhp = x_vals_bmhp.clone() * gamma_b1h1;
705            san(&x_gamma_bmhp);
706            let x_beta_bmhp = xs_vals_bmhp * beta_b1h1;
707            san(&x_beta_bmhp);
708
709            // einsum('bmhp,bmhr->bhpr', x_gamma, B_cur):
710            let xbt_state_bhpr = {
711                let b_bhmr = b_bmhr.clone().permute([0, 2, 1, 3]);
712                let xg_bhpm = x_gamma_bmhp.permute([0, 2, 3, 1]);
713                xg_bhpm.matmul(b_bhmr)
714            };
715            san(&xbt_state_bhpr);
716            let xbt_prev_bhpr = {
717                let b_state_bhmr = cache.k_state_bmhr.clone().permute([0, 2, 1, 3]);
718                let xb_bhpm = x_beta_bmhp.permute([0, 2, 3, 1]);
719                xb_bhpm.matmul(b_state_bhmr)
720            };
721            san(&xbt_prev_bhpr);
722
723            let alpha_bh11 = alpha_bh.unsqueeze_dims::<4>(&[2, 3]);
724            let new_state_bhpr =
725                alpha_bh11 * cache.ssm_bhpr.clone() + xbt_state_bhpr + xbt_prev_bhpr;
726            san(&new_state_bhpr);
727
728            // ── Output ────────────────────────────────────────────────────────
729            // outₘ[b, m, h, p] = sumᵣ C[b, m, h, r] * state[b, h, p, r] + D * x_vals[b, m, h, p]
730            let out_m_bmhp = Self::step_readout(new_state_bhpr.clone(), c_bmhr);
731            san(&out_m_bmhp);
732
733            // ── D skip, gate (or gated norm), rank aggregation, out-projection ─
734            let out_bm = self.step_finish(out_m_bmhp, x_vals_bmhp, z_bi);
735
736            // ── Update cache ──────────────────────────────────────────────────
737            cache.ssm_bhpr = new_state_bhpr;
738            cache.k_state_bmhr = b_bmhr;
739            cache.v_state_bhp = x_bhp;
740            cache.rotation = new_rotation;
741
742            (out_bm, cache)
743        }
744    }
745}
