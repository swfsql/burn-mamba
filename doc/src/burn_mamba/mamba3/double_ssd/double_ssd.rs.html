1//! # Mamba-3 — Double-Pass SSD Forward
2//!
3//! This module provides the [`Mamba3::forward_double_ssd`] method:
4//! The burn-mamba implementation of the [`VikramLex/mamba3-minimal`](https://github.com/VikramLex/mamba3-minimal) decomposition:
5//!
6//! ```text
7//!   hₜ = αₜ hₜ₋₁ + βₜ Bₜ₋₁ ⊗ xₜ₋₁ + γₜ Bₜ ⊗ xₜ      (original double-ssd trapezoidal)
8//!
9//!   forward:    h = SSD(γ-scaled V, B)   +   SSD(β-scaled V_shifted, B_shifted)
10//! ```
11//!
12//! This is simple to derive and to verify (everything reuses the standard SSD)
13//! but increases the intra-chunk and chunk-state memory during training.
14//!
15//! See also: [`crate::mamba3::mamba3`] and [`crate::mamba3::single_ssd::single_ssd`].
16

28/// Apply rotary position embeddings to `x` along its last dimension.
29///
30/// Two pairing conventions are supported, selected by `rotate_pairwise`:
31///
32/// - `rotate_pairwise = true` — **interleaved** (NeoX / Triton style): adjacent
33///   pairs `(0,1)`, `(2,3)`, … are rotated together. Used by the SISO Triton
34///   kernel (`mamba3_siso_*.py`).
35/// - `rotate_pairwise = false` — **half-and-half** (GPT-J style): position `n`
36///   is paired with `n + state_rank/2`. Used by the MIMO Tilelang kernel
37///   (`mamba3_mimo_fwd.py`).
38///
39/// Reference: `mamba3.py:335` sets `rotate_pairwise = not self.is_mimo`.
40///
41/// # Shapes
42/// - `x`:      `[..., state_rank]` where `state_rank` is even
43/// - `angles`: `[..., state_rank / 2]`  (one angle per pair)
44/// - output:   same shape as `x`
45pub fn apply_rope<B: Backend, const D: usize>(
46    x: Tensor<B, D>,
47    angles: Tensor<B, D>,
48    rotate_pairwise: bool,
49) -> Tensor<B, D> {
50    let dims = x.dims();
51    let n = dims[D - 1];
52    let n2 = n / 2;
53    let leading: usize = dims[..D - 1].iter().product();
54
55    let angles_flat = angles.reshape([leading, n2]);
56    let cos = angles_flat.clone().cos();
57    let sin = angles_flat.sin();
58
59    if rotate_pairwise {
60        // Interleaved: reshape to [leading, n2, 2], pairs along last axis.
61        let x_pairs = x.reshape([leading, n2, 2]);
62        let x0 = x_pairs.clone().narrow(2, 0, 1).squeeze_dim(2);
63        let x1 = x_pairs.narrow(2, 1, 1).squeeze_dim(2);
64
65        let x0r = cos.clone() * x0.clone() - sin.clone() * x1.clone();
66        let x1r = sin * x0 + cos * x1;
67
68        Tensor::cat(
69            vec![x0r.unsqueeze_dim::<3>(2), x1r.unsqueeze_dim::<3>(2)],
70            2,
71        )
72        .reshape(dims)
73    } else {
74        // Half-and-half: reshape to [leading, 2, n2], halves along middle axis.
75        let x_halves = x.reshape([leading, 2, n2]);
76        let x0 = x_halves.clone().narrow(1, 0, 1).squeeze_dim(1);
77        let x1 = x_halves.narrow(1, 1, 1).squeeze_dim(1);
78
79        let x0r = cos.clone() * x0.clone() - sin.clone() * x1.clone();
80        let x1r = sin * x0 + cos * x1;
81
82        Tensor::cat(
83            vec![x0r.unsqueeze_dim::<3>(1), x1r.unsqueeze_dim::<3>(1)],
84            1,
85        )
86        .reshape(dims)
87    }
88}

90/// Apply RoPE to only the rotation-active entries of the last dimension; the
91/// remainder passes through unchanged. Falls back to [`apply_rope`] when
92/// `rope_dim == state_rank` (full RoPE).
93///
94/// Pairing scheme (must match the reference kernels — see Section
95/// "Data-Dependent RoPE" in the paper, and `mamba3_siso_fwd.py` /
96/// `mamba3_mimo_fwd.py`):
97///
98/// - `rotate_pairwise = true` (SISO, interleaved/NeoX): pairs `(0,1), (2,3), …`.
99///   Only pairs `0..num_rope_angles` are rotated; pairs beyond are passed
100///   through. Equivalent to slicing the first `rope_dim` entries and rotating
101///   them.
102/// - `rotate_pairwise = false` (MIMO, half-and-half/GPT-J): pair distance is
103///   always `state_rank/2`, i.e. element `n` is paired with element
104///   `state_rank/2 + n`. With partial RoPE only the first `num_rope_angles`
105///   pairs are rotated; the remaining elements in both halves pass through.
106pub(crate) fn apply_rope_partial<B: Backend, const D: usize>(
107    x: Tensor<B, D>,
108    angles: Tensor<B, D>,
109    rope_dim: usize,
110    rotate_pairwise: bool,
111) -> Tensor<B, D> {
112    let state_rank = x.dims()[D - 1];
113    if rope_dim == state_rank {
114        return apply_rope::<B, D>(x, angles, rotate_pairwise);
115    }
116
117    if rotate_pairwise {
118        // Pairs are local — slicing the first rope_dim entries gives the same
119        // result as the reference (which rotates the whole headdim but with
120        // identity cos/sin for the tail pairs).
121        let x_rope = x.clone().narrow(D - 1, 0, rope_dim);
122        let x_rest = x.narrow(D - 1, rope_dim, state_rank - rope_dim);
123        let x_rope_rotated = apply_rope::<B, D>(x_rope, angles, true);
124        return Tensor::cat(vec![x_rope_rotated, x_rest], D - 1);
125    }
126
127    // Half-and-half partial RoPE: pair distance must be `state_rank/2`, not
128    // `rope_dim/2`. Slicing the first `rope_dim` entries and calling
129    // `apply_rope` would pair within the slice and produce the wrong rotation.
130    let half = state_rank / 2;
131    let num_rope_angles = rope_dim / 2;
132    debug_assert!(
133        num_rope_angles < half,
134        "partial RoPE requires rope_dim < state_rank here"
135    );
136
137    // Split x into the two halves, then within each half separate the
138    // rotation-active prefix from the pass-through suffix.
139    let x_h1 = x.clone().narrow(D - 1, 0, half);
140    let x_h2 = x.narrow(D - 1, half, half);
141    let x_h1_rope = x_h1.clone().narrow(D - 1, 0, num_rope_angles);
142    let x_h1_pass = x_h1.narrow(D - 1, num_rope_angles, half - num_rope_angles);
143    let x_h2_rope = x_h2.clone().narrow(D - 1, 0, num_rope_angles);
144    let x_h2_pass = x_h2.narrow(D - 1, num_rope_angles, half - num_rope_angles);
145
146    // angles: [..., num_rope_angles] — broadcasts element-wise against the rope-active slices.
147    let cos = angles.clone().cos();
148    let sin = angles.sin();
149    let x_h1_rot = cos.clone() * x_h1_rope.clone() - sin.clone() * x_h2_rope.clone();
150    let x_h2_rot = sin * x_h1_rope + cos * x_h2_rope;
151
152    // Reassemble: [ first-half-rotated | first-half-passthrough | second-half-rotated | second-half-passthrough ]
153    let x_h1_out = Tensor::cat(vec![x_h1_rot, x_h1_pass], D - 1);
154    let x_h2_out = Tensor::cat(vec![x_h2_rot, x_h2_pass], D - 1);
155    Tensor::cat(vec![x_h1_out, x_h2_out], D - 1)
156}

159// Mamba3::forward  (chunkwise double-SSD — training / prefill)
160// ---------------------------------------------------------------------------

162impl<B: Backend + Mamba3BackendExt> Mamba3<B> {
163    /// Process a full input sequence using the (double-ssd) trapezoidal algorithm.
164    ///
165    /// For SISO (mimo_rank=1), this is the standard double-SSD decomposition.
166    /// For MIMO (mimo_rank>1), B/C have mimo_rank parallel rank channels.
167    /// The hidden state is shared across mimo ranks; each mimo rank contributes independently.
168    ///
169    /// # Shapes
170    /// - `input_bsm` : `[batch, sequence, d_model]`
171    /// - output      : `[batch, sequence, d_model]`
172    #[allow(non_snake_case)]
173    pub fn forward_double_ssd(
174        &self,
175        input_bsm: Tensor<B, 3>,
176        cache: Option<Mamba3DoubleSsdCache<B>>,
177        ssd_path: Mamba3DoubleSsdPath,
178    ) -> (Tensor<B, 3>, Mamba3DoubleSsdCache<B>) {
179        println!("double-ssd forward");
180        let [batch, sequence, _d_model] = input_bsm.dims();
181        let d_inner = self.d_inner();
182        let nheads = self.nheads();
183        let ngroups = self.ngroups;
184        let per_head_dim = self.per_head_dim();
185        let state_rank = self.state_rank;
186        let num_rope_angles = self.num_rope_angles;
187        let mimo_rank = self.mimo_rank;
188        let device = input_bsm.device();
189
190        assert!(sequence > 0, "sequence length must be at least 1");
191        assert_eq!(nheads % ngroups, 0);
192        san(&input_bsm);
193
194        // ── Initialise cache if not provided ──────────────────────────────────
195        let mut cache = cache.unwrap_or_else(|| {
196            let ssm_bhpr = Tensor::zeros([batch, nheads, per_head_dim, state_rank], &device);
197            let k_state_bmhr = Tensor::zeros([batch, mimo_rank, nheads, state_rank], &device);
198            let v_state_bhp = Tensor::zeros([batch, nheads, per_head_dim], &device);
199            let cum_angle_bha = Tensor::zeros([batch, nheads, num_rope_angles], &device);
200            Mamba3DoubleSsdCache {
201                ssm_bhpr,
202                k_state_bmhr,
203                v_state_bhp,
204                cum_angle_bha,
205            }
206        });
207
208        // ── Step 1: In-projection ─────────────────────────────────────────────
209        let proj_bsd = self.in_proj.forward(input_bsm);
210        let bc_size = ngroups * state_rank * mimo_rank;
211
212        // [batch, sequence, *] split along channel dim.
213        // b_raw_bsMGR / c_raw_bsMGR have channel size `mimo_rank * ngroups * state_rank`.
214        #[rustfmt::skip]
215        let [
216                z_bsi, x_bsi,
217                b_raw_bsMGR, c_raw_bsMGR,
218                dd_dt_bsh, dd_A_raw_bsh, lambda_raw_bsh,
219                theta_bsa
220        ] = crate::utils::split::split_into(
221            proj_bsd,
222            [
223                d_inner, d_inner,
224                bc_size, bc_size,
225                nheads, nheads, nheads,
226                num_rope_angles,
227            ],
228            2,
229        );
230
231        san(&z_bsi);
232        san(&x_bsi);
233        san(&dd_dt_bsh);
234
235        // ── Step 2: Discretisation + trapezoidal coefficients ─────────────────
236        let helpers::TrapezoidCoeffs {
237            dt: dt_bsh,
238            da: da_bsh,
239            alpha: _alpha_bsh,
240            beta: beta_bsh,
241            gamma: gamma_bsh,
242        } = helpers::trapezoidal_coefficients(
243            dd_dt_bsh,
244            dd_A_raw_bsh,
245            lambda_raw_bsh,
246            self.dt_bias_h.val(),
247            self.dt_limit,
248            self.a_floor,
249        );
250
251        san(&dt_bsh);
252        san(&da_bsh);
253        san(&beta_bsh);
254        san(&gamma_bsh);
255
256        // ── Step 3: Reshape x ─────────────────────────────────────────────────
257        let x_bshp = x_bsi.reshape([batch, sequence, nheads, per_head_dim]);
258
259        // ── Step 4: QK-Norm on B and C  ───────────────────────────────────────
260        // QK-Norm over state_rank, then expand ngroups→nheads, then add per-(head,
261        // mimo-rank) bias [nheads, mimo_rank, state_rank]. Group dim is axis 3 of
262        // `_bsmgr` (D = 5).
263        let b_bsmhr = helpers::qk_norm_expand_bias::<_, 5, 6>(
264            b_raw_bsMGR.reshape([batch, sequence, mimo_rank, ngroups, state_rank]),
265            &self.b_norm,
266            self.b_bias_hmr.val(),
267            3,
268            nheads,
269        );
270        let c_bsmhr = helpers::qk_norm_expand_bias::<_, 5, 6>(
271            c_raw_bsMGR.reshape([batch, sequence, mimo_rank, ngroups, state_rank]),
272            &self.c_norm,
273            self.c_bias_hmr.val(),
274            3,
275            nheads,
276        );
277        assert_eq!(
278            [batch, sequence, mimo_rank, nheads, state_rank],
279            b_bsmhr.dims()
280        );
281        assert_eq!(
282            [batch, sequence, mimo_rank, nheads, state_rank],
283            c_bsmhr.dims()
284        );
285
286        // ── Step 5: Data-dependent cumulative RoPE angles ─────────────────────
287        let theta_scaled_bsa = theta_bsa.tanh() * std::f32::consts::PI;
288        let raw_angles_bsha =
289            dt_bsh.clone().unsqueeze_dim::<4>(3) // dt_bsh1
290            *
291            theta_scaled_bsa.unsqueeze_dim::<4>(2) // theta_scaled_bs1a
292            ;
293
294        let cumsum_bsha = raw_angles_bsha.cumsum(1);
295        let cum_angles_bsha = cache.cum_angle_bha.clone()
296            .unsqueeze_dim::<4>(1) // cum_angle_b1ha
297            + cumsum_bsha;
298        assert_eq!(
299            [batch, sequence, nheads, num_rope_angles],
300            cum_angles_bsha.dims()
301        );
302        san(&cum_angles_bsha);
303
304        // Apply RoPE to B and C: angles broadcast over the mimo_rank dim.
305        let cum_angles_bsmha = cum_angles_bsha
306            .clone()
307            .unsqueeze_dim::<5>(2) // cum_angles_bs1ha
308            .expand([batch, sequence, mimo_rank, nheads, num_rope_angles]); // cum_angles_bsmha
309        // SISO uses interleaved (pairwise) pairing; MIMO uses half-and-half.
310        // Partial RoPE: rotate only the first `rope_dim` entries of B/C.
311        let rotate_pairwise = mimo_rank == 1;
312        let rope_dim = self.rope_dim;
313        let b_bsmhr = apply_rope_partial::<B, 5>(
314            b_bsmhr,
315            cum_angles_bsmha.clone(),
316            rope_dim,
317            rotate_pairwise,
318        );
319        let c_bsmhr =
320            apply_rope_partial::<B, 5>(c_bsmhr, cum_angles_bsmha, rope_dim, rotate_pairwise);
321        san(&b_bsmhr);
322        san(&c_bsmhr);
323
324        // ── Step 6: Build shifted inputs for β term ───────────────────────────
325        //
326        // "Shift-Before-Chunking": prepend the cached xₜ₋₁ / Bₜ₋₁ at the
327        // sequence level (before SSD chunking) so the β term at t=0 sees the
328        // prior token from a continued cache. For a fresh (zero) cache this is
329        // equivalent to zero-padding.
330        let x_prev_first_b1hp = cache.v_state_bhp.clone().unsqueeze_dim::<4>(1);
331        let x_prev_bshp = if sequence == 1 {
332            x_prev_first_b1hp
333        } else {
334            Tensor::cat(
335                vec![x_prev_first_b1hp, x_bshp.clone().narrow(1, 0, sequence - 1)],
336                1,
337            )
338        };
339        let b_prev_first_b1mhr = cache.k_state_bmhr.clone().unsqueeze_dim::<5>(1);
340        let b_prev_bsmhr = if sequence == 1 {
341            b_prev_first_b1mhr
342        } else {
343            Tensor::cat(
344                vec![
345                    b_prev_first_b1mhr,
346                    b_bsmhr.clone().narrow(1, 0, sequence - 1),
347                ],
348                1,
349            )
350        };
351
352        // ── Step 7: Scale inputs by trapezoidal coefficients ──────────────────
353        // gamma and beta are per-head scalars, broadcast over mimo_rank and per_head_dim:
354        let gamma_bsh1 = gamma_bsh.unsqueeze_dim::<4>(3);
355        let beta_bsh1 = beta_bsh.unsqueeze_dim::<4>(3);
356        let x_gamma_bshp = x_bshp.clone() * gamma_bsh1; // γₜ · xₜ
357        let x_beta_bshp = x_prev_bshp * beta_bsh1; // βₜ · xₜ₋₁
358
359        // ── Save last-token B for cache ───────────────────────────────────────
360        let b_last_bmhr = b_bsmhr
361            .clone()
362            .narrow(1, sequence - 1, 1)
363            .reshape([batch, mimo_rank, nheads, state_rank]);
364
365        // ── Step 8: Pad sequence to multiple of chunk_len ─────────────────────
366        let chunk_len = ssd_path.chunk_len_or_optimal(state_rank, per_head_dim);
367        let sequence_padded = sequence.next_multiple_of(chunk_len);
368        let pad = sequence_padded - sequence;
369
370        #[rustfmt::skip]
371        let (x_gamma_bShp, x_beta_bShp, da_bSh, b_bSmhr, b_prev_bSmhr, c_bSmhr) = if pad == 0 {
372            (x_gamma_bshp, x_beta_bshp, da_bsh, b_bsmhr, b_prev_bsmhr, c_bsmhr)
373        } else {
374            let pad_bShp = Tensor::zeros([batch, pad, nheads, per_head_dim], &device);
375            let pad_bSh = Tensor::zeros([batch, pad, nheads], &device);
376            let pad_bSmhr = Tensor::zeros([batch, pad, mimo_rank, nheads, state_rank], &device);
377            (
378                Tensor::cat(vec![x_gamma_bshp, pad_bShp.clone()], 1),
379                Tensor::cat(vec![x_beta_bshp, pad_bShp], 1),
380                Tensor::cat(vec![da_bsh, pad_bSh], 1),
381                Tensor::cat(vec![b_bsmhr, pad_bSmhr.clone()], 1),
382                Tensor::cat(vec![b_prev_bsmhr, pad_bSmhr.clone()], 1),
383                Tensor::cat(vec![c_bsmhr, pad_bSmhr], 1),
384            )
385        };
386
387        // ── Reshape into chunks ───────────────────────────────────────────────
388        let nchunks = sequence_padded / chunk_len;
389        let x_gamma_bnlhp = x_gamma_bShp.reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
390        let x_beta_bnlhp = x_beta_bShp.reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
391        let da_bnlh = da_bSh.reshape([batch, nchunks, chunk_len, nheads]);
392        let b_bnlmhr = b_bSmhr.reshape([batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]);
393        let b_prev_bnlmhr =
394            b_prev_bSmhr.reshape([batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]);
395        let c_bnlmhr = c_bSmhr.reshape([batch, nchunks, chunk_len, mimo_rank, nheads, state_rank]);
396
397        // ── Step 9: Double MIMO-SSD calls ────────────────────────────────────────
398        // Build V tensors — insert the mimo_rank axis at position 3 of `_bnlhp`.
399        let mimo_x_hmp = self.mimo_x_hmp.as_ref().map(|p| p.val());
400        let v_gamma_bnlmhp =
401            helpers::build_v_with_mimo::<_, 5, 6>(x_gamma_bnlhp.clone(), mimo_x_hmp.as_ref(), 3);
402        let v_beta_bnlmhp =
403            helpers::build_v_with_mimo::<_, 5, 6>(x_beta_bnlhp, mimo_x_hmp.as_ref(), 3);
404
405        let input_gamma = Mamba3DoubleSsdInput {
406            v_bnlmhp: v_gamma_bnlmhp,
407            da_bnlh: da_bnlh.clone(),
408            b_bnlmhr: b_bnlmhr.clone(),
409            c_bnlmhr: c_bnlmhr.clone(),
410            initial_state_bhpr: cache.ssm_bhpr,
411            init_state_hpr: self.init_state_hpr.as_ref().map(|s| s.val()),
412        };
413        let (y_gamma_bnlmhp, final_state_gamma_bhpr) = ssd_path.clone().run(input_gamma);
414
415        let input_beta = Mamba3DoubleSsdInput {
416            v_bnlmhp: v_beta_bnlmhp,
417            da_bnlh,
418            b_bnlmhr: b_prev_bnlmhr,
419            c_bnlmhr,
420            initial_state_bhpr: Tensor::zeros([batch, nheads, per_head_dim, state_rank], &device),
421            init_state_hpr: None,
422        };
423        let (y_beta_bnlmhp, final_state_beta_bhpr) = ssd_path.run(input_beta);
424
425        let y_bnlmhp = y_gamma_bnlmhp + y_beta_bnlmhp;
426        let final_state_bhpr = final_state_gamma_bhpr + final_state_beta_bhpr;
427
428        san(&y_bnlmhp);
429        san(&final_state_bhpr);
430
431        cache.ssm_bhpr = final_state_bhpr;
432
433        // ── Step 10: Unpad ────────────────────────────────────────────────────
434        let y_bSmhp = y_bnlmhp.reshape([batch, sequence_padded, mimo_rank, nheads, per_head_dim]);
435        let y_bsmhp = if pad == 0 {
436            y_bSmhp
437        } else {
438            y_bSmhp.narrow(1, 0, sequence)
439        };
440
441        // ── Step 11: D skip + gate + aggregate ranks ──────────────────────────
442        // D skip uses raw x * mimo_x_hmp (not gamma-scaled)
443        // Insert the mimo_rank axis at position 2 of `_bshp`.
444        let v_raw_bsmhp =
445            helpers::build_v_with_mimo::<_, 4, 5>(x_bshp.clone(), mimo_x_hmp.as_ref(), 2);
446
447        let d_111h1 = self.d_h.val().unsqueeze_dims::<5>(&[0, 1, 2, 4]);
448        let y_bsmhp = y_bsmhp + d_111h1 * v_raw_bsmhp.clone();
449
450        // ── Gate (or gated norm) and rank aggregation ─────────────────────────
451        // When `out_norm` is set, the SiLU gate is replaced by a per-head
452        // gated RMSNorm: `RmsNormGated(y, z) = norm(y) * silu(z)`.
453        let y_bsi = if mimo_rank > 1 {
454            let mimo_z_hmp = self.mimo_z_hmp.as_ref().map(|p| p.val()).unwrap();
455            let mimo_o_hmp = self.mimo_o_hmp.as_ref().map(|p| p.val()).unwrap();
456
457            let z_bshp = z_bsi
458                .clone()
459                .reshape([batch, sequence, nheads, per_head_dim]);
460            let z_bsmhp = {
461                let z_bsmhp = z_bshp
462                    .unsqueeze_dim::<5>(2) // z_bs1hp
463                    .expand([batch, sequence, mimo_rank, nheads, per_head_dim]); // z_bsmhp
464                let mimo_z_bsmhp = mimo_z_hmp
465                    .permute([1, 0, 2]) // mimo_z_mhp
466                    .unsqueeze_dims::<5>(&[0, 1]) // mimo_z_11mhp
467                    .expand([batch, sequence, mimo_rank, nheads, per_head_dim]); // mimo_z_bsmhp
468                z_bsmhp * mimo_z_bsmhp
469            };
470
471            // gate or gated norm:
472            //   without out_norm: y_r * silu(z_r)
473            //   with    out_norm: norm(y_r) * silu(z_r)  (norm over per_head_dim)
474            let y_combined_bsmhp = match &self.out_norm {
475                Some(norm) => norm.forward(y_bsmhp, z_bsmhp),
476                None => y_bsmhp * Silu::new().forward(z_bsmhp),
477            };
478
479            // Down-project with mimoₒ_hmp: out = sumₘ mimoₒ_hmp[h, r, p] * yᵣ
480            let mimo_o_bsmhp = mimo_o_hmp
481                .permute([1, 0, 2]) // mimo_o_mhp
482                .unsqueeze_dims::<5>(&[0, 1]) // mimo_o_11mhp
483                .expand([batch, sequence, mimo_rank, nheads, per_head_dim]); // mimo_o_bsmhp
484            // sum over mimo rank dim
485            let y_bshp: Tensor<B, 4> = (y_combined_bsmhp * mimo_o_bsmhp)
486                .sum_dim(2) // y_bs1hp
487                .squeeze_dim(2); // y_bshp
488            y_bshp.reshape([batch, sequence, d_inner])
489        } else {
490            // SISO: squeeze rank dim, apply gate (or gated norm) over per_head_dim.
491            let y_bshp: Tensor<B, 4> = y_bsmhp.squeeze_dim(2); // mimo_rank == 1
492            let z_bshp = z_bsi.reshape([batch, sequence, nheads, per_head_dim]);
493            let y_combined_bshp = match &self.out_norm {
494                Some(norm) => norm.forward(y_bshp, z_bshp),
495                None => y_bshp * Silu::new().forward(z_bshp),
496            };
497            y_combined_bshp.reshape([batch, sequence, d_inner])
498        };
499        san(&y_bsi);
500
501        // ── Out-projection ────────────────────────────────────────────────────
502        let out_bsm = self.out_proj.forward(y_bsi);
503        san(&out_bsm);
504
505        // ── Update remaining cache fields ─────────────────────────────────────
506        // k_state = B at last token
507        cache.k_state_bmhr = b_last_bmhr;
508
509        // v_state = x at last token
510        cache.v_state_bhp = x_bshp
511            .narrow(1, sequence - 1, 1) // x_b1hp
512            .squeeze_dim::<3>(1); // x_bhp
513
514        // cum_angle at last token
515        cache.cum_angle_bha = cum_angles_bsha
516            .narrow(1, sequence - 1, 1) // cum_angles_b1ha
517            .squeeze_dim::<3>(1); // cum_angles_bha
518
519        (out_bsm, cache)
520    }
521}

523// ---------------------------------------------------------------------------
524// Mamba3::step  (recurrent SSM — token-by-token decoding)
525// ---------------------------------------------------------------------------
526
527mod step {
528    use super::*;
529
530    impl<B: Backend> Mamba3<B> {
531        /// Process a **single token** using the pure recurrent form.
532        ///
533        /// For SISO (mimo_rank=1):
534        /// ```text
535        ///   hₜ = αₜ hₜ₋₁ + βₜ Bₜ₋₁ ⊗ xₜ₋₁ + γₜ Bₜ ⊗ xₜ
536        ///   yₜ = Cₜᵀ hₜ + D xₜ
537        /// ```
538        ///
539        /// For MIMO (mimo_rank>1):
540        /// ```text
541        ///   hₜ = αₜ hₜ₋₁ + Σₘ βₜ Bₜ₋₁[m] ⊗ (xₜ₋₁ ⊙ mimo_x_hmp[m]) + Σₘ γₜ Bₜ[m] ⊗ (xₜ ⊙ mimo_x_hmp[m])
542        ///   yₜ[r] = Cₜ[r]ᵀ hₜ + D xₜ ⊙ mimo_x_hmp[r]
543        ///   outₜ = Σₘ mimo_o_hmp[m] ⊙ silu(zₜ ⊙ mimo_z_hmp[m]) ⊙ yₜ[m]
544        /// ```
545        ///
546        /// # Shapes
547        /// - `input_bd` : `[batch, d_model]`
548        /// - output     : `[batch, d_model]`
549        #[allow(non_snake_case)]
550        pub fn step_double_ssd(
551            &self,
552            input_bd: Tensor<B, 2>,
553            cache: Option<Mamba3DoubleSsdCache<B>>,
554        ) -> (Tensor<B, 2>, Mamba3DoubleSsdCache<B>) {
555            let [batch, d_model] = input_bd.dims();
556            let d_inner = self.d_inner();
557            let nheads = self.nheads();
558            let ngroups = self.ngroups;
559            let per_head_dim = self.per_head_dim();
560            let state_rank = self.state_rank;
561            let num_rope_angles = self.num_rope_angles;
562            let mimo_rank = self.mimo_rank;
563            let device = &input_bd.device();
564            let ssm_shape = [batch, nheads, per_head_dim, state_rank];
565
566            assert_eq!(nheads % ngroups, 0);
567
568            let mut cache = cache.unwrap_or_else(|| {
569                let ssm_bhpr = Tensor::zeros(ssm_shape, device);
570                let k_state_bmhr = Tensor::zeros([batch, mimo_rank, nheads, state_rank], device);
571                let v_state_bhp = Tensor::zeros([batch, nheads, per_head_dim], device);
572                let cum_angle_bha = Tensor::zeros([batch, nheads, num_rope_angles], device);
573                Mamba3DoubleSsdCache {
574                    ssm_bhpr,
575                    k_state_bmhr,
576                    v_state_bhp,
577                    cum_angle_bha,
578                }
579            });
580
581            // ── In-projection ─────────────────────────────────────────────────
582            let proj_bd = self.in_proj.forward(input_bd);
583            let bc_size = ngroups * state_rank * mimo_rank;
584            // [batch, *] split along channel dim.
585            // b_raw_bMGR / c_raw_bMGR have channel size `mimo_rank * ngroups * state_rank`.
586            #[rustfmt::skip]
587            let [
588                    z_bi, x_bi,
589                    b_raw_bMGR, c_raw_bMGR,
590                    dd_dt_bh, dd_a_raw_bh, lambda_raw_bh,
591                    theta_ba,
592            ] = crate::utils::split::split_into(
593                proj_bd,
594                [
595                    d_inner, d_inner,
596                    bc_size, bc_size,
597                    nheads, nheads, nheads,
598                    num_rope_angles,
599                ],
600                1,
601            );
602
603            // ── Reshape x ─────────────────────────────────────────────────────
604            let x_bhp = x_bi.reshape([batch, nheads, per_head_dim]);
605
606            // ── Discretisation + trapezoidal coefficients ─────────────────────
607            let helpers::TrapezoidCoeffs {
608                dt: dt_bh,
609                da: _da_bh,
610                alpha: alpha_bh,
611                beta: beta_bh,
612                gamma: gamma_bh,
613            } = helpers::trapezoidal_coefficients(
614                dd_dt_bh,
615                dd_a_raw_bh,
616                lambda_raw_bh,
617                self.dt_bias_h.val(),
618                self.dt_limit,
619                self.a_floor,
620            );
621
622            // ── QK-Norm on B and C ────────────────────────────────────────────
623            // Group dim is axis 2 of `_bmgr` (D = 4).
624            let b_bmhr = helpers::qk_norm_expand_bias::<_, 4, 5>(
625                b_raw_bMGR.reshape([batch, mimo_rank, ngroups, state_rank]),
626                &self.b_norm,
627                self.b_bias_hmr.val(),
628                2,
629                nheads,
630            );
631            let c_bmhr = helpers::qk_norm_expand_bias::<_, 4, 5>(
632                c_raw_bMGR.reshape([batch, mimo_rank, ngroups, state_rank]),
633                &self.c_norm,
634                self.c_bias_hmr.val(),
635                2,
636                nheads,
637            );
638            assert_eq!([batch, mimo_rank, nheads, state_rank], b_bmhr.dims());
639
640            // ── RoPE: update cumulative angle, rotate B and C ──────────────────
641            let theta_scaled_ba = theta_ba.tanh() * std::f32::consts::PI;
642            let raw_angle_bha = dt_bh.unsqueeze_dim::<3>(2) * theta_scaled_ba.unsqueeze_dim::<3>(1);
643            let new_cum_angle_bha = cache.cum_angle_bha.clone() + raw_angle_bha;
644
645            // Broadcast angles over mimo_rank
646            let new_cum_angle_bmha = new_cum_angle_bha
647                .clone()
648                .unsqueeze_dim::<4>(1) // new_cum_angle_b1ha
649                .expand([batch, mimo_rank, nheads, num_rope_angles]); // new_cum_angle_bmha
650            // SISO uses interleaved (pairwise) pairing; MIMO uses half-and-half.
651            // Partial RoPE: rotate only the first `rope_dim` entries of B/C.
652            let rotate_pairwise = mimo_rank == 1;
653            let rope_dim = self.rope_dim;
654            let b_bmhr = apply_rope_partial::<B, 4>(
655                b_bmhr,
656                new_cum_angle_bmha.clone(),
657                rope_dim,
658                rotate_pairwise,
659            );
660            let c_bmhr =
661                apply_rope_partial::<B, 4>(c_bmhr, new_cum_angle_bmha, rope_dim, rotate_pairwise);
662
663            // ── Build MIMO value tensors ───────────────────────────────────────
664            // Insert the mimo_rank axis at position 1 of `_bhp`.
665            let mimo_x_hmp = self.mimo_x_hmp.as_ref().map(|p| p.val());
666            let x_vals_bmhp =
667                helpers::build_v_with_mimo::<_, 3, 4>(x_bhp.clone(), mimo_x_hmp.as_ref(), 1);
668            let xs_vals_bmhp = helpers::build_v_with_mimo::<_, 3, 4>(
669                cache.v_state_bhp.clone(),
670                mimo_x_hmp.as_ref(),
671                1,
672            );
673
674            // ── SSM state update ───────────────────────────────────────────────
675            // new_state[b, h, p, r] = alpha * state
676            //   + sumₘ gamma * x_vals[m] ⊗ B_cur[m]
677            //   + sumₘ beta  * xs_vals[m] ⊗ B_state[m]
678            //
679            // For the outer product sum:
680            //   xBt[b, h, p, r] = sumₘ coeff[m, h, p] * B[m, h, n]
681            //   = einsum('bmhp,bmhr->bhpr', coeff*x_vals, B)
682            //   = matmul over m: [b, h, p, m] @ [b, h, m, r]
683            // x_vals_bmhp * gamma_b1h1
684            // Need gamma as [b, 1, h, 1] to broadcast over m and p:
685            let gamma_b1h1 = gamma_bh.clone().unsqueeze_dims::<4>(&[1, 3]);
686            let beta_b1h1 = beta_bh.clone().unsqueeze_dims::<4>(&[1, 3]);
687
688            let x_gamma_bmhp = x_vals_bmhp.clone() * gamma_b1h1;
689            let x_beta_bmhp = xs_vals_bmhp * beta_b1h1;
690
691            // einsum('bmhp,bmhr->bhpr', x_gamma, B_cur):
692            let xbt_state_bhpr = {
693                let b_bhmr = b_bmhr.clone().permute([0, 2, 1, 3]);
694                let xg_bhpm = x_gamma_bmhp.permute([0, 2, 3, 1]);
695                xg_bhpm.matmul(b_bhmr)
696            };
697            let xbt_prev_bhpr = {
698                let b_state_bhmr = cache.k_state_bmhr.clone().permute([0, 2, 1, 3]);
699                let xb_bhpm = x_beta_bmhp.permute([0, 2, 3, 1]);
700                xb_bhpm.matmul(b_state_bhmr)
701            };
702
703            let alpha_bh11 = alpha_bh.unsqueeze_dims::<4>(&[2, 3]);
704            let new_state_bhpr =
705                alpha_bh11 * cache.ssm_bhpr.clone() + xbt_state_bhpr + xbt_prev_bhpr;
706
707            // ── Output ────────────────────────────────────────────────────────
708            // outₘ[b, m, h, p] = sumᵣ C[b, m, h, r] * state[b, h, p, r] + D * x_vals[b, m, h, p]
709            // = einsum('bhpr,bmhr->bmhp', state, C)
710            let out_m_bmhp = {
711                let c_bhrm = c_bmhr.permute([0, 2, 3, 1]);
712                let out_bhpm = new_state_bhpr.clone().matmul(c_bhrm);
713                out_bhpm.permute([0, 3, 1, 2])
714            };
715
716            // D skip
717            let d_bmhp = self
718                .d_h
719                .val()
720                .unsqueeze_dims::<4>(&[0, 1, 3]) // d_11h1
721                .expand([batch, mimo_rank, nheads, per_head_dim]); // d_bmhp
722            let out_m_bmhp = out_m_bmhp + d_bmhp * x_vals_bmhp;
723
724            // ── Gate (or gated norm) and rank aggregation ─────────────────────
725            // When `out_norm` is set, the SiLU gate is replaced by a per-head
726            // gated RMSNorm: `RmsNormGated(y, z) = norm(y) * silu(z)`.
727            let z_bhp = z_bi.reshape([batch, nheads, per_head_dim]);
728            let y_bi = if mimo_rank > 1 {
729                let mimo_z_hmp = self.mimo_z_hmp.as_ref().map(|p| p.val()).unwrap();
730                let mimo_o_hmp = self.mimo_o_hmp.as_ref().map(|p| p.val()).unwrap();
731
732                // zₘ = z * mimo_z_hmp[m]
733                let z_bmhp = z_bhp
734                    .unsqueeze_dim::<4>(1) // z_b1hp
735                    .expand([batch, mimo_rank, nheads, per_head_dim]); // z_bmhp
736                // mimo_z_hmp
737                let mimo_z_bmhp = mimo_z_hmp
738                    .permute([1, 0, 2]) // mimo_z_mhp
739                    .unsqueeze_dim::<4>(0) // mimo_z_1mhp
740                    .expand([batch, mimo_rank, nheads, per_head_dim]); // mimo_z_bmhp
741                let z_bmhp = z_bmhp * mimo_z_bmhp;
742
743                // Per-rank gate or gated norm.
744                let combined_bmhp = match &self.out_norm {
745                    Some(norm) => norm.forward(out_m_bmhp, z_bmhp),
746                    None => out_m_bmhp * Silu::new().forward(z_bmhp),
747                };
748
749                // Project down: out = sumₘ mimo_o_hmp[m] * combined_bmhp[m]
750                let mimo_o_bmhp = mimo_o_hmp
751                    .permute([1, 0, 2]) // mimo_o_mhp
752                    .unsqueeze_dim::<4>(0) // mimo_o_1mhp
753                    .expand([batch, mimo_rank, nheads, per_head_dim]); // mimo_o_bmhp
754                let out_bhp: Tensor<B, 3> = (combined_bmhp * mimo_o_bmhp)
755                    .sum_dim(1) // out_b1hp
756                    .squeeze_dim(1); // out_bhp
757                out_bhp.reshape([batch, d_inner]) // y_bi
758            } else {
759                // SISO: squeeze rank dim, gate (or gated norm) over per_head_dim.
760                let y_bhp: Tensor<B, 3> = out_m_bmhp.squeeze_dim(1);
761                let combined = match &self.out_norm {
762                    Some(norm) => norm.forward(y_bhp, z_bhp),
763                    None => y_bhp * Silu::new().forward(z_bhp),
764                };
765                combined.reshape([batch, d_inner])
766            };
767
768            // ── Out-projection ────────────────────────────────────────────────
769            let out_bm = self.out_proj.forward(y_bi);
770            assert_eq!([batch, d_model], out_bm.dims());
771
772            // ── Update cache ──────────────────────────────────────────────────
773            cache.ssm_bhpr = new_state_bhpr;
774            cache.k_state_bmhr = b_bmhr;
775            cache.v_state_bhp = x_bhp;
776            cache.cum_angle_bha = new_cum_angle_bha;
777
778            (out_bm, cache)
779        }
780    }
781}

817    /// A bundle of input + model-parameter gradients extracted from one
818    /// forward+backward run.  Each `check_grads_match` call compares these
819    /// across two runs that should be mathematically equivalent.
820    struct RunGrads {
821        out: Tensor<InnerB, 3>,
822        /// Final SSM hidden state from the returned cache.
823        final_ssm: Tensor<InnerB, 4>,
824        /// Final previous-token B state from the returned cache.
825        final_k: Tensor<InnerB, 4>,
826        /// Final previous-token x state from the returned cache.
827        final_v: Tensor<InnerB, 3>,
828        /// Final cumulative RoPE angle from the returned cache.
829        final_angle: Tensor<InnerB, 3>,
830        d_input: Tensor<InnerB, 3>,
831        d_in_proj_w: Tensor<InnerB, 2>,
832        d_dt_bias: Tensor<InnerB, 1>,
833        d_d: Tensor<InnerB, 1>,

836        d_b_bias: Tensor<InnerB, 3>,
837        d_c_bias: Tensor<InnerB, 3>,
838        d_out_proj_w: Tensor<InnerB, 2>,
839    }
840
841    /// Fixed (non-tracked) random "downstream heads" used to form a scalar loss
842    /// from the output **and** every final cache field, so the backward pass
843    /// exercises both the output and the state path.
844    struct Heads {
845        out: Tensor<InnerB, 3>,
846        ssm: Tensor<InnerB, 4>,
847        k: Tensor<InnerB, 4>,
848        v: Tensor<InnerB, 3>,
849        angle: Tensor<InnerB, 3>,
850    }
851
852    /// Build the initial cache passed to both `forward` and the `step`
853    /// unrolling. With `random = false` it is zero (the standard fresh start);
854    /// with `random = true` every field (SSM state, previous-token B/x, and
855    /// cumulative RoPE angle) holds random values, exercising parity from an
856    /// arbitrary initial state.
857    fn build_init_cache(cfg: &Mamba3Config, batch: usize, random: bool) -> Mamba3DoubleSsdCache<B> {
858        let device: Device = Default::default();
859        let nheads = cfg.nheads();
860        let per_head_dim = cfg.per_head_dim;
861        let state_rank = cfg.state_rank;
862        let mimo_rank = cfg.mimo_rank;
863        let num_rope_angles = cfg.num_rope_angles();
864        let dist = Distribution::Normal(0.0, 1.0);
865        let mk4 = |shape: [usize; 4]| {
866            let t = if random {
867                Tensor::<InnerB, 4>::random(shape, dist, &device)
868            } else {
869                Tensor::<InnerB, 4>::zeros(shape, &device)
870            };
871            Tensor::from_inner(t)
872        };
873        let mk3 = |shape: [usize; 3]| {
874            let t = if random {
875                Tensor::<InnerB, 3>::random(shape, dist, &device)
876            } else {
877                Tensor::<InnerB, 3>::zeros(shape, &device)
878            };
879            Tensor::from_inner(t)
880        };
881        Mamba3DoubleSsdCache {
882            ssm_bhpr: mk4([batch, nheads, per_head_dim, state_rank]),
883            k_state_bmhr: mk4([batch, mimo_rank, nheads, state_rank]),
884            v_state_bhp: mk3([batch, nheads, per_head_dim]),
885            cum_angle_bha: mk3([batch, nheads, num_rope_angles]),
886        }
887    }
888
889    /// Compare the output and every final cache field of two runs.
890    fn assert_outputs_match(label: &str, a: &RunGrads, b: &RunGrads, tol: f32) {
891        use crate::utils::test_helpers::max_abs_diff;
892        let checks = [
893            ("output", max_abs_diff(a.out.clone(), b.out.clone())),
894            (
895                "final ssm",
896                max_abs_diff(a.final_ssm.clone(), b.final_ssm.clone()),
897            ),
898            (
899                "final k_state",
900                max_abs_diff(a.final_k.clone(), b.final_k.clone()),
901            ),
902            (
903                "final v_state",
904                max_abs_diff(a.final_v.clone(), b.final_v.clone()),
905            ),
906            (
907                "final cum_angle",
908                max_abs_diff(a.final_angle.clone(), b.final_angle.clone()),
909            ),
910        ];
911        for (name, d) in checks {
912            assert!(d < tol, "{label}: {name} max abs diff = {d:.6} (tol {tol})");
913        }
914    }
915
916    /// Run a closure that produces an output tensor from a model and an input
917    /// (wrapped as a `Param` so it has its own autodiff leaf), then derive a
918    /// scalar loss with a fixed (non-tracked) random "head" and return the
919    /// gradients of the input and a representative set of model parameters.
920    fn run_with_grads(
921        model: &Mamba3<B>,
922        input: &Param<Tensor<B, 3>>,
923        heads: &Heads,
924        forward: impl FnOnce(&Mamba3<B>, Tensor<B, 3>) -> (Tensor<B, 3>, Mamba3DoubleSsdCache<B>),
925    ) -> RunGrads {
926        let (out, cache) = forward(model, input.val());
927        let out_inner = out.clone().inner();
928        let ssm = cache.ssm_bhpr;
929        let k = cache.k_state_bmhr;
930        let v = cache.v_state_bhp;
931        let angle = cache.cum_angle_bha;
932        let final_ssm = ssm.clone().inner();
933        let final_k = k.clone().inner();
934        let final_v = v.clone().inner();
935        let final_angle = angle.clone().inner();
936
937        // Loss couples the output and every final cache field (each via its own
938        // random head) so parameter gradients reflect both output and state.
939        let out_head = Tensor::from_inner(heads.out.clone());
940        let ssm_head = Tensor::from_inner(heads.ssm.clone());
941        let k_head = Tensor::from_inner(heads.k.clone());
942        let v_head = Tensor::from_inner(heads.v.clone());
943        let angle_head = Tensor::from_inner(heads.angle.clone());
944        let loss = (out * out_head).sum()
945            + (ssm * ssm_head).sum()
946            + (k * k_head).sum()
947            + (v * v_head).sum()
948            + (angle * angle_head).sum();
949        let grads = loss.backward();
950
951        RunGrads {
952            out: out_inner,
953            final_ssm,
954            final_k,
955            final_v,
956            final_angle,
957            d_input: input.val().grad(&grads).expect("grad input"),
958            d_in_proj_w: model
959                .in_proj
960                .weight
961                .val()
962                .grad(&grads)
963                .expect("grad in_proj.weight"),
964            d_dt_bias: model.dt_bias_h.val().grad(&grads).expect("grad dt_bias_h"),
965            d_d: model.d_h.val().grad(&grads).expect("grad d_h"),
966            d_b_norm_gamma: model
967                .b_norm
968                .gamma
969                .val()
970                .grad(&grads)
971                .expect("grad b_norm.gamma"),
972            d_c_norm_gamma: model

997    /// Assert that every entry in `a` and `b` agrees to within `grad_tol`,
998    /// printing every comparison so a failure dump shows the full picture
999    /// (instead of stopping at the first mismatch).
1000    fn check_grads_match(label: &str, a: &RunGrads, b: &RunGrads, grad_tol: f32) {
1001        let mut failures: Vec<String> = Vec::new();
1002        macro_rules! check {
1003            ($field:ident, $name:expr) => {{
1004                let d = (a.$field.clone() - b.$field.clone())
1005                    .abs()
1006                    .max()
1007                    .into_scalar();
1008                eprintln!("{:>40} {:>16} | max abs diff = {:>10.6}", label, $name, d);
1009                if d >= grad_tol {
1010                    failures.push(format!(
1011                        "{}: grad of {} max abs diff = {:.6} (tol {})",
1012                        label, $name, d, grad_tol
1013                    ));
1014                }
1015            }};
1016        }
1017        check!(d_input, "input");
1018        check!(d_in_proj_w, "in_proj.weight");
1019        check!(d_dt_bias, "dt_bias_h");
1020        check!(d_d, "d_h");
1021        check!(d_b_norm_gamma, "b_norm.gamma");
1022        check!(d_c_norm_gamma, "c_norm.gamma");
1023        check!(d_b_bias, "b_bias_hmr");
1024        check!(d_c_bias, "c_bias_hmr");
1025        check!(d_out_proj_w, "out_proj.weight");
1026        assert!(
1027            failures.is_empty(),
1028            "gradient mismatches:\n  {}",
1029            failures.join("\n  ")
1030        );
1031    }

1033    /// Build a fresh `Param<Tensor>` from a stable inner tensor.
1034    /// A new Param is needed per run so that the autodiff leaf has a fresh
1035    /// node, isolating each backward pass to its own forward graph.
1036    fn param_input(input: &Tensor<InnerB, 3>) -> Param<Tensor<B, 3>> {
1037        Param::from_tensor(Tensor::from_inner(input.clone()))
1038    }

1040    /// `forward(x)` is mathematically equivalent to repeatedly calling `step`
1041    /// token-by-token from the **same** initial cache. Outputs, every final
1042    /// cache field (SSM state, previous-token B/x, cumulative RoPE angle), and
1043    /// parameter gradients must all agree up to float-summation-order noise.
1044    ///
1045    /// With `random_init = true` the shared initial cache is random rather than
1046    /// zero. Parity from an arbitrary initial state subsumes the chunked-prefill
1047    /// (split-vs-full) guarantee: if `forward` from any state matches the
1048    /// recurrent unrolling from that same state — outputs *and* final cache —
1049    /// then feeding a `forward`-produced cache back in continues correctly.
1050    fn run_step_matches_forward(cfg: Mamba3Config, random_init: bool) {
1051        let device: Device = Default::default();
1052        let model = cfg.init::<B>(&device);
1053
1054        let batch = 2;
1055        let seq_len = 5;
1056        let d_model = cfg.d_model;
1057        let nheads = cfg.nheads();
1058        let per_head_dim = cfg.per_head_dim;
1059        let state_rank = cfg.state_rank;
1060        let mimo_rank = cfg.mimo_rank;
1061        let num_rope_angles = cfg.num_rope_angles();
1062        let normal = Distribution::Normal(0.0, 1.0);
1063
1064        let input = Tensor::<InnerB, 3>::random([batch, seq_len, d_model], normal, &device);
1065        let heads = Heads {
1066            out: Tensor::<InnerB, 3>::random([batch, seq_len, d_model], normal, &device),
1067            ssm: Tensor::<InnerB, 4>::random(
1068                [batch, nheads, per_head_dim, state_rank],
1069                normal,
1070                &device,
1071            ),
1072            k: Tensor::<InnerB, 4>::random([batch, mimo_rank, nheads, state_rank], normal, &device),
1073            v: Tensor::<InnerB, 3>::random([batch, nheads, per_head_dim], normal, &device),
1074            angle: Tensor::<InnerB, 3>::random([batch, nheads, num_rope_angles], normal, &device),
1075        };
1076
1077        let ssd_path = Mamba3DoubleSsdPath::Minimal(Some(4));
1078        let init_cache = build_init_cache(&cfg, batch, random_init);
1079
1080        let input_fwd = param_input(&input);
1081        let cache_fwd = init_cache.clone();
1082        let path_fwd = ssd_path.clone();
1083        let r_fwd = run_with_grads(&model, &input_fwd, &heads, |m, x| {
1084            m.forward_double_ssd(x, Some(cache_fwd), path_fwd)
1085        });
1086
1087        let input_step = param_input(&input);
1088        let cache_step = init_cache;
1089        let r_step = run_with_grads(&model, &input_step, &heads, |m, x| {
1090            let mut cache: Option<Mamba3DoubleSsdCache<B>> = Some(cache_step);
1091            let mut outs: Vec<Tensor<B, 2>> = Vec::with_capacity(seq_len);
1092            for t in 0..seq_len {
1093                let token = x.clone().narrow(1, t, 1).squeeze_dim(1);
1094                let (out_t, new_cache) = m.step_double_ssd(token, cache);
1095                cache = Some(new_cache);
1096                outs.push(out_t);
1097            }
1098            (Tensor::stack(outs, 1), cache.unwrap())
1099        });
1100
1101        assert_outputs_match("step vs forward", &r_fwd, &r_step, 1e-4);
1102        check_grads_match("step vs forward", &r_fwd, &r_step, 1e-3);
1103
1104        // ── Guard: the random initial state must actually be consumed ─────
1105        // Re-run forward from a *zero* initial cache; its output must differ
1106        // from the random-init output. Otherwise the initial state is being
1107        // silently ignored and forward/step would match trivially.
1108        if random_init {
1109            use crate::utils::test_helpers::max_abs_diff;
1110            let (out_zero, _) = model.forward_double_ssd(
1111                Tensor::from_inner(input.clone()),
1112                Some(build_init_cache(&cfg, batch, false)),
1113                ssd_path.clone(),
1114            );
1115            let d = max_abs_diff(r_fwd.out.clone(), out_zero.inner());
1116            assert!(
1117                d > 1e-3,
1118                "random initial state appears ignored: random-init vs zero-init \
1119                 output max abs diff = {d:.6} (expected a clear difference)"
1120            );
1121        }
1122    }
