1//! # Pathway-agnostic SSD algorithm selection (Mamba-3)
2//!
3//! [`Mamba3SsdPath`] picks the chunkwise SSD *algorithm* (Minimal / Serial /
4//! SerialRecalculated) and chunk length, independent of the double-vs-single
5//! *pathway* (which the supplied cache variant selects).  It converts into the
6//! per-pathway path types via `From` and is threaded by
7//! [`Mamba3::forward`](crate::mamba3::mamba3::Mamba3::forward) into whichever
8//! pathway the cache implies.
9
10use crate::mamba3::prelude::*;
11use burn::backend::Backend;
12use burn::prelude::*;
13
14/// Algorithm selection for the Mamba-3 chunkwise SSD.
15///
16/// This selects the chunkwise SSD *algorithm*. The *pathway* (double- vs
17/// single-ssd) is selected separately, by the supplied cache variant (see
18/// [`crate::mamba3::cache::Mamba3Caches`]); [`Mamba3::forward`] threads this
19/// same selection into whichever pathway the cache implies, converting it into
20/// the per-pathway input bundle ([`crate::mamba3::double_ssd::ssd::Mamba3DoubleSsdInput`]
21/// or [`crate::mamba3::single_ssd::ssd::Mamba3SingleSsdInput`]) and calling
22/// that bundle's `run`.
23///
24/// Each variant carries an optional chunk length. Larger values increase the
25/// intra-chunk GEMM work and reduce the inter-chunk scan length; the optimal
26/// value is approximately `√(state_rank · per_head_dim)` (see
27/// [`Self::optimal_chunk_len`]). `None` falls back to that optimal value.
28///
29/// If no path is specified, the cache defaults to
30/// [`crate::mamba3::cache::Mamba3Caches::SingleSsd`] with [`Self::default`]
31/// (i.e. [`Self::SerialRecalculated`] with an unset chunk length).
32#[derive(Debug, Clone)]
33pub enum Mamba3SsdPath {
34    /// Minimal/segsum SSD: mostly batched matmuls; backward via autodiff.
35    ///
36    /// See [`crate::mamba3::double_ssd::ssd::Mamba3DoubleSsdInput::double_ssd_minimal`]
37    /// / [`crate::mamba3::single_ssd::ssd::Mamba3SingleSsdInput::single_ssd_minimal`].
38    /// For training, prefer [`Self::SerialRecalculated`].
39    Minimal(Option<usize>),
40
41    /// (Hybrid) serial SSD: a serial loop over the chunks plus batched matmuls;
42    /// backward via autodiff.
43    ///
44    /// See [`crate::mamba3::double_ssd::ssd::Mamba3DoubleSsdInput::double_ssd_serial`]
45    /// / [`crate::mamba3::single_ssd::ssd::Mamba3SingleSsdInput::single_ssd_serial`].
46    /// For a memory-saving custom backward, see [`Self::SerialRecalculated`].
47    Serial(Option<usize>),
48
49    /// (Hybrid) serial SSD with a custom, memory-efficient backward that
50    /// recomputes the forward intermediates instead of storing them.
51    ///
52    /// See [`crate::mamba3::double_ssd::ssd::Mamba3DoubleSsdInput::double_ssd_serial_recalculated`]
53    /// / [`crate::mamba3::single_ssd::ssd::Mamba3SingleSsdInput::single_ssd_serial_recalculated`].
54    /// For a plain autodiff backward, see [`Self::Serial`].
55    SerialRecalculated(Option<usize>),
56}
57
58impl Mamba3SsdPath {
59    /// Optimal chunk length, approximately `√(state_rank · per_head_dim)`,
60    /// rounded up to a multiple of 32 and capped at 512.
61    pub fn optimal_chunk_len(state_rank: usize, per_head_dim: usize) -> usize {
62        (state_rank * per_head_dim)
63            .isqrt()
64            .next_multiple_of(32) // rule-of-thumb: common plane dimension.
65            .min(512) // rule-of-thumb: ceiling at 512.
66    }
67
68    /// The chunk length carried by this variant, if any.
69    pub fn chunk_len(&self) -> Option<usize> {
70        match self {
71            Self::Minimal(chunk_len)
72            | Self::Serial(chunk_len)
73            | Self::SerialRecalculated(chunk_len) => *chunk_len,
74        }
75    }
76
77    /// The chunk length carried by this variant, or [`Self::optimal_chunk_len`]
78    /// when unset.
79    pub fn chunk_len_or_optimal(&self, state_rank: usize, per_head_dim: usize) -> usize {
80        self.chunk_len()
81            .unwrap_or_else(|| Self::optimal_chunk_len(state_rank, per_head_dim))
82    }
83
84    /// The recommended default path for a given block: [`Self::SerialRecalculated`]
85    /// with [`Self::optimal_chunk_len`] for the block's dimensions.
86    pub fn default_optimal_from_block(block: &Mamba3) -> Self {
87        let chunk_len = Self::optimal_chunk_len(block.state_rank, block.per_head_dim());
88        Self::SerialRecalculated(Some(chunk_len))
89    }
90}
91
92impl Default for Mamba3SsdPath {
93    fn default() -> Self {
94        // Defaults to the SerialRecalculated algorithm with the optimal chunk length.
95        Self::SerialRecalculated(None)
96    }
97}
