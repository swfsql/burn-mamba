81    // Implement our custom backend trait for the generic `CubeBackend`.
82    //
83    // TODO: see if burn/crates/burn-vision/src/backends/cube/connected_components/prefix_sum.rs
84    // can be used here
85    impl<R: CubeRuntime, F: FloatElement, I: IntElement, BT: BoolElement> BackendExt
86        for CubeBackend<R, F, I, BT>
87    {
88        /// Returns `da_cumsum_bhnl`.
89        fn k1_ssd_chunk_cumsum(
90            dt_discretized_bhnl: FloatTensor<Self>,
91            a_decay_h: FloatTensor<Self>,
92        ) -> FloatTensor<Self> {
93            // Note: `burn::tensor::ops::FloatTensor` is concretized to `burn::cubecl::tensor::CubeTensor`.
94
95            let client = dt_discretized_bhnl.client.clone();
96            let device = dt_discretized_bhnl.device.clone();
97            dt_discretized_bhnl.assert_is_on_same_device(&a_decay_h);
98
99            // For simplicity, make sure each tensor is contiguous.
100            let dt_discretized_bhnl = into_contiguous(dt_discretized_bhnl);
101            let a_decay_h = into_contiguous(a_decay_h);
102
103            // Get relevant shapes.
104            let [batch, nheads, nchunks, chunk_len] = dt_discretized_bhnl.meta.shape().dims();
105
106            // Create output tensor.
107            let da_cumsum_shape_bhnl = Shape::from([batch, nheads, nchunks, chunk_len]);
108            let da_cumsum_buffer_bhnl =
109                client.empty(da_cumsum_shape_bhnl.num_elements() * core::mem::size_of::<F>());
110            let da_cumsum_bhnl = CubeTensor::new_contiguous(
111                client.clone(),
112                device.clone(),
113                da_cumsum_shape_bhnl,
114                da_cumsum_buffer_bhnl,
115                F::dtype(),
116            );
117
118            // Compute number of planes per cube.
119            let num_planes = {
120                use crate::mamba2::gpu::USUAL_PLANE_DIM; // 32
121                assert_eq!(
122                    chunk_len as u32 % USUAL_PLANE_DIM,
123                    0,
124                    "only filled lanes are supported. chunk_len % USUAL_PLANE_DIM = {chunk_len} % {USUAL_PLANE_DIM} != 0"
125                );
126                assert!(
127                    chunk_len as u32 <= USUAL_PLANE_DIM * USUAL_PLANE_DIM,
128                    "chunk_len ({}) exceeds supported limit ({}). Current implementation requires chunk_len <= USUAL_PLANE_DIM * USUAL_PLANE_DIM",
129                    chunk_len,
130                    USUAL_PLANE_DIM * USUAL_PLANE_DIM
131                );
132                chunk_len as u32 / USUAL_PLANE_DIM
133            };
134
135            unsafe {
136                kernel::forward::launch_unchecked::<F, R>(
137                    &client,
138                    CubeCount::Static(batch as u32, nheads as u32, nchunks as u32),
139                    CubeDim::new_1d(chunk_len as u32),
140                    dt_discretized_bhnl.into_tensor_arg(),
141                    a_decay_h.into_tensor_arg(),
142                    da_cumsum_bhnl.clone().into_tensor_arg(),
143                    num_planes, // jit time
144                )
145            };
146
147            da_cumsum_bhnl
148        }
149    }

203/// Kernel implementation in `cubecl`.
204#[cfg(feature = "cubecl")]
205pub mod kernel {
206    use burn::cubecl;
207    use cubecl::prelude::*;
208
209    /// # Shapes
210    /// - `dt_discretized_bhnl`   : `[batch, nheads, nchunks, chunk_len]`
211    /// - `a_decay_h`             : `[nheads]`
212    /// - `da_cumsum_bhnl`        : `[batch, nheads, nchunks, chunk_len]`
213    #[cube(launch_unchecked)]
214    pub fn forward<F: Float>(
215        dt_discretized_bhnl: &Tensor<F>,
216        a_decay_h: &Tensor<F>,
217        da_cumsum_bhnl: &mut Tensor<F>,
218        #[comptime] num_planes: u32, // jit time
219    ) {
220        // Note: all plane operations require filled planes.
221        // Operation examples: plane_exclusive_sum, plane_shuffle.
222
223        let dt = dt_discretized_bhnl;
224        let a = a_decay_h;
225        let num_planes = num_planes as usize;
226
227        let batch_idx = CUBE_POS_X as usize; // 0..batch
228        let nheads_idx = CUBE_POS_Y as usize; // 0..nheads
229        let nchunks_idx = CUBE_POS_Z as usize; // 0..nchunks
230        let thread_id = UNIT_POS_X as usize; // 0..chunk_len
231
232        let dt_batch_stride = dt.stride(0);
233        let dt_nheads_stride = dt.stride(1);
234        let dt_nchunks_stride = dt.stride(2);
235        let dt_chunk_len_stride = dt.stride(3);
236        let a_nheads_stride = a.stride(0);
237
238        // Base linear offset for the current (batch, head, chunk) slice.
239        let base_pos = batch_idx * dt_batch_stride
240            + nheads_idx * dt_nheads_stride
241            + nchunks_idx * dt_nchunks_stride;
242
243        // Scalar value a[h] (identical for all units in this cube).
244        let a_offset = nheads_idx * a_nheads_stride;
245        let a_1 = a[a_offset];
246
247        // Local position within the chunk
248        let plane_idx = thread_id / PLANE_DIM as usize; // 0..num_planes
249        let plane_thread_idx = UNIT_POS_PLANE as usize; // 0..PLANE_DIM
250
251        // Position of the element processed by this unit.
252        let my_pos = base_pos + (thread_id as usize) * dt_chunk_len_stride;
253
254        // Per-position increment for the prefix sum.
255        // eg. increment        = 01 02 03 04 ; 05 06 07 08 ; 09 10 11 12 ; 013 014 015 016
256        let increment = dt[my_pos] * a_1;
257
258        // Shared memory for plane totals / offsets (size known at jit time).
259        let mut shared_totals = SharedMemory::<F>::new(num_planes as usize);
260
261        // 1. Exclusive prefix sum within the current plane.
262        // e.g. local_scan       = 00 01 03 06 ; 00 05 11 18 ; 00 09 19 30 ; 000 013 027 042
263        let local_scan = plane_exclusive_sum(increment);
264
265        // 2. Compute inclusive sum of the entire plane and store in shared memory
266        //    (only the first thread of each plane writes).
267        // e.g. plane_last_scan  =          06 ;          18 ;          30 ;             042
268        let plane_last_scan = plane_shuffle(local_scan, PLANE_DIM - 1);
269        // e.g. plane_last_inc   =          04 ;          08 ;          12 ;             016
270        let plane_last_increment = plane_shuffle(increment, PLANE_DIM - 1);
271        // e.g. plane_total      =          10 ;          26 ;          42 ;             058
272        let plane_total = plane_last_scan + plane_last_increment;
273        // e.g. shared_totals    =          10 ;          26 ;          42 ;             058
274        if plane_thread_idx == 0 {
275            shared_totals[plane_idx] = plane_total;
276        }
277        let () = sync_cube(); // for shared_totals
278
279        // 3. Exclusive prefix sum over the plane totals (performed by plane 0).
280        comptime! { // jit time
281            if num_planes > PLANE_DIM as usize {
282                panic!("num_planes exceeds PLANE_DIM. The current plane-based reduction assumes PLANE_DIM >= num_planes.");
283            }
284            // i.e. requires chunk_len <= PLANE_DIM * PLANE_DIM.
285        }
286        // e.g. offset           =           0 ;          10 ;          36 ;             078
287        // e.g. shared_totals    =           0 ;          10 ;          36 ;             078
288        if plane_idx == 0 {
289            // The number of threads capacity in plane 0 may exceed the required num_planes,
290            // so the extra threads need to compute the identity operation.
291            let val = if plane_thread_idx < num_planes {
292                shared_totals[plane_thread_idx]
293            } else {
294                F::new(0.0)
295            };
296            // Note: All threads in plane 0 must call plane_exclusive_sum together (no intra-plane divergence).
297            let offset = plane_exclusive_sum(val);
298            // Only write back for slots that actually exist.
299            if plane_thread_idx < num_planes {
300                shared_totals[plane_thread_idx] = offset;
301            }
302        }
303        let () = sync_cube(); // for shared_totals
304
305        // 4. Full-cube exclusive prefix sum.
306        // e.g. exclusive_result = 00 01 03 06 ; 10 15 21 28 ; 36 45 55 66 ; 078 091 105 120
307        let exclusive_result = local_scan + shared_totals[plane_idx];
308
309        // Convert to inclusive prefix sum (required by the original algorithm).
310        // e.g. inclusive_result = 01 03 06 10 ; 15 21 28 36 ; 45 55 66 78 ; 091 105 120 136
311        let inclusive_result = exclusive_result + increment;
312
313        // Write result.
314        da_cumsum_bhnl[my_pos] = inclusive_result;
315    }
316}

318/// Naive (forward) implementation in `burn`.
319pub mod naive {
320    use burn::prelude::*;
321
322    /// Returns `da_cumsum_bhnl`.
323    pub fn forward<B: Backend>(
324        dt_discretized_bhnl: Tensor<B, 4>,
325        a_decay_h: Tensor<B, 1>,
326    ) -> Tensor<B, 4> {
327        let [batch, nheads, nchunks, chunk_len] = dt_discretized_bhnl.dims();
328        let a_decay_bhnl = a_decay_h
329            .unsqueeze_dims::<4>(&[0, 1, 2]) // 111h
330            .expand([batch, nheads, nchunks, chunk_len]);
331        (dt_discretized_bhnl * a_decay_bhnl).cumsum(3) // da_cumsum_bhnl
332    }
333}

337mod test {
338    use super::*;
339    pub type TestBackend = burn::backend::Cuda<f32, i32>;
340    use burn::tensor::Tolerance;
341    use burn::tensor::{Distribution, Tensor};
342
343    #[test]
344    fn kernel_naive_comparison_01() {
345        let device = Default::default();
346        let (batch, nheads, nchunks, chunk_len) = (2, 2, 2, 32);
347
348        // input setup
349        let dt_discretized_bhnl: Tensor<TestBackend, 1, burn::prelude::Int> = Tensor::arange(
350            1..(1 + batch * nheads * nchunks * chunk_len) as i64,
351            &device,
352        );
353        let dt_discretized_bhnl: Tensor<TestBackend, 4> = dt_discretized_bhnl
354            .float()
355            .reshape([batch, nheads, nchunks, chunk_len]);
356        let a_decay_h: Tensor<TestBackend, 1> = Tensor::ones([nheads], &device);
357
358        // naive
359        let da_cumsum_naive_bhnl = naive::forward(dt_discretized_bhnl.clone(), a_decay_h.clone());
360        assert_eq!(
361            [batch, nheads, nchunks, chunk_len],
362            da_cumsum_naive_bhnl.dims()
363        );
364        println!("naive: {}", &da_cumsum_naive_bhnl);
365
366        // kernel
367        let da_cumsum_cubecl_bhnl: Tensor<TestBackend, 4> =
368            backend_ext::k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), a_decay_h.clone());
369        assert_eq!(
370            [batch, nheads, nchunks, chunk_len],
371            da_cumsum_cubecl_bhnl.dims()
372        );
373        println!("cubecl: {}", &da_cumsum_cubecl_bhnl);
374
375        // comparison
376        da_cumsum_naive_bhnl
377            .to_data()
378            .assert_approx_eq::<f32>(&da_cumsum_cubecl_bhnl.to_data(), Tolerance::default());
379    }
