1// mamba2-pre-gpu.rs
2//
3// GPU-naive but Triton-faithful forward implementation of the Mamba-2 SSD module.
4//
5// This file mirrors `_mamba_chunk_scan_combined_fwd` from `ssd_combined.py`,
6// which orchestrates the following five GPU kernels, each described in detail below:
7//
8//   Kernel 1: _chunk_cumsum_fwd_kernel  (ssd_chunk_state.py)
9//   Kernel 2: _bmm_chunk_fwd_kernel     (ssd_bmm.py)
10//   Kernel 3: _chunk_state_fwd_kernel   (ssd_chunk_state.py)
11//   Kernel 4: _state_passing_fwd_kernel (ssd_state_passing.py)
12//   Kernel 5: _chunk_scan_fwd_kernel    (ssd_chunk_scan.py)
13//
14// Parallelism conventions used in comments:
15//   blockIdx.{x,y,z} — the outer for-loops in this file; in a real GPU each iteration
16//                       would be a separate thread block running concurrently.
17//   threadIdx / SRAM  — the inner slice/matmul operations inside a loop body; in a real GPU
18//                       these are the per-warp register operations within a thread block.
19//   autotune(N)       — a Triton `@triton.autotune` BLOCK_SIZE_* constant; N is the typical
20//                       best value so a GPU translation can use it as a starting point.
21//
22// Each for-loop carries a `// blockIdx.*` or `// threadIdx` comment to make the
23// mapping explicit. Tile-size variables are given full descriptive names instead of
24// single letters so a reader can immediately understand what dimension they tile.
25
26use crate::mamba2::prelude::*;
27use burn::prelude::*;
28
29/// Returns the Triton autotune fallback value for a BLOCK_SIZE_* constant.
30///
31/// In a real GPU kernel, Triton picks the best block size at JIT-compile time
32/// by benchmarking configurations listed in `@triton.autotune(configs=[...])`.
33/// Here we record the typical winning value so that CUDA/WGSL/Metal translations
34/// can use it as a sensible starting tiling without re-running the search.
35
36impl<B: Backend> Mamba2<B>
37where
38    B: crate::mamba2::gpu::BackendExt,
39{
40    /// GPU-naive, Triton-faithful forward pass for the Mamba-2 SSD module.
41    ///
42    /// Mirrors `_mamba_chunk_scan_combined_fwd` from `ssd_combined.py`,
43    /// which is called from `Mamba2.forward()` via `mamba_chunk_scan_combined()`.
44    ///
45    /// # Tensor shapes flowing through the function
46    ///
47    /// ```text
48    /// x_bshp:                  [batch, sequence, nheads, per_head_dim] OK
49    /// dt_bsh:                  [batch, sequence, nheads] OK
50    /// a_decay_h:               [nheads] OK
51    /// b_bsgr:                  [batch, sequence, ngroups, state_rank]
52    /// c_bsgr:                  [batch, sequence, ngroups, state_rank]
53    /// d_h:                     [nheads]
54    /// ssm_initial_state_bhpr   [batch, nheads, per_head_dim, state_rank]
55    /// _init_states_hpr         [nheads, per_head_dim, state_rank]
56    ///
57    /// K1 outputs:
58    ///   dA_cumsum          [batch, nheads, nchunks, chunk_len]  (intra-chunk cumsum)
59    ///   dA_chunk_end       [batch, nheads, nchunks]              (last dA_cumsum per chunk)
60    ///
61    /// K2 output:
62    ///   cb_mat             [batch, nchunks, ngroups, chunk_len, chunk_len]
63    ///                      cb_mat[b,c,g,l,s] = sum_k C[b,c*cs+l,g,k] * B[b,c*cs+s,g,k]
64    ///
65    /// K3 output:
66    ///   intra_chunk_states [batch, nchunks, nheads, per_head_dim, d_state]
67    ///                      state assuming zero initial state at each chunk boundary
68    ///
69    /// K4 outputs:
70    ///   chunk_input_states [batch, nchunks, nheads, per_head_dim, d_state]
71    ///                      chunk_input_states[c] = true initial state entering chunk c
72    ///   final_ssm_state    [batch, nheads, per_head_dim, d_state]
73    ///
74    /// K5 output:
75    ///   y_chunked          [batch, nchunks, chunk_len, nheads, per_head_dim]
76    ///   → reshaped →      [batch, sequence, nheads, per_head_dim]
77    /// ```
78    #[allow(non_snake_case)]
79    pub fn chunked_selective_scan_hybrid_naive(
80        x_bshp: Tensor<B, 4>,
81        dt_bsh: Tensor<B, 3>,
82        a_decay_h: Tensor<B, 1>,
83        b_bsgr: Tensor<B, 4>,
84        c_bsgr: Tensor<B, 4>,
85        d_h: Tensor<B, 1>,
86        ssm_initial_state_bhpr: Tensor<B, 4>,
87        _init_states_hpr: Option<Tensor<B, 3>>,
88        ngroups: usize,
89        state_rank: usize,
90        // currently must be set by caller
91        chunk_len: usize,
92    ) -> (Tensor<B, 4>, Tensor<B, 4>) {
93        let [batch, sequence, nheads, per_head_dim] = x_bshp.dims();
94        let device = x_bshp.device();
95        assert_ne!(ngroups, 0);
96        assert_eq!(nheads % ngroups, 0);
97        assert!(sequence > 0, "sequence length must be at least 1");
98        // `heads_per_group` is called `nheads_ngroups_ratio` in every Triton kernel.
99        // It is the compile-time constant used by GQA (Grouped Query Attention) to map
100        // a head index to its B/C group: `group_idx = head_idx / heads_per_group`.
101        let heads_per_group = nheads / ngroups;
102
103        // We require seqlen % chunk_len == 0 for simplicity (the Triton kernel handles a
104        // partial last chunk via `chunk_len_limit = min(chunk_len, seqlen - pid_c * chunk_len)`).
105        debug_assert_eq!(sequence % chunk_len, 0);
106        let nchunks = sequence / chunk_len;
107
108        assert!(
109            _init_states_hpr.is_none(),
110            "init_states_hpr not yet implemented"
111        );
112
113        // ── Reshape into chunks ───────────────────────────────────────────────────────
114        let x_bnlhp = x_bshp.reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
115        let dt_bnlh = dt_bsh.reshape([batch, nchunks, chunk_len, nheads]);
116        let b_bnlgr = b_bsgr.reshape([batch, nchunks, chunk_len, ngroups, state_rank]);
117        let c_bnlgr = c_bsgr.reshape([batch, nchunks, chunk_len, ngroups, state_rank]);
118
119        // ── Permutes ──────────────────────────────────────────────────────────────────
120        // Note: dt_bnlh calculation (originally in Kernel 1) moved to Step 4 (before padding).
121        let dt_discretized_bhnl = dt_bnlh.permute([0, 3, 1, 2]);
122        assert_eq!(
123            [batch, nheads, nchunks, chunk_len],
124            dt_discretized_bhnl.dims()
125        );
126
127        // ── Kernel 1 ──────────────────────────────────────────────────────────────────
128        let da_cumsum_bhnl: Tensor<B, 4> = crate::mamba2::gpu::k1_ssd_chunk_cumsum::naive::forward(
129            dt_discretized_bhnl.clone(),
130            a_decay_h.clone(),
131        );
132        assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
133
134        // dA_chunk_end[b, h, c] = dA_cumsum[b, h, c, chunk_len-1]
135        // = total log-decay accumulated over the entire chunk c for head h.
136        // Used in K4 as the inter-chunk decay exponent.
137        // Triton in _mamba_chunk_scan_combined_fwd: `dA_cumsum[:, :, :, -1]`
138        let da_chunk_end_bhn = da_cumsum_bhnl
139            .clone()
140            .slice(s![.., .., .., -1])
141            .squeeze_dim::<3>(3);
142        assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
143
144        // ── Kernel 2 ──────────────────────────────────────────────────────────────────
145        let c_bnglr = c_bnlgr.clone().permute([0, 1, 3, 2, 4]);
146        let b_bngrl = b_bnlgr.clone().permute([0, 1, 3, 4, 2]);
147        let cb_bngll: Tensor<B, 5> = c_bnglr.matmul(b_bngrl);
148        assert_eq!(
149            [batch, nchunks, ngroups, chunk_len, chunk_len],
150            cb_bngll.dims()
151        );
152        // Note: cb_bngll is then only used by Kernel 5.
153
154        // ── Kernel 3 ──────────────────────────────────────────────────────────────────
155        let intra_chunk_states_bnhpr: Tensor<B, 5> = {
156            use burn::tensor::s;
157
158            let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
159            let [_batch, _nchunks, _chunk_len, ngroups, state_rank] = b_bnlgr.dims();
160
161            // permute b and x to prepare them for the mamtul
162            let x_bnhpl = x_bnlhp.clone().permute([0, 1, 3, 4, 2]);
163            assert_eq!(
164                [batch, nchunks, nheads, per_head_dim, chunk_len],
165                x_bnhpl.dims()
166            );
167            let b_bnglr = b_bnlgr.permute([0, 1, 3, 2, 4]); // note: still in groups instead of heads
168            assert_eq!(
169                [batch, nchunks, ngroups, chunk_len, state_rank],
170                b_bnglr.dims()
171            );
172
173            // Expand B from ngroups to nheads by repeating each group's
174            // projection across all heads_per_group heads in that group.
175            let heads_per_group = nheads / ngroups;
176            let b_bnhlr = b_bnglr
177                .unsqueeze_dim::<6>(3) // bng1lr
178                .expand([
179                    batch,
180                    nchunks,
181                    ngroups,
182                    heads_per_group,
183                    chunk_len,
184                    state_rank,
185                ])
186                .reshape([batch, nchunks, nheads, chunk_len, state_rank]);
187
188            // scale b
189            let b_scaled_bnhlr = {
190                let b_bar_scale_bhnl = {
191                    let da_cumsum_last_in_chunk_bhn1 =
192                        da_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
193                    assert_eq!(
194                        [batch, nheads, nchunks, 1],
195                        da_cumsum_last_in_chunk_bhn1.dims()
196                    );
197
198                    let forward_decay_to_chunk_end_bhnl = (da_cumsum_last_in_chunk_bhn1
199                        .expand([batch, nheads, nchunks, chunk_len])
200                        - da_cumsum_bhnl.clone())
201                    .exp();
202                    assert_eq!(
203                        [batch, nheads, nchunks, chunk_len],
204                        forward_decay_to_chunk_end_bhnl.dims()
205                    );
206
207                    forward_decay_to_chunk_end_bhnl * dt_discretized_bhnl.clone()
208                };
209                assert_eq!([batch, nheads, nchunks, chunk_len], b_bar_scale_bhnl.dims());
210
211                let b_bar_scale_bnhl = b_bar_scale_bhnl.permute([0, 2, 1, 3]);
212                assert_eq!([batch, nchunks, nheads, chunk_len], b_bar_scale_bnhl.dims());
213                let b_bar_scale_bnhlr = b_bar_scale_bnhl
214                    .unsqueeze_dim::<5>(4) // bnhl1
215                    .expand([batch, nchunks, nheads, chunk_len, state_rank]);
216                b_bnhlr * b_bar_scale_bnhlr
217            };
218            assert_eq!(
219                [batch, nchunks, nheads, chunk_len, state_rank],
220                b_scaled_bnhlr.dims()
221            );
222
223            x_bnhpl.matmul(b_scaled_bnhlr) // intra_chunk_states_bnhpr  
224        };
225        assert_eq!(
226            [batch, nchunks, nheads, per_head_dim, state_rank],
227            intra_chunk_states_bnhpr.dims()
228        );
229
230        // ── Kernel 4 ──────────────────────────────────────────────────────────────────
231        let flat_state_dim = per_head_dim * state_rank;
232        let (chunk_input_states_bnhpr, final_ssm_state_bhpr): (Tensor<B, 5>, Tensor<B, 4>) =
233            crate::mamba2::gpu::k4_ssd_state_passing::naive::forward(
234                intra_chunk_states_bnhpr.clone(),
235                da_chunk_end_bhn.clone(),
236                ssm_initial_state_bhpr.clone(),
237            );
238        assert_eq!(
239            [batch, nchunks, nheads, per_head_dim, state_rank],
240            chunk_input_states_bnhpr.dims()
241        );
242        assert_eq!(
243            [batch, nheads, per_head_dim, state_rank],
244            final_ssm_state_bhpr.dims()
245        );
246
247        // ── Kernel 5 ──────────────────────────────────────────────────────────────────
248        let y_bnlhp: Tensor<B, 5> = crate::mamba2::gpu::k5_ssd_chunk_scan::naive::forward(
249            da_cumsum_bhnl.clone(),
250            dt_discretized_bhnl.clone(),
251            x_bnlhp.clone(),
252            c_bnlgr.clone(),
253            cb_bngll.clone(),
254            chunk_input_states_bnhpr.clone(),
255            d_h.clone(),
256        );
257        assert_eq!(
258            [batch, nchunks, chunk_len, nheads, per_head_dim],
259            y_bnlhp.dims()
260        );
261
262        // Reshape from chunked layout back to flat sequence.
263        let y_bshp = y_bnlhp.reshape([batch, sequence, nheads, per_head_dim]);
264
265        (y_bshp, final_ssm_state_bhpr)
266    }
267}
