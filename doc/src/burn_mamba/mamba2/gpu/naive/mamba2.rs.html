1// mamba2-pre-gpu.rs
2//
3// GPU-naive but Triton-faithful forward implementation of the Mamba-2 SSD module.
4//
5// This file mirrors `_mamba_chunk_scan_combined_fwd` from `ssd_combined.py`,
6// which orchestrates the following five GPU kernels, each described in detail below:
7//
8//   Kernel 1: _chunk_cumsum_fwd_kernel  (ssd_chunk_state.py)
9//   Kernel 2: _bmm_chunk_fwd_kernel     (ssd_bmm.py)
10//   Kernel 3: _chunk_state_fwd_kernel   (ssd_chunk_state.py)
11//   Kernel 4: _state_passing_fwd_kernel (ssd_state_passing.py)
12//   Kernel 5: _chunk_scan_fwd_kernel    (ssd_chunk_scan.py)
13//
14// Parallelism conventions used in comments:
15//   blockIdx.{x,y,z} — the outer for-loops in this file; in a real GPU each iteration
16//                       would be a separate thread block running concurrently.
17//   threadIdx / SRAM  — the inner slice/matmul operations inside a loop body; in a real GPU
18//                       these are the per-warp register operations within a thread block.
19//   autotune(N)       — a Triton `@triton.autotune` BLOCK_SIZE_* constant; N is the typical
20//                       best value so a GPU translation can use it as a starting point.
21//
22// Each for-loop carries a `// blockIdx.*` or `// threadIdx` comment to make the
23// mapping explicit. Tile-size variables are given full descriptive names instead of
24// single letters so a reader can immediately understand what dimension they tile.
25
26use crate::mamba2::prelude::*;
27use burn::prelude::*;

29/// Returns the Triton autotune fallback value for a BLOCK_SIZE_* constant.
30///
31/// In a real GPU kernel, Triton picks the best block size at JIT-compile time
32/// by benchmarking configurations listed in `@triton.autotune(configs=[...])`.
33/// Here we record the typical winning value so that CUDA/WGSL/Metal translations
34/// can use it as a sensible starting tiling without re-running the search.
35fn autotune(fallback: usize) -> usize {
36    fallback
37}

39impl<B: Backend> Mamba2<B> {
40    /// GPU-naive, Triton-faithful forward pass for the Mamba-2 SSD module.
41    ///
42    /// Mirrors `_mamba_chunk_scan_combined_fwd` from `ssd_combined.py`,
43    /// which is called from `Mamba2.forward()` via `mamba_chunk_scan_combined()`.
44    ///
45    /// # Tensor shapes flowing through the function
46    ///
47    /// ```text
48    /// x_bshp:                  [batch, sequence, nheads, per_head_dim] OK
49    /// dt_bsh:                  [batch, sequence, nheads] OK
50    /// a_decay_h:               [nheads] OK
51    /// b_bsgr:                  [batch, sequence, ngroups, state_rank]
52    /// c_bsgr:                  [batch, sequence, ngroups, state_rank]
53    /// d_h:                     [nheads]
54    /// ssm_initial_state_bhpr   [batch, nheads, per_head_dim, state_rank]
55    /// _init_states_hpr         [nheads, per_head_dim, state_rank]
56    ///
57    /// K1 outputs:
58    ///   dA_cumsum          [batch, nheads, nchunks, chunk_len]  (intra-chunk cumsum)
59    ///   dA_chunk_end       [batch, nheads, nchunks]              (last dA_cumsum per chunk)
60    ///
61    /// K2 output:
62    ///   cb_mat             [batch, nchunks, ngroups, chunk_len, chunk_len]
63    ///                      cb_mat[b,c,g,l,s] = sum_k C[b,c*cs+l,g,k] * B[b,c*cs+s,g,k]
64    ///
65    /// K3 output:
66    ///   intra_chunk_states [batch, nchunks, nheads, per_head_dim, d_state]
67    ///                      state assuming zero initial state at each chunk boundary
68    ///
69    /// K4 outputs:
70    ///   chunk_input_states [batch, nchunks, nheads, per_head_dim, d_state]
71    ///                      chunk_input_states[c] = true initial state entering chunk c
72    ///   final_ssm_state    [batch, nheads, per_head_dim, d_state]
73    ///
74    /// K5 output:
75    ///   y_chunked          [batch, nchunks, chunk_len, nheads, per_head_dim]
76    ///   → reshaped →      [batch, sequence, nheads, per_head_dim]
77    /// ```
78    #[allow(non_snake_case)]
79    pub fn chunked_selective_scan_hybrid_naive(
80        x_bshp: Tensor<B, 4>,
81        dt_bsh: Tensor<B, 3>,
82        a_decay_h: Tensor<B, 1>,
83        b_bsgr: Tensor<B, 4>,
84        c_bsgr: Tensor<B, 4>,
85        d_h: Tensor<B, 1>,
86        ssm_initial_state_bhpr: Tensor<B, 4>,
87        _init_states_hpr: Option<Tensor<B, 3>>,
88        ngroups: usize,
89        state_rank: usize,
90        // currently must be set by caller
91        chunk_len: usize,
92    ) -> (Tensor<B, 4>, Tensor<B, 4>) {
93        let [batch, sequence, nheads, per_head_dim] = x_bshp.dims();
94        let device = x_bshp.device();
95        assert_ne!(ngroups, 0);
96        assert_eq!(nheads % ngroups, 0);
97        assert!(sequence > 0, "sequence length must be at least 1");
98        // `heads_per_group` is called `nheads_ngroups_ratio` in every Triton kernel.
99        // It is the compile-time constant used by GQA (Grouped Query Attention) to map
100        // a head index to its B/C group: `group_idx = head_idx / heads_per_group`.
101        let heads_per_group = nheads / ngroups;
102
103        // We require seqlen % chunk_len == 0 for simplicity (the Triton kernel handles a
104        // partial last chunk via `chunk_len_limit = min(chunk_len, seqlen - pid_c * chunk_len)`).
105        debug_assert_eq!(sequence % chunk_len, 0);
106        let nchunks = sequence / chunk_len;
107
108        assert!(
109            _init_states_hpr.is_none(),
110            "init_states_hpr not yet implemented"
111        );
112
113        // ── Reshape into chunks ───────────────────────────────────────────────────────
114        let x_bnlhp = x_bshp.reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
115        let dt_bnlh = dt_bsh.reshape([batch, nchunks, chunk_len, nheads]);
116        let b_bnlgr = b_bsgr.reshape([batch, nchunks, chunk_len, ngroups, state_rank]);
117        let c_bnlgr = c_bsgr.reshape([batch, nchunks, chunk_len, ngroups, state_rank]);
118
119        // ================================================================
120        // KERNEL 1: _chunk_cumsum_fwd_kernel  (ssd_chunk_state.py)
121        //
122        // Triton compile-time constants for this call (from ssd_combined.py):
123        //   DT_SOFTPLUS = True
124        //   HAS_DT_BIAS = True  (dt_bias is provided)
125        //
126        // Grid: (batch, nchunks, ceil(nheads / BLOCK_SIZE_H))
127        //   blockIdx.x = pid_b         — batch index
128        //   blockIdx.y = pid_c         — chunk index
129        //   blockIdx.z = pid_h_block   — block of BLOCK_SIZE_H heads
130        //
131        // threadIdx work per block:
132        //   - Loads BLOCK_SIZE_H heads × BLOCK_SIZE_CHUNK positions from dt_raw.
133        //   - Adds dt_bias, applies softplus, clamps to dt_limit.
134        //   - Multiplies by A (scalar per head) to get dA per position.
135        //   - Computes cumulative sum of dA along the chunk_len axis (axis=1 in Triton).
136        //   - Stores dt_discretized and dA_cumsum back to VRAM.
137        //
138        // Outputs are stored in Triton's transposed layout [batch, nheads, nchunks, chunk_len],
139        // which puts the heads dimension before chunks to allow coalesced per-head access
140        // in the downstream kernels.
141        // ================================================================
142
143        // BLOCK_SIZE_H: number of heads processed by one thread block in K1.
144        // The autotune key is ['chunk_len', 'nheads'].
145        let _heads_per_k1_block = autotune(8);
146        // BLOCK_SIZE_CHUNK: positions within a chunk processed by one thread block.
147        // Must be a power-of-two ≥ chunk_len (next_power_of_2 in Triton).
148        let _chunk_pos_k1_block = autotune(128);
149        // These tile sizes are used only for GPU layout annotation; the Rust loops below
150        // iterate per-head per-chunk for clarity.
151        let _ = (_heads_per_k1_block, _chunk_pos_k1_block);
152
153        // Outputs in Triton layout [batch, nheads, nchunks, chunk_len].
154        // Note: dt_bnlh calculation moved to Step 4 (before padding).
155        let dt_discretized_bhnl = dt_bnlh.permute([0, 3, 1, 2]);
156        assert_eq!(
157            [batch, nheads, nchunks, chunk_len],
158            dt_discretized_bhnl.dims()
159        );
160        let mut da_cumsum_bhnl =
161            Tensor::<B, 4>::zeros([batch, nheads, nchunks, chunk_len], &device);
162
163        // blockIdx.x = pid_b
164        for i_batch in 0..batch {
165            // blockIdx.y = pid_c
166            for i_chunk in 0..nchunks {
167                // blockIdx.z = pid_h_block (one block per BLOCK_SIZE_H heads).
168                // We expand one-per-head here; a GPU translation would use:
169                //   offs_h = pid_h_block * BLOCK_SIZE_H + tl.arange(0, BLOCK_SIZE_H)
170                for i_head in 0..nheads {
171                    // threadIdx: load dt_raw tile for this (batch, chunk, head).
172                    // Triton ptr: dt_ptr + pid_b*stride_dt_batch + pid_c*chunk_len*stride_dt_seqlen
173                    //             + pid_h_block*BLOCK_H*stride_dt_head
174                    //   offs_c spans BLOCK_CHUNK positions (threadIdx.x covers seqlen within block)
175                    // Note: dt_l calculation moved to Step 4 (before padding).
176                    let dt_l = dt_discretized_bhnl
177                        .clone()
178                        .slice(s![i_batch, i_head, i_chunk, ..,])
179                        .squeeze_dims::<1>(&[0, 1, 2]); // tile
180                    assert_eq!([chunk_len], dt_l.dims());
181
182                    let da_cumsum_l = {
183                        let a_decay_1 = a_decay_h.clone().slice(s![i_head]);
184                        assert_eq!([1], a_decay_1.dims());
185                        // Per-position decay increment: dA[s] = dt_clamped[s] * A[h].
186                        // A[h] < 0 so dA[s] < 0; exp(dA[s]) ∈ (0, 1) is a damping factor.
187                        let da_decay_increment_l = dt_l.clone() * a_decay_1.expand([chunk_len]);
188                        assert_eq!([chunk_len], da_decay_increment_l.dims());
189                        // Cumulative sum along chunk positions (Triton axis=1).
190                        // dA_cumsum[s] = Σ_{t=0}^{s} dA[t] = total log-decay from chunk start to s.
191                        // Triton: `dA_cs = tl.cumsum(dA, axis=1)`
192                        da_decay_increment_l.cumsum(0) // tile
193                    };
194                    assert_eq!([chunk_len], da_cumsum_l.dims());
195
196                    // threadIdx: write dt_discretized and dA_cumsum to VRAM.
197                    // Triton: tl.store(dt_out_ptrs, dt, mask=(offs_h < nheads) & (offs_c < chunk_len))
198                    //         tl.store(dA_cs_ptrs,  dA_cs, ...)
199                    // Note: dt_discretized already calculated at Step 4.
200                    da_cumsum_bhnl = da_cumsum_bhnl.slice_assign(
201                        s![i_batch, i_head, i_chunk, ..],
202                        da_cumsum_l.unsqueeze_dims::<4>(&[0, 1, 2]),
203                    );
204                } // n_heads loop
205            } // nchunks loop
206        } // batch loop
207
208        // dA_chunk_end[b, h, c] = dA_cumsum[b, h, c, chunk_len-1]
209        // = total log-decay accumulated over the entire chunk c for head h.
210        // Used in K4 as the inter-chunk decay exponent.
211        // Triton in _mamba_chunk_scan_combined_fwd: `dA_cumsum[:, :, :, -1]`
212        let da_chunk_end_bhn = da_cumsum_bhnl
213            .clone()
214            .slice(s![.., .., .., -1])
215            .squeeze_dim::<3>(3);
216        assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
217
218        // ================================================================
219        // KERNEL 2: _bmm_chunk_fwd_kernel  (ssd_bmm.py)
220        //
221        // Triton compile-time constants:
222        //   IS_CAUSAL = False  (full chunk_len × chunk_len matrix, causal mask applied in K5)
223        //   HAS_SEQ_IDX = False  (single-sequence batch, no packing)
224        //
225        // Computes: cb_mat[b, c, g, target_pos, source_pos]
226        //         = Σ_k  C[b, c·cs + target_pos, g, k] · B[b, c·cs + source_pos, g, k]
227        //         = C[target, :] · B[source, :]   (dot product over d_state)
228        //
229        // This is the "CB" product whose (l, s) entry weights the intra-chunk contribution
230        // of source position s to output position l (the off-diagonal orange blocks and the
231        // diagonal blocks of the semiseparable matrix M from Part II of the paper).
232        //
233        // Grid: (ceil(cs/BM) * ceil(cs/BN), batch * nchunks, ngroups)
234        //   blockIdx.x = pid_tile = pid_m * n_tiles_N + pid_n
235        //   blockIdx.y = pid_b * nchunks + pid_c
236        //   blockIdx.z = pid_g   — group index
237        //
238        // threadIdx: standard Tensor Core GEMM tile:
239        //   accumulates a [BLOCK_M, BLOCK_N] output sub-tile over BLOCK_K steps of d_state.
240        //
241        // Output dtype: float32  (`output_dtype=torch.float32` in the Python call).
242        // ================================================================
243
244        // autotune key: ['chunk_len', 'K', 'IS_CAUSAL']
245        let cb_target_pos_tile = autotune(64); // BLOCK_SIZE_M  (target chunk_pos dimension)
246        let cb_source_pos_tile = autotune(64); // BLOCK_SIZE_N  (source chunk_pos dimension)
247        let cb_state_rank_tile = autotune(32); // BLOCK_SIZE_K  (inner state_rank dimension)
248
249        // outputs
250        let mut cb_bngll =
251            Tensor::<B, 5>::zeros([batch, nchunks, ngroups, chunk_len, chunk_len], &device);
252
253        // blockIdx.y part-1: pid_b
254        for i_batch in 0..batch {
255            // blockIdx.y part-2: pid_c
256            for i_chunk in 0..nchunks {
257                // blockIdx.z: pid_g  — one group per block
258                for i_group in 0..ngroups {
259                    // Full [chunk_len, chunk_len] accumulator for this (batch, chunk, group).
260                    // In the GPU, each (pid_m, pid_n) pair is a separate thread block;
261                    // here we accumulate into a single tensor for clarity.
262                    let mut cb_tile_ll = Tensor::<B, 2>::zeros([chunk_len, chunk_len], &device);
263
264                    // blockIdx.x: pid_m — output target-position tile
265                    for target_start in (0..chunk_len).step_by(cb_target_pos_tile) {
266                        let target_end = (target_start + cb_target_pos_tile).min(chunk_len);
267                        let target_tile_width = target_end - target_start;
268                        let target_range = target_start..target_end;
269                        assert_ne!(target_tile_width, 0);
270
271                        // blockIdx.x: pid_n — output source-position tile
272                        for source_start in (0..chunk_len).step_by(cb_source_pos_tile) {
273                            let source_end = (source_start + cb_source_pos_tile).min(chunk_len);
274                            let source_tile_width = source_end - source_start;
275                            let source_range = source_start..source_end;
276                            assert_ne!(source_tile_width, 0);
277
278                            // threadIdx: accumulator for this [target_tile × source_tile] output block.
279                            // Triton: acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=float32)
280                            let mut cb_subtile_accum_LL = Tensor::<B, 2>::zeros(
281                                [target_tile_width, source_tile_width],
282                                &device,
283                            );
284
285                            // threadIdx K-loop: stream through d_state in BLOCK_K steps.
286                            // Triton: `for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):`
287                            //           acc += tl.dot(c_tile, b_tile_transposed)
288                            for state_rank_start in (0..state_rank).step_by(cb_state_rank_tile) {
289                                let state_rank_end =
290                                    (state_rank_start + cb_state_rank_tile).min(state_rank);
291                                let state_rank_width = state_rank_end - state_rank_start;
292                                let state_rank_range = state_rank_start..state_rank_end;
293                                assert_ne!(state_rank_width, 0);
294
295                                // SRAM load: C tile [target_tile_width, state_rank_block_width]
296                                // Triton ptr: a_ptr + offs_m * stride_a_seqlen + offs_k * stride_ak
297                                //   offs_m → target chunk positions, offs_k → d_state indices
298                                let c_LR = c_bnlgr
299                                    .clone()
300                                    .slice(s![
301                                        i_batch,
302                                        i_chunk,
303                                        target_range.clone(),
304                                        i_group,
305                                        state_rank_range.clone()
306                                    ])
307                                    .squeeze_dims::<2>(&[0, 1, 3]); // SRAM subtile
308                                assert_eq!([target_tile_width, state_rank_width], c_LR.dims());
309
310                                // SRAM load: B tile, loaded transposed as [state_rank_block_width, source_tile_width]
311                                // Triton ptr: b_ptr + offs_k * stride_bk + offs_n * stride_b_seqlen
312                                //   offs_k → d_state (fast stride), offs_n → source chunk positions
313                                // The Triton layout means b[state_rank, source_pos], i.e. B is stored
314                                // with d_state as the fast axis, yielding B^T naturally in the matmul.
315                                let b_RL = b_bnlgr
316                                    .clone()
317                                    .slice(s![
318                                        i_batch,
319                                        i_chunk,
320                                        source_range.clone(),
321                                        i_group,
322                                        state_rank_range.clone()
323                                    ])
324                                    .squeeze_dims::<2>(&[0, 1, 3])
325                                    // transpose
326                                    .permute([1, 0]);
327                                assert_eq!([state_rank_width, source_tile_width], b_RL.dims());
328
329                                // Tensor Core: C[target, state_rank] @ B^T[state_rank, source] = [target, source]
330                                // Triton: `acc += tl.dot(a, b)`
331                                cb_subtile_accum_LL = cb_subtile_accum_LL + c_LR.matmul(b_RL);
332                            } // state_rank tile loop
333
334                            // threadIdx: store completed subtile back (SRAM → VRAM).
335                            // Triton: `tl.store(out_ptrs, out, mask=...)`
336                            cb_tile_ll = cb_tile_ll.slice_assign(
337                                s![target_range.clone(), source_range.clone()],
338                                cb_subtile_accum_LL,
339                            );
340                        } // source tile loop
341                    } // target tile loop
342
343                    // Write the full [chunk_len, chunk_len] group-chunk tile to VRAM.
344                    cb_bngll = cb_bngll.slice_assign(
345                        s![i_batch, i_chunk, i_group, .., ..],
346                        cb_tile_ll.unsqueeze_dims::<5>(&[0, 1, 2]),
347                    ); // tile to VRAM
348                } // ngroups loop
349            } // nchunks loop
350        } // batch loop
351
352        // ================================================================
353        // KERNEL 3: _chunk_state_fwd_kernel  (ssd_chunk_state.py)
354        //
355        // Triton compile-time constants:
356        //   HAS_SEQ_IDX = False
357        //
358        // Computes the SSM state at the END of each chunk, assuming the state
359        // was ZERO at the START of that chunk (the zero-initial-state assumption).
360        // Kernel 4 will later correct for the actual non-zero initial state.
361        //
362        //   intra_state[b, c, h, per_head_dim_p, state_rank_n]
363        //     = Σ_{s=0}^{chunk_len-1}  X[b, c, s, h, p]
364        //                                * exp(dA_cumsum[b,h,c,chunk_len-1] - dA_cumsum[b,h,c,s])
365        //                                * dt_discretized[b,h,c,s]
366        //                                * B[b, c, s, group(h), n]
367        //
368        // The factor  exp(dA_last - dA[s]) * dt[s]  is the "scale" per position s:
369        //   - exp(dA_last - dA[s]): decay from position s forward to end-of-chunk
370        //   - dt[s]: discretization factor converting continuous B into discrete B-bar
371        //
372        // Grid: (ceil(per_head_dim/BM)*ceil(d_state/BN), batch*nchunks, nheads)
373        //   blockIdx.x = pid_m * n_tiles_N + pid_n  — tile over per_head_dim × d_state
374        //   blockIdx.y = pid_b * nchunks + pid_c    — batch-chunk pair
375        //   blockIdx.z = pid_h                      — head index
376        //
377        // GQA: B is indexed as B[..., group(h), :] where group(h) = pid_h / heads_per_group.
378        //
379        // states_in_fp32=True in the Python call → output is float32 accumulation.
380        // ================================================================
381
382        // autotune key: ['hdim', 'state_rank', 'chunk_len']
383        let k3_per_head_dim_tile = autotune(64); // BLOCK_SIZE_M  (per_head_dim output dimension)
384        let k3_state_rank_tile = autotune(64); // BLOCK_SIZE_N  (state_rank output dimension)
385        let k3_chunk_pos_tile = autotune(64); // BLOCK_SIZE_K  (chunk_len inner dimension)
386
387        // output
388        let mut intra_chunk_states_bnhpr =
389            Tensor::<B, 5>::zeros([batch, nchunks, nheads, per_head_dim, state_rank], &device);
390
391        // blockIdx.y part-1: pid_b
392        for i_batch in 0..batch {
393            // blockIdx.y part-2: pid_c
394            for i_chunk in 0..nchunks {
395                // blockIdx.z: pid_h
396                for i_head in 0..nheads {
397                    // GQA: map head → group.
398                    // Triton: `b_ptr += (pid_h // nheads_ngroups_ratio) * stride_b_head`
399                    let i_group = i_head / heads_per_group;
400
401                    // dA_cumsum at the LAST position within this chunk (scalar).
402                    // Triton: `dA_cs_last = tl.load(dA_cumsum_ptr + (chunk_len - 1) * stride)`
403                    let da_cumsum_last_in_chunk_1 = da_cumsum_bhnl
404                        .clone()
405                        .slice(s![i_batch, i_head, i_chunk, -1])
406                        .squeeze_dims::<1>(&[0, 1, 2]);
407                    assert_eq!([1], da_cumsum_last_in_chunk_1.dims());
408
409                    // dA_cumsum at every position within this chunk.
410                    // Triton: `dA_cumsum_ptrs = dA_cumsum_ptr + offs_k * stride_dA_cs_csize`
411                    let da_cumsum_within_chunk_l = da_cumsum_bhnl
412                        .clone()
413                        .slice(s![i_batch, i_head, i_chunk, ..])
414                        .squeeze_dims::<1>(&[0, 1, 2]);
415                    assert_eq!([chunk_len], da_cumsum_within_chunk_l.dims());
416
417                    // dt_discretized at every position within this chunk.
418                    // Triton: `dt_ptrs = dt_ptr + offs_k * stride_dt_csize`
419                    let dt_within_chunk_l = dt_discretized_bhnl
420                        .clone()
421                        .slice(s![i_batch, i_head, i_chunk, ..])
422                        .squeeze_dims::<1>(&[0, 1, 2]);
423                    assert_eq!([chunk_len], dt_within_chunk_l.dims());
424
425                    // scale[s] = exp(dA_last - dA[s]) * dt[s]
426                    // = "forward decay from position s to end-of-chunk, times B-bar factor"
427                    // Triton: `scale = tl.exp((dA_cs_last - dA_cs_k)) * dt_k`
428                    let forward_decay_to_chunk_end_l =
429                        (da_cumsum_last_in_chunk_1.expand([chunk_len]) - da_cumsum_within_chunk_l)
430                            .exp();
431                    assert_eq!([chunk_len], forward_decay_to_chunk_end_l.dims());
432
433                    let b_bar_scale_l = forward_decay_to_chunk_end_l * dt_within_chunk_l;
434                    assert_eq!([chunk_len], b_bar_scale_l.dims());
435
436                    // Tiled matmul: X[per_head_dim, chunk_len] @ (B ⊙ scale)[chunk_len, d_state]
437                    // where ⊙ denotes element-wise multiplication over the chunk_len axis.
438                    //
439                    // blockIdx.x: pid_m — per_head_dim output tile
440                    for per_head_dim_start in (0..per_head_dim).step_by(k3_per_head_dim_tile) {
441                        let per_head_dim_end =
442                            (per_head_dim_start + k3_per_head_dim_tile).min(per_head_dim);
443                        let per_head_dim_tile_width = per_head_dim_end - per_head_dim_start;
444                        let per_head_dim_range = per_head_dim_start..per_head_dim_end;
445                        assert_ne!(per_head_dim_tile_width, 0);
446
447                        // blockIdx.x: pid_n — d_state output tile
448                        for state_rank_start in (0..state_rank).step_by(k3_state_rank_tile) {
449                            let state_rank_end =
450                                (state_rank_start + k3_state_rank_tile).min(state_rank);
451                            let state_rank_tile_width = state_rank_end - state_rank_start;
452                            let state_rank_range = state_rank_start..state_rank_end;
453                            assert_ne!(state_rank_tile_width, 0);
454
455                            // threadIdx: accumulator [per_head_dim_tile, state_rank_tile].
456                            // Triton: `acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), float32)`
457                            let mut state_subtile_accum_PR = Tensor::<B, 2>::zeros(
458                                [per_head_dim_tile_width, state_rank_tile_width],
459                                &device,
460                            );
461
462                            // threadIdx K-loop: tile the chunk_len inner dimension.
463                            // Triton: `for k in range(0, chunk_len_limit, BLOCK_SIZE_K):`
464                            for chunk_start in (0..chunk_len).step_by(k3_chunk_pos_tile) {
465                                let chunk_end = (chunk_start + k3_chunk_pos_tile).min(chunk_len);
466                                let chunk_width = chunk_end - chunk_start;
467                                let chunk_range = chunk_start..chunk_end;
468                                assert_ne!(chunk_width, 0);
469
470                                // SRAM load: X tile [per_head_dim_tile, chunk_block]
471                                // Triton ptr: x_ptrs = offs_m * stride_x_hdim + offs_k * stride_x_seqlen
472                                //   (offs_m → per_head_dim, offs_k → seqlen within chunk)
473                                let x_PL = x_bnlhp
474                                    .clone()
475                                    .slice(s![
476                                        i_batch,
477                                        i_chunk,
478                                        chunk_range.clone(),
479                                        i_head,
480                                        per_head_dim_range.clone()
481                                    ]) // SRAM load
482                                    .squeeze_dims::<2>(&[0, 1, 3])
483                                    .permute([1, 0]);
484                                assert_eq!([per_head_dim_tile_width, chunk_width], x_PL.dims());
485
486                                // SRAM load: B tile [chunk_block, state_rank_tile]
487                                // Triton ptr: b_ptrs = offs_n * stride_b_state_rank + offs_k * stride_b_seqlen
488                                //   GQA: group_idx = pid_h / nheads_ngroups_ratio
489                                let b_LR = b_bnlgr
490                                    .clone()
491                                    .slice(s![
492                                        i_batch,
493                                        i_chunk,
494                                        chunk_range.clone(),
495                                        i_group,
496                                        state_rank_range.clone()
497                                    ])
498                                    .squeeze_dims::<2>(&[0, 1, 3]); // SRAM load
499                                assert_eq!([chunk_width, state_rank_tile_width], b_LR.dims());
500
501                                // Scale B by b_bar_scale[s] (broadcast over d_state columns).
502                                // Triton: `b *= scale[:, None]`
503                                let scale_for_block_LR = b_bar_scale_l
504                                    .clone()
505                                    .slice(s![chunk_range.clone()])
506                                    .unsqueeze_dim::<2>(1)
507                                    .expand([chunk_width, state_rank_tile_width]);
508
509                                let b_scaled_LR = b_LR * scale_for_block_LR;
510                                assert_eq!(
511                                    [chunk_width, state_rank_tile_width],
512                                    b_scaled_LR.dims()
513                                );
514
515                                // Tensor Core: X^T[per_head_dim, chunk] @ B_scaled[chunk, state_rank]
516                                // Triton: `acc += tl.dot(x, b_scaled)`
517                                state_subtile_accum_PR =
518                                    state_subtile_accum_PR + x_PL.matmul(b_scaled_LR);
519                            } // chunk tile loop
520
521                            // threadIdx: store completed subtile to VRAM.
522                            // Triton: `tl.store(states_ptrs, states, mask=...)`
523                            intra_chunk_states_bnhpr = intra_chunk_states_bnhpr.slice_assign(
524                                s![
525                                    i_batch,
526                                    i_chunk,
527                                    i_head,
528                                    per_head_dim_range.clone(),
529                                    state_rank_range.clone()
530                                ],
531                                state_subtile_accum_PR.unsqueeze_dims::<5>(&[0, 0, 0]),
532                            );
533                        } // state_rank tile loop
534                    } // per_head_dim tile loop
535                } // nheads loop
536            } // chunk loop
537        } // batch loop
538
539        // ================================================================
540        // KERNEL 4: _state_passing_fwd_kernel  (ssd_state_passing.py)
541        //
542        // This is the ONLY inherently serial kernel in the SSD pipeline.
543        // It performs the SSM recurrence BETWEEN chunks, accumulating history.
544        //
545        // Triton compile-time constants:
546        //   HAS_INITSTATES = True   (initial states from cache.ssm)
547        //   HAS_SEQ_IDX    = False
548        //
549        // The kernel works on the state flattened to a 1D vector per (batch, head):
550        //   flat_state_dim = per_head_dim * d_state
551        //
552        // In the Python call:
553        //   states_flat, _ = _state_passing_fwd(
554        //       rearrange(intra_chunk_states, "... p n -> ... (p n)"),
555        //       dA_chunk_end,
556        //       initial_states=rearrange(cache.ssm, "... p n -> ... (p n)"),
557        //       out_dtype=C.dtype)    ← output cast to C.dtype (typically fp16/bf16)
558        //   chunk_input_states = rearrange(states_flat, "... (p n) -> ... p n", n=d_state)
559        //
560        // Algorithm (per batch-head pair, serial over chunks):
561        //   prev_state = initial_state
562        //   out[0] = prev_state                    ← state entering chunk 0
563        //   for c in 0..nchunks:
564        //     decay = exp(dA_chunk_end[c])
565        //     prev_state = decay * prev_state + intra_chunk_states[c]
566        //     if c < nchunks - 1:
567        //         out[c+1] = prev_state            ← state entering chunk c+1
568        //     else:
569        //         final_ssm_state = prev_state     ← state after all chunks
570        //
571        // Grid: (ceil(flat_state_dim / BLOCK_SIZE), batch, nheads)
572        //   blockIdx.x = pid_dim_block — tile over the flat state dimension
573        //   blockIdx.y = pid_b
574        //   blockIdx.z = pid_h
575        //
576        // threadIdx: each block owns BLOCK_SIZE contiguous elements of the flat state.
577        // The serial loop over chunks is INSIDE the thread block (not a new grid dispatch),
578        // which is what makes this kernel inherently serial.
579        // ================================================================
580
581        // autotune key: ['dim']  where dim = per_head_dim * d_state
582        let flat_state_dim = per_head_dim * state_rank;
583        // BLOCK_SIZE tiles the flat_state_dim dimension across thread blocks.
584        let _state_flat_block_size = autotune(256); // BLOCK_SIZE
585        // The Rust loop below processes the entire flat state at once, corresponding to
586        // all pid_dim_block blocks combined.
587        let _ = _state_flat_block_size;
588
589        // Flatten intra_chunk_states for state-passing arithmetic.
590        // Python: `rearrange(states, "... p n -> ... (p n)")`
591        let intra_states_bnhf =
592            intra_chunk_states_bnhpr.reshape([batch, nchunks, nheads, flat_state_dim]);
593
594        // Initial state from cache, also flattened.
595        // Python: `rearrange(initial_states, "... p n -> ... (p n)")`
596        let initial_ssm_state_bhf =
597            ssm_initial_state_bhpr
598                .clone()
599                .reshape([batch, nheads, flat_state_dim]);
600
601        // outputs
602        // chunk_input_states_flat[b, c, h, :] = state entering chunk c for (batch b, head h).
603        let mut chunk_input_states_bnhf =
604            Tensor::<B, 4>::zeros([batch, nchunks, nheads, flat_state_dim], &device);
605        let mut final_ssm_state_bnf =
606            Tensor::<B, 3>::zeros([batch, nheads, flat_state_dim], &device);
607
608        // blockIdx.y = pid_b
609        for i_batch in 0..batch {
610            // blockIdx.z = pid_h
611            for i_head in 0..nheads {
612                // blockIdx.x = pid_dim_block tiles the flat state dimension.
613                // In a GPU translation, each thread block processes
614                //   offs_m = pid_dim_block * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
615                // Here we process the full flat state to preserve algorithmic clarity.
616
617                // threadIdx: load initial state for this (batch, head) pair.
618                // Triton: `if HAS_INITSTATES: states = tl.load(initstates_ptrs, ...)`
619                let mut running_state_f = initial_ssm_state_bhf
620                    .clone()
621                    .slice(s![i_batch, i_head, ..])
622                    .squeeze_dims::<1>(&[0, 1]);
623                assert_eq!([flat_state_dim], running_state_f.dims());
624
625                // Store the initial state as the state entering chunk 0.
626                // Triton: `tl.store(out_ptrs, states, ...)` BEFORE the loop,
627                //         then `out_ptrs += stride_out_chunk`.
628                chunk_input_states_bnhf = chunk_input_states_bnhf.slice_assign(
629                    s![i_batch, 0, i_head, ..],
630                    running_state_f.clone().unsqueeze_dims::<4>(&[0, 1, 2]),
631                );
632
633                // Serial loop over chunks — this loop CANNOT be parallelised.
634                // Triton: `for c in range(nchunks):`
635                for i_chunk in 0..nchunks {
636                    // threadIdx: load intra-chunk state for this chunk.
637                    // Triton: `new_states = tl.load(states_ptrs, ...)`
638                    let intra_state_for_chunk_f = intra_states_bnhf
639                        .clone()
640                        .slice(s![i_batch, i_chunk, i_head, ..])
641                        .squeeze_dims::<1>(&[0, 1, 2]);
642                    assert_eq!([flat_state_dim], intra_state_for_chunk_f.dims());
643
644                    // Inter-chunk decay factor: exp(dA_chunk_end[b, h, c]).
645                    // Triton: `dA_cs = tl.load(dA_cs_ptr); scale = tl.exp(dA_cs)`
646                    // dA_chunk_end is a scalar per (batch, head, chunk).
647                    let chunk_decay_f = da_chunk_end_bhn
648                        .clone()
649                        .slice(s![i_batch, i_head, i_chunk])
650                        .squeeze_dims::<1>(&[0, 1]) // [1]
651                        .exp()
652                        .expand([flat_state_dim]);
653
654                    // SSM recurrence: running_state = decay * running_state + intra_state
655                    // Triton: `states = scale * states + new_states`
656                    running_state_f = chunk_decay_f * running_state_f + intra_state_for_chunk_f;
657
658                    if i_chunk < nchunks - 1 {
659                        // Store the state entering chunk pid_c + 1.
660                        // Triton: `if c < nchunks - 1: tl.store(out_ptrs, states, ...)`
661                        chunk_input_states_bnhf = chunk_input_states_bnhf.slice_assign(
662                            s![i_batch, i_chunk + 1, i_head, ..],
663                            running_state_f.clone().unsqueeze_dims::<4>(&[0, 1, 2]),
664                        );
665                    } else {
666                        // Last chunk: store as the final SSM state.
667                        // Triton: `else: tl.store(final_states_ptrs, states, ...)`
668                        final_ssm_state_bnf = final_ssm_state_bnf.slice_assign(
669                            s![i_batch, i_head, ..],
670                            running_state_f.clone().unsqueeze_dims::<3>(&[0, 1]),
671                        );
672                    }
673                } // nchunks serial loop
674            } // nheads loop
675        } // batch loop
676
677        // Restore the 5D shape after state-passing.
678        // Python: `rearrange(states_flat, "... (p n) -> ... p n", n=state_rank)`
679        let chunk_input_states_bnhpr =
680            chunk_input_states_bnhf.reshape([batch, nchunks, nheads, per_head_dim, state_rank]);
681        let final_ssm_state_bnpr =
682            final_ssm_state_bnf.reshape([batch, nheads, per_head_dim, state_rank]);
683
684        // ================================================================
685        // KERNEL 5: _chunk_scan_fwd_kernel  (ssd_chunk_scan.py)
686        //
687        // Triton compile-time constants for this call (from ssd_combined.py):
688        //   IS_CAUSAL     = True   — output pos l only sees source pos s ≤ l
689        //   HAS_D         = True   — the skip connection D·x is included
690        //   D_HAS_HDIM    = False  — D is [nheads] (scalar per head, not per channel)
691        //   HAS_Z         = False  — z gating is NOT done here; it is applied separately
692        //                            via self.norm.forward(y, z) after this kernel
693        //   HAS_SEQ_IDX   = False  — single-sequence batches
694        //
695        // For each output position l in chunk c, this kernel accumulates THREE contributions
696        // corresponding to the three colour regions of the semiseparable matrix M
697        // (see Part III of the paper, "Semiseparable Matrix M Block Decomposition"):
698        //
699        //   BLUE  — history contribution (off-diagonal, state entering from previous chunks):
700        //     y_blue[l, p] = exp(dA_cumsum[l]) * Σ_n  C[l, n] * chunk_input_states[c, p, n]
701        //     Triton code:
702        //       scale_m = tl.exp(dA_cs_m)
703        //       acc = tl.dot(C, prev_states) * scale_m[:, None]
704        //
705        //   ORANGE — intra-chunk causal contribution (diagonal blocks of M):
706        //     y_orange[l, p] = Σ_{s ≤ l}  CB[l,s] · exp(dA[l] - dA[s]) · dt[s] · x[s, p]
707        //     Triton code (IS_CAUSAL=True):
708        //       K_MAX = min((pid_m + 1) * BLOCK_SIZE_M, chunk_len_limit)
709        //       for k in range(0, K_MAX, BLOCK_SIZE_K):
710        //           cb *= tl.exp(dA_cs_k - dA_cs_m)  ← note sign: source minus target
711        //           cb *= dt_k
712        //           mask = offs_m[:, None] >= k + offs_k[None, :]
713        //           cb = tl.where(mask, cb, 0.0)
714        //           acc += tl.dot(cb, x)
715        //     Note: the Triton sign convention for the decay in the orange part is
716        //     exp(dA_cs_k - dA_cs_m) which equals exp(dA[source] - dA[target]).
717        //     This is the INVERSE of the blue convention and encodes "decay from target
718        //     back to source" — equivalent to exp(-(dA[target] - dA[source])).
719        //
720        //   SKIP  — D residual (outside the state-space recurrence):
721        //     y_skip[l, p] = D[h] · x[l, p]
722        //     D_HAS_HDIM=False means D is a scalar per head (shape [nheads]).
723        //
724        // Grid: (ceil(chunk_len/BM) * ceil(per_head_dim/BN), batch * nchunks, nheads)
725        //   blockIdx.x = pid_m * n_tiles_N + pid_n  — tile over (chunk_len × per_head_dim)
726        //   blockIdx.y = pid_b * nchunks + pid_c    — batch-chunk pair
727        //   blockIdx.z = pid_h                      — head index
728        // ================================================================
729
730        // autotune key: ['chunk_len', 'hdim', 'state_rank', 'IS_CAUSAL']
731        let k5_output_pos_tile = autotune(64); // BLOCK_SIZE_M  (output chunk_pos dimension)
732        let k5_per_head_dim_tile = autotune(32); // BLOCK_SIZE_N  (per_head_dim output dimension)
733        let k5_state_rank_tile = autotune(64); // BLOCK_SIZE_state_rank (d_state for blue part)
734        let k5_source_pos_tile = autotune(32); // BLOCK_SIZE_K  (source chunk_pos for orange)
735
736        // output
737        let mut y_bnlhp =
738            Tensor::<B, 5>::zeros([batch, nchunks, chunk_len, nheads, per_head_dim], &device);
739
740        // blockIdx.y part-1: pid_b
741        for i_batch in 0..batch {
742            // blockIdx.y part-2: pid_c
743            for i_chunk in 0..nchunks {
744                // blockIdx.z: pid_h
745                for i_head in 0..nheads {
746                    // GQA: CB and C use group_idx for the head dimension.
747                    // Triton: `(pid_h // nheads_ngroups_ratio) * stride_cb_head`
748                    let i_group = i_head / heads_per_group;
749
750                    // Load dA_cumsum for this (batch, head, chunk) — shape [chunk_len].
751                    // Triton ptr: dA_cumsum_ptr + pid_b*stride_batch + pid_c*stride_chunk + pid_h*stride_head
752                    let da_cumsum_l = da_cumsum_bhnl
753                        .clone()
754                        .slice(s![i_batch, i_head, i_chunk, ..])
755                        .squeeze_dims::<1>(&[0, 1, 2]);
756                    assert_eq!([chunk_len], da_cumsum_l.dims());
757
758                    // Load dt_discretized for this (batch, head, chunk) — shape [chunk_len].
759                    // Triton ptr: dt_ptr + pid_b*stride + pid_c*stride + pid_h*stride
760                    let dt_l = dt_discretized_bhnl
761                        .clone()
762                        .slice(s![i_batch, i_head, i_chunk, ..])
763                        .squeeze_dims::<1>(&[0, 1, 2]);
764                    assert_eq!([chunk_len], dt_l.dims());
765
766                    // Load the state entering this chunk: shape [per_head_dim, d_state].
767                    // Triton ptr: prev_states_ptr + pid_b*stride_batch + pid_c*stride_chunk + pid_h*stride_head
768                    // Layout: prev_states[per_head_dim_p, state_rank_n] (row=per_head_dim, col=d_state)
769                    let state_entering_pr = chunk_input_states_bnhpr
770                        .clone()
771                        .slice(s![i_batch, i_chunk, i_head, .., ..])
772                        .squeeze_dims::<2>(&[0, 1, 2]);
773                    assert_eq!([per_head_dim, state_rank], state_entering_pr.dims());
774
775                    // Load x for this (batch, chunk, head): shape [chunk_len, per_head_dim].
776                    // Triton ptr: x_ptr + pid_b*stride + pid_c*chunk_len*stride_seqlen + pid_h*stride_head
777                    let x_lp = x_bnlhp
778                        .clone()
779                        .slice(s![i_batch, i_chunk, .., i_head, ..])
780                        .squeeze_dims::<2>(&[0, 1, 3]);
781                    assert_eq!([chunk_len, per_head_dim], x_lp.dims());
782
783                    // Load CB for this (batch, chunk, group): shape [chunk_len, chunk_len].
784                    // Triton ptr: cb_ptr + ... (pid_h // ratio) * stride_cb_head
785                    let cb_ll = cb_bngll
786                        .clone()
787                        .slice(s![i_batch, i_chunk, i_group, .., ..])
788                        .squeeze_dims::<2>(&[0, 1, 2]);
789                    assert_eq!([chunk_len, chunk_len], cb_ll.dims());
790                    // [target, source]
791
792                    // D skip coefficient for this head.
793                    // D_HAS_HDIM=False: D is [nheads], one scalar per head.
794                    // Triton: `D = tl.load(D_ptr + pid_h * stride_D_head)` (single scalar load)
795                    let d_skip_1 = d_h.clone().slice(s![i_head]);
796                    assert_eq!([1], d_skip_1.dims());
797
798                    // Accumulate the full output for this (batch, chunk, head).
799                    // In the GPU, each (pid_m, pid_n) sub-block is a separate thread block.
800                    // Here we use a single [chunk_len, per_head_dim] tensor.
801                    let mut y_lp = Tensor::<B, 2>::zeros([chunk_len, per_head_dim], &device);
802
803                    // blockIdx.x: pid_m — output target-position tile
804                    for output_pos_start in (0..chunk_len).step_by(k5_output_pos_tile) {
805                        let output_pos_end = (output_pos_start + k5_output_pos_tile).min(chunk_len);
806                        let output_pos_tile_width = output_pos_end - output_pos_start;
807                        let output_pos_range = output_pos_start..output_pos_end;
808                        assert_ne!(output_pos_tile_width, 0);
809
810                        // blockIdx.x: pid_n — per_head_dim output tile
811                        for per_head_dim_start in (0..per_head_dim).step_by(k5_per_head_dim_tile) {
812                            let per_head_dim_end =
813                                (per_head_dim_start + k5_per_head_dim_tile).min(per_head_dim);
814                            let per_head_dim_tile_width = per_head_dim_end - per_head_dim_start;
815                            let per_head_dim_range = per_head_dim_start..per_head_dim_end;
816                            assert_ne!(per_head_dim_tile_width, 0);
817
818                            // threadIdx: accumulator for this [output_pos_tile × per_head_dim_tile] block.
819                            // Triton: `acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=float32)`
820                            let mut output_subtile_accum_LP = Tensor::<B, 2>::zeros(
821                                [output_pos_tile_width, per_head_dim_tile_width],
822                                &device,
823                            );
824
825                            // ----------------------------------------------------------------
826                            // BLUE contribution — prev-state propagated forward to each output pos.
827                            //
828                            // Mathematical form:
829                            //   y_blue[l, p] = exp(dA_cumsum[l]) * Σ_n  C[l, n] * state[p, n]
830                            //
831                            // Triton loads and stores:
832                            //   C_ptrs      = offs_m * stride_C_seqlen + offs_k_state_rank * stride_C_state_rank
833                            //   prev_states_ptrs = offs_n * stride_states_hdim + offs_k_state_rank * stride_states_state_rank
834                            //
835                            // Note the index roles:
836                            //   offs_m  → output chunk positions (l),  shapes C as [output_pos, d_state]
837                            //   offs_n  → per_head_dim output (p),          shapes prev_states as [d_state, per_head_dim]
838                            //   offs_k_state_rank → inner d_state dim (n)
839                            //
840                            // Matmul: C[l, n] @ prev_states[n, p] = [output_pos_tile, per_head_dim_tile]
841                            // ----------------------------------------------------------------
842                            // threadIdx K-loop over d_state (BLOCK_SIZE_state_rank steps).
843                            for state_rank_start in (0..state_rank).step_by(k5_state_rank_tile) {
844                                let state_rank_end =
845                                    (state_rank_start + k5_state_rank_tile).min(state_rank);
846                                let state_rank_block_width = state_rank_end - state_rank_start;
847                                let state_rank_range = state_rank_start..state_rank_end;
848                                assert_ne!(state_rank_block_width, 0);
849
850                                // SRAM load: C tile [output_pos_tile, state_rank_block]
851                                // Triton: C_ptrs = C_ptr + offs_m * stride_C_seqlen + offs_k * stride_C_state_rank
852                                //         GQA: group_idx applied
853                                let c_LR = c_bnlgr
854                                    .clone()
855                                    .slice(s![
856                                        i_batch,
857                                        i_chunk,
858                                        output_pos_range.clone(),
859                                        i_group,
860                                        state_rank_range.clone()
861                                    ])
862                                    .squeeze_dims::<2>(&[0, 1, 3]);
863                                assert_eq!(
864                                    [output_pos_tile_width, state_rank_block_width],
865                                    c_LR.dims()
866                                );
867
868                                // SRAM load: prev_state tile [state_rank_block, per_head_dim_tile]
869                                // Triton: prev_states_ptrs = offs_n * stride_states_hdim
870                                //                           + offs_k * stride_states_state_rank
871                                // state_entering_this_chunk has shape [per_head_dim, d_state].
872                                // We slice [per_head_dim_start..per_head_dim_end, state_rank_start..state_rank_end]
873                                // = [per_head_dim_tile, state_rank_block], then transpose to [state_rank_block, per_head_dim_tile].
874                                // This matches the Triton access pattern where offs_n indexes per_head_dim
875                                // and offs_k indexes d_state.
876                                let state_entering_RP = state_entering_pr
877                                    .clone()
878                                    .slice(s![per_head_dim_range.clone(), state_rank_range.clone()])
879                                    .permute([1, 0]); // transpose
880                                assert_eq!(
881                                    [state_rank_block_width, per_head_dim_tile_width],
882                                    state_entering_RP.dims()
883                                );
884
885                                // Tensor Core: C[output_pos, state_rank] @ prev_state^T[state_rank, per_head_dim]
886                                // Triton: `acc = tl.dot(C, prev_states) * scale_m[:, None]`
887                                // (scale is applied after the d_state accumulation loop)
888                                output_subtile_accum_LP =
889                                    output_subtile_accum_LP + c_LR.matmul(state_entering_RP);
890                            } // state_rank tile loop
891
892                            // Apply the state decay to the blue term.
893                            // exp(dA_cumsum[l]) = how much the initial state has decayed
894                            // by the time we reach output position l in this chunk.
895                            // Triton: `acc = tl.dot(...) * scale_m[:, None]`
896                            //   where `scale_m = tl.exp(dA_cs_m)`
897                            let state_decay_LP = da_cumsum_l
898                                .clone()
899                                .slice(s![output_pos_range.clone()])
900                                .exp()
901                                .unsqueeze_dim::<2>(1)
902                                .expand([output_pos_tile_width, per_head_dim_tile_width]);
903                            output_subtile_accum_LP = output_subtile_accum_LP * state_decay_LP;
904
905                            // ----------------------------------------------------------------
906                            // ORANGE contribution — intra-chunk causal attention.
907                            //
908                            // IS_CAUSAL=True: source position s can only contribute to output
909                            // position l when s ≤ l.
910                            //
911                            // K_MAX (Triton): `min((pid_m + 1) * BLOCK_SIZE_M, chunk_len_limit)`
912                            //   = output_pos_end in our notation.
913                            // Source positions beyond output_pos_end cannot causally contribute
914                            // to any output in [output_pos_start, output_pos_end), so we skip them.
915                            //
916                            // Triton orange loop (from _chunk_scan_fwd_kernel in ssd_chunk_scan.py):
917                            //   for k in range(0, K_MAX, BLOCK_SIZE_K):
918                            //     cb = tl.load(cb_ptrs, ...)
919                            //     dA_cs_k = tl.load(dA_cumsum_ptrs, ...)   ← source positions
920                            //     cb *= tl.exp(dA_cs_m[:, None] - dA_cs_k[None, :])
921                            //                  ^^^^^^^^^^^^^^^^^^^^^^^^^^^
922                            //                  target_cumsum - source_cumsum
923                            //                  Since dA < 0 and target ≥ source in causal region,
924                            //                  dA_cumsum[target] ≤ dA_cumsum[source], so the
925                            //                  difference is ≤ 0 and exp(⋅) ∈ (0, 1].
926                            //     dt_k = tl.load(dt_ptrs, ...)
927                            //     cb *= dt_k
928                            //     if IS_CAUSAL:
929                            //         mask = offs_m[:, None] >= k + offs_k[None, :]
930                            //         cb = tl.where(mask, cb, 0.0)
931                            //     acc += tl.dot(cb, x)
932                            // ----------------------------------------------------------------
933                            let source_pos_upper_limit = output_pos_end; // K_MAX = (pid_m+1)*BM
934                            for source_pos_start in
935                                (0..source_pos_upper_limit).step_by(k5_source_pos_tile)
936                            {
937                                let source_pos_end = (source_pos_start + k5_source_pos_tile)
938                                    .min(source_pos_upper_limit);
939                                let source_pos_tile_width = source_pos_end - source_pos_start;
940                                let source_pos_range = source_pos_start..source_pos_end;
941                                assert_ne!(source_pos_tile_width, 0);
942
943                                // SRAM load: CB tile [output_pos_tile, source_pos_tile]
944                                // Triton: cb_ptrs = cb_ptr + offs_m * stride_cb_csize_m
945                                //                          + offs_k * stride_cb_csize_k
946                                //         GQA: (pid_h // ratio) * stride_cb_head
947                                let cb_LL = cb_ll
948                                    .clone()
949                                    .slice(s![output_pos_range.clone(), source_pos_range.clone()]);
950                                assert_eq!(
951                                    [output_pos_tile_width, source_pos_tile_width],
952                                    cb_LL.dims()
953                                );
954
955                                // Inter-position decay: exp(dA_cumsum[target] - dA_cumsum[source])
956                                //
957                                // Triton: `cb *= tl.exp(dA_cs_m[:, None] - dA_cs_k[None, :])`
958                                //   dA_cs_m → dA_cumsum at target (output) positions  (loaded outside K-loop)
959                                //   dA_cs_k → dA_cumsum at source positions            (loaded inside K-loop)
960                                //
961                                // Physical meaning: the state at source position s has decayed
962                                // by factor exp(sum of dA[s+1..=l]) when it reaches target l.
963                                // Using the inclusive cumsum identity:
964                                //   dA_cumsum[l] - dA_cumsum[s] = sum(dA[s+1..=l])
965                                // So exp(dA_cumsum[target] - dA_cumsum[source]) encodes exactly that
966                                // decay.  Because dA < 0 and target ≥ source (causal), the difference
967                                // is ≤ 0 and the factor lies in (0, 1]. This is consistent with the
968                                // BLUE contribution which uses exp(dA_cumsum[l]) = decay from chunk
969                                // start (before position 0) to position l.
970                                let da_cumsum_at_target_LL = da_cumsum_l
971                                    .clone()
972                                    .slice(s![output_pos_range.clone()])
973                                    .unsqueeze_dim::<2>(1)
974                                    .expand([output_pos_tile_width, source_pos_tile_width]);
975                                let da_cumsum_at_source_LL = da_cumsum_l
976                                    .clone()
977                                    .slice(s![source_pos_range.clone()])
978                                    .unsqueeze_dim::<2>(0)
979                                    .expand([output_pos_tile_width, source_pos_tile_width]);
980
981                                // dA_at_source: [output_pos_tile_width, source_pos_tile_width]
982                                //
983                                // IMPORTANT: sign is (target - source), NOT (source - target).
984                                // target_cumsum ≤ source_cumsum in the causal region (both are
985                                // non-positive and target has accumulated more negative dA steps),
986                                // so the difference is ≤ 0 and exp(⋅) ∈ (0, 1] as required.
987                                let causal_intra_chunk_decay_LL =
988                                    (da_cumsum_at_target_LL - da_cumsum_at_source_LL).exp();
989                                assert_eq!(
990                                    [output_pos_tile_width, source_pos_tile_width],
991                                    causal_intra_chunk_decay_LL.dims()
992                                );
993
994                                // Discretization scale at each source position.
995                                // Triton: `dt_k = tl.load(dt_ptrs, ...); cb *= dt_k`
996                                let dt_at_source_LL = dt_l
997                                    .clone()
998                                    .slice([source_pos_range.clone()])
999                                    .unsqueeze_dim::<2>(0)
1000                                    .expand([output_pos_tile_width, source_pos_tile_width]);
1001
1002                                let cb_weighted_LL =
1003                                    cb_LL * causal_intra_chunk_decay_LL * dt_at_source_LL;
1004                                assert_eq!(
1005                                    [output_pos_tile_width, source_pos_tile_width],
1006                                    cb_weighted_LL.dims()
1007                                );
1008
1009                                // Causal mask: output position (absolute) >= source position (absolute).
1010                                // Triton: `mask = offs_m[:, None] >= k + offs_k[None, :]`
1011                                //   where k = source_pos_start in our notation.
1012                                // In CUDA this would be a per-thread integer comparison:
1013                                //   valid = (output_pos_start + threadIdx_m) >= (source_pos_start + threadIdx_k)
1014                                // We approximate it with float broadcasting:
1015                                let causal_mask_LL = {
1016                                    let output_abs_indices: Vec<f32> =
1017                                        output_pos_range.clone().map(|i| i as f32).collect();
1018                                    let source_abs_indices: Vec<f32> =
1019                                        source_pos_range.clone().map(|j| j as f32).collect();
1020                                    let output_abs_float_LL = Tensor::<B, 1>::from_floats(
1021                                        output_abs_indices.as_slice(),
1022                                        &device,
1023                                    )
1024                                    .unsqueeze_dim::<2>(1)
1025                                    .expand([output_pos_tile_width, source_pos_tile_width]);
1026                                    let source_abs_float_LL = Tensor::<B, 1>::from_floats(
1027                                        source_abs_indices.as_slice(),
1028                                        &device,
1029                                    )
1030                                    .unsqueeze_dim::<2>(0)
1031                                    .expand([output_pos_tile_width, source_pos_tile_width]);
1032                                    // causal_mask[target, source] = True when target_abs >= source_abs
1033                                    output_abs_float_LL.greater_equal(source_abs_float_LL)
1034                                };
1035
1036                                // Zero out anti-causal entries.
1037                                // Triton: `cb = tl.where(mask, cb, 0.0)`
1038                                let cb_causal_LL =
1039                                    cb_weighted_LL.mask_fill(causal_mask_LL.bool_not(), 0.0);
1040
1041                                // SRAM load: X tile [source_pos_tile, per_head_dim_tile]
1042                                // Triton: x_ptrs = offs_k * stride_x_seqlen + offs_n * stride_x_hdim
1043                                let x_LP = x_lp
1044                                    .clone()
1045                                    .slice([source_pos_range.clone(), per_head_dim_range.clone()]);
1046                                assert_eq!(
1047                                    [source_pos_tile_width, per_head_dim_tile_width],
1048                                    x_LP.dims()
1049                                );
1050
1051                                // Tensor Core: CB_causal[output, source] @ X[source, per_head_dim]
1052                                // Triton: `acc += tl.dot(cb_causal.to(x.dtype), x)`
1053                                output_subtile_accum_LP =
1054                                    output_subtile_accum_LP + cb_causal_LL.matmul(x_LP);
1055                            } // source_pos tile loop
1056
1057                            // ----------------------------------------------------------------
1058                            // SKIP contribution — D residual connection.
1059                            //
1060                            // HAS_D=True, D_HAS_HDIM=False: D is a scalar per head.
1061                            // Mathematical form: y_skip[l, p] = D[h] · x[l, p]
1062                            //
1063                            // Triton:
1064                            //   D = tl.load(D_ptr + pid_h * stride_D_head)  ← scalar
1065                            //   x_residual = tl.load(x_ptr + offs_m * stride_seqlen
1066                            //                                + offs_n * stride_hdim)
1067                            //   acc += x_residual * D
1068                            // ----------------------------------------------------------------
1069                            let x_skip_LP = x_lp
1070                                .clone()
1071                                .slice(s![output_pos_range.clone(), per_head_dim_range.clone()]);
1072                            assert_eq!(
1073                                [output_pos_tile_width, per_head_dim_tile_width],
1074                                x_skip_LP.dims()
1075                            );
1076
1077                            let d_skip_LP = d_skip_1
1078                                .clone()
1079                                .unsqueeze_dim::<2>(0)
1080                                .expand([output_pos_tile_width, per_head_dim_tile_width]);
1081
1082                            output_subtile_accum_LP =
1083                                output_subtile_accum_LP + d_skip_LP * x_skip_LP;
1084
1085                            // threadIdx: store the completed [output_pos_tile × per_head_dim_tile]
1086                            // subtile back to the y_for_chunk accumulator (SRAM → registers).
1087                            // Triton: `tl.store(out_ptrs, acc, mask=...)`
1088                            y_lp = y_lp.slice_assign(
1089                                s![output_pos_range.clone(), per_head_dim_range.clone()],
1090                                output_subtile_accum_LP,
1091                            );
1092                        } // per_head_dim tile loop
1093                    } // output_pos tile loop
1094
1095                    // Write this (batch, chunk, head) output tile to VRAM.
1096                    // Triton: out_ptr += pid_b*stride_batch + pid_c*chunk_len*stride_seqlen
1097                    //                  + pid_h*stride_head
1098                    y_bnlhp = y_bnlhp.slice_assign(
1099                        s![i_batch, i_chunk, 0..chunk_len, i_head, 0..per_head_dim],
1100                        y_lp
1101                            // TOD0: unsqueeze_dims::<5>(&[0, 1, 3]) currently panics
1102                            .unsqueeze_dim::<3>(0)
1103                            .unsqueeze_dim::<4>(1)
1104                            .unsqueeze_dim::<5>(3),
1105                    );
1106                } // nheads loop
1107            } // nchunks loop
1108        } // batch loop
1109
1110        // Reshape from chunked layout back to flat sequence.
1111        let y_bshp = y_bnlhp.reshape([batch, sequence, nheads, per_head_dim]);
1112
1113        (y_bshp, final_ssm_state_bnpr)
1114    }
