7    // Wraps the extended backend for `burn`.
8    //
9    /// Returns `(chunk_input_states_bnhpr, final_ssm_state_bhpr)`.
10    ///
11    /// # Shapes
12    /// - `intra_chunk_states_bnhpr` : `[batch, nchunks, nheads, per_head_dim, state_rank]`
13    /// - `da_chunk_end_bhn`         : `[batch, nheads, nchunks]`
14    /// - `initial_states_bhpr`      : `[batch, nheads, per_head_dim, state_rank]`
15    #[cfg(feature = "cubecl")]
16    pub fn k4_ssd_state_passing<B: BackendExt>(
17        intra_chunk_states_bnhpr: Tensor<B, 5>,
18        da_chunk_end_bhn: Tensor<B, 3>,
19        initial_states_bhpr: Tensor<B, 4>,
20    ) -> (Tensor<B, 5>, Tensor<B, 4>) {
21        let [batch, nchunks, nheads, per_head_dim, state_rank] = intra_chunk_states_bnhpr.dims();
22        let flat_state_dim = per_head_dim * state_rank;
23
24        // Flatten the (per_head_dim, state_rank) trailing dims into one flat_state_dim axis.
25        // Python: `rearrange(states, "... p n -> ... (p n)")`
26        let intra_chunk_states_bnhf =
27            intra_chunk_states_bnhpr.reshape([batch, nchunks, nheads, flat_state_dim]);
28        let initial_states_bhf = initial_states_bhpr.reshape([batch, nheads, flat_state_dim]);
29
30        let (chunk_input_states_bnhf, final_ssm_state_bhf) = B::k4_ssd_state_passing(
31            intra_chunk_states_bnhf.into_primitive().tensor(),
32            da_chunk_end_bhn.into_primitive().tensor(),
33            initial_states_bhf.into_primitive().tensor(),
34        );
35
36        let chunk_input_states_bnhf: Tensor<B, 4> =
37            Tensor::from_primitive(TensorPrimitive::Float(chunk_input_states_bnhf));
38        let final_ssm_state_bhf: Tensor<B, 5> =
39            Tensor::from_primitive(TensorPrimitive::Float(final_ssm_state_bhf));
40
41        // Restore the 5D shape after state-passing.
42        // Python: `rearrange(states_flat, "... (p n) -> ... p n", n=state_rank)`
43        let chunk_input_states_bnhpr: Tensor<B, 5> =
44            chunk_input_states_bnhf.reshape([batch, nchunks, nheads, per_head_dim, state_rank]);
45        let final_ssm_state_bhpr: Tensor<B, 4> =
46            final_ssm_state_bhf.reshape([batch, nheads, per_head_dim, state_rank]);
47
48        (chunk_input_states_bnhpr, final_ssm_state_bhpr)
49    }

111pub mod forward {

120    // Implement our custom backend trait for the generic `CubeBackend`.
121    impl<R: CubeRuntime, F: FloatElement, I: IntElement, BT: BoolElement> BackendExt
122        for CubeBackend<R, F, I, BT>
123    {
124        /// Returns `(chunk_input_states_bnhf, final_states_bhf)`.
125        ///
126        /// # Shapes (flat representation — caller handles reshape)
127        /// - `intra_chunk_states_bnhf` : `[batch, nchunks, nheads, flat_state_dim]`
128        /// - `da_chunk_end_bhn`        : `[batch, nheads, nchunks]`
129        /// - `initial_states_bhf`      : `[batch, nheads, flat_state_dim]`
130        fn k4_ssd_state_passing(
131            intra_chunk_states_bnhf: FloatTensor<Self>,
132            da_chunk_end_bhn: FloatTensor<Self>,
133            initial_states_bhf: FloatTensor<Self>,
134        ) -> (FloatTensor<Self>, FloatTensor<Self>) {
135            let client = intra_chunk_states_bnhf.client.clone();
136            let device = intra_chunk_states_bnhf.device.clone();
137            intra_chunk_states_bnhf.assert_is_on_same_device(&da_chunk_end_bhn);
138            intra_chunk_states_bnhf.assert_is_on_same_device(&initial_states_bhf);
139
140            // For simplicity, make sure each tensor is contiguous.
141            let intra = into_contiguous(intra_chunk_states_bnhf);
142            let da = into_contiguous(da_chunk_end_bhn);
143            let init = into_contiguous(initial_states_bhf);
144
145            // Get relevant shapes.
146            let [batch, nchunks, nheads, flat_state_dim] = intra.meta.shape().dims();
147
148            // Tile the flat_state_dim across cubes; each unit owns one element.
149            //
150            //   CUBE_POS_X → tile over flat_state_dim   (0 .. ceil(flat / BLOCK_SIZE))
151            //   CUBE_POS_Y → batch index                (0 .. batch)
152            //   CUBE_POS_Z → head index                 (0 .. nheads)
153            //   UNIT_POS_X → position within tile       (0 .. BLOCK_SIZE)
154            //
155            // The serial loop over `nchunks` lives inside each unit; it cannot be
156            // parallelised because each iteration depends on the previous.
157            //
158            // BLOCK_SIZE = 256 matches the Triton autotune default.
159            pub const STATE_FLAT_BLOCK_SIZE: u32 = 256;
160            let block_size = STATE_FLAT_BLOCK_SIZE;
161            let num_tiles =
162                flat_state_dim.next_multiple_of(block_size as usize) as u32 / block_size;
163
164            // Create output tensors.
165            //
166            // chunk_input_states[b, c, h, f] = SSM state *entering* chunk c.
167            let cis_shape = Shape::from([batch, nchunks, nheads, flat_state_dim]);
168            let cis_buffer = client.empty(cis_shape.num_elements() * core::mem::size_of::<F>());
169            let chunk_input_states = CubeTensor::new_contiguous(
170                client.clone(),
171                device.clone(),
172                cis_shape,
173                cis_buffer,
174                F::dtype(),
175            );
176
177            // final_states[b, h, f] = SSM state *after* the last chunk.
178            let fs_shape = Shape::from([batch, nheads, flat_state_dim]);
179            let fs_buffer = client.empty(fs_shape.num_elements() * core::mem::size_of::<F>());
180            let final_states = CubeTensor::new_contiguous(
181                client.clone(),
182                device.clone(),
183                fs_shape,
184                fs_buffer,
185                F::dtype(),
186            );
187
188            unsafe {
189                kernel::forward::launch_unchecked::<F, R>(
190                    &client,
191                    CubeCount::Static(num_tiles, batch as u32, nheads as u32),
192                    CubeDim::new_1d(block_size),
193                    intra.into_tensor_arg(),
194                    da.into_tensor_arg(),
195                    init.into_tensor_arg(),
196                    chunk_input_states.clone().into_tensor_arg(),
197                    final_states.clone().into_tensor_arg(),
198                    block_size, // #[comptime]
199                )
200            };
201
202            (chunk_input_states, final_states)
203        }
204    }

261/// Kernel implementation in `cubecl`.
262#[cfg(feature = "cubecl")]
263pub mod kernel {
264    use burn::cubecl;
265    use cubecl::prelude::*;
266
267    /// # Shapes
268    /// - `intra_states`        : `[batch, nchunks, nheads, flat_state_dim]`
269    /// - `da_chunk_end`        : `[batch, nheads, nchunks]`
270    /// - `initial_states`      : `[batch, nheads, flat_state_dim]`
271    /// - `chunk_input_states`  : `[batch, nchunks, nheads, flat_state_dim]` (output)
272    /// - `final_states`        : `[batch, nheads, flat_state_dim]`         (output)
273    ///
274    /// # Dispatch
275    /// ```text
276    ///   CubeCount  = (ceil(flat_state_dim / BLOCK_SIZE), batch, nheads)
277    ///   CubeDim    = (BLOCK_SIZE, 1, 1)
278    ///
279    ///   CUBE_POS_X  = tile index over flat_state_dim
280    ///   CUBE_POS_Y  = batch index
281    ///   CUBE_POS_Z  = head index
282    ///   UNIT_POS_X  = lane within tile  →  flat_idx = CUBE_POS_X * BLOCK_SIZE + UNIT_POS_X
283    /// ```
284    ///
285    /// Each unit independently walks the serial chunk-loop for its own flat_state_dim
286    /// element.  No synchronisation is required because units never share state.
287    #[cube(launch_unchecked)]
288    pub fn forward<F: Float>(
289        intra_states: &Tensor<F>,
290        da_chunk_end: &Tensor<F>,
291        initial_states: &Tensor<F>,
292        chunk_input_states: &mut Tensor<F>,
293        final_states: &mut Tensor<F>,
294        #[comptime] block_size: u32,
295    ) {
296        // ── Index derivation ────────────────────────────────────────────────────
297        let batch_idx = CUBE_POS_Y as usize;
298        let head_idx = CUBE_POS_Z as usize;
299        let tile_idx = CUBE_POS_X as usize;
300        let local_idx = UNIT_POS_X as usize;
301
302        // Global position in the flat state vector for this unit.
303        let flat_idx = tile_idx * block_size as usize + local_idx;
304
305        // Guard: the last tile may be partially filled.
306        let flat_state_dim = initial_states.shape(2); // == intra_states.shape(3)
307        if flat_idx >= flat_state_dim {
308            terminate!();
309        }
310
311        // ── Stride pre-computation ──────────────────────────────────────────────
312        // intra_states  : [batch, nchunks, nheads, flat_state_dim]
313        let is_s0 = intra_states.stride(0); // batch stride
314        let is_s1 = intra_states.stride(1); // chunk stride
315        let is_s2 = intra_states.stride(2); // head stride
316        let is_s3 = intra_states.stride(3); // flat stride
317
318        // da_chunk_end  : [batch, nheads, nchunks]
319        let da_s0 = da_chunk_end.stride(0); // batch stride
320        let da_s1 = da_chunk_end.stride(1); // head stride
321        let da_s2 = da_chunk_end.stride(2); // chunk stride
322
323        // initial_states : [batch, nheads, flat_state_dim]
324        let init_s0 = initial_states.stride(0); // batch stride
325        let init_s1 = initial_states.stride(1); // head stride
326        let init_s2 = initial_states.stride(2); // flat stride
327
328        // chunk_input_states : [batch, nchunks, nheads, flat_state_dim]
329        //   (same logical layout as intra_states)
330        let cis_s0 = chunk_input_states.stride(0);
331        let cis_s1 = chunk_input_states.stride(1);
332        let cis_s2 = chunk_input_states.stride(2);
333        let cis_s3 = chunk_input_states.stride(3);
334
335        // final_states : [batch, nheads, flat_state_dim]
336        let fs_s0 = final_states.stride(0);
337        let fs_s1 = final_states.stride(1);
338        let fs_s2 = final_states.stride(2);
339
340        // ── Base offsets (shared across all chunks for this unit) ───────────────
341        // Using usize throughout to avoid overflow on large tensors.
342        let intra_base = batch_idx * is_s0 + head_idx * is_s2 + flat_idx * is_s3;
343        let da_base = batch_idx * da_s0 + head_idx * da_s1;
344        let init_base = batch_idx * init_s0 + head_idx * init_s1 + flat_idx * init_s2;
345        let cis_base = batch_idx * cis_s0 + head_idx * cis_s2 + flat_idx * cis_s3;
346        let fs_base = batch_idx * fs_s0 + head_idx * fs_s1 + flat_idx * fs_s2;
347
348        // ── Load initial state ──────────────────────────────────────────────────
349        // Triton: `if HAS_INITSTATES: states = tl.load(initstates_ptrs, ...)`
350        let mut running_state = initial_states[init_base];
351
352        // Store the initial state as the input for chunk 0.
353        // Triton: `tl.store(out_ptrs, states, ...)` BEFORE the chunk loop.
354        chunk_input_states[cis_base] = running_state; // chunk index 0 → offset 0
355
356        // ── Serial recurrence over chunks ───────────────────────────────────────
357        // This loop is the ONLY inherently serial part of the SSD pipeline.
358        // Triton: `for c in range(nchunks):`
359        //
360        // Each iteration:
361        //   decay          = exp(dA_chunk_end[b, h, c])
362        //   running_state  = decay * running_state + intra_state[b, c, h, f]
363        //
364        // Then the updated state is stored as the *input* for the next chunk
365        // (or as the final SSM state when c == nchunks - 1).
366        let nchunks = intra_states.shape(1); // runtime value
367
368        let mut i_chunk: u32 = 0;
369        while i_chunk < nchunks as u32 {
370            let i_chunk_us = i_chunk as usize;
371
372            // Triton: `new_states = tl.load(states_ptrs, ...)`
373            let intra_pos = intra_base + i_chunk_us * is_s1;
374            let intra_state = intra_states[intra_pos];
375
376            // Triton: `dA_cs = tl.load(dA_cs_ptr); scale = tl.exp(dA_cs)`
377            // da_chunk_end is a scalar per (batch, head, chunk) — all units in this
378            // cube load the same value, which is fine (read-only broadcast).
379            let da_pos = da_base + i_chunk_us * da_s2;
380            let decay = F::exp(da_chunk_end[da_pos]);
381
382            // Triton: `states = scale * states + new_states`
383            running_state = decay * running_state + intra_state;
384
385            // Use `i_chunk + 1 < nchunks` (instead of `i_chunk < nchunks - 1`)
386            // to stay safe with u32 arithmetic when nchunks == 1.
387            if i_chunk + 1 < nchunks as u32 {
388                // Store state entering chunk (i_chunk + 1).
389                // Triton: `tl.store(out_ptrs, states, ...); out_ptrs += stride_out_chunk`
390                let cis_pos = cis_base + (i_chunk_us + 1) * cis_s1;
391                chunk_input_states[cis_pos] = running_state;
392            } else {
393                // Last chunk: this running_state is the final SSM state.
394                // Triton: `tl.store(final_states_ptrs, states, ...)`
395                final_states[fs_base] = running_state;
396            }
397
398            i_chunk += 1;
399        }
400    }

403/// Naive (forward) implementation in `burn`.
404pub mod naive {
405    use burn::prelude::*;
406    use burn::tensor::s;
407
408    /// Returns `(chunk_input_states_bnhpr, final_ssm_state_bhpr)`.
409    ///
410    /// # Shapes
411    /// - `intra_chunk_states_bnhpr` : `[batch, nchunks, nheads, per_head_dim, state_rank]`
412    /// - `da_chunk_end_bhn`         : `[batch, nheads, nchunks]`
413    /// - `ssm_initial_state_bhpr`   : `[batch, nheads, per_head_dim, state_rank]`
414    pub fn forward<B: Backend>(
415        intra_chunk_states_bnhpr: Tensor<B, 5>,
416        da_chunk_end_bhn: Tensor<B, 3>,
417        ssm_initial_state_bhpr: Tensor<B, 4>,
418    ) -> (Tensor<B, 5>, Tensor<B, 4>) {
419        let [batch, nchunks, nheads, per_head_dim, state_rank] = intra_chunk_states_bnhpr.dims();
420
421        let flat_state_dim = per_head_dim * state_rank;
422        let mut running_state_bhpr = ssm_initial_state_bhpr;
423        assert_eq!(
424            [batch, nheads, per_head_dim, state_rank],
425            running_state_bhpr.dims()
426        );
427
428        let mut chunk_input_states_vec_bhpr = Vec::with_capacity(nchunks + 1);
429        chunk_input_states_vec_bhpr.push(running_state_bhpr.clone());
430
431        // serial loop
432        for i_chunk in 0..nchunks {
433            let intra_state_bhpr = intra_chunk_states_bnhpr
434                .clone()
435                .slice(s![.., i_chunk, .., .., ..])
436                .squeeze_dim::<4>(1);
437            assert_eq!(
438                [batch, nheads, per_head_dim, state_rank],
439                intra_state_bhpr.dims()
440            );
441
442            let decay_bhpr = da_chunk_end_bhn
443                .clone()
444                .slice(s![.., .., i_chunk]) // bh1
445                .exp()
446                .unsqueeze_dim::<4>(3) // bh11
447                .expand([batch, nheads, per_head_dim, state_rank]);
448
449            // SSM recurrence: running_state = decay * running_state + intra_state
450            running_state_bhpr = decay_bhpr * running_state_bhpr + intra_state_bhpr;
451            chunk_input_states_vec_bhpr.push(running_state_bhpr.clone());
452        }
453
454        let final_ssm_state_bhpr = chunk_input_states_vec_bhpr.pop().unwrap();
455        let chunk_input_states_bnhpr = Tensor::stack(chunk_input_states_vec_bhpr, 1);
456        assert_eq!(
457            [batch, nchunks, nheads, per_head_dim, state_rank],
458            chunk_input_states_bnhpr.dims()
459        );
460
461        (chunk_input_states_bnhpr, final_ssm_state_bhpr)
462    }
