1//! Kernel 5: SSD chunk-scan forward pass.
2//!
3//! For each output position `l` in chunk `c`, accumulates three contributions
4//! corresponding to the three colour regions of the semiseparable matrix M
5//! (see Part III of the paper, "Semiseparable Matrix M Block Decomposition"):
6//!
7//!   **BLUE**   – history from the SSM state entering this chunk:
8//!     `y_blue[l, p]  = exp(dA_cumsum[l]) * Σ_n  C[l,n] * state_in[p,n]`
9//!
10//!   **ORANGE** – causal intra-chunk attention (source positions s ≤ l):
11//!     `y_orange[l, p] = Σ_{s≤l} CB[l,s] · exp(dA[l]−dA[s]) · dt[s] · x[s,p]`
12//!
13//!   **SKIP**   – D residual connection (D_HAS_HDIM=False: scalar per head):
14//!     `y_skip[l, p]  = D[h] · x[l,p]`
15//!
16//! Reference: `_chunk_scan_fwd_kernel` in `ssd_chunk_scan.py`.
17//!
18//! # Optimisation road-map (first iteration: naive scalar kernel)
19//!
20//! Both BLUE and ORANGE reduce to batched matrix multiplications that can be
21//! accelerated with `cubek::matmul` in a future tiled implementation:
22//!
23//! * **BLUE** (per batch·chunk·head slice):
24//!   `C[L, R]  @  state_in^T[R, P]  →  [L, P]`
25//!   Requires a preprocessing step to broadcast C from *ngroups* to *nheads*
26//!   (GQA) and to transpose `state_in`.  After the matmul, each row `l` is
27//!   scaled by `exp(dA_cumsum[l])`.
28//!
29//! * **ORANGE** (per batch·chunk·head slice):
30//!   `CB_weighted[L, L]  @  X[L, P]  →  [L, P]`
31//!   Requires a preprocessing kernel that fills the lower-triangular
32//!   `CB_weighted[l, s] = CB[l,s] · exp(dA[l]−dA[s]) · dt[s]  (s ≤ l, else 0)`.
33//!   After that the matmul is a standard dense×dense product.
34//!
35//! Once those kernels exist, the per-element loops in `kernel::forward` can be
36//! replaced by calls to `NaiveBatchMatmulFamily::launch_unchecked` (or
37//! `cubek::matmul::launch::launch_ref` at the host level), reducing the
38//! arithmetic intensity per unit and enabling higher occupancy.
39

146/// Backend implementation.
147///
148/// Wraps the `cubecl` kernel for `burn_cubecl`.
149#[cfg(feature = "cubecl")]
150pub mod forward {
151    use super::backend_ext::BackendExt;
152    use burn::cubecl::{CubeCount, CubeDim};
153    use burn::tensor::{Shape, ops::FloatTensor};
154    use burn_cubecl::{
155        CubeBackend, CubeRuntime, FloatElement, IntElement, element::BoolElement,
156        kernel::into_contiguous, tensor::CubeTensor,
157    };
158
159    // Number of GPU units (threads) per cube.
160    //
161    // Each unit computes exactly one element of the output tile, identified by a
162    // flat index `flat_idx = CUBE_POS_X * BLOCK_SIZE + UNIT_POS_X`.  The pair
163    // `(l, p) = (flat_idx / per_head_dim, flat_idx % per_head_dim)` is the
164    // output chunk-position and per-head channel for that unit.
165    //
166    // FUTURE: switching to a 2-D CubeDim (TILE_M × TILE_N) and using shared
167    // memory to cache C, state_in, CB, and X tiles would allow each cube to
168    // leverage `NaiveMatmul::execute` for the BLUE and ORANGE reductions,
169    // amortising global-memory latency across the tile.
170    const BLOCK_SIZE: u32 = 256;
171
172    impl<R: CubeRuntime, F: FloatElement, I: IntElement, BT: BoolElement> BackendExt
173        for CubeBackend<R, F, I, BT>
174    {
175        /// Returns `y_bnlhp : [batch, nchunks, chunk_len, nheads, per_head_dim]`.
176        fn k5_ssd_chunk_scan(
177            da_cumsum_bhnl: FloatTensor<Self>,
178            dt_discretized_bhnl: FloatTensor<Self>,
179            x_bnlhp: FloatTensor<Self>,
180            c_bnlgr: FloatTensor<Self>,
181            cb_bngll: FloatTensor<Self>,
182            chunk_input_states_bnhpr: FloatTensor<Self>,
183            d_h: FloatTensor<Self>,
184        ) -> FloatTensor<Self> {
185            let client = x_bnlhp.client.clone();
186            let device = x_bnlhp.device.clone();
187
188            // Assert same device (spot-check a subset).
189            x_bnlhp.assert_is_on_same_device(&da_cumsum_bhnl);
190            x_bnlhp.assert_is_on_same_device(&chunk_input_states_bnhpr);
191
192            // Ensure all inputs are contiguous before handing to the kernel.
193            let da_cumsum_bhnl = into_contiguous(da_cumsum_bhnl);
194            let dt_discretized_bhnl = into_contiguous(dt_discretized_bhnl);
195            let x_bnlhp = into_contiguous(x_bnlhp);
196            let c_bnlgr = into_contiguous(c_bnlgr);
197            let cb_bngll = into_contiguous(cb_bngll);
198            let chunk_input_states_bnhpr = into_contiguous(chunk_input_states_bnhpr);
199            let d_h = into_contiguous(d_h);
200
201            // x_bnlhp : [batch, nchunks, chunk_len, nheads, per_head_dim]
202            let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.meta.shape().dims();
203
204            // ── Dispatch ─────────────────────────────────────────────────────────
205            //
206            // Each unit owns one (l, p) output element for a fixed (batch, chunk, head).
207            //
208            // flat range   = [0, chunk_len * per_head_dim)
209            // CUBE_POS_X   = tile index over that flat range
210            // CUBE_POS_Y   = batch_idx * nchunks + chunk_idx
211            // CUBE_POS_Z   = head_idx
212            let elems_per_bch = (chunk_len * per_head_dim) as u32;
213            let cube_count_x = elems_per_bch.div_ceil(BLOCK_SIZE);
214
215            // Allocate output tensor.
216            let y_shape = Shape::from([batch, nchunks, chunk_len, nheads, per_head_dim]);
217            let y_buffer = client.empty(y_shape.num_elements() * core::mem::size_of::<F>());
218            let y_bnlhp = CubeTensor::new_contiguous(
219                client.clone(),
220                device.clone(),
221                y_shape,
222                y_buffer,
223                F::dtype(),
224            );
225
226            unsafe {
227                super::kernel::forward::launch_unchecked::<F, R>(
228                    &client,
229                    CubeCount::Static(cube_count_x, (batch * nchunks) as u32, nheads as u32),
230                    CubeDim::new_1d(BLOCK_SIZE),
231                    da_cumsum_bhnl.into_tensor_arg(),
232                    dt_discretized_bhnl.into_tensor_arg(),
233                    x_bnlhp.into_tensor_arg(),
234                    c_bnlgr.into_tensor_arg(),
235                    cb_bngll.into_tensor_arg(),
236                    chunk_input_states_bnhpr.into_tensor_arg(),
237                    d_h.into_tensor_arg(),
238                    y_bnlhp.clone().into_tensor_arg(),
239                    BLOCK_SIZE, // #[comptime]
240                )
241            };
242
243            y_bnlhp
244        }
245    }

316    /// SSD chunk-scan: compute `y[batch, chunk, l, head, p]` for all elements.
317    ///
318    /// Each unit independently computes one output element as:
319    /// ```text
320    ///   y[l, p] = BLUE[l, p] + ORANGE[l, p] + SKIP[l, p]
321    /// ```
322    ///
323    /// # Tensor shapes
324    /// | arg                   | layout                                               |
325    /// |-----------------------|------------------------------------------------------|
326    /// | `da_cumsum`           | `[batch, nheads, nchunks, chunk_len]`                |
327    /// | `dt`                  | `[batch, nheads, nchunks, chunk_len]`                |
328    /// | `x`                   | `[batch, nchunks, chunk_len, nheads, per_head_dim]`  |
329    /// | `c`                   | `[batch, nchunks, chunk_len, ngroups, state_rank]`   |
330    /// | `cb`                  | `[batch, nchunks, ngroups, chunk_len, chunk_len]`    |
331    /// | `chunk_input_states`  | `[batch, nchunks, nheads, per_head_dim, state_rank]` |
332    /// | `d`                   | `[nheads]`                                           |
333    /// | `y`                   | `[batch, nchunks, chunk_len, nheads, per_head_dim]`  |
334    ///
335    /// # Dispatch
336    /// ```text
337    /// CubeCount  = (ceil(chunk_len * per_head_dim / block_size), batch * nchunks, nheads)
338    /// CubeDim    = (block_size, 1, 1)
339    ///
340    /// flat_idx   = CUBE_POS_X * block_size + UNIT_POS_X
341    /// l          = flat_idx / per_head_dim   (output chunk position)
342    /// p          = flat_idx % per_head_dim   (per-head channel index)
343    ///
344    /// batch_idx  = CUBE_POS_Y / nchunks
345    /// chunk_idx  = CUBE_POS_Y % nchunks
346    /// head_idx   = CUBE_POS_Z
347    /// ```
348    #[cube(launch_unchecked)]
349    pub fn forward<F: Float>(
350        da_cumsum: &Tensor<F>,          // [batch, nheads, nchunks, chunk_len]
351        dt: &Tensor<F>,                 // [batch, nheads, nchunks, chunk_len]
352        x: &Tensor<F>,                  // [batch, nchunks, chunk_len, nheads, per_head_dim]
353        c: &Tensor<F>,                  // [batch, nchunks, chunk_len, ngroups, state_rank]
354        cb: &Tensor<F>,                 // [batch, nchunks, ngroups, chunk_len(tgt), chunk_len(src)]
355        chunk_input_states: &Tensor<F>, // [batch, nchunks, nheads, per_head_dim, state_rank]
356        d: &Tensor<F>,                  // [nheads]
357        y: &mut Tensor<F>,              // [batch, nchunks, chunk_len, nheads, per_head_dim]
358        #[comptime] block_size: u32,
359    ) {
360        // ── Index derivation ────────────────────────────────────────────────────
361        let flat_idx = (CUBE_POS_X * block_size + UNIT_POS_X) as usize;
362        let batch_chunk_idx = CUBE_POS_Y as usize;
363        let head_idx = CUBE_POS_Z as usize;
364
365        // Runtime shape queries.
366        // x            : [batch, nchunks, chunk_len, nheads, per_head_dim]
367        // da_cumsum    : [batch, nheads,  nchunks,   chunk_len]
368        let chunk_len = x.shape(2);
369        let per_head_dim = x.shape(4);
370        let state_rank = chunk_input_states.shape(4);
371        let nchunks = da_cumsum.shape(2);
372        let nheads = da_cumsum.shape(1);
373        let ngroups = c.shape(3);
374
375        // Decompose flat_idx → (output chunk position l, per-head channel p).
376        let output_pos = flat_idx / per_head_dim; // l ∈ [0, chunk_len)
377        let per_head_dim_idx = flat_idx % per_head_dim; // p ∈ [0, per_head_dim)
378
379        // Guard: the last cube may be partially filled.
380        if output_pos >= chunk_len {
381            terminate!();
382        }
383
384        // Decompose CUBE_POS_Y → (batch_idx, chunk_idx).
385        let batch_idx = batch_chunk_idx / nchunks;
386        let chunk_idx = batch_chunk_idx % nchunks;
387
388        // GQA: map head index → group index.
389        let heads_per_group = nheads / ngroups;
390        let group_idx = head_idx / heads_per_group;
391
392        // ── Pre-compute stride-scaled base offsets ──────────────────────────────
393        //
394        // da_cumsum : [batch, nheads, nchunks, chunk_len]
395        let da_s0 = da_cumsum.stride(0); // batch
396        let da_s1 = da_cumsum.stride(1); // head
397        let da_s2 = da_cumsum.stride(2); // chunk
398        let da_s3 = da_cumsum.stride(3); // position
399        // Base for (batch, head, chunk); position l or s added in loops.
400        let da_base = batch_idx * da_s0 + head_idx * da_s1 + chunk_idx * da_s2;
401
402        // dt : [batch, nheads, nchunks, chunk_len]
403        let dt_s2 = dt.stride(2);
404        let dt_s3 = dt.stride(3);
405        let dt_base = batch_idx * dt.stride(0) + head_idx * dt.stride(1) + chunk_idx * dt_s2;
406
407        // x : [batch, nchunks, chunk_len, nheads, per_head_dim]
408        let x_s2 = x.stride(2); // chunk-position stride (the inner loop stride)
409        // Base for (batch, chunk, head, p); chunk-pos l or s added in loops.
410        let x_base_bhp = batch_idx * x.stride(0)
411            + chunk_idx * x.stride(1)
412            + head_idx * x.stride(3)
413            + per_head_dim_idx * x.stride(4);
414
415        // c : [batch, nchunks, chunk_len, ngroups, state_rank]
416        let c_s4 = c.stride(4); // state_rank stride
417        // Base for (batch, chunk, l, group); state index n added in BLUE loop.
418        let c_base = batch_idx * c.stride(0)
419            + chunk_idx * c.stride(1)
420            + output_pos * c.stride(2)
421            + group_idx * c.stride(3);
422
423        // cb : [batch, nchunks, ngroups, chunk_len(target), chunk_len(source)]
424        let cb_s4 = cb.stride(4); // source-position stride
425        // Base for (batch, chunk, group, target=l); source s added in ORANGE loop.
426        let cb_base = batch_idx * cb.stride(0)
427            + chunk_idx * cb.stride(1)
428            + group_idx * cb.stride(2)
429            + output_pos * cb.stride(3);
430
431        // chunk_input_states : [batch, nchunks, nheads, per_head_dim, state_rank]
432        let st_s4 = chunk_input_states.stride(4); // state_rank stride
433        // Base for (batch, chunk, head, p); state index n added in BLUE loop.
434        let st_base = batch_idx * chunk_input_states.stride(0)
435            + chunk_idx * chunk_input_states.stride(1)
436            + head_idx * chunk_input_states.stride(2)
437            + per_head_dim_idx * chunk_input_states.stride(3);
438
439        // y : [batch, nchunks, chunk_len, nheads, per_head_dim]
440        let y_idx = batch_idx * y.stride(0)
441            + chunk_idx * y.stride(1)
442            + output_pos * y.stride(2)
443            + head_idx * y.stride(3)
444            + per_head_dim_idx * y.stride(4);
445
446        // ── dA_cumsum at output position l ──────────────────────────────────────
447        //
448        // Used in both BLUE (as exp scale) and ORANGE (as decay reference).
449        let da_cumsum_l = da_cumsum[da_base + output_pos * da_s3];
450
451        // ── BLUE contribution ───────────────────────────────────────────────────
452        //
453        //   y_blue[l, p] = exp(dA_cumsum[l]) * Σ_n  C[l, n] * state_in[p, n]
454        //
455        // The inner Σ_n is a dot-product of row C[l, :] with column state_in[p, :].
456        //
457        // FUTURE tiled version: load tiles of C and state_in into shared memory
458        // and compute via NaiveMatmul::execute (see module-level doc-comment).
459        let mut blue = F::new(0.0);
460        let mut n: u32 = 0;
461        while n < state_rank as u32 {
462            let n_us = n as usize;
463            // C[batch, chunk, l, group, n]
464            let c_val = c[c_base + n_us * c_s4];
465            // state_in[batch, chunk, head, p, n]
466            let st_val = chunk_input_states[st_base + n_us * st_s4];
467            blue += c_val * st_val;
468            n += 1;
469        }
470        // Scale by the state-decay accumulated from chunk start to position l.
471        // Triton: `acc = tl.dot(C, prev_states) * scale_m[:, None]`
472        //         where `scale_m = tl.exp(dA_cs_m)`
473        blue *= F::exp(da_cumsum_l);
474
475        // ── ORANGE contribution ─────────────────────────────────────────────────
476        //
477        //   y_orange[l, p] = Σ_{s≤l} CB[l,s] · exp(dA[l]−dA[s]) · dt[s] · x[s,p]
478        //
479        // IS_CAUSAL=True: only source positions s ≤ l contribute (lower-triangular).
480        //
481        // Triton inner loop (orange):
482        //   for k in range(0, K_MAX, BLOCK_K):        K_MAX = (pid_m+1)*BM
483        //     cb     *= exp(dA_cs_m[:,None] - dA_cs_k[None,:])
484        //     cb     *= dt_k
485        //     mask    = offs_m[:,None] >= offs_k[None,:]
486        //     cb      = where(mask, cb, 0)
487        //     acc    += dot(cb, x)
488        //
489        // FUTURE tiled version: precompute cb_weighted in a separate kernel,
490        // then call NaiveMatmul::execute on cb_weighted @ X.
491        let mut orange = F::new(0.0);
492        // Source positions s ∈ [0, l].
493        let mut s: u32 = 0;
494        while s <= output_pos as u32 {
495            let s_us = s as usize;
496            // CB[batch, chunk, group, l, s]  (target=l, source=s)
497            let cb_val = cb[cb_base + s_us * cb_s4];
498            // dA_cumsum[batch, head, chunk, s]
499            let da_cumsum_s = da_cumsum[da_base + s_us * da_s3];
500            // dt[batch, head, chunk, s]
501            let dt_val = dt[dt_base + s_us * dt_s3];
502            // x[batch, chunk, s, head, p]
503            let x_val = x[x_base_bhp + s_us * x_s2];
504            // Decay: exp(dA_cumsum[l] - dA_cumsum[s])
505            // Both dA_cumsum values are ≤ 0 with cumsum[l] ≤ cumsum[s]
506            // (target accumulated more decay steps), so the difference is ≤ 0
507            // and exp(·) ∈ (0, 1] as required by the causal structure.
508            orange += cb_val * F::exp(da_cumsum_l - da_cumsum_s) * dt_val * x_val;
509            s += 1;
510        }
511
512        // ── SKIP contribution ───────────────────────────────────────────────────
513        //
514        //   y_skip[l, p] = D[h] · x[l, p]
515        //
516        // D_HAS_HDIM=False: D is a scalar per head, shape [nheads].
517        // Triton: `D = tl.load(D_ptr + pid_h * stride_D_head)`
518        let x_lp = x[x_base_bhp + output_pos * x_s2];
519        // d is 1-D contiguous: d[head_idx] = d[head_idx * stride(0)].
520        // For contiguous storage stride(0) == 1, but we use the stride to be safe.
521        let skip = d[head_idx * d.stride(0)] * x_lp;
522
523        // ── Write output ────────────────────────────────────────────────────────
524        y[y_idx] = blue + orange + skip;
525    }
526}

528/// Naive (forward) implementation in `burn`.
529pub mod naive {
530    use burn::prelude::*;
531
532    /// Returns `y_bnlhp`.
533    ///
534    /// # Shapes
535    /// | argument                   | layout                                               |
536    /// |----------------------------|------------------------------------------------------|
537    /// | `da_cumsum_bhnl`           | `[batch, nheads, nchunks, chunk_len]`                |
538    /// | `dt_discretized_bhnl`      | `[batch, nheads, nchunks, chunk_len]`                |
539    /// | `x_bnlhp`                  | `[batch, nchunks, chunk_len, nheads, per_head_dim]`  |
540    /// | `c_bnlgr`                  | `[batch, nchunks, chunk_len, ngroups, state_rank]`   |
541    /// | `cb_bngll`                 | `[batch, nchunks, ngroups, chunk_len, chunk_len]`    |
542    /// | `chunk_input_states_bnhpr` | `[batch, nchunks, nheads, per_head_dim, state_rank]` |
543    /// | `d_h`                      | `[nheads]`                                           |
544    pub fn forward<B: Backend>(
545        da_cumsum_bhnl: Tensor<B, 4>,
546        dt_discretized_bhnl: Tensor<B, 4>,
547        x_bnlhp: Tensor<B, 5>,
548        c_bnlgr: Tensor<B, 5>,
549        cb_bngll: Tensor<B, 5>,
550        chunk_input_states_bnhpr: Tensor<B, 5>,
551        d_h: Tensor<B, 1>,
552    ) -> Tensor<B, 5> {
553        let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
554        let [_b, _n, _l, ngroups, state_rank] = c_bnlgr.dims();
555        let heads_per_group = nheads / ngroups;
556        let device = x_bnlhp.device();
557
558        // Rearrange inputs to the common [batch, nchunks, nheads, ...] ordering used below.
559        let da_cumsum_bnhl = da_cumsum_bhnl.permute([0, 2, 1, 3]); // [b,n,h,l]
560        let dt_bnhl = dt_discretized_bhnl.permute([0, 2, 1, 3]); // [b,n,h,l]
561        let x_bnhlp = x_bnlhp.clone().permute([0, 1, 3, 2, 4]); // [b,n,h,l,p]
562
563        // GQA: expand C  [b,n,l,g,r] → [b,n,h,l,r].
564        let c_bnhlr = c_bnlgr
565            .unsqueeze_dim::<6>(4) // [b,n,l,g,1,r]
566            .expand([
567                batch,
568                nchunks,
569                chunk_len,
570                ngroups,
571                heads_per_group,
572                state_rank,
573            ])
574            .reshape([batch, nchunks, chunk_len, nheads, state_rank])
575            .permute([0, 1, 3, 2, 4]); // [b,n,h,l,r]
576
577        // GQA: expand CB [b,n,g,l,l] → [b,n,h,l,l].
578        let cb_bnhll = cb_bngll
579            .unsqueeze_dim::<6>(3) // [b,n,g,1,l,l]
580            .expand([
581                batch,
582                nchunks,
583                ngroups,
584                heads_per_group,
585                chunk_len,
586                chunk_len,
587            ])
588            .reshape([batch, nchunks, nheads, chunk_len, chunk_len]);
589
590        // ── BLUE: exp(dA[l]) · C[l,:] @ state_in^T ─────────────────────────────
591        //
592        //   blue[b,n,h,l,p] = exp(da[b,n,h,l]) · Σ_r  c[b,n,h,l,r] · state[b,n,h,p,r]
593        //
594        //   [b,n,h,l,r] @ [b,n,h,r,p]  →  [b,n,h,l,p]
595        let blue_bnhlp = c_bnhlr
596            .matmul(chunk_input_states_bnhpr.permute([0, 1, 2, 4, 3]))  // state^T [b,n,h,r,p]
597            * da_cumsum_bnhl
598                .clone()
599                .exp()
600                .unsqueeze_dim::<5>(4)
601                .expand([batch, nchunks, nheads, chunk_len, per_head_dim]);
602
603        // ── ORANGE: causal CB_weighted @ X ──────────────────────────────────────
604        //
605        //   orange[b,n,h,l,p] = Σ_{s≤l} CB[l,s] · exp(da[l]-da[s]) · dt[s] · x[s,p]
606        //
607        // Precompute the full lower-triangular weight matrix, then do a single matmul.
608        let da_target_bnhll = da_cumsum_bnhl
609            .clone()
610            .unsqueeze_dim::<5>(4) // [b,n,h,l,1]
611            .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
612        let da_source_bnhll = da_cumsum_bnhl
613            .unsqueeze_dim::<5>(3) // [b,n,h,1,l]
614            .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
615        let dt_source_bnhll = dt_bnhl
616            .unsqueeze_dim::<5>(3) // [b,n,h,1,l]  (broadcast over target positions)
617            .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
618
619        // Causal mask (0 above the main diagonal, 1 elsewhere).
620        let causal_mask_bnhll =
621            Tensor::ones([batch, nchunks, nheads, chunk_len, chunk_len], &device).tril(0);
622
623        //   [b,n,h,l,l] @ [b,n,h,l,p]  →  [b,n,h,l,p]
624        let orange_bnhlp = (cb_bnhll
625            * (da_target_bnhll - da_source_bnhll).exp()
626            * dt_source_bnhll
627            * causal_mask_bnhll)
628            .matmul(x_bnhlp);
629
630        // ── SKIP: D[h] · x[l,p] ─────────────────────────────────────────────────
631        //
632        //   D_HAS_HDIM = False: D is a scalar per head, shape [nheads].
633        //   Triton: `acc += x_residual * D`
634        let skip_bnlhp = d_h.reshape([1, 1, 1, nheads, 1]).expand([
635            batch,
636            nchunks,
637            chunk_len,
638            nheads,
639            per_head_dim,
640        ]) * x_bnlhp;
641
642        // Permute BLUE + ORANGE from [b,n,h,l,p] back to [b,n,l,h,p], then add SKIP.
643        (blue_bnhlp + orange_bnhlp).permute([0, 1, 3, 2, 4]) + skip_bnlhp // y_bnlhp
644    }
645}

649mod test {
650    use super::*;
651    pub type TestBackend = burn::backend::Cuda<f32, i32>;
652    use burn::tensor::Tolerance;
653    use burn::tensor::{Distribution, Tensor};
654
655    /// Smoke test: cubecl kernel output matches the burn naive reference.
656    #[test]
657    fn kernel_naive_comparison_k5() {
658        let device = Default::default();
659        // Small shapes for quick iteration.
660        let (batch, nheads, ngroups, nchunks, chunk_len, per_head_dim, state_rank) =
661            (2, 4, 2, 3, 16, 8, 4);
662        let heads_per_group = nheads / ngroups;
663        assert_eq!(heads_per_group * ngroups, nheads);
664
665        let da_cumsum_bhnl: Tensor<TestBackend, 4> = Tensor::random(
666            [batch, nheads, nchunks, chunk_len],
667            Distribution::Uniform(-1.0, 0.0),
668            &device,
669        );
670        let dt_bhnl: Tensor<TestBackend, 4> = Tensor::random(
671            [batch, nheads, nchunks, chunk_len],
672            Distribution::Uniform(0.0, 1.0),
673            &device,
674        );
675        let x_bnlhp: Tensor<TestBackend, 5> = Tensor::random(
676            [batch, nchunks, chunk_len, nheads, per_head_dim],
677            Distribution::Default,
678            &device,
679        );
680        let c_bnlgr: Tensor<TestBackend, 5> = Tensor::random(
681            [batch, nchunks, chunk_len, ngroups, state_rank],
682            Distribution::Default,
683            &device,
684        );
685        let cb_bngll: Tensor<TestBackend, 5> = Tensor::random(
686            [batch, nchunks, ngroups, chunk_len, chunk_len],
687            Distribution::Default,
688            &device,
689        );
690        let states_bnhpr: Tensor<TestBackend, 5> = Tensor::random(
691            [batch, nchunks, nheads, per_head_dim, state_rank],
692            Distribution::Default,
693            &device,
694        );
695        let d_h: Tensor<TestBackend, 1> = Tensor::random([nheads], Distribution::Default, &device);
696
697        // Naive (burn reference).
698        let y_naive = naive::forward::<TestBackend>(
699            da_cumsum_bhnl.clone(),
700            dt_bhnl.clone(),
701            x_bnlhp.clone(),
702            c_bnlgr.clone(),
703            cb_bngll.clone(),
704            states_bnhpr.clone(),
705            d_h.clone(),
706        );
707        assert_eq!(
708            [batch, nchunks, chunk_len, nheads, per_head_dim],
709            y_naive.dims()
710        );
711        println!("naive: {}", &y_naive);
712
713        // CubeCL kernel.
714        let y_cubecl: Tensor<TestBackend, 5> = backend_ext::k5_ssd_chunk_scan(
715            da_cumsum_bhnl.clone(),
716            dt_bhnl.clone(),
717            x_bnlhp.clone(),
718            c_bnlgr.clone(),
719            cb_bngll.clone(),
720            states_bnhpr.clone(),
721            d_h.clone(),
722        );
723        assert_eq!(
724            [batch, nchunks, chunk_len, nheads, per_head_dim],
725            y_cubecl.dims()
726        );
727        println!("cubecl: {}", &y_cubecl);
728
729        // Numerical comparison.
730        y_naive
731            .to_data()
732            .assert_approx_eq::<f32>(&y_cubecl.to_data(), Tolerance::default());
733    }
