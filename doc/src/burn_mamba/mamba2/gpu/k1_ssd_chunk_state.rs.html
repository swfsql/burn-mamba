5/// # Shapes
6/// - `dt_discretized_bhnl`   : `[batch, nheads, nchunks, chunk_len]`
7/// - `a_decay_h`             : `[nheads]`
8/// - `da_cumsum_bhnl`        : `[batch, nheads, nchunks, chunk_len]`
9#[cube(launch_unchecked)]
10pub fn forward<F: Float>(
11    dt_discretized_bhnl: &Tensor<F>,
12    a_decay_h: &Tensor<F>,
13    da_cumsum_bhnl: &mut Tensor<F>,
14    #[comptime] num_planes: u32,
15) {
16    let dt = dt_discretized_bhnl;
17    let a = a_decay_h;
18    let num_planes = num_planes as usize;
19
20    let batch_stride = dt.stride(0);
21    let nheads_stride = dt.stride(1);
22    let nchunks_stride = dt.stride(2);
23    let chunk_len_stride = dt.stride(3);
24
25    // Base linear offset for the current (batch, head, chunk) slice.
26    let base_pos = CUBE_POS_X as usize * batch_stride
27        + CUBE_POS_Y as usize * nheads_stride
28        + CUBE_POS_Z as usize * nchunks_stride;
29
30    // Scalar value a[h] (identical for all units in this cube).
31    let a_offset = CUBE_POS_Y as usize * a.stride(0);
32    let a_1 = a[a_offset];
33
34    // Local position within the chunk (0 .. chunk_len-1).
35    let thread_id = UNIT_POS_X;
36    let plane_thread_idx = UNIT_POS_PLANE as usize;
37    let plane_idx = (thread_id / PLANE_DIM) as usize;
38
39    // Effective plane size (handles cases where chunk_len < PLANE_DIM).
40    let plane_size = if CUBE_DIM_X < PLANE_DIM {
41        CUBE_DIM_X
42    } else {
43        PLANE_DIM
44    };
45
46    // Position of the element processed by this unit.
47    let my_pos = base_pos + (thread_id as usize) * chunk_len_stride;
48
49    // Per-position increment for the prefix sum.
50    // eg. increment        = 01 02 03 04 ; 05 06 07 08 ; 09 10 11 12 ; 013 014 015 016
51    let increment = dt[my_pos] * a_1;
52
53    // Shared memory for plane totals / offsets (size known at compile time).
54    let mut shared_totals = SharedMemory::<F>::new(num_planes as usize);
55
56    // 1. Exclusive prefix sum within the current plane.
57    // e.g. local_scan       = 00 01 03 06 ; 00 05 11 18 ; 00 09 19 30 ; 000 013 027 042
58    let local_scan = plane_exclusive_sum(increment);
59
60    // 2. Compute inclusive sum of the entire plane and store in shared memory
61    //    (only the first thread of each plane writes).
62    // e.g. plane_last_scan  =          06 ;          18 ;          30 ;             042
63    let plane_last_scan = plane_shuffle(local_scan, plane_size - 1);
64    // e.g. plane_last_inc   =          04 ;          08 ;          12 ;             016
65    let plane_last_increment = plane_shuffle(increment, plane_size - 1);
66    // e.g. plane_total      =          10 ;          26 ;          42 ;             058
67    let plane_total = plane_last_scan + plane_last_increment;
68    // e.g. shared_totals    =          10 ;          26 ;          42 ;             058
69    if plane_thread_idx == 0 {
70        shared_totals[plane_idx] = plane_total;
71    }
72    sync_cube();
73
74    // 3. Exclusive prefix sum over the plane totals (performed by plane 0).
75    comptime! {
76        if num_planes > PLANE_DIM as usize {
77            panic!("num_planes exceeds PLANE_DIM. The current plane-based reduction assumes PLANE_DIM >= num_planes.");
78        }
79    }
80    // e.g. offset           =           0 ;          10 ;          36 ;             078
81    // e.g. shared_totals    =           0 ;          10 ;          36 ;             078
82    if plane_idx == 0 && plane_thread_idx < num_planes {
83        let offset = plane_exclusive_sum(shared_totals[plane_thread_idx]);
84        shared_totals[plane_thread_idx] = offset;
85    }
86    sync_cube();
87
88    // 4. Full-cube exclusive prefix sum.
89    // e.g. exclusive_result = 00 01 03 06 ; 10 15 21 28 ; 36 45 55 66 ; 078 091 105 120
90    let exclusive_result = local_scan + shared_totals[plane_idx];
91
92    // Convert to inclusive prefix sum (required by the original algorithm).
93    // e.g. inclusive_result = 01 03 06 10 ; 15 21 28 36 ; 45 55 66 78 ; 091 105 120 136
94    let inclusive_result = exclusive_result + increment;
95
96    // Write result.
97    da_cumsum_bhnl[my_pos] = inclusive_result;
98}

100#[allow(unused_mut)]
101pub fn launch<R: Runtime, F: Float + CubeElement>(
102    device: &R::Device,
103    dt_discretized_bhnl: TensorArg<R>,
104    a_decay_h: TensorArg<R>,
105    mut da_cumsum_bhnl: TensorArg<R>,
106    batch: usize,
107    nheads: usize,
108    nchunks: usize,
109    chunk_len: usize,
110) {
111    let client = R::client(device);
112
113    // Compute number of planes per cube (comptime parameter).
114    // Typical plane dimension is 32 on most backends; adjust if a backend-specific
115    // query is available (shared memory is over-allocated safely if needed).
116    let plane_dim = 32u32; // conservative default for CUDA/WebGPU/Metal
117    let num_planes = ((chunk_len as u32) + plane_dim - 1) / plane_dim;
118
119    unsafe {
120        forward::launch_unchecked::<F, R>(
121            &client,
122            CubeCount::Static(batch as u32, nheads as u32, nchunks as u32),
123            CubeDim::new_1d(chunk_len as u32),
124            dt_discretized_bhnl,
125            a_decay_h,
126            da_cumsum_bhnl,
127            num_planes, // #[comptime]
128        )
129    };
