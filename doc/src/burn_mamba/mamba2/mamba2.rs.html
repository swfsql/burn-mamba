1//! # Mamba-2 SSM Block — Structured State Space Duality (SSD)
2//!
3//! This module implements the core **SSD layer** from the paper
4//! *"Transformers are SSMs: Generalized Models and Efficient Algorithms
5//! through Structured State Space Duality"* (Dao & Gu, 2024).
6//!
7//! ## The SSD Model
8//!
9//! The SSD layer is a multi-head selective SSM.  Each head processes a
10//! sequence of `per_head_dim`-dimensinal inputs `X ∈ ℝ^{sequence×per_head_dim}` through the recurrence
11//! (Eq. 1–2 of the paper):
12//!
13//! ```text
14//!   hₜ = Āₜ hₜ₋₁ + B̄ₜ xₜ          (state update)
15//!   yₜ = Cₜᵀ hₜ                  (output readout)
16//! ```
17//!
18//! where:
19//! - `hₜ ∈ ℝ^{`state_rank`×per_head_dim}` is the hidden state
20//! - `Āₜ = exp(Δₜ A) ∈ ℝ` is a scalar decay (the key SSD constraint:
21//!   Āₜ = αₜ · I, i.e. scalar times identity, rather than a diagonal matrix)
22//! - `B̄ₜ = Δₜ Bₜ ∈ ℝᴺ`  is the (discretised) input projection
23//! - `Cₜ ∈ ℝᴺ`           is the output projection
24//! - `Δₜ > 0`             is the (input-dependent) discretisation step size
25//! - `A < 0`              is a learnable scalar decay rate per head
26//!
27//! ## State Space Duality
28//!
29//! Unrolling the recurrence yields an equivalent **attention-like** form
30//! (Eq. 6–7):
31//!
32//! ```text
33//!   M = L ∘ (C Bᵀ)      ∈ ℝ^{sequence×sequence}
34//!   Y = M · X
35//! ```
36//!
37//! where `L` is the **1-semiseparable mask** (Eq. 4–5):
38//!
39//! ```text
40//!   Lᵢⱼ = āᵢ · āᵢ₋₁ · ... · āⱼ₊₁    (i ≥ j)
41//!   Lᵢⱼ = 0                     (i < j)
42//! ```
43//!
44//! This makes the layer equivalent to causal linear attention
45//! `Y = (L ∘ QKᵀ) V` under the renaming `(C, B, X) ↦ (Q, K, V)`.
46//!
47//! ## The Chunkwise SSD Algorithm
48//!
49//! See [`ssd::minimal`](crate::mamba2::ssd::minimal) for more information.
50//!
51//! ## Relationship to the other families
52//!
53//! [`Mamba-1`](crate::mamba1) is the predecessor: a sequential selective scan
54//! with a diagonal (rather than scalar) state transition.  [`Mamba-3`](crate::mamba3)
55//! extends the SSD layer below with trapezoidal discretisation, data-dependent
56//! RoPE on B/C, and MIMO rank expansion.
57//!
58//! ## Notation / Dimension Keys
59//!
60//! Throughout all Mamba-2 files, tensor names carry a suffix representing their shape.
61//! The letters used differ from the reference paper and the python implementation.
62//! The "Paper" column gives the symbol from the Mamba-2 paper (Dao & Gu, 2024); the
63//! "Python" column gives the field/variable name in the reference implementation
64//! (`refs/state-spaces/mamba/mamba_ssm/modules/mamba2.py`).
65//!
66//! | Letter | Dimension | Paper | Python | Typical value |
67//! |--------|-----------|-------|--------|---------------|
68//! | `b`    | `batch` | — | `batch` | varies |
69//! | `s`    | `sequence` length | `T` | `seqlen` | varies |
70//! | `d`    | `d_model` | `D` | `d_model` | 768, 1024 |
71//! | `i`    | `d_inner` = `expand`·`d_model` | `E·D` | `d_inner` | 2·`d_model` |
72//! | `h`    | `nheads` | `H` | `nheads` | `d_inner` / `per_head_dim` |
73//! | `p`    | `per_head_dim` | `P` | `headdim` | 64, 128 |
74//! | `r`    | `state_rank` | `N` | `d_state` | 64, 256 |
75//! | `v`    | `conv_dim` | — | `conv_dim` | `d_inner` + 2·`ngroups`·`state_rank` |
76//! | `k`    | `conv_kernel` | — | `d_conv` | 4 |
77//! | `g`    | `ngroups` | `G` | `ngroups` | 1, .., `nheads` |
78//! | `n`    | `nchunks` = `sequence`/`chunk_len` | — | `nchunks` | varies |
79//! | `l`    | `chunk_len` | `Q` | `chunk_size` | 64, .., 256 |
80//!
81//! Uppercase letters represent a relation (e.g. offset, multiple, concat, stacking)
82//! of the lowercase letters. e.g. `X` may represent `x+1`, `x-1`, `x*2`, etc.
83//! `XY` may also represent `x+y`, `x*y`, etc.

101// Mamba2  (the SSM block)
102// ---------------------------------------------------------------------------
103
104/// The Mamba-2 SSM block.
105///
106/// Implements the full SSD layer as described in §5 of the paper.  Supports
107/// two execution modes:
108///
109/// - [`Self::forward`] — chunkwise SSD for training / prefill
110///   (exploits tensor cores; linear in sequence length)
111/// - [`Self::step`]    — pure recurrent form for token-by-token decoding
112///   (O(`nheads`·`per_head_dim`·`state_rank`) per step)
113#[derive(Module, Debug)]
114pub struct Mamba2 {
115    /// Input projection: maps `d_model → d_inner + conv_dim + nheads`.
116    ///
117    /// The output is split into three parts:
118    /// - `z      [batch, sequence, d_inner]`  — multiplicative gate for the output norm
119    /// - `xbc    [batch, sequence, conv_dim]` — input to the causal convolution, which
120    ///   is then split into (x, B, C) after activation
121    /// - `dt_raw [batch, sequence, nheads]`   — raw (pre-softplus) discretisation step Δ
122    pub in_proj: Linear,
123
124    /// Causal depthwise Conv1d applied to the `xbc` projection.
125    ///
126    /// - Input/output channels: `conv_dim`
127    /// - Kernel size: `conv_kernel` (typically 4)
128    /// - Groups: `conv_dim` (fully depthwise — each channel is independent)
129    /// - Padding: **none** (left-padding is applied manually so the convolution
130    ///   is strictly causal)
131    ///
132    /// The convolution provides a local `conv_kernel`-token context window
133    /// before the SSM, which helps the model capture short-range dependencies
134    /// that the SSM's recurrent form handles less efficiently.
135    pub conv1d: Conv1d,
136
137    /// Per-head bias for the discretisation step size Δ.
138    ///
139    /// Shape: `[nheads]`
140    ///
141    /// At inference time, `Δₜ = softplus(dt_rawₜ + dt_bias)`.
142    /// Initialised such that the corresponding initial `Δ` values are
143    /// log-uniformly distributed in `[dt_min, dt_max]`.
144    pub dt_bias_h: Param<Tensor<1>>,
145
146    /// Hard clamp applied to Δ after softplus:  `Δ ∈ [dt_limit.0, dt_limit.1]`.
147    ///
148    /// Prevents degenerate discretisations (e.g. Δ → 0 causes Ā → 1, meaning
149    /// the state never decays; Δ → ∞ causes Ā → 0, meaning the state is
150    /// immediately wiped each step).
151    pub dt_limit: (f64, f64),
152
153    /// Per-head log-magnitude of the continuous-time decay parameter A.
154    ///
155    /// Shape: `[nheads]`
156    ///
157    /// The actual (negative) decay rate is `A = -exp(a_log)`.  The discrete
158    /// decay is `Āₜ = exp(Δₜ · A) = exp(-Δₜ · exp(a_log)) ∈ (0, 1)`.
159    ///
160    /// Storing the *log* of the magnitude and negating ensures A < 0
161    /// (decaying system) unconditionally and avoids any sign-constraint
162    /// during gradient descent.
163    pub a_log_h: Param<Tensor<1>>,
164
165    /// Per-head skip (D) coefficient.
166    ///
167    /// Shape: `[nheads]`
168    ///
169    /// Adds a direct path from the (post-convolution, pre-SSM) input to the
170    /// output:  `yₜ += D · xₜ`.  Initialised to ones.
171    pub d_h: Param<Tensor<1>>,
172
173    /// Gated RMSNorm applied to the SSM output, conditioned on the gate `z`.
174    ///
175    /// Input channel dimension: `d_inner`.
176    ///
177    /// This combines the multiplicative gate (from `z`) and a normalisation
178    /// step into a single fused operation, matching the architecture in §5.2
179    /// of the paper.
180    pub norm: RmsNormGated,
181
182    /// Output projection: maps `d_inner → d_model`.
183    pub out_proj: Linear,
184
185    /// Optional learnable initial hidden state `h₀`.
186    ///
187    /// Shape: `[nheads, per_head_dim, state_rank]`.
188    ///
189    /// When `None`, the initial state is zero (the standard default).
190    /// When `Some`, the stored tensor is used as the initial condition for
191    /// *every* forward call (not per-batch; it is broadcast over the batch
192    /// dimension).
193    pub init_state_hpr: Option<Param<Tensor<3>>>,
194
195    /// `state_rank` — the number of latent dimensions in the SSM hidden
196    /// state `h ∈ ℝ^{state_rank×per_head_dim}` per head.
197    ///
198    /// Paper: `N`. Python: `d_state`.
199    pub state_rank: usize,
200
201    /// Number of B/C groups `ngroups` for grouped SSM heads (analogous to
202    /// grouped-query attention). `ngroups` divides `nheads`; all `nheads/ngroups` heads
203    /// within a group share the same B and C projections while having
204    /// independent X, A, and Z projections.
205    ///
206    /// Paper: `G`. Python: `ngroups`.
207    pub ngroups: usize,
208}
209
210impl Mamba2 {
211    /// `d_inner = expand · d_model`.  Inferred from the norm's weight shape.
212    pub fn d_inner(&self) -> usize {
213        let [d_inner] = self.norm.gamma.dims();
214        d_inner
215    }
216
217    /// `nheads = d_inner / per_head_dim`.
218    pub fn nheads(&self) -> usize {
219        // Inferred from `a_log_h`
220        let [nheads] = self.a_log_h.dims();
221        nheads
222    }
223
224    /// `per_head_dim = d_inner / nheads`.
225    pub fn per_head_dim(&self) -> usize {
226        self.d_inner() / self.nheads()
227    }
228
229    /// `conv_dim = d_inner + 2 · ngroups · state_rank`.
230    pub fn conv_dim(&self) -> usize {
231        self.d_inner() + 2 * self.ngroups * self.state_rank
232    }
233}

235// ---------------------------------------------------------------------------
236// Mamba2Config  (hyperparameters and factory)
237// ---------------------------------------------------------------------------
238
239/// Hyperparameters for the Mamba-2 SSM block.
240///
241/// All computed quantities (e.g. `nheads`, `d_inner`, `conv_dim`) are derived
242/// from the stored fields; see the helper methods on [`Mamba2Config`].
243#[derive(Config, Debug)]
244pub struct Mamba2Config {
245    /// Model (hidden) dimension. Every token is represented as a
246    /// `d_model`-dimensional vector at the input and output of the block.
247    ///
248    /// Paper: `D`. Python: `d_model`.
249    pub d_model: usize,
250
251    /// State rank — the latent dimension of the SSM hidden state.
252    ///
253    /// Larger `state_rank` gives a more expressive state but increases memory and compute
254    /// per step. The paper uses N ∈ {64, 128, 256} for most experiments.
255    ///
256    /// Paper: `N`. Python: `d_state`.
257    #[config(default = 128)]
258    pub state_rank: usize,
259
260    /// Causal convolution window length. Typically 4.
261    ///
262    /// Python: `d_conv`.
263    #[config(default = 4)]
264    pub conv_kernel: usize,
265
266    /// Expansion factor for `d_inner = expand · d_model`.
267    ///
268    /// An expansion of 2 doubles the internal width, keeping the parameter
269    /// count of the SSM block comparable to a standard attention layer.
270    ///
271    /// Paper: `E`. Python: `expand`.
272    #[config(default = 2)]
273    pub expand: usize,
274
275    /// Head dimension. The total `d_inner` is split into
276    /// `nheads = d_inner / per_head_dim` independent SSM heads.
277    ///
278    /// Typical values: 64 or 128.
279    ///
280    /// Paper: `P`. Python: `headdim`.
281    #[config(default = 64)]
282    pub per_head_dim: usize,
283
284    /// Number of B/C groups. Must divide `nheads`.
285    ///
286    /// Setting `ngroups < nheads` reduces the B and C projection sizes (analogous to
287    /// GQA in attention), saving memory without a large accuracy cost.
288    ///
289    /// Paper: `G`. Python: `ngroups`.
290    #[config(default = 1)]
291    pub ngroups: usize,
292
293    /// Range `[lo, hi]` for the uniform initialisation of the magnitude of A.
294    ///
295    /// `A = -Uniform(lo, hi)`, stored as `a_log = log(Uniform(lo, hi))`.
296    /// The paper uses `[1, 16]` by default.
297    #[config(default = "(1., 16.)")]
298    pub a_init_range: (f64, f64),
299
300    /// Gated RMSNorm mode: when `true` the norm is applied *before* the gate;
301    /// when `false` (default) the gate is applied first (SiLU-gated norm).
302    #[config(default = false)]
303    pub is_norm_before_gate: bool,
304
305    /// Minimum value of the initial Δ distribution.  Used to set `dt_bias`.
306    #[config(default = 1e-3)]
307    pub dt_min: f64,
308
309    /// Maximum value of the initial Δ distribution.  Used to set `dt_bias`.
310    #[config(default = 0.1)]
311    pub dt_max: f64,
312
313    /// Floor clamped onto the sampled initial Δ values before inverting to
314    /// obtain `dt_bias`.  Prevents numerical issues with very small Δ.
315    #[config(default = 1e-4)]
316    pub dt_init_floor: f64,
317
318    /// Hard clamp limits for Δ at runtime: `Δ ∈ [dt_limit.0, dt_limit.1]`.
319    ///
320    /// Defaults to `(0, f16::MAX ≈ 65504)`, effectively only clamping at 0.
321    #[config(default = "(0., 6.5504e+4)")]
322    pub dt_limit: (f64, f64),
323
324    /// Whether to add a bias term to the `in_proj` and `out_proj` projections.
325    #[config(default = false)]
326    pub has_proj_bias: bool,
327
328    /// Whether to add a bias term to the depthwise convolution.
329    #[config(default = true)]
330    pub has_conv_bias: bool,
331
332    /// Whether to allocate a learnable initial SSM state `h₀`.
333    ///
334    /// When `false` (default), the hidden state starts at zero for every
335    /// sequence.  When `true`, `init_state_hpr` is allocated as a trainable
336    /// parameter of shape `[nheads, per_head_dim, state_rank]`.
337    #[config(default = false)]
338    pub has_learnable_init_state: bool,
339}
340
341impl Mamba2Config {
342    // -----------------------------------------------------------------------
343    // Computed dimensions
344    // -----------------------------------------------------------------------
345
346    /// `d_inner = expand · d_model`.
347    pub fn d_inner(&self) -> usize {
348        self.expand * self.d_model
349    }
350
351    /// `nheads = d_inner / per_head_dim`.
352    pub fn nheads(&self) -> usize {
353        self.d_inner() / self.per_head_dim
354    }
355
356    /// `conv_dim = d_inner + 2 · ngroups · state_rank`.
357    ///
358    /// The depthwise convolution processes `x`, `B`, and `C` concatenated:
359    /// x contributes `d_inner` channels, B and C each contribute
360    /// `ngroups · state_rank` channels.
361    pub fn conv_dim(&self) -> usize {
362        self.d_inner() + 2 * self.ngroups * self.state_rank
363    }
364
365    // -----------------------------------------------------------------------
366    // Initialisation
367    // -----------------------------------------------------------------------
368
369    /// Allocate and initialise all Mamba-2 block parameters on `device`.
370    pub fn init(&self, device: &Device) -> Mamba2 {
371        let d_inner = self.d_inner();
372        let nheads = self.nheads();
373        let conv_dim = self.conv_dim();
374
375        assert!(self.per_head_dim > 0, "per_head_dim must be positive");
376        assert_eq!(
377            nheads * self.per_head_dim,
378            d_inner,
379            "d_inner must be divisible by per_head_dim"
380        );
381        assert_eq!(
382            nheads % self.ngroups,
383            0,
384            "nheads must be divisible by ngroups"
385        );
386
387        // Uniform initialiser matching PyTorch's default: U(-1/√fan_in, 1/√fan_in).
388        let uniform_init = |fan_in: usize| {
389            let bound = 1.0 / (fan_in as f64).sqrt();
390            Initializer::Uniform {
391                min: -bound,
392                max: bound,
393            }
394        };
395
396        // ── in_proj ──────────────────────────────────────────────────────────
397        // Projects d_model → (z, xbc, dt_raw).
398        // Size:  d_inner  +  conv_dim  +  nheads
399        let d_in_proj_out = d_inner + conv_dim + nheads;
400        let in_proj = LinearConfig::new(self.d_model, d_in_proj_out)
401            .with_bias(self.has_proj_bias)
402            .with_initializer(uniform_init(self.d_model))
403            .init(device);
404
405        // ── conv1d ───────────────────────────────────────────────────────────
406        // Causal depthwise convolution.  Left-padding is applied manually in
407        // `forward` and `step`, so we request "Valid" (no automatic padding).
408        // The initialiser fan_in is `in_channels / groups * kernel_size = 1 * conv_kernel`.
409        let conv1d = Conv1dConfig::new(conv_dim, conv_dim, self.conv_kernel)
410            .with_padding(burn::nn::PaddingConfig1d::Valid)
411            .with_groups(conv_dim)
412            .with_bias(self.has_conv_bias)
413            .with_initializer(uniform_init(self.conv_kernel))
414            .init(device);
415
416        // ── dt_bias ──────────────────────────────────────────────────────────
417        // We want the initial Δ values (after softplus) to be log-uniformly
418        // distributed in [dt_min, dt_max].  The inverse softplus (inverse of
419        // softplus(x) = ln(1 + exp(x))) is used to back-solve for the bias:
420        //   dt_bias = softplus⁻¹(dt) = dt + ln(1 - exp(-dt)) ≈ dt + ln(dt)
421        // which simplifies to `dt + log(exp(dt) - 1)` in the formula below.
422        let expm1 = |t: Tensor<1>| t.exp() - 1.;
423        let dt_h = Tensor::random(
424            [nheads],
425            burn::tensor::Distribution::Uniform(self.dt_min.ln(), self.dt_max.ln()),
426            device,
427        )
428        .exp();
429        let dt_h = dt_h.clamp(self.dt_init_floor, f64::INFINITY);
430        // Inverse softplus: softplus⁻¹(y) = y + log(1 - e^{-y}) = y + log(e^y - 1) - y = log(e^y - 1)
431        let inv_dt_h = dt_h.clone() + (-expm1(-dt_h)).log();
432        let dt_bias_h = Param::from_tensor(inv_dt_h);
433
434        // ── a_log ─────────────────────────────────────────────────────────────
435        // A is constrained to be negative (decaying system).
436        // We store a_log = log(|A|) and recover A = -exp(a_log) at runtime.
437        // This parameterisation ensures A < 0 unconditionally.
438        assert!(
439            self.a_init_range.0 > 0.0,
440            "a_init_range lower bound must be > 0"
441        );
442        assert!(
443            self.a_init_range.0 < self.a_init_range.1,
444            "a_init_range must satisfy lo < hi"
445        );
446        let a_h = Tensor::random(
447            [nheads],
448            burn::tensor::Distribution::Uniform(self.a_init_range.0, self.a_init_range.1),
449            device,
450        );
451        let a_log_h = Param::from_tensor(a_h.log());
452
453        // ── D (skip connection) ───────────────────────────────────────────────
454        // Initialised to ones, adding a direct residual path from input to output.
455        let d_h = Initializer::Ones.init::<1, _>([nheads], device);
456
457        // ── norm (gated RMSNorm) and out_proj ─────────────────────────────────
458        let norm = RmsNormGatedConfig::new(d_inner)
459            .with_norm_before_gate(self.is_norm_before_gate)
460            .init(device);
461        let out_proj = LinearConfig::new(d_inner, self.d_model)
462            .with_bias(self.has_proj_bias)
463            .with_initializer(uniform_init(d_inner))
464            .init(device);
465
466        // ── learnable initial state (optional) ────────────────────────────────
467        let init_state_hpr = self.has_learnable_init_state.then(|| {
468            Initializer::Zeros.init::<3, _>([nheads, self.per_head_dim, self.state_rank], device)
469        });
470
471        Mamba2 {
472            in_proj,
473            conv1d,
474            dt_bias_h,
475            dt_limit: self.dt_limit,
476            a_log_h,
477            d_h,
478            norm,
479            out_proj,
480            init_state_hpr,
481            state_rank: self.state_rank,
482            ngroups: self.ngroups,
483        }
484    }
485}

487// ---------------------------------------------------------------------------
488// Mamba2::forward  (chunkwise SSD — training / prefill)
489// ---------------------------------------------------------------------------
490
491impl Mamba2 {
492    /// Process a full input sequence using the chunkwise SSD algorithm.
493    ///
494    /// This is the primary training and prefill path.  The computation is
495    /// **linear in sequence** but uses batched matrix multiplications (GEMMs) that
496    /// can exploit GPU tensor cores — unlike the naive sequential recurrence,
497    /// which requires O(sequence) serial steps.
498    ///
499    /// ## Full dataflow
500    ///
501    /// 1. **In-projection**: `u → (z, xbc, dt_raw)` via a single linear layer.
502    /// 2. **Causal Conv1d + SiLU**: local context mixing over `xbc`.
503    /// 3. **Split**: `xbc → (x, B, C)`.
504    /// 4. **Discretise**: `Δ = softplus(dt_raw + dt_bias)`;
505    ///    `Ā = exp(Δ · A)`;  `B̄ = Δ · B`.
506    /// 5. **Padding**: sequence padding.
507    /// 6. **SSD Algorithm**: chunkwise selective scan algorithm selection.
508    ///    See [`Mamba2SsdPath`] for more info.
509    /// 7. **Gated RMSNorm**: `y = RMSNorm(y) · σ(z)`.
510    /// 8. **Out-projection**: `y → output`.
511    ///
512    /// ## Sequence padding
513    ///
514    /// If `sequence_unpadded % chunk_len ≠ 0`, the sequence is zero-padded
515    /// to the next multiple of chunk_len.  Zero-padding is equivalent to inserting
516    /// identity steps (`Δ = 0  ⇒  Ā = exp(0) = 1,  B̄ = 0`), so the SSM
517    /// state is carried forward unchanged through the pad — making it safe to
518    /// read the final state of the padded last chunk as the true final state.
519    ///
520    /// ## Shapes
521    /// - `input_bsm` : `[batch, sequence, d_model]`
522    /// - output      : `[batch, sequence, d_model]`
523    /// - cache (out) : updated convolution window and SSM state
524    #[allow(non_snake_case)]
525    pub fn forward(
526        &self,
527        input_bsm: Tensor<3>,
528        cache: Option<Mamba2Cache>,
529        ssd_path: Mamba2SsdPath,
530    ) -> (Tensor<3>, Mamba2Cache) {
531        let [batch, sequence, _d_model] = input_bsm.dims();
532        let d_inner = self.d_inner();
533        let ngroups = self.ngroups;
534        let nheads = self.nheads();
535        let per_head_dim = self.per_head_dim();
536        let conv_dim = self.conv_dim();
537        let state_rank = self.state_rank;
538        let [_conv_dim, _, conv_kernel] = self.conv1d.weight.dims();
539        let [_d_model, d_in_proj_out] = self.in_proj.weight.dims();
540        let device = input_bsm.device();
541        assert_eq!(conv_dim, _conv_dim);
542        assert_ne!(ngroups, 0);
543        assert_eq!(conv_dim, _conv_dim);
544        assert_eq!(nheads % ngroups, 0);
545        assert!(sequence > 0, "sequence length must be at least 1");
546        san(&input_bsm);
547
548        // ── Initialise cache if not provided ──────────────────────────────────
549        let mut cache = cache.unwrap_or_else(|| {
550            let conv_bvk = Tensor::zeros(Shape::new([batch, conv_dim, conv_kernel]), &device);
551            let ssm_bhpr = Tensor::zeros(
552                Shape::new([batch, nheads, per_head_dim, state_rank]),
553                &device,
554            );
555            Mamba2Cache { conv_bvk, ssm_bhpr }
556        });
557        cache.sanity();
558
559        // ── Step 1: In-projection ─────────────────────────────────────────────
560        // One linear layer projects the input to all SSM parameters at once.
561        // This "parallel projection" structure (vs. Mamba-1's sequential
562        // projections) enables tensor parallelism with only 1 all-reduce per
563        // layer instead of 2.
564        //
565        // Projection output:  [z | xbc | dt_raw]
566        //   `z      [batch, sequence, d_inner]`  — gate for the output RMSNorm
567        //   `xbc    [batch, sequence, conv_dim]` — will become (x, B, C) after conv + split
568        //   `dt_raw [batch, sequence, nheads]`   — raw discretisation step (pre-softplus)
569        let (z_gate_bsi, xbc_bsv, dt_raw_bsh) = {
570            let z_xbc_dt_bsd = self.in_proj.forward(input_bsm);
571            assert_eq!([batch, sequence, d_in_proj_out], z_xbc_dt_bsd.dims());
572            assert_eq!(
573                [batch, sequence, d_inner + conv_dim + nheads],
574                z_xbc_dt_bsd.dims(),
575            );
576
577            let [z_gate_bsi, xbc_bsv, dt_raw_bsh] =
578                crate::utils::split::split_into(z_xbc_dt_bsd, [d_inner, conv_dim, nheads], 2);
579            (z_gate_bsi, xbc_bsv, dt_raw_bsh)
580        };
581        assert_eq!([batch, sequence, d_inner], z_gate_bsi.dims());
582        assert_eq!([batch, sequence, conv_dim], xbc_bsv.dims());
583        assert_eq!([batch, sequence, nheads], dt_raw_bsh.dims());
584        san(&z_gate_bsi);
585        san(&xbc_bsv);
586        san(&dt_raw_bsh);
587
588        // ── Step 2: Causal depthwise Conv1d ───────────────────────────────────
589        // Apply the causal 1-dimensional depthwise convolution to `xbc`.  To maintain
590        // strict causality, the input is left-padded with the last
591        // `(conv_kernel - 1)` columns from the cache (the tail of the previous
592        // chunk), giving a padded input of length `(conv_kernel-1) + sequence`.
593        // After the convolution the output has length `sequence` (Valid padding).
594        //
595        // The right-most `conv_kernel` columns of the padded input become the
596        // new convolution cache for the next call.
597        let xbc_bvs = xbc_bsv.permute([0, 2, 1]);
598        assert_eq!([batch, conv_dim, sequence], xbc_bvs.dims());
599
600        // Build the causally-padded input: [cached tail | new input]
601        let xbc_padded_bvS = if conv_kernel >= 2 {
602            // Drop the oldest (leftmost) element of the cache, keeping the
603            // last (conv_kernel - 1) columns.
604            let tail_bvK = cache.conv_bvk.slice(s![.., .., 1..]);
605            assert_eq!([batch, conv_dim, conv_kernel - 1], tail_bvK.dims());
606            Tensor::cat(vec![tail_bvK, xbc_bvs], 2)
607        } else {
608            // conv_kernel == 1: no causal padding needed.
609            xbc_bvs
610        };
611        assert_eq!(
612            [batch, conv_dim, (conv_kernel - 1) + sequence],
613            xbc_padded_bvS.dims()
614        );
615        san(&xbc_padded_bvS);
616
617        // Update the cache: save the last `conv_kernel` columns of the padded
618        // input (i.e. starting at position `sequence - 1` from the new input).
619        cache.conv_bvk = xbc_padded_bvS.clone().slice(s![.., .., (sequence - 1)..]);
620        assert_eq!([batch, conv_dim, conv_kernel], cache.conv_bvk.dims());
621
622        // Apply the depthwise convolution and transpose back to [batch, sequence, conv_dim].
623        let xbc_bvs = self.conv1d.forward(xbc_padded_bvS);
624        assert_eq!([batch, conv_dim, sequence], xbc_bvs.dims());
625        san(&xbc_bvs);
626
627        let xbc_bsv = xbc_bvs.permute([0, 2, 1]);
628        assert_eq!([batch, sequence, conv_dim], xbc_bsv.dims());
629
630        // SiLU activation (element-wise).
631        let xbc_bsv = Silu::new().forward(xbc_bsv);
632        assert_eq!([batch, sequence, conv_dim], xbc_bsv.dims());
633        san(&xbc_bsv);
634
635        // ── Step 3: Split xbc into (x, B, C) ──────────────────────────────────
636        // After activation, xbc is partitioned along the channel dimension:
637        //   x_bsi          → reshaped to x_bshp   (input)
638        //   b_bsGR        → reshaped to b_bsgr   (state input proj)
639        //   c_bsGR        → reshaped to c_bsgr   (state output proj)
640        //
641        // Note: in the SSM/attention duality, C ↔ Q, B ↔ K, x ↔ V.
642        let (x_bshp, b_bsgr, c_bsgr) = {
643            let [x_bsi, b_bsGR, c_bsGR] = crate::utils::split::split_into(
644                xbc_bsv,
645                [d_inner, ngroups * state_rank, ngroups * state_rank],
646                2,
647            );
648            (
649                x_bsi.reshape([batch, sequence, nheads, per_head_dim]), // x_bshp
650                b_bsGR.reshape([batch, sequence, ngroups, state_rank]), // b_bsgr
651                c_bsGR.reshape([batch, sequence, ngroups, state_rank]), // c_bsgr
652            )
653        };
654        // No shape assertions on reshapes (shapes are algebraically guaranteed).
655
656        // ── Step 4: Discretisation ────────────────────────────────────────────
657        // Compute the discrete-time parameters from the continuous-time A
658        // and input-dependent step size Δ (ZOH discretisation, §4.5):
659        //
660        //   Δₜ = softplus(dt_rawₜ + dt_bias)     ∈ (0, ∞)
661        //   Āₜ = exp(Δₜ · A)                       ∈ (0, 1)   [scalar per head]
662        //   B̄ₜ = Δₜ · Bₜ                           ∈ ℝᴺ       [Euler approx]
663        //
664        // The Euler approximation B̄ ≈ ΔB (instead of the exact ZOH formula)
665        // is standard in Mamba-1 and Mamba-2 (see §4.5 of the reference).
666        //
667        // `a_head_decay_h` = A = -exp(a_log) < 0 (negative, one scalar per head).
668        // Note: we pass this negative value to the ssd algo; inside
669        // that function it is multiplied by Δ > 0, giving a negative exponent
670        // which produces Āₜ = exp(Δₜ·A) ∈ (0,1) as required.
671        let dt_bias_11h = self.dt_bias_h.val().unsqueeze_dims(&[0, 1]);
672        assert_eq!([1, 1, nheads], dt_bias_11h.dims());
673
674        let dt_bsh = softplus(dt_raw_bsh + dt_bias_11h).clamp(self.dt_limit.0, self.dt_limit.1);
675        assert_eq!([batch, sequence, nheads], dt_bsh.dims());
676        san(&dt_bsh);
677
678        let a_head_decay_h = -self.a_log_h.val().exp(); // A = -exp(a_log) < 0
679        assert_eq!([nheads], a_head_decay_h.dims());
680        san(&a_head_decay_h);
681
682        // ── Step 5: Pad sequence to a multiple of chunk_len ───────────────────
683        // Zeros are the correct pad: Δ=0  ⇒  Ā=exp(0·A)=1, B̄=0·B=0.
684        // The state is thus carried through unchanged, so the final state of
685        // the padded last chunk equals the state after the last real token.
686        let chunk_len = ssd_path.chunk_len_or_optimal(state_rank, per_head_dim);
687        assert!(chunk_len > 0);
688        let sequence_padded = sequence.next_multiple_of(chunk_len);
689        let pad = sequence_padded - sequence;
690        let (x_bShp, dt_bSh, b_bSgr, c_bSgr) = if pad == 0 {
691            (x_bshp.clone(), dt_bsh, b_bsgr, c_bsgr)
692        } else {
693            let pad_bShp = Tensor::zeros(Shape::new([batch, pad, nheads, per_head_dim]), &device);
694            let pad_bSh = Tensor::zeros(Shape::new([batch, pad, nheads]), &device);
695            let pad_bSgr = Tensor::zeros(Shape::new([batch, pad, ngroups, state_rank]), &device);
696            let x_bshp = Tensor::cat(vec![x_bshp.clone(), pad_bShp], 1);
697            let dt_bsh = Tensor::cat(vec![dt_bsh, pad_bSh], 1);
698            let b_bsgr = Tensor::cat(vec![b_bsgr, pad_bSgr.clone()], 1);
699            let c_bsgr = Tensor::cat(vec![c_bsgr, pad_bSgr], 1);
700            (x_bshp.clone(), dt_bsh, b_bsgr, c_bsgr)
701        };
702
703        // ── Reshapes into chunks ───────────────────────────────────────────────
704        let nchunks = sequence_padded / chunk_len;
705        let x_bnlhp = x_bShp.reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
706        let dt_bnlh = dt_bSh.reshape([batch, nchunks, chunk_len, nheads]);
707        let b_bnlgr = b_bSgr.reshape([batch, nchunks, chunk_len, ngroups, state_rank]);
708        let c_bnlgr = c_bSgr.reshape([batch, nchunks, chunk_len, ngroups, state_rank]);
709
710        // GQA expansion: B and C are produced per-group, but the SSD algorithms expect
711        // them per-head. Group dim is at axis 3 (`[b, n, l, g, r]`).
712        let b_bnlhr = crate::utils::gqa::gqa_expand_to_heads::<5, 6>(b_bnlgr, 3, nheads);
713        let c_bnlhr = crate::utils::gqa::gqa_expand_to_heads::<5, 6>(c_bnlgr, 3, nheads);
714
715        // ── Step 6: Selective Scan ────────────────────────────────────────────
716        let ssd_input = crate::mamba2::ssd::Mamba2SsdInput {
717            x_bnlhp,
718            dt_bnlh,
719            a_decay_h: a_head_decay_h,
720            b_bnlhr,
721            c_bnlhr,
722            d_h: self.d_h.val(),
723            initial_state_bhpr: cache.ssm_bhpr,
724            init_state_hpr: self.init_state_hpr.as_ref().map(|s| s.val()),
725        };
726        ssd_input.sanity();
727        let (y_bnlhp, final_state_bhpr) = ssd_input.run(&ssd_path);
728        assert_eq!(
729            [batch, nchunks, chunk_len, nheads, per_head_dim],
730            y_bnlhp.dims()
731        );
732        san(&y_bnlhp);
733        san(&final_state_bhpr);
734
735        // Update the SSM state in the cache for the next call.
736        cache.ssm_bhpr = final_state_bhpr;
737        assert_eq!(
738            [batch, nheads, per_head_dim, state_rank],
739            cache.ssm_bhpr.dims()
740        );
741
742        // Remove zero-pad columns that were added at Step 5.
743        let y_bShp = y_bnlhp.reshape([batch, sequence_padded, nheads, per_head_dim]);
744        let y_bshp = if pad == 0 {
745            y_bShp
746        } else {
747            y_bShp.slice(s![.., 0..sequence, .., ..])
748        };
749
750        // Reshape into sequence.
751        let y_bsi = y_bshp.reshape([batch, sequence, d_inner]);
752        assert_eq!([batch, sequence, d_inner], y_bsi.dims());
753
754        // ── Step 7: Gated RMSNorm ─────────────────────────────────────────────
755        let y_bsi = self.norm.forward(y_bsi, z_gate_bsi);
756        assert_eq!([batch, sequence, d_inner], y_bsi.dims());
757        san(&y_bsi);
758
759        // ── Step 8: Out-projection ────────────────────────────────────────────
760        let out_bsm = self.out_proj.forward(y_bsi);
761        assert_eq!([batch, sequence, _d_model], out_bsm.dims());
762        san(&out_bsm);
763
764        (out_bsm, cache)
765    }
766}

768// ---------------------------------------------------------------------------
769// Mamba2::step  (recurrent SSM — token-by-token decoding)
770// ---------------------------------------------------------------------------
771
772mod step {
773    use super::*;
774
775    impl Mamba2 {
776        /// Process a **single token** using the pure recurrent SSM form.
777        ///
778        /// This is the O(nheads·per_head_dim·state_rank)-per-token decoding path.  It runs one tick of
779        /// the discretised Mamba-2 recurrence:
780        ///
781        /// ```text
782        ///   Āₜ  = exp(Δₜ · A)           scalar per head, ∈ (0, 1)
783        ///   B̄ₜ  = Δₜ · Bₜ               ∈ ℝᴺ   (Euler discretisation)
784        ///   hₜ  = Āₜ · hₜ₋₁ + B̄ₜ · xₜᵀ  ∈ ℝ^{per_head_dim×state_rank}   (outer product update)
785        ///   yₜ  = Cₜᵀ · hₜ + D · xₜ     ∈ ℝᴾ   (output)
786        /// ```
787        ///
788        /// The convolution is handled by manually sliding the cache window:
789        /// the oldest input column is dropped and the new token's projection
790        /// is appended.
791        ///
792        /// The SSM hidden state `cache.ssm_bhpr` is updated in-place via
793        /// the recurrence above.
794        ///
795        /// # Shapes
796        /// - `input_bm` : `[batch, d_model]`
797        /// - output     : `[batch, d_model]`
798        #[allow(non_snake_case)]
799        pub fn step(
800            &self,
801            input_bm: Tensor<2>,
802            cache: Option<Mamba2Cache>,
803        ) -> (Tensor<2>, Mamba2Cache) {
804            let [batch, d_model] = input_bm.dims();
805            let d_inner = self.d_inner();
806            let ngroups = self.ngroups;
807            let nheads = self.nheads();
808            let per_head_dim = self.per_head_dim();
809            let conv_dim = self.conv_dim();
810            let state_rank = self.state_rank;
811            let [_conv_dim, _, conv_kernel] = self.conv1d.weight.dims();
812            let [_d_model, d_in_proj_out] = self.in_proj.weight.dims();
813
814            assert_eq!(conv_dim, _conv_dim);
815            assert_eq!(nheads % ngroups, 0);
816
817            let mut cache = cache.unwrap_or_else(|| {
818                let device = &input_bm.device();
819                let conv_bvk = Tensor::zeros(Shape::new([batch, conv_dim, conv_kernel]), device);
820                let ssm_bhpr = Tensor::zeros(
821                    Shape::new([batch, nheads, per_head_dim, state_rank]),
822                    device,
823                );
824                Mamba2Cache { conv_bvk, ssm_bhpr }
825            });
826
827            // ── In-projection ─────────────────────────────────────────────────
828            let (z_gate_bi, xbc_bv, dt_raw_bh) = {
829                let z_xbc_dt_bd = self.in_proj.forward(input_bm);
830                assert_eq!([batch, d_in_proj_out], z_xbc_dt_bd.dims());
831                assert_eq!([batch, d_inner + conv_dim + nheads], z_xbc_dt_bd.dims());
832
833                let [z_gate_bi, xbc_bv, dt_raw_bh] =
834                    crate::utils::split::split_into(z_xbc_dt_bd, [d_inner, conv_dim, nheads], 1);
835                (z_gate_bi, xbc_bv, dt_raw_bh)
836            };
837            assert_eq!([batch, d_inner], z_gate_bi.dims());
838            assert_eq!([batch, conv_dim], xbc_bv.dims());
839            assert_eq!([batch, nheads], dt_raw_bh.dims());
840
841            // ── Causal convolution (single step) ──────────────────────────────
842            // Slide the cache window left by one position, then insert the new
843            // token's projection `xbc_bv` as the rightmost column.
844            cache.conv_bvk = {
845                let conv_bvk = cache.conv_bvk;
846                assert_eq!([batch, conv_dim, conv_kernel], conv_bvk.dims());
847
848                // Drop the oldest (leftmost) column.
849                let tail_bvK = conv_bvk.slice(s![.., .., 1..]);
850                assert_eq!([batch, conv_dim, conv_kernel - 1], tail_bvK.dims());
851
852                // Append the new token as the rightmost column.
853                let updated_bvk = Tensor::cat([tail_bvK, xbc_bv.unsqueeze_dim(2)].to_vec(), 2);
854                assert_eq!([batch, conv_dim, conv_kernel], updated_bvk.dims());
855                updated_bvk
856            };
857
858            // Apply the depthwise convolution manually (one step = dot product
859            // of the cached window with the conv weight along the kernel axis).
860            let xbc_bv = {
861                let conv1d_v1k = self.conv1d.weight.val(); // [conv_dim, 1, conv_kernel]
862                assert_eq!([conv_dim, 1, conv_kernel], conv1d_v1k.dims());
863
864                let conv1d_bvk = conv1d_v1k
865                    .permute([1, 0, 2]) // conv1d_1vk
866                    .expand([batch, conv_dim, conv_kernel]); // conv1d_bvk
867                assert_eq!([batch, conv_dim, conv_kernel], conv1d_bvk.dims());
868
869                // Element-wise multiply and sum over the kernel axis.
870                let product_bvk = cache.conv_bvk.clone() * conv1d_bvk;
871                let mut xbc_bv = product_bvk.sum_dim(2).squeeze_dim(2);
872                assert_eq!([batch, conv_dim], xbc_bv.dims());
873
874                // Add the (optional) bias.
875                if let Some(bias_v) = &self.conv1d.bias {
876                    assert_eq!([conv_dim], bias_v.dims());
877                    xbc_bv = xbc_bv + bias_v.val().unsqueeze();
878                }
879
880                Silu::new().forward(xbc_bv)
881            };
882            assert_eq!([batch, conv_dim], xbc_bv.dims());
883
884            // ── Split (x, B, C) ───────────────────────────────────────────────
885            assert_eq!(d_inner, nheads * per_head_dim);
886            let (x_bhp, b_bgr, c_bgr) = {
887                let [x_bi, b_bGR, c_bGR] = crate::utils::split::split_into(
888                    xbc_bv,
889                    [d_inner, ngroups * state_rank, ngroups * state_rank],
890                    1,
891                );
892                (
893                    x_bi.reshape([batch, nheads, per_head_dim]), // x_bhp
894                    b_bGR.reshape([batch, ngroups, state_rank]), // b_bgr
895                    c_bGR.reshape([batch, ngroups, state_rank]), // c_bgr
896                )
897            };
898
899            // ── Discretisation ────────────────────────────────────────────────
900            // Δₜ = softplus(dt_raw + dt_bias)
901            let dt_bias_1h = self.dt_bias_h.val().unsqueeze_dim(0);
902            assert_eq!([1, nheads], dt_bias_1h.dims());
903            let dt_bh = softplus(dt_raw_bh + dt_bias_1h).clamp(self.dt_limit.0, self.dt_limit.1);
904            assert_eq!([batch, nheads], dt_bh.dims());
905
906            // A = -exp(a_log) < 0   (negative, decaying)
907            let a_head_decay_h = -self.a_log_h.val().exp();
908            assert_eq!([nheads], a_head_decay_h.dims());
909
910            // Āₜ = exp(Δₜ · A) ∈ (0, 1)   scalar per [batch, nheads]
911            let dta_bh = (dt_bh.clone() * a_head_decay_h.unsqueeze()).exp();
912            assert_eq!([batch, nheads], dta_bh.dims());
913
914            // ── SSM state update:  hₜ = Āₜ hₜ₋₁ + B̄ₜ xₜᵀ ───────────────────
915            // The cache holds h_{t-1} with shape _bhpr.
916            // Āₜ is a scalar per head, so we broadcast it over per_head_dim and state_rank.
917            // B̄ₜ xₜᵀ is an outer product producing a _pr matrix per _bh.
918
919            let ssm_shape_bhpr = [batch, nheads, per_head_dim, state_rank];
920
921            let dta_bhpr = dta_bh.unsqueeze_dims::<4>(&[2, 3]).expand(ssm_shape_bhpr);
922
923            // B̄ₜ xₜᵀ = (Δₜ Bₜ) xₜᵀ
924            let heads_per_group = nheads / ngroups;
925            let dtbx_bhpr = {
926                let x_bhpr = x_bhp.clone().unsqueeze_dim::<4>(3).expand(ssm_shape_bhpr);
927
928                // Expand B from _bgr → _bhpr, matching the SSD forward path:
929                // each group's projection is replicated across the heads_per_group heads of
930                // that group so that heads 0..(nheads/ngroups) belong to group 0, etc.
931                let b_bhpr = b_bgr
932                    .unsqueeze_dim::<4>(2) // b_bg1r
933                    .expand([batch, ngroups, heads_per_group, state_rank]) // b_bgHr
934                    .reshape([batch, nheads, state_rank]) // b_bhr
935                    .unsqueeze_dim::<4>(2) // b_bh1r
936                    .expand(ssm_shape_bhpr); // b_bhpr
937
938                let dt_bhpr = dt_bh.unsqueeze_dims::<4>(&[2, 3]).expand(ssm_shape_bhpr);
939
940                dt_bhpr * b_bhpr * x_bhpr // B̄ₜ xₜᵀ
941            };
942            assert_eq!(ssm_shape_bhpr, dtbx_bhpr.dims());
943
944            // hₜ = Āₜ hₜ₋₁ + B̄ₜ xₜᵀ
945            cache.ssm_bhpr = cache.ssm_bhpr * dta_bhpr + dtbx_bhpr;
946            assert_eq!(ssm_shape_bhpr, cache.ssm_bhpr.dims());
947
948            // ── Output:  yₜ = Cₜ hₜ + D xₜ ──────────────────────────────────
949            let y_bi = {
950                // Cₜ hₜ:  element-wise multiply C
951                // with hₜ, then sum over state_rank.
952                let c_bhpr = c_bgr
953                    .unsqueeze_dim::<4>(2) // c_bg1r
954                    .expand([batch, ngroups, heads_per_group, state_rank]) // c_bgHr
955                    .reshape([batch, nheads, state_rank]) // c_bhr
956                    .unsqueeze_dim::<4>(2) // c_bh1r
957                    .expand(ssm_shape_bhpr); // c_bhpr
958                assert_eq!(ssm_shape_bhpr, c_bhpr.dims());
959
960                let ch_bhp = (cache.ssm_bhpr.clone() * c_bhpr).sum_dim(3).squeeze_dim(3);
961                assert_eq!([batch, nheads, per_head_dim], ch_bhp.dims());
962
963                // D xₜ:  per-head scalar skip.
964                let d_1h1 = self.d_h.val().unsqueeze_dims(&[0, 2]);
965                assert_eq!([1, nheads, 1], d_1h1.dims());
966                let skip_bhp = d_1h1.expand([batch, nheads, per_head_dim]) * x_bhp;
967                assert_eq!([batch, nheads, per_head_dim], skip_bhp.dims());
968
969                let y_bhp = ch_bhp + skip_bhp;
970                assert_eq!([batch, nheads, per_head_dim], y_bhp.dims());
971
972                // Flatten heads, then apply gated RMSNorm.
973                let y_bi = y_bhp.reshape([batch, d_inner]);
974                self.norm.forward(y_bi, z_gate_bi)
975            };
976            assert_eq!([batch, d_inner], y_bi.dims());
977
978            // ── Out-projection ────────────────────────────────────────────────
979            let out_bm = self.out_proj.forward(y_bi);
980            assert_eq!([batch, d_model], out_bm.dims());
981
982            (out_bm, cache)
983        }
984    }
985}
