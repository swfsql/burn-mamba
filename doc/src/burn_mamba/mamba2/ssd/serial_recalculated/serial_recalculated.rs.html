1//! # Serial SSD with a custom, memory-efficient backward (Mamba-2)
2//!
3//! This is the `SerialRecalculated` path.  The forward is the same five-kernel
4//! serial scan as [`super::super::serial`], but it is routed through the
5//! [`Mamba2BackendExt`] trait so that `Autodiff` backends can substitute a
6//! **custom backward** that recomputes the per-chunk intermediates instead of
7//! storing them (see [`super::backward`] / [`super::combined_backward`]),
8//! trading a little extra compute for ~⅓ less training memory.
9//!
10//! Every plain (non-autodiff) backend uses the trait's default body, which
11//! simply replays the [`super::super::serial`] kernels K1–K5.  The
12//! [`crate::impl_ssd_backend_ext_for_burn_backends!`] /
13//! [`crate::decl_ssd_autodiff_backend_ext!`] macros wire up the per-backend
14//! impls and the autodiff marker trait.
15
16use crate::mamba2::prelude::*;
17use crate::mamba2::ssd::serial;
18use crate::utils::primitive::mk;
19use burn::prelude::*;
20use burn::tensor::{Tensor, TensorPrimitive, ops::FloatTensor};

22impl<B: Backend + Mamba2BackendExt> Mamba2SsdInput<B> {
23    /// Forward pass for the Mamba-2 SSD module (recompute-backward path).
24    ///
25    /// Returns:
26    /// - `y_bnlhp`.
27    /// - `final_state_bhpr`.
28    #[allow(non_snake_case)]
29    pub fn ssd_serial_recalculated(self) -> (Tensor<B, 5>, Tensor<B, 4>) {
30        let input = self;
31        // Must use a backend-dependent method.
32        //
33        // For inference, this will ultimately replicate Mamba2::ssd_serial;
34        // For autodiff, this will call the custom implementation.
35
36        let [batch, nchunks, chunk_len, nheads, _per_head_dim] = input.x_bnlhp.dims();
37        assert!(nchunks > 0, "sequence length must be at least 1");
38
39        assert!(
40            input.init_state_hpr.is_none(),
41            "init_state_hpr not yet implemented"
42        );
43
44        // ── Permutes ──────────────────────────────────────────────────────────────────
45        // Note: dt_bnlh calculation (originally in Kernel 1) moved to Step 4 (before padding).
46        let dt_discretized_bhnl = input.dt_bnlh.permute([0, 3, 1, 2]);
47        assert_eq!(
48            [batch, nheads, nchunks, chunk_len],
49            dt_discretized_bhnl.dims()
50        );
51
52        // K1 is now computed inside the custom op (both forward and backward).
53        // a_decay_h is passed directly; da_cumsum is no longer an autodiff-tracked
54        // intermediate crossing the boundary.
55        let (y_bnlhp, final_state_bhpr) = <B as Mamba2BackendExt>::ssd_serial_recalculated(
56            input.x_bnlhp.into_primitive().tensor(),
57            dt_discretized_bhnl.into_primitive().tensor(),
58            input.b_bnlhr.into_primitive().tensor(),
59            input.c_bnlhr.into_primitive().tensor(),
60            input.d_h.into_primitive().tensor(),
61            input.initial_state_bhpr.into_primitive().tensor(),
62            input.a_decay_h.into_primitive().tensor(),
63        );
64        let y_bnlhp = Tensor::from_primitive(TensorPrimitive::Float(y_bnlhp));
65        let final_state_bhpr = Tensor::from_primitive(TensorPrimitive::Float(final_state_bhpr));
66        (y_bnlhp, final_state_bhpr)
67    }
68}

70/// Extends the backend and wraps it for `burn`.
71pub trait Mamba2BackendExt: burn::tensor::backend::Backend {
72    /// Returns:
73    /// - `y_bnlhp`.
74    /// - `final_state_bhpr`.
75    fn ssd_serial_recalculated(
76        x_bnlhp: FloatTensor<Self>,
77        dt_discretized_bhnl: FloatTensor<Self>,
78        b_bnlhr: FloatTensor<Self>,
79        c_bnlhr: FloatTensor<Self>,
80        d_h: FloatTensor<Self>,
81        initial_state_bhpr: FloatTensor<Self>,
82        a_decay_h: FloatTensor<Self>,
83    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
84        // Default impl essentially replicates Mamba2SsdInput::ssd_serial.
85
86        let x_bnlhp: Tensor<Self, 5> = mk(x_bnlhp);
87        let dt_discretized_bhnl: Tensor<Self, 4> = mk(dt_discretized_bhnl);
88        let b_bnlhr: Tensor<Self, 5> = mk(b_bnlhr);
89        let c_bnlhr: Tensor<Self, 5> = mk(c_bnlhr);
90        let d_h: Tensor<Self, 1> = mk(d_h);
91        let initial_state_bhpr: Tensor<Self, 4> = mk(initial_state_bhpr);
92        let a_decay_h: Tensor<Self, 1> = mk(a_decay_h);
93
94        let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
95        let [.., state_rank] = b_bnlhr.dims();
96        assert!(nchunks > 0, "sequence length must be at least 1");
97
98        // ── Kernel 1 ──────────────────────────────────────────────────────────
99        let (da_cumsum_bhnl, da_chunk_end_bhn) =
100            serial::k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), a_decay_h);
101        assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
102        assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
103
104        // ── Kernel 2 ──────────────────────────────────────────────────────────
105        // IO: (..) -> (cb_bnhll [used in K5][!])
106        let cb_bnhll: Tensor<Self, 5> = serial::k2_ssd_bmm(c_bnlhr.clone(), b_bnlhr.clone());
107        assert_eq!(
108            [batch, nchunks, nheads, chunk_len, chunk_len],
109            cb_bnhll.dims()
110        );
111
112        // ── Kernel 3 ──────────────────────────────────────────────────────────
113        // IO: (..) -> (intra_chunk_state_bnhpr [used in K4][!])
114        let intra_chunk_state_bnhpr: Tensor<Self, 5> = serial::k3_ssd_chunk_state(
115            x_bnlhp.clone(),
116            b_bnlhr.clone(),
117            da_cumsum_bhnl.clone(),
118            dt_discretized_bhnl.clone(),
119        );
120        assert_eq!(
121            [batch, nchunks, nheads, per_head_dim, state_rank],
122            intra_chunk_state_bnhpr.dims()
123        );
124
125        // ── Kernel 4 ──────────────────────────────────────────────────────────
126        // IO: (..) -> (chunk_input_state_bnhpr [used in K5][!], final_state_bhpr [final output])
127        let (chunk_input_state_bnhpr, final_state_bhpr): (Tensor<Self, 5>, Tensor<Self, 4>) =
128            serial::k4_ssd_state_passing(
129                intra_chunk_state_bnhpr.clone(),
130                da_chunk_end_bhn.clone(),
131                initial_state_bhpr,
132            );
133        assert_eq!(
134            [batch, nchunks, nheads, per_head_dim, state_rank],
135            chunk_input_state_bnhpr.dims()
136        );
137        assert_eq!(
138            [batch, nheads, per_head_dim, state_rank],
139            final_state_bhpr.dims()
140        );
141
142        // ── Kernel 5 ──────────────────────────────────────────────────────────
143        let y_bnlhp: Tensor<Self, 5> = serial::k5_ssd_chunk_scan(
144            da_cumsum_bhnl,
145            dt_discretized_bhnl,
146            x_bnlhp,
147            c_bnlhr,
148            cb_bnhll,
149            chunk_input_state_bnhpr,
150            d_h,
151        );
152        assert_eq!(
153            [batch, nchunks, chunk_len, nheads, per_head_dim],
154            y_bnlhp.dims()
155        );
156
157        let y_bnlhp = y_bnlhp.into_primitive().tensor();
158        let final_state_bhpr = final_state_bhpr.into_primitive().tensor();
159        (y_bnlhp, final_state_bhpr)
160    }
161}
