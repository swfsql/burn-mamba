8impl<B: Backend + Mamba2BackendExt> Mamba2<B> {
9    /// Forward pass for the Mamba-2 SSD module.
10    ///
11    /// # Tensor shapes flowing through the function
12    ///
13    /// ```text
14    /// x_bnlhp:                  [batch, nchunks, chunk_len, nheads, per_head_dim]
15    /// dt_bnlh:                  [batch, nchunks, chunk_len, nheads]
16    /// a_decay_h:               [nheads]
17    /// b_bnlgr:                  [batch, nchunks, chunk_len, ngroups, state_rank]
18    /// c_bnlgr:                  [batch, nchunks, chunk_len, ngroups, state_rank]
19    /// d_h:                     [nheads]
20    /// initial_state_bhpr   [batch, nheads, per_head_dim, state_rank]
21    /// _init_state_hpr         [nheads, per_head_dim, state_rank]
22    /// ```
23    ///
24    /// Returns:
25    /// - `y_bnlhp`.
26    /// - `final_state_bhpr`.
27    #[allow(non_snake_case)]
28    pub fn ssd_serial_recalculated(
29        x_bnlhp: Tensor<B, 5>,
30        dt_bnlh: Tensor<B, 4>,
31        a_decay_h: Tensor<B, 1>,
32        b_bnlgr: Tensor<B, 5>,
33        c_bnlgr: Tensor<B, 5>,
34        d_h: Tensor<B, 1>,
35        initial_state_bhpr: Tensor<B, 4>,
36        init_state_hpr: Option<Tensor<B, 3>>,
37    ) -> (Tensor<B, 5>, Tensor<B, 4>) {
38        // Must use a backend-dependent method.
39        //
40        // For inference, this will untimately replicate Mamba2::ssd_serial;
41        // For autodiff, this will call the custom implementation.
42
43        let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
44        let [.., ngroups, state_rank] = b_bnlgr.dims();
45        let device = x_bnlhp.device();
46        assert_ne!(ngroups, 0);
47        assert_eq!(nheads % ngroups, 0);
48        assert!(nchunks > 0, "sequence length must be at least 1");
49
50        assert!(
51            init_state_hpr.is_none(),
52            "init_state_hpr not yet implemented"
53        );
54
55        // ── Permutes ──────────────────────────────────────────────────────────────────
56        // Note: dt_bnlh calculation (originally in Kernel 1) moved to Step 4 (before padding).
57        let dt_discretized_bhnl = dt_bnlh.permute([0, 3, 1, 2]);
58        assert_eq!(
59            [batch, nheads, nchunks, chunk_len],
60            dt_discretized_bhnl.dims()
61        );
62
63        // Always executes K1 in autodiff.
64        // ── Kernel 1 ──────────────────────────────────────────────────────────────────
65        // IO: (..) -> (da_cumsum_bhnl [used in K3+K5][*], da_chunk_end_bhn [used in K4][omitted][*])
66        let (da_cumsum_bhnl, _da_chunk_end_bhn): (Tensor<B, 4>, Tensor<B, 3>) =
67            serial::k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), a_decay_h.clone());
68        assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
69        assert_eq!([batch, nheads, nchunks], _da_chunk_end_bhn.dims());
70
71        let (y_bnlhp, final_state_bhpr) = <B as Mamba2BackendExt>::ssd_serial_recalculated(
72            x_bnlhp.into_primitive().tensor(),
73            dt_discretized_bhnl.into_primitive().tensor(),
74            b_bnlgr.into_primitive().tensor(),
75            c_bnlgr.into_primitive().tensor(),
76            d_h.into_primitive().tensor(),
77            initial_state_bhpr.into_primitive().tensor(),
78            da_cumsum_bhnl.into_primitive().tensor(),
79        );
80        let y_bnlhp = Tensor::from_primitive(TensorPrimitive::Float(y_bnlhp));
81        let final_state_bhpr = Tensor::from_primitive(TensorPrimitive::Float(final_state_bhpr));
82        (y_bnlhp, final_state_bhpr)
83    }
84}

86/// Extends the backend and wraps it for `burn`.
87pub trait Mamba2BackendExt: burn::tensor::backend::Backend {
88    /// Returns:
89    /// - `y_bnlhp`.
90    /// - `final_state_bhpr`.
91    fn ssd_serial_recalculated(
92        x_bnlhp: FloatTensor<Self>,
93        // dt_bnlh: FloatTensor<Self>,
94        dt_discretized_bhnl: FloatTensor<Self>,
95        // a_decay_h: FloatTensor<Self>,
96        b_bnlgr: FloatTensor<Self>,
97        c_bnlgr: FloatTensor<Self>,
98        d_h: FloatTensor<Self>,
99        initial_state_bhpr: FloatTensor<Self>,
100        da_cumsum_bhnl: FloatTensor<Self>, // from K1
101    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
102        // Default impl essentially replicates Mamba2::ssd_serial.
103
104        let x_bnlhp: Tensor<Self, 5> = mk(x_bnlhp);
105        let dt_discretized_bhnl: Tensor<Self, 4> = mk(dt_discretized_bhnl);
106        let b_bnlgr: Tensor<Self, 5> = mk(b_bnlgr);
107        let c_bnlgr: Tensor<Self, 5> = mk(c_bnlgr);
108        let d_h: Tensor<Self, 1> = mk(d_h);
109        let initial_state_bhpr: Tensor<Self, 4> = mk(initial_state_bhpr);
110        let da_cumsum_bhnl: Tensor<Self, 4> = mk(da_cumsum_bhnl);
111
112        let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
113        let [.., ngroups, state_rank] = b_bnlgr.dims();
114        let device = x_bnlhp.device();
115        assert_ne!(ngroups, 0);
116        assert_eq!(nheads % ngroups, 0);
117        assert!(nchunks > 0, "sequence length must be at least 1");
118        // `heads_per_group` is called `nheads_ngroups_ratio` in every Triton kernel.
119        // It is the compile-time constant used by GQA (Grouped Query Attention) to map
120        // a head index to its B/C group: `group_idx = head_idx / heads_per_group`.
121        let heads_per_group = nheads / ngroups;
122
123        // ── Kernel 1 ──────────────────────────────────────────────────────────
124        // Note: This kernel is assumed to already have been called.
125        // Just recalculate da_chunk_end_bhn.
126        let da_chunk_end_bhn = da_cumsum_bhnl
127            .clone()
128            .slice(s![.., .., .., -1]) // da_cumsum_bhn1 // replay forward step 5
129            .squeeze_dim::<3>(3); // replay forward step 6
130        assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
131
132        // ── Kernel 2 ──────────────────────────────────────────────────────────
133        // IO: (..) -> (cb_bngll [used in K5][!])
134        let cb_bngll: Tensor<Self, 5> = serial::k2_ssd_bmm(c_bnlgr.clone(), b_bnlgr.clone());
135        assert_eq!(
136            [batch, nchunks, ngroups, chunk_len, chunk_len],
137            cb_bngll.dims()
138        );
139        // Note: cb_bngll is then only used by Kernel 5.
140
141        // ── Kernel 3 ──────────────────────────────────────────────────────────
142        // IO: (..) -> (intra_chunk_state_bnhpr [used in K4][!])
143        let intra_chunk_state_bnhpr: Tensor<Self, 5> = serial::k3_ssd_chunk_state(
144            x_bnlhp.clone(),
145            b_bnlgr.clone(),
146            da_cumsum_bhnl.clone(),
147            dt_discretized_bhnl.clone(),
148        );
149        assert_eq!(
150            [batch, nchunks, nheads, per_head_dim, state_rank],
151            intra_chunk_state_bnhpr.dims()
152        );
153
154        // ── Kernel 4 ──────────────────────────────────────────────────────────
155        // IO: (..) -> (chunk_input_state_bnhpr [used in K5][!], final_state_bhpr [final output])
156        let flat_state_dim = per_head_dim * state_rank;
157        let (chunk_input_state_bnhpr, final_state_bhpr): (Tensor<Self, 5>, Tensor<Self, 4>) =
158            serial::k4_ssd_state_passing(
159                intra_chunk_state_bnhpr.clone(),
160                da_chunk_end_bhn.clone(),
161                initial_state_bhpr,
162            );
163        assert_eq!(
164            [batch, nchunks, nheads, per_head_dim, state_rank],
165            chunk_input_state_bnhpr.dims()
166        );
167        assert_eq!(
168            [batch, nheads, per_head_dim, state_rank],
169            final_state_bhpr.dims()
170        );
171
172        // ── Kernel 5 ──────────────────────────────────────────────────────────
173        let y_bnlhp: Tensor<Self, 5> = serial::k5_ssd_chunk_scan(
174            da_cumsum_bhnl,
175            dt_discretized_bhnl,
176            x_bnlhp,
177            c_bnlgr,
178            cb_bngll,
179            chunk_input_state_bnhpr,
180            d_h,
181        );
182        assert_eq!(
183            [batch, nchunks, chunk_len, nheads, per_head_dim],
184            y_bnlhp.dims()
185        );
186
187        let y_bnlhp = y_bnlhp.into_primitive().tensor();
188        let final_state_bhpr = final_state_bhpr.into_primitive().tensor();
189        (y_bnlhp, final_state_bhpr)
190    }
191}
