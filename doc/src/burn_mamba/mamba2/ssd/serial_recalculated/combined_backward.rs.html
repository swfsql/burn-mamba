7/// Per-input gradients produced by [`combined_backward`].
8#[non_exhaustive]
9pub struct CombinedGrads<B: Backend> {
10    pub d_x_bnlhp: Tensor<B, 5>,
11    pub d_dt_discretized_bhnl: Tensor<B, 4>,
12    pub d_b_bnlhr: Tensor<B, 5>,
13    pub d_c_bnlhr: Tensor<B, 5>,
14    pub d_d_h: Tensor<B, 1>,
15    pub d_initial_state_bhpr: Tensor<B, 4>,
16    pub d_a_decay_h: Tensor<B, 1>,
17    /// Local same-chunk contribution to `d_da_cumsum` from BLUE+ORANGE only
18    /// (excludes K3 and K4 cross-chunk contributions). Exposed for the
19    /// `out_x · dout − ddt · dt` oracle test from Tri Dao's reference
20    /// (`_chunk_scan_bwd_ddAcs_unstable`). Test-only; absent in release builds.
21    #[cfg(test)]
22    pub d_da_local_bhnl: Tensor<B, 4>,
23    /// Same-chunk d_dt contribution from ORANGE only (= what Tri Dao calls
24    /// `ddt` in `_chunk_scan_bwd_ddAcs_unstable`). Test-only; absent in release
25    /// builds.
26    #[cfg(test)]
27    pub d_dt_orange_bhnl: Tensor<B, 4>,
28}

30/// Same as [`serial::k3_ssd_chunk_state`] but also returns intermediates needed
31/// by the custom backward:
32/// - `intra_chunk_state_bnhpr` — chunk-end state assuming zero initial state
33/// - `b_bar_scale_bhnl` — the K3 scaling factor `dt · exp(cumA_last − cumA)`
34/// - `forward_decay_to_chunk_end_bhnl` — the decay factor `exp(cumA_last − cumA)`
35/// - `b_scaled_bnhlr` — B already scaled by `b_bar_scale`
36pub fn k3_ssd_chunk_state_extended<B: Backend>(
37    x_bnlhp: Tensor<B, 5>,
38    b_bnlhr: Tensor<B, 5>,
39    da_cumsum_bhnl: Tensor<B, 4>,
40    dt_discretized_bhnl: Tensor<B, 4>,
41) -> (Tensor<B, 5>, Tensor<B, 4>, Tensor<B, 4>, Tensor<B, 5>) {
42    use burn::tensor::s;
43
44    let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
45    let [.., state_rank] = b_bnlhr.dims();
46
47    let x_bnhpl = x_bnlhp.permute([0, 1, 3, 4, 2]);
48    let b_bnhlr = b_bnlhr.permute([0, 1, 3, 2, 4]);
49
50    // K3 scaling factor: dt · exp(cumA_last − cumA)
51    let da_cumsum_last_bhn1 = da_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
52    let forward_decay_to_chunk_end_bhnl =
53        (da_cumsum_last_bhn1.expand([batch, nheads, nchunks, chunk_len]) - da_cumsum_bhnl).exp();
54    san(&forward_decay_to_chunk_end_bhnl);
55    let b_bar_scale_bhnl = forward_decay_to_chunk_end_bhnl.clone() * dt_discretized_bhnl;
56    san(&b_bar_scale_bhnl);
57
58    let b_bar_scale_bnhlr = b_bar_scale_bhnl
59        .clone()
60        .permute([0, 2, 1, 3]) // b_bar_scale_bnhl
61        .unsqueeze_dim::<5>(4) // b_bar_scale_bnhl1
62        .expand([batch, nchunks, nheads, chunk_len, state_rank]); // b_bar_scale_bnhlr
63    let b_scaled_bnhlr = b_bnhlr * b_bar_scale_bnhlr;
64    san(&b_scaled_bnhlr);
65
66    let intra_chunk_state_bnhpr = x_bnhpl.matmul(b_scaled_bnhlr.clone());
67    assert_eq!(
68        [batch, nchunks, nheads, per_head_dim, state_rank],
69        intra_chunk_state_bnhpr.dims()
70    );
71    san(&intra_chunk_state_bnhpr);
72
73    (
74        intra_chunk_state_bnhpr,
75        b_bar_scale_bhnl,
76        forward_decay_to_chunk_end_bhnl,
77        b_scaled_bnhlr,
78    )
79}

81/// Memory-efficient backward for the Mamba-2 chunkwise SSD.
82///
83/// Recomputes the forward intermediates (K1-K4) from the saved inputs, then
84/// runs a reverse per-chunk loop that fuses the K5 (BLUE + ORANGE) backward
85/// with the K4 state-passing backward. K3/K2/K1 backwards run as single
86/// batched ops once the loop has collected all per-chunk slices.
87///
88/// # Arguments
89/// - `d_y_bnlhp` — upstream gradient of the SSD output
90/// - `d_final_bhpr` — upstream gradient of the final SSM state
91/// - `x_bnlhp`, `dt_discretized_bhnl`, `b_bnlhr`, `c_bnlhr`, `d_h`,
92///   `initial_state_bhpr`, `a_decay_h` — the seven saved forward inputs
93///
94/// # Returns
95/// One [`CombinedGrads`] struct containing gradients for all 7 inputs.
96#[allow(clippy::too_many_arguments)]
97pub fn combined_backward<B: Backend>(
98    d_y_bnlhp: Tensor<B, 5>,
99    d_final_bhpr: Tensor<B, 4>,
100    //
101    x_bnlhp: Tensor<B, 5>,
102    dt_discretized_bhnl: Tensor<B, 4>,
103    b_bnlhr: Tensor<B, 5>,
104    c_bnlhr: Tensor<B, 5>,
105    d_h: Tensor<B, 1>,
106    initial_state_bhpr: Tensor<B, 4>,
107    a_decay_h: Tensor<B, 1>,
108) -> CombinedGrads<B> {
109    use burn::tensor::s;
110
111    let [batch, nheads, nchunks, chunk_len] = dt_discretized_bhnl.dims();
112    let [.., per_head_dim] = x_bnlhp.dims();
113    let [.., state_rank] = b_bnlhr.dims();
114    let device = dt_discretized_bhnl.device();
115
116    san(&d_y_bnlhp);
117    san(&d_final_bhpr);
118    san(&x_bnlhp);
119    san(&dt_discretized_bhnl);
120    san(&b_bnlhr);
121    san(&c_bnlhr);
122    san(&d_h);
123    san(&initial_state_bhpr);
124    san(&a_decay_h);
125
126    // ═══════════════════════════════════════════════════════════════════════
127    // RECOMPUTE FORWARD INTERMEDIATES (the memory-saving heart of this op)
128    // ═══════════════════════════════════════════════════════════════════════
129
130    // K1 — pre-combined Δ·A → intra-chunk cumsum
131    let (da_cumsum_bhnl, da_chunk_end_bhn) =
132        serial::k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), a_decay_h.clone());
133    san(&da_cumsum_bhnl);
134
135    // K2 — CB matrix used in K5 ORANGE
136    let cb_bnhll = serial::k2_ssd_bmm(c_bnlhr.clone(), b_bnlhr.clone());
137    san(&cb_bnhll);
138
139    // K3 — intra-chunk state + decay/decayed-B intermediates
140    let (
141        intra_chunk_state_bnhpr,
142        b_bar_scale_bhnl,
143        forward_decay_to_chunk_end_bhnl,
144        b_scaled_bnhlr,
145    ) = k3_ssd_chunk_state_extended(
146        x_bnlhp.clone(),
147        b_bnlhr.clone(),
148        da_cumsum_bhnl.clone(),
149        dt_discretized_bhnl.clone(),
150    );
151
152    // K4 — chunk-input state stream consumed by K5 BLUE
153    let (chunk_input_state_bnhpr, _final_state_bhpr) = serial::k4_ssd_state_passing(
154        intra_chunk_state_bnhpr.clone(),
155        da_chunk_end_bhn.clone(),
156        initial_state_bhpr,
157    );
158    san(&chunk_input_state_bnhpr);
159
160    // ═══════════════════════════════════════════════════════════════════════
161    // SKIP backward — y += D · x
162    // ═══════════════════════════════════════════════════════════════════════
163    let d_d_h = (d_y_bnlhp.clone() * x_bnlhp.clone())
164        .permute([3, 0, 1, 2, 4]) // _hbnlp
165        .reshape([nheads, batch * nchunks * chunk_len * per_head_dim]) // _hBNLP
166        .sum_dim(1) // _h1
167        .reshape([nheads]);
168    san(&d_d_h);
169
170    let d_x_skip_bnlhp = d_y_bnlhp.clone()
171        * d_h
172            .clone()
173            .unsqueeze_dims::<5>(&[0, 1, 2, 4]) // _111h1
174            .expand([batch, nchunks, chunk_len, nheads, per_head_dim]); // _bnlhp
175    san(&d_x_skip_bnlhp);
176
177    // ═══════════════════════════════════════════════════════════════════════
178    // REVERSE PER-CHUNK LOOP — K5 (BLUE + ORANGE) + K4 fused
179    //
180    // Per-iteration working set is _bhll (not _bnhll).
181    // ═══════════════════════════════════════════════════════════════════════
182
183    // Reusable [chunk_len, chunk_len] upper-triangle base mask for ORANGE.
184    let causal_mask_ll: Tensor<B, 2, burn::prelude::Bool> =
185        Tensor::<B, 2, burn::prelude::Bool>::tril_mask([chunk_len, chunk_len], 0, &device);
186
187    let mut vec_orange_d_x_bhlp: Vec<Tensor<B, 4>> = Vec::with_capacity(nchunks);
188    let mut vec_orange_d_dt_bhl: Vec<Tensor<B, 3>> = Vec::with_capacity(nchunks);
189    let mut vec_orange_d_da_bhl: Vec<Tensor<B, 3>> = Vec::with_capacity(nchunks);
190    let mut vec_d_cb_bhll: Vec<Tensor<B, 4>> = Vec::with_capacity(nchunks);
191    let mut vec_blue_d_c_bhlr: Vec<Tensor<B, 4>> = Vec::with_capacity(nchunks);
192    let mut vec_blue_d_da_bhl: Vec<Tensor<B, 3>> = Vec::with_capacity(nchunks);
193    let mut vec_d_intra_bhpr: Vec<Tensor<B, 4>> = Vec::with_capacity(nchunks);
194    let mut vec_d_da_end_bh: Vec<Tensor<B, 2>> = Vec::with_capacity(nchunks);
195
196    let mut d_running_state_bhpr: Tensor<B, 4> = d_final_bhpr;
197
198    for i_chunk in (0..nchunks).rev() {
199        // ── Per-chunk slices ───────────────────────────────────────────────
200        let da_cumsum_bhl: Tensor<B, 3> = da_cumsum_bhnl
201            .clone()
202            .slice(s![.., .., i_chunk, ..]) // _bh1l
203            .squeeze_dim::<3>(2); // _bhl
204        let dt_bhl: Tensor<B, 3> = dt_discretized_bhnl
205            .clone()
206            .slice(s![.., .., i_chunk, ..]) // _bh1l
207            .squeeze_dim::<3>(2); // _bhl
208        let x_bhlp: Tensor<B, 4> = x_bnlhp
209            .clone()
210            .slice(s![.., i_chunk, .., .., ..]) // _b1lhp
211            .squeeze_dim::<4>(1) // _blhp
212            .permute([0, 2, 1, 3]); // _bhlp
213        let d_y_bhlp: Tensor<B, 4> = d_y_bnlhp
214            .clone()
215            .slice(s![.., i_chunk, .., .., ..]) // _b1lhp
216            .squeeze_dim::<4>(1) // _blhp
217            .permute([0, 2, 1, 3]); // _bhlp
218        let c_bhlr: Tensor<B, 4> = c_bnlhr
219            .clone()
220            .slice(s![.., i_chunk, .., .., ..]) // _b1lhr
221            .squeeze_dim::<4>(1) // _blhr
222            .permute([0, 2, 1, 3]); // _bhlr
223        let cb_bhll: Tensor<B, 4> = cb_bnhll
224            .clone()
225            .slice(s![.., i_chunk, .., .., ..]) // _b1hll
226            .squeeze_dim::<4>(1); // _bhll
227        let chunk_input_state_bhpr: Tensor<B, 4> = chunk_input_state_bnhpr
228            .clone()
229            .slice(s![.., i_chunk, .., .., ..]) // _b1hpr
230            .squeeze_dim::<4>(1); // _bhpr
231        san(&chunk_input_state_bhpr);
232
233        // ── BLUE backward ──────────────────────────────────────────────────
234        //   blue[l,p] = exp(cumA[l]) · Σᵣ C[l,r] · state[p,r]
235        let exp_da_cumsum_bhl: Tensor<B, 3> = da_cumsum_bhl.clone().exp();
236        let exp_da_cumsum_bhlp: Tensor<B, 4> = exp_da_cumsum_bhl
237            .clone()
238            .unsqueeze_dim::<4>(3) // _bhl1
239            .expand([batch, nheads, chunk_len, per_head_dim]); // _bhlp
240        let d_ch_bhlp: Tensor<B, 4> = d_y_bhlp.clone() * exp_da_cumsum_bhlp.clone();
241        san(&d_ch_bhlp);
242
243        // d_chunk_input_state = C^T @ d_ch
244        let d_chunk_input_state_bhpr: Tensor<B, 4> = c_bhlr
245            .clone()
246            .permute([0, 1, 3, 2]) // c_bhrl
247            .matmul(d_ch_bhlp.clone()) // d_chunk_input_state_bhrp
248            .permute([0, 1, 3, 2]); // _bhpr
249        san(&d_chunk_input_state_bhpr);
250
251        // d_C_blue = d_ch @ state
252        let d_c_blue_bhlr: Tensor<B, 4> = d_ch_bhlp.clone().matmul(chunk_input_state_bhpr.clone());
253        san(&d_c_blue_bhlr);
254        vec_blue_d_c_bhlr.push(d_c_blue_bhlr);
255
256        // d_da from BLUE:  d_da[l] = (Σₚ d_y[l,p] · ch[l,p]) · exp_da[l]
257        let ch_bhlp: Tensor<B, 4> = c_bhlr.clone().matmul(
258            chunk_input_state_bhpr.clone().permute([0, 1, 3, 2]), // _bhrp
259        ); // _bhlp
260        let d_da_blue_bhl: Tensor<B, 3> = (d_y_bhlp.clone() * ch_bhlp * exp_da_cumsum_bhlp)
261            .sum_dim(3) // _bhl1
262            .squeeze_dim::<3>(3); // _bhl
263        san(&d_da_blue_bhl);
264        vec_blue_d_da_bhl.push(d_da_blue_bhl);
265
266        // ── ORANGE backward ────────────────────────────────────────────────
267        //   w[lₜ,lₛ] = CB[lₜ,lₛ] · exp(cumA[lₜ] − cumA[lₛ]) · dt[lₛ]   (causal)
268        //   orange[lₜ,p] = Σ_{lₛ} w[lₜ,lₛ] · x[lₛ,p]
269        let diff_bhll = {
270            let target_bhll = da_cumsum_bhl
271                .clone()
272                .unsqueeze_dim::<4>(3) // _bhlt1
273                .expand([batch, nheads, chunk_len, chunk_len]); // _bhltls
274            let source_bhll = da_cumsum_bhl
275                .unsqueeze_dim::<4>(2) // _bh1ls
276                .expand([batch, nheads, chunk_len, chunk_len]); // _bhltls
277            target_bhll - source_bhll
278        };
279        let causal_mask_bhll: Tensor<B, 4, burn::prelude::Bool> = causal_mask_ll
280            .clone()
281            .unsqueeze_dims::<4>(&[0, 1]) // _11ll
282            .expand([batch, nheads, chunk_len, chunk_len]); // _bhll
283        let decay_bhll = diff_bhll
284            .mask_fill(causal_mask_bhll.clone(), f32::NEG_INFINITY)
285            .exp();
286        san(&decay_bhll);
287
288        let dt_source_bhll: Tensor<B, 4> = dt_bhl
289            .unsqueeze_dim::<4>(2) // _bh1l
290            .expand([batch, nheads, chunk_len, chunk_len]); // _bhll
291        let cb_decay_bhll = cb_bhll.clone() * decay_bhll.clone();
292        let w_bhll = cb_decay_bhll.clone() * dt_source_bhll.clone();
293
294        let d_orange_bhlp = d_y_bhlp; // = d_y_partial
295        // d_w = d_orange @ x^T
296        let d_w_bhll: Tensor<B, 4> = d_orange_bhlp.clone().matmul(
297            x_bhlp.permute([0, 1, 3, 2]), // x_bhpl
298        );
299        san(&d_w_bhll);
300
301        // d_x = w^T @ d_orange
302        let d_x_orange_bhlp: Tensor<B, 4> = w_bhll
303            .permute([0, 1, 3, 2]) // w_bhlslt
304            .matmul(d_orange_bhlp);
305        san(&d_x_orange_bhlp);
306        vec_orange_d_x_bhlp.push(d_x_orange_bhlp);
307
308        // Mask off above-diagonal contributions, then split d_w into its factors.
309        let d_w_masked_bhll = d_w_bhll.mask_fill(causal_mask_bhll, 0.);
310        let d_cb_decay_bhll = d_w_masked_bhll.clone() * dt_source_bhll;
311        san(&d_cb_decay_bhll);
312
313        // d_dt[s] = Σ_{lₜ ≥ lₛ} d_w[lₜ,lₛ] · CB[lₜ,lₛ] · decay[lₜ,lₛ]
314        let d_dt_orange_bhl: Tensor<B, 3> = (d_w_masked_bhll * cb_decay_bhll)
315            .sum_dim(2) // _bh1ls
316            .squeeze_dim::<3>(2); // _bhls
317        san(&d_dt_orange_bhl);
318        vec_orange_d_dt_bhl.push(d_dt_orange_bhl);
319
320        // d_cb = d_cb_decay · decay ;  d_decay = d_cb_decay · cb
321        let d_cb_bhll = d_cb_decay_bhll.clone() * decay_bhll.clone();
322        vec_d_cb_bhll.push(d_cb_bhll);
323        let d_decay_bhll = d_cb_decay_bhll * cb_bhll;
324        let d_diff_bhll = d_decay_bhll * decay_bhll;
325
326        // d_da from ORANGE:  d_da_tgt[l] += Σₛ d_diff[l,s]; d_da_src[s] −= Σₗ d_diff[l,s].
327        let d_da_tgt_bhl: Tensor<B, 3> = d_diff_bhll
328            .clone()
329            .sum_dim(3) // _bhlt1
330            .squeeze_dim::<3>(3); // _bhlt
331        let d_da_src_bhl: Tensor<B, 3> = d_diff_bhll
332            .sum_dim(2) // _bh1ls
333            .squeeze_dim::<3>(2); // _bhls
334        let d_da_orange_bhl = d_da_tgt_bhl - d_da_src_bhl;
335        san(&d_da_orange_bhl);
336        vec_orange_d_da_bhl.push(d_da_orange_bhl);
337
338        // ── K4 backward step for chunk i ───────────────────────────────────
339        //   Forward: sᵢ₊₁ = decayᵢ · sᵢ + intra_stateᵢ
340        //   - d_intra_stateᵢ      = d_sᵢ₊₁  (current d_running_state)
341        //   - d_decayᵢ            = d_sᵢ₊₁ · sᵢ
342        //   - d_sᵢ (propagated)   = decayᵢ · d_sᵢ₊₁ + d_chunk_input_state
343        vec_d_intra_bhpr.push(d_running_state_bhpr.clone());
344
345        let decay_chunk_bhpr: Tensor<B, 4> = da_chunk_end_bhn
346            .clone()
347            .slice(s![.., .., i_chunk]) // _bh1
348            .exp() // _bh
349            .unsqueeze_dim::<4>(3) // _bh11
350            .expand([batch, nheads, per_head_dim, state_rank]); // _bhpr
351        san(&decay_chunk_bhpr);
352
353        let d_decay_chunk_bhpr = d_running_state_bhpr.clone() * chunk_input_state_bhpr;
354        // d_da_chunk_end[b,h] = Σ_{p,r} d_decay · decay   (decay = exp(da_chunk_end))
355        let d_da_chunk_end_bh: Tensor<B, 2> = (d_decay_chunk_bhpr * decay_chunk_bhpr.clone())
356            .reshape([batch, nheads, per_head_dim * state_rank]) // _bhPR
357            .sum_dim(2) // _bh1
358            .squeeze_dim::<2>(2); // _bh
359        san(&d_da_chunk_end_bh);
360        vec_d_da_end_bh.push(d_da_chunk_end_bh);
361
362        d_running_state_bhpr = decay_chunk_bhpr * d_running_state_bhpr + d_chunk_input_state_bhpr;
363        san(&d_running_state_bhpr);
364    }
365    // d_initial_state = the trailing d_running_state after the reverse loop.
366    let d_initial_state_bhpr = d_running_state_bhpr;
367
368    // ── Restore natural (forward) chunk order ─────────────────────────────
369    vec_orange_d_x_bhlp.reverse();
370    vec_orange_d_dt_bhl.reverse();
371    vec_orange_d_da_bhl.reverse();
372    vec_d_cb_bhll.reverse();
373    vec_blue_d_c_bhlr.reverse();
374    vec_blue_d_da_bhl.reverse();
375    vec_d_intra_bhpr.reverse();
376    vec_d_da_end_bh.reverse();
377
378    // ── Stack per-chunk slices back into batched tensors ──────────────────
379    let d_x_orange_bnlhp: Tensor<B, 5> =
380        Tensor::stack::<5>(vec_orange_d_x_bhlp, 1).permute([0, 1, 3, 2, 4]);
381    let d_dt_orange_bhnl: Tensor<B, 4> = Tensor::stack(vec_orange_d_dt_bhl, 2);
382    let d_da_orange_bhnl: Tensor<B, 4> = Tensor::stack(vec_orange_d_da_bhl, 2);
383    let d_cb_bnhll: Tensor<B, 5> = Tensor::stack(vec_d_cb_bhll, 1);
384    let d_da_blue_bhnl: Tensor<B, 4> = Tensor::stack(vec_blue_d_da_bhl, 2);
385    let d_intra_chunk_state_bnhpr: Tensor<B, 5> = Tensor::stack(vec_d_intra_bhpr, 1);
386    let d_c_blue_bnhlr: Tensor<B, 5> = Tensor::stack(vec_blue_d_c_bhlr, 1);
387    let d_da_end_bhn: Tensor<B, 3> = Tensor::stack(vec_d_da_end_bh, 2);
388    san(&d_x_orange_bnlhp);
389    san(&d_dt_orange_bhnl);
390    san(&d_da_orange_bhnl);
391    san(&d_cb_bnhll);
392    san(&d_da_blue_bhnl);
393    san(&d_intra_chunk_state_bnhpr);
394    san(&d_c_blue_bnhlr);
395
396    // d_da_cumsum from K4: only the last-l position of each chunk gets the
397    // d_da_chunk_end contribution (da_chunk_end = cumA[chunk_len-1]).
398    let d_da_cumsum_k4_bhnl = {
399        let zeros = Tensor::<B, 4>::zeros([batch, nheads, nchunks, chunk_len - 1], &device);
400        Tensor::cat(vec![zeros, d_da_end_bhn.unsqueeze_dim::<4>(3)], 3)
401    };
402
403    let d_c_blue_bnlhr = d_c_blue_bnhlr.permute([0, 1, 3, 2, 4]);
404
405    // ═══════════════════════════════════════════════════════════════════════
406    // K3 BACKWARD
407    //
408    // Forward (recap):
409    //   b_bnhlr = b_bnlhr.permute
410    //   x_bnhpl = x_bnlhp.permute
411    //   forward_decay_to_chunk_end = exp(cumA_last − cumA)
412    //   b_bar_scale = forward_decay · dt
413    //   b_scaled = b · b_bar_scale_broadcast
414    //   intra_state = x_bnhpl @ b_scaled
415    // ═══════════════════════════════════════════════════════════════════════
416    let x_bnhpl = x_bnlhp.permute([0, 1, 3, 4, 2]);
417
418    // d_x_k3 = d_intra_state @ b_scaled^T
419    let d_x_k3_bnlhp = d_intra_chunk_state_bnhpr
420        .clone()
421        .matmul(
422            b_scaled_bnhlr.clone().permute([0, 1, 2, 4, 3]), // b_scaled_bnhrl
423        ) // d_x_k3_bnhpl
424        .permute([0, 1, 4, 2, 3]); // d_x_k3_bnlhp
425    san(&d_x_k3_bnlhp);
426
427    // d_b_scaled = x^T @ d_intra_state
428    let d_b_scaled_bnhlr = x_bnhpl
429        .permute([0, 1, 2, 4, 3]) // x_bnhlp
430        .matmul(d_intra_chunk_state_bnhpr);
431    san(&d_b_scaled_bnhlr);
432
433    // Split d_b_scaled via the product rule: d_b = d_b_scaled · b_bar_scale ; d_b_bar_scale = d_b_scaled · b
434    let b_bar_scale_bnhlr = b_bar_scale_bhnl
435        .clone()
436        .permute([0, 2, 1, 3]) // _bnhl
437        .unsqueeze_dim::<5>(4) // _bnhl1
438        .expand([batch, nchunks, nheads, chunk_len, state_rank]);
439    let d_b_k3_bnhlr = d_b_scaled_bnhlr.clone() * b_bar_scale_bnhlr;
440    let d_b_k3_bnlhr = d_b_k3_bnhlr.permute([0, 1, 3, 2, 4]);
441    san(&d_b_k3_bnlhr);
442
443    let b_bnhlr = b_bnlhr.clone().permute([0, 1, 3, 2, 4]);
444    let d_b_bar_scale_bhnl = (d_b_scaled_bnhlr * b_bnhlr)
445        .sum_dim(4) // _bnhl1
446        .squeeze_dim::<4>(4) // _bnhl
447        .permute([0, 2, 1, 3]); // _bhnl
448    san(&d_b_bar_scale_bhnl);
449
450    // Through the product rule for b_bar_scale = forward_decay · dt:
451    //   d_forward_decay = d_b_bar_scale · dt
452    //   d_dt_k3         = d_b_bar_scale · forward_decay
453    let d_forward_decay_bhnl = d_b_bar_scale_bhnl.clone() * dt_discretized_bhnl.clone();
454    let d_dt_discretized_k3_bhnl = d_b_bar_scale_bhnl * forward_decay_to_chunk_end_bhnl.clone();
455
456    // Through exp: d_(cumA_last − cumA) = d_forward_decay · forward_decay
457    let d_da_delta_bhnl = d_forward_decay_bhnl * forward_decay_to_chunk_end_bhnl;
458    // Subtraction splits into:
459    //   d_cumA_last = +Σ_l d_da_delta (it was broadcast over l), scattered into last-l position.
460    //   d_cumA      = −d_da_delta
461    let d_da_cumsum_sub_bhnl = -d_da_delta_bhnl.clone();
462    let d_da_cumsum_last_bhn = d_da_delta_bhnl
463        .sum_dim(3) // _bhn1
464        .squeeze_dim::<3>(3); // _bhn
465    let d_da_cumsum_k3_bhnl = {
466        let zeros = Tensor::<B, 4>::zeros([batch, nheads, nchunks, chunk_len - 1], &device);
467        d_da_cumsum_sub_bhnl
468            + Tensor::cat(vec![zeros, d_da_cumsum_last_bhn.unsqueeze_dim::<4>(3)], 3)
469    };
470    san(&d_da_cumsum_k3_bhnl);
471
472    // ═══════════════════════════════════════════════════════════════════════
473    // K2 BACKWARD (from d_cb_bnhll)
474    //
475    //   cb_bnhll = c_bnhlr @ b_bnhrl
476    //   d_c = d_cb @ b ;  d_b = d_cb^T @ c
477    // ═══════════════════════════════════════════════════════════════════════
478    let c_bnhlr = c_bnlhr.permute([0, 1, 3, 2, 4]);
479    let b_bnhlr = b_bnlhr.permute([0, 1, 3, 2, 4]);
480
481    let d_c_k2_bnhlr = d_cb_bnhll.clone().matmul(b_bnhlr.clone());
482    let d_c_k2_bnlhr = d_c_k2_bnhlr.permute([0, 1, 3, 2, 4]);
483    san(&d_c_k2_bnlhr);
484
485    let d_b_k2_bnhlr = d_cb_bnhll
486        .permute([0, 1, 2, 4, 3]) // d_cb_bnhsltl (target/source swap)
487        .matmul(c_bnhlr);
488    let d_b_k2_bnlhr = d_b_k2_bnhlr.permute([0, 1, 3, 2, 4]);
489    san(&d_b_k2_bnlhr);
490
491    // ═══════════════════════════════════════════════════════════════════════
492    // SUM GRADIENT CONTRIBUTIONS + K1 BACKWARD
493    // ═══════════════════════════════════════════════════════════════════════
494
495    // Test-only: local same-chunk d_da contribution (BLUE + ORANGE) snapshot
496    // for the `out_x · dout − ddt · dt` oracle. Production builds skip the
497    // extra add and the retained _bhnl tensor.
498    #[cfg(test)]
499    let d_da_local_bhnl = d_da_blue_bhnl.clone() + d_da_orange_bhnl.clone();
500    #[cfg(test)]
501    san(&d_da_local_bhnl);
502
503    let d_da_cumsum_bhnl =
504        d_da_blue_bhnl + d_da_orange_bhnl + d_da_cumsum_k3_bhnl + d_da_cumsum_k4_bhnl;
505    san(&d_da_cumsum_bhnl);
506
507    // K1 forward:  da_cumsum[l] = cumsumₗ(dt[l] · a_decay)
508    // Reverse:     d_da[l] = Σ_{k ≥ l} d_da_cumsum[k]   (suffix sum)
509    //                      = total − cumsum(d_da_cumsum)[l−1]   (cumsum[−1] = 0)
510    let d_da_bhnl = {
511        let d_total_bhnl = d_da_cumsum_bhnl
512            .clone()
513            .sum_dim(3) // _bhn1
514            .expand([batch, nheads, nchunks, chunk_len]);
515        let prefix_bhnl = d_da_cumsum_bhnl.cumsum(3);
516        let zeros_bhn1 = Tensor::<B, 4>::zeros([batch, nheads, nchunks, 1], &device);
517        let prefix_shifted_bhnl =
518            Tensor::cat(vec![zeros_bhn1, prefix_bhnl.narrow(3, 0, chunk_len - 1)], 3);
519        d_total_bhnl - prefix_shifted_bhnl
520    };
521    san(&d_da_bhnl);
522
523    // d_dt from K1: d_dt = d_da · a_decay
524    let a_decay_111h1 = a_decay_h
525        .unsqueeze_dims::<4>(&[0, 2, 3])
526        .expand([batch, nheads, nchunks, chunk_len]);
527    let d_dt_k1_bhnl = d_da_bhnl.clone() * a_decay_111h1;
528    san(&d_dt_k1_bhnl);
529
530    // d_a_decay[h] = Σ_{b,n,l} d_da[b,h,n,l] · dt[b,h,n,l]
531    let d_a_decay_h = (d_da_bhnl * dt_discretized_bhnl.clone())
532        .permute([1, 0, 2, 3]) // _hbnl
533        .reshape([nheads, batch * nchunks * chunk_len]) // _hBNL
534        .sum_dim(1) // _h1
535        .squeeze_dim::<1>(1); // _h
536    san(&d_a_decay_h);
537
538    // ── Combine per-input gradient contributions ──────────────────────────
539    #[cfg(test)]
540    let d_dt_orange_bhnl_save = d_dt_orange_bhnl.clone();
541    let d_dt_discretized_bhnl = d_dt_orange_bhnl + d_dt_discretized_k3_bhnl + d_dt_k1_bhnl;
542    san(&d_dt_discretized_bhnl);
543
544    let d_x_bnlhp = d_x_skip_bnlhp + d_x_k3_bnlhp + d_x_orange_bnlhp;
545    san(&d_x_bnlhp);
546
547    let d_b_bnlhr = d_b_k2_bnlhr + d_b_k3_bnlhr;
548    san(&d_b_bnlhr);
549    let d_c_bnlhr = d_c_k2_bnlhr + d_c_blue_bnlhr;
550    san(&d_c_bnlhr);
551
552    CombinedGrads {
553        d_a_decay_h,
554        d_dt_discretized_bhnl,
555        d_x_bnlhp,
556        d_b_bnlhr,
557        d_c_bnlhr,
558        d_d_h,
559        d_initial_state_bhpr,
560        #[cfg(test)]
561        d_da_local_bhnl,
562        #[cfg(test)]
563        d_dt_orange_bhnl: d_dt_orange_bhnl_save,
564    }
565}

568#[cfg(feature = "backend-flex")]
569mod tests {
570    use super::*;
571    use crate::mamba2::ssd::serial;
572    use burn::backend::Flex;
573    use burn::tensor::Distribution;
574
575    type B = Flex;
576
577    /// Oracle for the local (BLUE+ORANGE) part of `d_da_cumsum`.
578    ///
579    /// Per Tri Dao's `_chunk_scan_bwd_ddAcs_unstable` (ssd_chunk_scan.py:1618):
580    /// ```text
581    /// d_da_cumsum_local[b,h,n,l]
582    ///   = Σ_p out_x[b,n,l,h,p] · d_y[b,n,l,h,p]
583    ///     − d_dt_orange[b,h,n,l] · dt[b,h,n,l]
584    /// ```
585    /// where `out_x = y − D·x` (output before D-skip) and `d_dt_orange` is
586    /// the same-chunk d_dt contribution from ORANGE only.
587    ///
588    /// This is a stronger correctness check than the autodiff comparison:
589    /// both sides are computed via independent analytical paths and should
590    /// match to fp32 precision on small inputs. The identity is numerically
591    /// unstable for large inputs (catastrophic cancellation between einsum
592    /// and ddt·dt).
593    #[test]
594    fn oracle_da_local_matches_einsum_minus_ddt_dt() {
595        let device = Default::default();
596        let batch = 2;
597        let nchunks = 3;
598        let chunk_len = 4;
599        let nheads = 4;
600        let per_head_dim = 4;
601        let state_rank = 6;
602
603        // ─── Random inputs (small, gentle distributions) ─────────────────
604        let x_bnlhp = Tensor::<B, 5>::random(
605            [batch, nchunks, chunk_len, nheads, per_head_dim],
606            Distribution::Normal(0.0, 1.0),
607            &device,
608        );
609        let dt_bnlh = Tensor::<B, 4>::random(
610            [batch, nchunks, chunk_len, nheads],
611            Distribution::Uniform(0.05, 0.3),
612            &device,
613        );
614        let a_decay_h =
615            Tensor::<B, 1>::random([nheads], Distribution::Uniform(-1.0, -0.5), &device);
616        let b_bnlhr = Tensor::<B, 5>::random(
617            [batch, nchunks, chunk_len, nheads, state_rank],
618            Distribution::Normal(0.0, 1.0),
619            &device,
620        );
621        let c_bnlhr = Tensor::<B, 5>::random(
622            [batch, nchunks, chunk_len, nheads, state_rank],
623            Distribution::Normal(0.0, 1.0),
624            &device,
625        );
626        let d_h = Tensor::<B, 1>::random([nheads], Distribution::Normal(0.0, 0.1), &device);
627        let initial_state_bhpr = Tensor::<B, 4>::random(
628            [batch, nheads, per_head_dim, state_rank],
629            Distribution::Normal(0.0, 0.1),
630            &device,
631        );
632        let dt_discretized_bhnl = dt_bnlh.permute([0, 3, 1, 2]);
633
634        // ─── Forward (Serial path) ───────────────────────────────────────
635        let (da_cumsum_bhnl, da_chunk_end_bhn) =
636            serial::k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), a_decay_h.clone());
637        let cb_bnhll = serial::k2_ssd_bmm(c_bnlhr.clone(), b_bnlhr.clone());
638        let intra_chunk_state_bnhpr = serial::k3_ssd_chunk_state(
639            x_bnlhp.clone(),
640            b_bnlhr.clone(),
641            da_cumsum_bhnl.clone(),
642            dt_discretized_bhnl.clone(),
643        );
644        let (chunk_input_state_bnhpr, _final_state_bhpr) = serial::k4_ssd_state_passing(
645            intra_chunk_state_bnhpr,
646            da_chunk_end_bhn,
647            initial_state_bhpr.clone(),
648        );
649        let y_bnlhp = serial::k5_ssd_chunk_scan(
650            da_cumsum_bhnl,
651            dt_discretized_bhnl.clone(),
652            x_bnlhp.clone(),
653            c_bnlhr.clone(),
654            cb_bnhll,
655            chunk_input_state_bnhpr,
656            d_h.clone(),
657        );
658
659        // ─── out_x = y − D·x  (output before D-skip) ─────────────────────
660        let skip_bnlhp = d_h
661            .clone()
662            .unsqueeze_dims::<5>(&[0, 1, 2, 4]) // d_111h1
663            .expand([batch, nchunks, chunk_len, nheads, per_head_dim]) // d_bnlhp
664            * x_bnlhp.clone();
665        let out_x_bnlhp = y_bnlhp - skip_bnlhp;
666
667        // ─── Random upstream gradients ────────────────────────────────────
668        let d_y_bnlhp = Tensor::<B, 5>::random(
669            [batch, nchunks, chunk_len, nheads, per_head_dim],
670            Distribution::Normal(0.0, 1.0),
671            &device,
672        );
673        let d_final_bhpr = Tensor::<B, 4>::random(
674            [batch, nheads, per_head_dim, state_rank],
675            Distribution::Normal(0.0, 1.0),
676            &device,
677        );
678
679        // ─── Run combined_backward (gets d_da_local + d_dt_orange) ───────
680        let grads = combined_backward(
681            d_y_bnlhp.clone(),
682            d_final_bhpr,
683            x_bnlhp,
684            dt_discretized_bhnl.clone(),
685            b_bnlhr,
686            c_bnlhr,
687            d_h,
688            initial_state_bhpr,
689            a_decay_h,
690        );
691
692        // ─── Oracle: einsum(out_x, d_y, "bnlhp,bnlhp->bhnl") − d_dt_orange·dt
693        let einsum_bhnl: Tensor<B, 4> = (out_x_bnlhp * d_y_bnlhp)
694            .sum_dim(4) // einsum_bnlh1
695            .squeeze_dim::<4>(4) // einsum_bnlh
696            .permute([0, 3, 1, 2]); // einsum_bhnl
697        let oracle_bhnl = einsum_bhnl - grads.d_dt_orange_bhnl * dt_discretized_bhnl;
698
699        // ─── Compare ─────────────────────────────────────────────────────
700        let diff: f32 = (grads.d_da_local_bhnl - oracle_bhnl)
701            .abs()
702            .max()
703            .into_scalar()
704            .elem();
705        assert!(
706            diff < 1e-3,
707            "d_da_local oracle identity violated; max abs diff = {diff}",
708        );
709    }
