6pub struct CombinedGrads<B: Backend> {
7    pub d_x_bnlhp: Tensor<B, 5>,
8    pub d_dt_discretized_bhnl: Tensor<B, 4>,
9    pub d_b_bnlgr: Tensor<B, 5>,

11    pub d_d_h: Tensor<B, 1>,
12    pub d_initial_state_bhpr: Tensor<B, 4>,
13    pub d_a_decay_h: Tensor<B, 1>,
14    /// Local same-chunk contribution to `d_da_cumsum` from BLUE+ORANGE only
15    /// (excludes K3 and K4 cross-chunk contributions). Exposed for the
16    /// `out_x · dout − ddt · dt` oracle test from Tri Dao's reference
17    /// (`_chunk_scan_bwd_ddAcs_unstable`). Test-only; absent in release builds.
18    #[cfg(test)]
19    pub d_da_local_bhnl: Tensor<B, 4>,
20    /// Same-chunk d_dt contribution from ORANGE only (= what Tri Dao calls
21    /// `ddt` in `_chunk_scan_bwd_ddAcs_unstable`). Test-only; absent in release
22    /// builds.
23    #[cfg(test)]
24    pub d_dt_orange_bhnl: Tensor<B, 4>,
25}

27/// Same as [k3_ssd_chunk_state](serial::k3_ssd_chunk_state) but return some intermediaries
28/// that are useful to the custom backward operation.
29///
30/// Returns:
31/// - intra_chunk_state_bnhpr
32/// - b_bar_scale_bhnl
33/// - forward_decay_to_chunk_end_bhnl
34/// - b_scaled_bnhlr
35pub fn k3_ssd_chunk_state_extended<B: Backend>(
36    x_bnlhp: Tensor<B, 5>,
37    b_bnlgr: Tensor<B, 5>,
38    da_cumsum_bhnl: Tensor<B, 4>,
39    dt_discretized_bhnl: Tensor<B, 4>,
40) -> (Tensor<B, 5>, Tensor<B, 4>, Tensor<B, 4>, Tensor<B, 5>) {
41    use burn::tensor::s;
42
43    let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
44    let [.., ngroups, state_rank] = b_bnlgr.dims();
45
46    // permute b and x to prepare them for the mamtul
47    // - 1/15: permute: (x_bnlhp [in][*]) -> (x_bnhpl)
48    let x_bnhpl = x_bnlhp.clone().permute([0, 1, 3, 4, 2]);
49    assert_eq!(
50        [batch, nchunks, nheads, per_head_dim, chunk_len],
51        x_bnhpl.dims()
52    );
53    // - 2: permute: (b_bnlgr [in][*]) -> (b_bnglr)
54    let b_bnglr = b_bnlgr.permute([0, 1, 3, 2, 4]); // note: still in groups instead of heads
55    assert_eq!(
56        [batch, nchunks, ngroups, chunk_len, state_rank],
57        b_bnglr.dims()
58    );
59
60    // Expand B from ngroups to nheads by repeating each group's
61    // projection across all heads_per_group heads in that group.
62    let heads_per_group = nheads / ngroups;
63    let b_bnhlr = b_bnglr
64        // - 3: unsqueeze: (b_bnglr) -> (b_bng1lr)
65        .unsqueeze_dim::<6>(3) // b_bng1lr
66        // - 4: expand: (b_bng1lr) -> (b_bngHlr)
67        .expand([
68            batch,
69            nchunks,
70            ngroups,
71            heads_per_group,
72            chunk_len,
73            state_rank,
74        ]) // b_bngHlr
75        // - 5: reshape: (b_bngHlr) -> (b_bnhlr)
76        .reshape([batch, nchunks, nheads, chunk_len, state_rank]); // b_bnhlr
77
78    // scale b
79    let da_cumsum_last_in_chunk_bhn1 =
80        // - 6: slice: (da_cumsum_bhnl [in][*]) -> (da_cumsum_last_in_chunk_bhn1)
81        da_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
82    assert_eq!(
83        [batch, nheads, nchunks, 1],
84        da_cumsum_last_in_chunk_bhn1.dims()
85    );
86
87    // - 7: expand: (da_cumsum_last_in_chunk_bhn1) -> (da_cumsum_last_bhnl)
88    let da_cumsum_last_bhnl =
89        da_cumsum_last_in_chunk_bhn1.expand([batch, nheads, nchunks, chunk_len]);
90    // - 8: sub: (da_cumsum_last_bhnl, da_cumsum_bhnl [from K1][*]) -> (da_delta_bhnl)
91    let da_delta_bhnl = da_cumsum_last_bhnl - da_cumsum_bhnl.clone();
92    san(&da_delta_bhnl);
93    // - 9: exp: (da_delta_bhnl) -> (forward_decay_to_chunk_end_bhnl [+])
94    let forward_decay_to_chunk_end_bhnl = da_delta_bhnl.exp();
95    assert_eq!(
96        [batch, nheads, nchunks, chunk_len],
97        forward_decay_to_chunk_end_bhnl.dims()
98    );
99    san(&forward_decay_to_chunk_end_bhnl);
100
101    // - 10: mul: (forward_decay_to_chunk_end_bhnl [+], dt_discretized_bhnl [in][*]) -> (b_bar_scale_bhnl [+])
102    let b_bar_scale_bhnl = forward_decay_to_chunk_end_bhnl.clone() * dt_discretized_bhnl.clone();
103    assert_eq!([batch, nheads, nchunks, chunk_len], b_bar_scale_bhnl.dims());
104    san(&b_bar_scale_bhnl);
105
106    // - 11: permute: (b_bar_scale_bhnl [+]) -> (b_bar_scale_bnhl)
107    let b_bar_scale_bnhl = b_bar_scale_bhnl.clone().permute([0, 2, 1, 3]);
108    assert_eq!([batch, nchunks, nheads, chunk_len], b_bar_scale_bnhl.dims());
109    let b_bar_scale_bnhlr = b_bar_scale_bnhl
110        // - 12: unsqueeze: (b_bar_scale_bnhl) -> (b_bar_scale_bnhl1)
111        .unsqueeze_dim::<5>(4) // b_bar_scale_bnhl1
112        // - 13: expand: (b_bar_scale_bnhl1) -> (b_bar_scale_bnhlr)
113        .expand([batch, nchunks, nheads, chunk_len, state_rank]); // b_bar_scale_bnhlr
114    // - 14: mul: (b_bnhlr, b_bar_scale_bnhlr) -> (b_scaled_bnhlr [+])
115    let b_scaled_bnhlr = b_bnhlr * b_bar_scale_bnhlr;
116    assert_eq!(
117        [batch, nchunks, nheads, chunk_len, state_rank],
118        b_scaled_bnhlr.dims()
119    );
120    san(&b_scaled_bnhlr);
121
122    // - 15/15: matmul: (x_bnhpl, b_scaled_bnhlr [+]) -> (intra_chunk_state_bnhpr [out][!])
123    let intra_chunk_state_bnhpr: Tensor<B, 5> = x_bnhpl.matmul(b_scaled_bnhlr.clone());
124    assert_eq!(
125        [batch, nchunks, nheads, per_head_dim, state_rank],
126        intra_chunk_state_bnhpr.dims()
127    );
128    san(&intra_chunk_state_bnhpr);
129    (
130        intra_chunk_state_bnhpr,
131        b_bar_scale_bhnl,
132        forward_decay_to_chunk_end_bhnl,
133        b_scaled_bnhlr,
134    )
135}

137/// Core gradient computation.  All arguments use the shapes from the forward.
138///
139/// `d_y_bnlhp`         : upstream gradient of the scan output  [batch,state_rank,chunk_len,nheads,per_head_dim]
140/// `d_final_bhpr`      : upstream gradient of the final state  [batch,nheads,per_head_dim,R]
141///
142/// Returns one `CombinedGrads` struct containing gradients for all 7 inputs.
143#[allow(clippy::too_many_arguments)]
144pub fn combined_backward<B: Backend>(
145    d_y_bnlhp: Tensor<B, 5>,
146    d_final_bhpr: Tensor<B, 4>,
147    // Saved forward inputs
148    x_bnlhp: Tensor<B, 5>,
149    dt_discretized_bhnl: Tensor<B, 4>,
150    b_bnlgr: Tensor<B, 5>,
151    c_bnlgr: Tensor<B, 5>,
152    d_h: Tensor<B, 1>,
153    initial_state_bhpr: Tensor<B, 4>,
154    a_decay_h: Tensor<B, 1>,
155) -> CombinedGrads<B> {
156    let [batch, nheads, nchunks, chunk_len] = dt_discretized_bhnl.dims();
157    let [.., per_head_dim] = x_bnlhp.dims();
158    let [.., ngroups, state_rank] = b_bnlgr.dims();
159    let heads_per_group = nheads / ngroups;
160    let device = dt_discretized_bhnl.device();
161
162    san(&d_y_bnlhp);
163    san(&d_final_bhpr);
164    san(&x_bnlhp);
165    san(&dt_discretized_bhnl);
166    san(&b_bnlgr);
167    san(&c_bnlgr);
168    san(&d_h);
169    san(&initial_state_bhpr);
170    san(&a_decay_h);
171
172    // ═══════════════════════════════════════════════════════════════════════
173    // RECOMPUTE FORWARD INTERMEDIATES (the memory-saving heart of this op)
174    // ═══════════════════════════════════════════════════════════════════════
175
176    // K1 recomputation ─────────────────────────────────────────────────────
177    // da_cumsum is not saved across the boundary; recompute from dt and a_decay.
178    let (da_cumsum_bhnl, da_chunk_end_bhn) =
179        serial::k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), a_decay_h.clone());
180    san(&da_cumsum_bhnl);
181    san(&da_chunk_end_bhn);
182
183    // K2 ───────────────────────────────────────────────────────────────────
184    let cb_bngll = serial::k2_ssd_bmm(c_bnlgr.clone(), b_bnlgr.clone());
185    san(&cb_bngll);
186
187    // K3 (with intermediates) ──────────────────────────────────────────────
188    let (
189        intra_chunk_state_bnhpr,
190        b_bar_scale_bhnl,
191        forward_decay_to_chunk_end_bhnl,
192        b_scaled_bnhlr,
193    ) = k3_ssd_chunk_state_extended(
194        x_bnlhp.clone(),
195        b_bnlgr.clone(),
196        da_cumsum_bhnl.clone(),
197        dt_discretized_bhnl.clone(),
198    );
199    san(&intra_chunk_state_bnhpr);
200    san(&b_bar_scale_bhnl);
201    san(&forward_decay_to_chunk_end_bhnl);
202    san(&b_scaled_bnhlr);
203
204    // K4 ───────────────────────────────────────────────────────────────────
205    let (chunk_input_state_bnhpr, _final_state_bhpr): (Tensor<B, 5>, Tensor<B, 4>) =
206        serial::k4_ssd_state_passing(
207            intra_chunk_state_bnhpr.clone(),
208            da_chunk_end_bhn.clone(),
209            initial_state_bhpr,
210        );
211    san(&chunk_input_state_bnhpr);
212    san(&_final_state_bhpr);
213
214    // ═══════════════════════════════════════════════════════════════════════
215    // K5 + K4 FUSED BACKWARD (reverse per-chunk loop)
216    // ═══════════════════════════════════════════════════════════════════════
217
218    // ── SKIP backward ─────────────────────────────────────────────────────
219    // - | 36/36: add: (y_partial_bnlhp, skip_bnlhp) -> (y_bnlhp [out])
220    // - | (d_skip_bnlhp = d_y_bnlhp)
221    let d_skip_bnlhp = d_y_bnlhp.clone();
222    //
223    //
224    // For d_h:
225    // - - | 33: mul: (d_bnlhp, x_bnlhp[*]) -> (skip_bnlhp)
226    // - - | (d_d_bnlhp = d_skip_bnlhp * x_bnlhp)
227    // - - | 32: expand: (d_111h1) -> (d_bnlhp)
228    // - - | 31: unsqueeze-dims: (d_h [*]) -> (d_111h1)
229    //
230    // - - | d_d[h] = Σ_{b,n,l,p} dy * x   — use permute+reshape to avoid chained sum_dim
231    let d_d_h = {
232        d_skip_bnlhp.clone()
233            .permute([3, 0, 1, 2, 4]) // d_y_hbnlp
234            .reshape([nheads, batch * nchunks * chunk_len * per_head_dim]) // d_y_hBNLP
235            * x_bnlhp
236                .clone()
237                .permute([3, 0, 1, 2, 4]) // x_hbnlp
238                .reshape([nheads, batch * nchunks * chunk_len * per_head_dim]) // x_hBNLP
239    }
240    .sum_dim(1) // d_d_h1
241    .reshape([nheads]); // d_d_h
242    san(&d_d_h);
243    //
244    // For d_x:
245    // - - | 33: mul: (d_bnlhp, x_bnlhp[*]) -> (skip_bnlhp)
246    // - - | (d_x_skip_bnlhp = d_skip_bnlhp * d_bnlhp)
247    let d_x_skip_bnlhp = d_skip_bnlhp
248        * d_h
249            .clone()
250            .unsqueeze_dims::<5>(&[0, 1, 2, 4]) // d_111h1
251            .expand([batch, nchunks, chunk_len, nheads, per_head_dim]) // d_bnlhp
252        ;
253    san(&d_x_skip_bnlhp);
254
255    // ── Fused K5 (BLUE + ORANGE) + K4 reverse loop ─────────────────────────
256    //
257    // Per-iteration working set is _bhll (not _bnhll).
258    // Per-chunk gradients are collected in vecs and stacked once the loop is done.
259
260    // _ll causal mask, broadcast as a view inside each iteration —
261    // value does not depend on (b,n,h).
262    let causal_mask_ll: Tensor<B, 2, burn::prelude::Bool> =
263        Tensor::<B, 2, burn::prelude::Bool>::tril_mask([chunk_len, chunk_len], 0, &device);
264
265    let mut vec_orange_d_x_bhlp: Vec<Tensor<B, 4>> = Vec::with_capacity(nchunks);
266    let mut vec_orange_d_dt_bhl: Vec<Tensor<B, 3>> = Vec::with_capacity(nchunks);
267    let mut vec_orange_d_da_bhl: Vec<Tensor<B, 3>> = Vec::with_capacity(nchunks);
268    let mut vec_d_cb_bgll: Vec<Tensor<B, 4>> = Vec::with_capacity(nchunks);
269    let mut vec_blue_d_c_bhlr: Vec<Tensor<B, 4>> = Vec::with_capacity(nchunks);
270    let mut vec_blue_d_da_bhl: Vec<Tensor<B, 3>> = Vec::with_capacity(nchunks);
271    let mut vec_d_intra_bhpr: Vec<Tensor<B, 4>> = Vec::with_capacity(nchunks);
272    let mut vec_d_da_end_bh: Vec<Tensor<B, 2>> = Vec::with_capacity(nchunks);
273
274    let mut d_running_state_bhpr: Tensor<B, 4> = d_final_bhpr;
275
276    for i_chunk in (0..nchunks).rev() {
277        // ── Per-chunk inputs (slice the batched tensors to [batch,nheads,...]) ──────
278        let da_cumsum_bhl: Tensor<B, 3> = da_cumsum_bhnl
279            .clone()
280            .slice(s![.., .., i_chunk, ..]) // da_cumsum_bh1l
281            .squeeze_dim::<3>(2); // da_cumsum_bhl
282        let dt_bhl: Tensor<B, 3> = dt_discretized_bhnl
283            .clone()
284            .slice(s![.., .., i_chunk, ..]) // dt_discretized_bh1l
285            .squeeze_dim::<3>(2); // dt_bhl
286        let x_bhlp: Tensor<B, 4> = x_bnlhp
287            .clone()
288            .slice(s![.., i_chunk, .., .., ..]) // x_b1lhp
289            .squeeze_dim::<4>(1) // x_blhp
290            .permute([0, 2, 1, 3]); // x_bhlp
291        let d_y_bhlp: Tensor<B, 4> = d_y_bnlhp
292            .clone()
293            .slice(s![.., i_chunk, .., .., ..]) // d_y_b1lhp
294            .squeeze_dim::<4>(1) // d_y_blhp
295            .permute([0, 2, 1, 3]); // d_y_bhlp
296
297        // GQA-expand C for this chunk: _blgr → _bhlr
298        let c_bhlr: Tensor<B, 4> = c_bnlgr
299            .clone()
300            .slice(s![.., i_chunk, .., .., ..]) // c_b1lgr
301            .squeeze_dim::<4>(1) // c_blgr
302            .unsqueeze_dim::<5>(3) // c_blg1r
303            .expand([batch, chunk_len, ngroups, heads_per_group, state_rank]) // c_blgHr
304            .reshape([batch, chunk_len, nheads, state_rank]) // c_blhr
305            .permute([0, 2, 1, 3]); // c_bhlr
306
307        // GQA-expand CB for this chunk: _bgll → _bhll
308        let cb_bhll: Tensor<B, 4> = cb_bngll
309            .clone()
310            .slice(s![.., i_chunk, .., .., ..]) // cb_b1gll
311            .squeeze_dim::<4>(1) // cb_bgll
312            .unsqueeze_dim::<5>(2) // cb_bg1ll
313            .expand([batch, ngroups, heads_per_group, chunk_len, chunk_len]) // cb_bgHll
314            .reshape([batch, nheads, chunk_len, chunk_len]); // cb_bhll
315
316        let causal_mask_bhll: Tensor<B, 4, burn::prelude::Bool> = causal_mask_ll
317            .clone()
318            .unsqueeze_dims::<4>(&[0, 1]) // causal_mask_11ll
319            .expand([batch, nheads, chunk_len, chunk_len]); // causal_mask_bhll
320
321        // Running state entering chunk i.
322        let chunk_input_state_bhpr: Tensor<B, 4> = chunk_input_state_bnhpr
323            .clone()
324            .slice(s![.., i_chunk, .., .., ..]) // chunk_input_state_b1hpr
325            .squeeze_dim::<4>(1); // chunk_input_state_bhpr
326        san(&chunk_input_state_bhpr);
327
328        // ── BLUE backward for chunk i ──────────────────────────────────────
329        // y_blue[l,p] = exp(da[l]) · Σ_r C[l,r] · state[p,r]
330        let exp_da_cumsum_bhl: Tensor<B, 3> = da_cumsum_bhl.clone().exp();
331        let exp_da_cumsum_bhlp: Tensor<B, 4> = exp_da_cumsum_bhl
332            .clone()
333            .unsqueeze_dim::<4>(3) // exp_da_cumsum_bhl1
334            .expand([batch, nheads, chunk_len, per_head_dim]); // exp_da_cumsum_bhlp
335        let d_blue_scaled_bhlp: Tensor<B, 4> = d_y_bhlp.clone(); // = d_y_partial
336        let d_blue_bhlp: Tensor<B, 4> = d_blue_scaled_bhlp.clone() * exp_da_cumsum_bhlp.clone();
337        san(&d_blue_bhlp);
338
339        // d_chunk_input_state = c^T @ d_blue
340        let d_chunk_input_state_bhpr: Tensor<B, 4> = c_bhlr
341            .clone()
342            .permute([0, 1, 3, 2]) // c_bhrl
343            .matmul(d_blue_bhlp.clone()) // d_chunk_input_state_bhrp
344            .permute([0, 1, 3, 2]); // d_chunk_input_state_bhpr
345        san(&d_chunk_input_state_bhpr);
346
347        // d_c_blue = d_blue @ chunk_input_state
348        let d_c_blue_bhlr: Tensor<B, 4> =
349            d_blue_bhlp.clone().matmul(chunk_input_state_bhpr.clone());
350        san(&d_c_blue_bhlr);
351        vec_blue_d_c_bhlr.push(d_c_blue_bhlr);
352
353        // d_da from BLUE:
354        //   blue_no_scale = c @ state^T
355        //   d_da[l] = Σ_p (d_blue_scaled[l,p] * blue_no_scale[l,p]) * exp_da[l]
356        let blue_bhlp: Tensor<B, 4> = c_bhlr.clone().matmul(
357            chunk_input_state_bhpr.clone().permute([0, 1, 3, 2]), // chunk_input_state_bhrp
358        ); // blue_bhlp
359        let d_exp_da_cumsum_bhlp: Tensor<B, 4> = d_blue_scaled_bhlp.clone() * blue_bhlp;
360        let d_da_blue_bhl: Tensor<B, 3> = (d_exp_da_cumsum_bhlp * exp_da_cumsum_bhlp)
361            .sum_dim(3) // d_da_blue_bhl1
362            .squeeze_dim::<3>(3); // d_da_blue_bhl
363        san(&d_da_blue_bhl);
364        vec_blue_d_da_bhl.push(d_da_blue_bhl);
365
366        // ── ORANGE backward for chunk i ────────────────────────────────────
367        // y_orange[l,p] = Σ_{s≤l} CB[l,s] · exp(da[l]-da[s]) · dt[s] · x[s,p]
368        //   CB_w[l,s] = CB[l,s] · exp_diff[l,s] · dt[s] · mask
369        let da_cumsum_target_bhll: Tensor<B, 4> = da_cumsum_bhl
370            .clone()
371            .unsqueeze_dim::<4>(3) // [batch,nheads,chunk_len_tgt,1]
372            .expand([batch, nheads, chunk_len, chunk_len]); // [batch,nheads,chunk_len_tgt,chunk_len_src]
373        let da_cumsum_source_bhll: Tensor<B, 4> = da_cumsum_bhl
374            .unsqueeze_dim::<4>(2) // [batch,nheads,1,chunk_len_src]
375            .expand([batch, nheads, chunk_len, chunk_len]); // [batch,nheads,chunk_len_tgt,chunk_len_src]
376        let da_cumsum_diff_bhll = da_cumsum_target_bhll - da_cumsum_source_bhll;
377        san(&da_cumsum_diff_bhll);
378
379        // Causal mask + exp stabiliser (-inf above the main diagonal).
380        let da_cumsum_diff_masked_bhll =
381            da_cumsum_diff_bhll.mask_fill(causal_mask_bhll.clone(), f32::NEG_INFINITY);
382        let da_cumsum_diff_exp_bhll = da_cumsum_diff_masked_bhll.exp();
383        san(&da_cumsum_diff_exp_bhll);
384
385        let dt_source_bhll: Tensor<B, 4> = dt_bhl
386            .unsqueeze_dim::<4>(2) // dt_bh1l
387            .expand([batch, nheads, chunk_len, chunk_len]);
388
389        let orange_lhs_partial1_bhll: Tensor<B, 4> =
390            cb_bhll.clone() * da_cumsum_diff_exp_bhll.clone();
391        san(&orange_lhs_partial1_bhll);
392        let orange_lhs_partial2_bhll: Tensor<B, 4> =
393            orange_lhs_partial1_bhll.clone() * dt_source_bhll.clone();
394        san(&orange_lhs_partial2_bhll);
395
396        let d_orange_bhlp: Tensor<B, 4> = d_y_bhlp; // = d_y_partial
397        // d_CB_w = d_orange @ x^T
398        let d_orange_lhs_partial2_bhll: Tensor<B, 4> = d_orange_bhlp.clone().matmul(
399            x_bhlp.permute([0, 1, 3, 2]), // x_bhpl
400        ); // d_orange_lhs_partial2_bhll
401        san(&d_orange_lhs_partial2_bhll);
402
403        // d_x = CB_w^T @ d_orange
404        let d_x_orange_bhlp: Tensor<B, 4> = orange_lhs_partial2_bhll
405            .permute([0, 1, 3, 2]) // TODO: describe this shape (contrast src and tgt)
406            .matmul(d_orange_bhlp); // d_x_orange_bhlp
407        san(&d_x_orange_bhlp);
408        vec_orange_d_x_bhlp.push(d_x_orange_bhlp);
409
410        // Mask off above-diagonal (set to 0 above the main diagonal).
411        let d_orange_lhs_partial2_bhll = d_orange_lhs_partial2_bhll.mask_fill(causal_mask_bhll, 0.);
412        san(&d_orange_lhs_partial2_bhll);
413        let d_orange_lhs_partial1_bhll = d_orange_lhs_partial2_bhll.clone() * dt_source_bhll;
414        san(&d_orange_lhs_partial1_bhll);
415
416        // d_dt[s] = Σ_{l≥s} d_CB_w[l,s] · CB[l,s] · decay[l,s]
417        let d_dt_orange_bhl: Tensor<B, 3> = (d_orange_lhs_partial2_bhll
418            * orange_lhs_partial1_bhll.clone()) // d_dt_orange_bhll
419        .sum_dim(2) // d_dt_orange_bh1l; TODO: describe this shape (contrast src and tgt)
420        .squeeze_dim::<3>(2); // d_dt_orange_bhl
421        san(&d_dt_orange_bhl);
422        vec_orange_d_dt_bhl.push(d_dt_orange_bhl);
423
424        // d_da from ORANGE: decay = exp(da_tgt - da_src)
425        //   d_da_tgt[l] += Σₛ (d_decay · decay)[l,s]
426        //   d_da_src[s] -= Σₗ (d_decay · decay)[l,s]
427        let d_da_cumsum_diff_exp_bhll = d_orange_lhs_partial1_bhll.clone() * cb_bhll;
428        let d_da_cumsum_diff_bhll = d_da_cumsum_diff_exp_bhll * da_cumsum_diff_exp_bhll.clone();
429        let d_da_tgt_bhl: Tensor<B, 3> =
430            d_da_cumsum_diff_bhll.clone().sum_dim(3).squeeze_dim::<3>(3);
431        let d_da_src_bhl: Tensor<B, 3> = d_da_cumsum_diff_bhll.sum_dim(2).squeeze_dim::<3>(2);
432        let d_da_orange_bhl: Tensor<B, 3> = d_da_tgt_bhl - d_da_src_bhl;
433        san(&d_da_orange_bhl);
434        vec_orange_d_da_bhl.push(d_da_orange_bhl);
435
436        // d_cb (GQA reduce nheads → ngroups): [batch,nheads,chunk_len,chunk_len] → [batch,ngroups,chunk_len,chunk_len]
437        let d_cb_bhll = d_orange_lhs_partial1_bhll * da_cumsum_diff_exp_bhll;
438        let d_cb_bgll: Tensor<B, 4> = d_cb_bhll
439            .reshape([batch, ngroups, heads_per_group, chunk_len, chunk_len]) // d_cb_bgHll
440            .sum_dim(2) // d_cb_bg1ll
441            .squeeze_dim::<4>(2); // d_cb_bgll
442        san(&d_cb_bgll);
443        vec_d_cb_bgll.push(d_cb_bgll);
444
445        // ── K4 backward step for chunk i ───────────────────────────────────
446        // d_intra[i] = current d_running_state (before the propagation step).
447        vec_d_intra_bhpr.push(d_running_state_bhpr.clone());
448
449        // Recompute decay for this chunk.
450        let decay_bhpr: Tensor<B, 4> = da_chunk_end_bhn
451            .clone()
452            .slice(s![.., .., i_chunk]) // da_chunk_end_bh1
453            .exp()
454            .unsqueeze_dim::<4>(3) // da_chunk_end_bh11
455            .expand([batch, nheads, per_head_dim, state_rank]); // decay_bhpr
456        san(&decay_bhpr);
457
458        // d_decay = d_running_state · running_state (running_state = chunk_input_state[i]).
459        let d_decay_bhpr = d_running_state_bhpr.clone() * chunk_input_state_bhpr;
460        san(&d_decay_bhpr);
461
462        // d_da_chunk_end[b,h] = Σ_{p,r} d_decay · decay   (decay = exp(da_chunk_end)).
463        let d_da_chunk_end_bh: Tensor<B, 2> = (d_decay_bhpr * decay_bhpr.clone())
464            .reshape([batch, nheads, per_head_dim * state_rank]) // d_da_chunk_end_bhPR
465            .sum_dim(2) // d_da_chunk_end_bh1
466            .squeeze_dim::<2>(2); // d_da_chunk_end_bh
467        san(&d_da_chunk_end_bh);
468        vec_d_da_end_bh.push(d_da_chunk_end_bh);
469
470        // Propagate to previous chunk:
471        //   d_running_state_prev = decay · d_running_state + d_chunk_input_state.
472        d_running_state_bhpr = decay_bhpr * d_running_state_bhpr + d_chunk_input_state_bhpr;
473        san(&d_running_state_bhpr);
474    }
475    // d_initial_state = the trailing d_running_state after the reverse loop.
476    let d_initial_state_bhpr = d_running_state_bhpr;
477
478    // ── Stack per-chunk gradients into the batched tensors K3/K2/K1 expect ──
479    // Restore natural (forward) chunk order — vecs were filled in reverse.
480    vec_orange_d_x_bhlp.reverse();
481    vec_orange_d_dt_bhl.reverse();
482    vec_orange_d_da_bhl.reverse();
483    vec_d_cb_bgll.reverse();
484    vec_blue_d_c_bhlr.reverse();
485    vec_blue_d_da_bhl.reverse();
486    vec_d_intra_bhpr.reverse();
487    vec_d_da_end_bh.reverse();
488
489    // Stack tensors
490    let d_x_orange_bnlhp: Tensor<B, 5> =
491        Tensor::stack::<5>(vec_orange_d_x_bhlp, 1).permute([0, 1, 3, 2, 4]);
492    let d_dt_orange_bhnl: Tensor<B, 4> = Tensor::stack(vec_orange_d_dt_bhl, 2);
493    let d_da_orange_bhnl: Tensor<B, 4> = Tensor::stack(vec_orange_d_da_bhl, 2);
494    let d_cb_bngll: Tensor<B, 5> = Tensor::stack(vec_d_cb_bgll, 1);
495    let d_da_blue_bhnl: Tensor<B, 4> = Tensor::stack(vec_blue_d_da_bhl, 2);
496    let d_intra_chunk_state_bnhpr: Tensor<B, 5> = Tensor::stack(vec_d_intra_bhpr, 1);
497    san(&d_x_orange_bnlhp);
498    san(&d_dt_orange_bhnl);
499    san(&d_da_orange_bhnl);
500    san(&d_cb_bngll);
501    san(&d_da_blue_bhnl);
502    san(&d_intra_chunk_state_bnhpr);
503    let d_c_blue_bnhlr: Tensor<B, 5> = Tensor::stack(vec_blue_d_c_bhlr, 1);
504    let d_c_blue_bnlgr: Tensor<B, 5> = d_c_blue_bnhlr
505        .reshape([
506            batch,
507            nchunks,
508            ngroups,
509            heads_per_group,
510            chunk_len,
511            state_rank,
512        ]) // d_c_blue_bngHlr
513        .sum_dim(3) // d_c_blue_bng1lr
514        .squeeze_dim::<5>(3) // d_c_blue_bnglr
515        .permute([0, 1, 3, 2, 4]); // d_c_blue_bnlgr
516    san(&d_c_blue_bnlgr);
517    let d_da_end_bhn: Tensor<B, 3> = Tensor::stack(vec_d_da_end_bh, 2);
518    let d_da_cumsum_k4_bhnl = {
519        let zeros = Tensor::<B, 4>::zeros([batch, nheads, nchunks, chunk_len - 1], &device);
520        let d_da_end_bhn1 = d_da_end_bhn.unsqueeze_dim::<4>(3);
521        Tensor::cat(vec![zeros, d_da_end_bhn1], 3)
522    };
523
524    // ═══════════════════════════════════════════════════════════════════════
525    // K3 BACKWARD
526    // ═══════════════════════════════════════════════════════════════════════
527    let x_bnhpl = x_bnlhp.clone().permute([0, 1, 3, 4, 2]);
528    // For d_x_bnlhp:
529    // - 15/15: matmul: (x_bnhpl, b_scaled_bnhlr [+]) -> (intra_chunk_state_bnhpr [out][!])
530    // - (d_x_bnhpl = d_intra_chunk_state_bnhpr @ b_scaled_bnhlr^T)
531    let d_x_k3_bnhpl = d_intra_chunk_state_bnhpr.clone().matmul(
532        b_scaled_bnhlr.clone().permute([0, 1, 2, 4, 3]), // b_scaled_bnhrl
533    );
534    san(&d_x_k3_bnhpl);
535    // - 1/15: permute: (x_bnlhp [in][*]) -> (x_bnhpl)
536    let d_x_k3_bnlhp = d_x_k3_bnhpl.permute([0, 1, 4, 2, 3]);
537    //
538    // - 15/15: matmul: (x_bnhpl, b_scaled_bnhlr [+]) -> (intra_chunk_state_bnhpr [out][!])
539    // (d_b_scaled_bnhlr = x_bnhpl^T @ d_intra_chunk_state_bnhpr)
540    let d_b_scaled_bnhlr = x_bnhpl
541        .permute([0, 1, 2, 4, 3]) // x_bnhlp
542        .matmul(d_intra_chunk_state_bnhpr); // d_b_scaled_bnhlr
543    san(&d_b_scaled_bnhlr);
544    //
545    // For d_b:
546    // - 14: mul: (b_bnhlr, b_bar_scale_bnhlr) -> (b_scaled_bnhlr [+])
547    // - (d_b_bnhlr = d_b_scaled_bnhlr * b_bar_scale_bnhlr)
548    let b_bar_scale_bnhlr = b_bar_scale_bhnl
549        .clone()
550        .permute([0, 2, 1, 3]) // b_bar_scale_bnhl // replay forward step 11
551        .unsqueeze_dim::<5>(4) // b_bar_scale_bnhl1 // replay forward step 12
552        .expand([batch, nchunks, nheads, chunk_len, state_rank]); // replay forward step 13
553    let d_b_k3_bnhlr = d_b_scaled_bnhlr.clone() * b_bar_scale_bnhlr;
554    san(&d_b_k3_bnhlr);
555    // - 5: reshape: (b_bngHlr) -> (b_bnhlr)
556    // - 4: expand: (b_bng1lr) -> (b_bngHlr)
557    // - 3: unsqueeze: (b_bnglr) -> (b_bng1lr)
558    // - 2: permute: (b_bnlgr [in][*]) -> (b_bnglr)
559    // GQA reduce: [batch,nchunks,H,chunk_len,R] → [batch,nchunks,ngroups,chunk_len,R] → [batch,nchunks,chunk_len,ngroups,R]
560    let d_b_k3_bnlgr = d_b_k3_bnhlr
561        .reshape([
562            batch,
563            nchunks,
564            ngroups,
565            heads_per_group,
566            chunk_len,
567            state_rank,
568        ]) // d_b_k3_bngHlr
569        .sum_dim(3) // d_b_k3_bng1lr
570        .squeeze_dim::<5>(3) // d_b_k3_bnglr
571        .permute([0, 1, 3, 2, 4]);
572    san(&d_b_k3_bnlgr);
573
574    // - 14: mul: (b_bnhlr, b_bar_scale_bnhlr) -> (b_scaled_bnhlr [+])
575    // - (d_b_bar_scale_bnhlr = d_b_scaled_bnhlr * b_bnhlr)
576    // GQA-expand B back to per-head for the product: [batch,nchunks,ngroups,chunk_len,R] → [batch,nchunks,H,chunk_len,R]
577    let b_bnhlr = b_bnlgr
578        .clone()
579        .permute([0, 1, 3, 2, 4]) // b_bnglr // replay forward step 2
580        .unsqueeze_dim::<6>(3) // b_bng1lr // replay forward step 3
581        .expand([
582            batch,
583            nchunks,
584            ngroups,
585            heads_per_group,
586            chunk_len,
587            state_rank,
588        ]) // b_bngHlr // replay forward step 4
589        .reshape([batch, nchunks, nheads, chunk_len, state_rank]); // b_bnhlr // replay forward step 5
590    let d_b_bar_scale_bnhlr = d_b_scaled_bnhlr.clone() * b_bnhlr;
591    san(&d_b_bar_scale_bnhlr);
592    // - 13: expand: (b_bar_scale_bnhl1) -> (b_bar_scale_bnhlr)
593    // - 12: unsqueeze: (b_bar_scale_bnhl) -> (b_bar_scale_bnhl1)
594    // - 11: permute: (b_bar_scale_bhnl [+]) -> (b_bar_scale_bnhl)
595    let d_b_bar_scale_bhnl = d_b_bar_scale_bnhlr
596        .sum_dim(4) // d_b_bar_scale_bnhl1
597        .squeeze_dim::<4>(4) // d_b_bar_scale_bnhl
598        .permute([0, 2, 1, 3]); // d_b_bar_scale_bhnl
599    san(&d_b_bar_scale_bhnl);
600    //
601    // For d_da_cumsum_bhnl:
602    // - 10: mul: (forward_decay_to_chunk_end_bhnl [+], dt_discretized_bhnl [in][*]) -> (b_bar_scale_bhnl [+])
603    // - (d_forward_decay_to_chunk_end_bhnl = d_b_bar_scale_bhnl * dt_discretized_bhnl)
604    let d_forward_decay_to_chunk_end_bhnl =
605        d_b_bar_scale_bhnl.clone() * dt_discretized_bhnl.clone();
606    san(&d_forward_decay_to_chunk_end_bhnl);
607    // - 9: exp: (da_delta_bhnl) -> (forward_decay_to_chunk_end_bhnl [+])
608    // - (d_da_delta_bhnl = d_forward_decay_to_chunk_end_bhnl * exp(da_delta_bhnl))
609    // note: forward_decay_to_chunk_end_bhnl = exp(da_delta_bhnl)
610    let d_da_delta_bhnl =
611        d_forward_decay_to_chunk_end_bhnl * forward_decay_to_chunk_end_bhnl.clone();
612    san(&d_da_delta_bhnl);
613    // - 8: sub: (da_cumsum_last_bhnl, da_cumsum_bhnl [from K1][*]) -> (da_delta_bhnl)
614    let d_da_cumsum_last_bhnl = d_da_delta_bhnl.clone();
615    let d_da_cumsum_sub_bhnl = -d_da_delta_bhnl.clone();
616    // - 7: expand: (da_cumsum_last_in_chunk_bhn1) -> (da_cumsum_last_bhnl)
617    // - 6: slice: (da_cumsum_bhnl [in][*]) -> (da_cumsum_last_in_chunk_bhn1)
618    let d_da_cumsum_last_bhn = d_da_cumsum_last_bhnl
619        .sum_dim(3) // d_da_cumsum_last_bhn1
620        .squeeze_dim::<3>(3); // d_da_cumsum_last_bhn
621    san(&d_da_cumsum_last_bhn);
622    //
623    // For d_dt_discretized_bhnl:
624    // - 10: mul: (forward_decay_to_chunk_end_bhnl [+], dt_discretized_bhnl [in][*]) -> (b_bar_scale_bhnl [+])
625    // - (d_dt_discretized_bhnl = d_b_bar_scale_bhnl * forward_decay_to_chunk_end_bhnl)
626    let d_dt_discretized_k3_bhnl = d_b_bar_scale_bhnl * forward_decay_to_chunk_end_bhnl;
627    san(&d_dt_discretized_k3_bhnl);
628    //
629
630    // TODO: understand this.
631    let d_da_cumsum_k3_bhnl = {
632        let zeros = Tensor::<B, 4>::zeros([batch, nheads, nchunks, chunk_len - 1], &device);
633        let d_last = d_da_cumsum_last_bhn.unsqueeze_dim::<4>(3);
634        d_da_cumsum_sub_bhnl + Tensor::cat(vec![zeros, d_last], 3)
635    };
636    san(&d_da_cumsum_k3_bhnl);
637
638    // ═══════════════════════════════════════════════════════════════════════
639    // K2 BACKWARD (from d_cb_bngll)
640    // ═══════════════════════════════════════════════════════════════════════
641    let c_bnglr = c_bnlgr.clone().permute([0, 1, 3, 2, 4]);
642    let b_bnglr = b_bnlgr.clone().permute([0, 1, 3, 2, 4]);
643    // - 3/3: matmul: (c_bnglr, b_bngrl) -> (cb_bngll [out][!])
644    // - cb[b,n,g,l,s] = Σ_r c[l,r]*b[s,r]  →  CB = C @ B^T
645    // -  d_C_bngls = d_CB @ B
646    // -  d_B_bngls = d_CB^T @ C
647    let d_c_k2_bnglr = d_cb_bngll.clone().matmul(b_bnglr.clone());
648    san(&d_c_k2_bnglr);
649    let d_c_k2_bnlgr = d_c_k2_bnglr.permute([0, 1, 3, 2, 4]);
650
651    let d_b_k2_bnglr = d_cb_bngll
652        .permute([0, 1, 2, 4, 3]) // [batch,nchunks,ngroups,chunk_len_src,chunk_len_tgt]
653        .matmul(c_bnglr.clone()); // d_b_k2_bnglr
654    san(&d_b_k2_bnglr);
655    let d_b_k2_bnlgr = d_b_k2_bnglr.permute([0, 1, 3, 2, 4]);
656
657    // ═══════════════════════════════════════════════════════════════════════
658    // SUM GRADIENT CONTRIBUTIONS
659    // ═══════════════════════════════════════════════════════════════════════
660
661    // Test-only: local same-chunk d_da contribution (BLUE + ORANGE) snapshot
662    // for the `out_x · dout − ddt · dt` oracle. Production builds skip the
663    // extra add and the retained _bhnl tensor.
664    #[cfg(test)]
665    let d_da_local_bhnl = d_da_blue_bhnl.clone() + d_da_orange_bhnl.clone();
666    #[cfg(test)]
667    san(&d_da_local_bhnl);
668    // Accumulated gradient of the cumulative sum produced by K1.
669    let d_da_cumsum_bhnl =
670        d_da_blue_bhnl + d_da_orange_bhnl + d_da_cumsum_k3_bhnl + d_da_cumsum_k4_bhnl;
671    san(&d_da_cumsum_bhnl);
672
673    // ── K1 BACKWARD ────────────────────────────────────────────────────────
674    // K1 forward: da_cumsum[l] = cumsumₗ(dt[l] * a_decay)
675    //
676    // Reverse cumsum (suffix sum) converts d_da_cumsum → d_da:
677    //   d_da[l] = sum_{k >= l} d_da_cumsum[k]
678    //           = total_sum - cumsum(d_da_cumsum)[l-1]   (cumsum[-1] == 0)
679    let d_da_cumsum_total_bhnl = d_da_cumsum_bhnl
680        .clone()
681        .sum_dim(3) // d_da_cumsum_bhn1
682        .expand([batch, nheads, nchunks, chunk_len]);
683    let prefix_sum_bhnl = d_da_cumsum_bhnl.clone().cumsum(3);
684    let zeros_bhn1 = Tensor::<B, 4>::zeros([batch, nheads, nchunks, 1], &device);
685    // prefix_sum shifted right by 1 (i.e., cumsum[l-1], with cumsum[-1] = 0)
686    let prefix_sum_shifted_bhnl = Tensor::cat(
687        vec![zeros_bhn1, prefix_sum_bhnl.narrow(3, 0, chunk_len - 1)],
688        3,
689    );
690    let d_da_bhnl = d_da_cumsum_total_bhnl - prefix_sum_shifted_bhnl; // suffix sum
691    san(&d_da_bhnl);
692    // d_dt from K1: d_dt = d_da * a_decay
693    let a_decay_expand = a_decay_h
694        .clone()
695        .unsqueeze_dims::<4>(&[0, 2, 3])
696        .expand([batch, nheads, nchunks, chunk_len]);
697    let d_dt_k1_bhnl = d_da_bhnl.clone() * a_decay_expand;
698    san(&d_dt_k1_bhnl);
699    // d_a_decay_h from K1: d_a[h] = sum_{b,n,l} d_da[b,h,n,l] * dt[b,h,n,l]
700    let d_a_decay_h = (d_da_bhnl * dt_discretized_bhnl.clone())
701        .permute([1, 0, 2, 3]) // d_a_decay_hbnl
702        .reshape([nheads, batch * nchunks * chunk_len]) // d_a_decay_hBNL
703        .sum_dim(1) // d_a_decay_h1
704        .squeeze_dim::<1>(1);
705    san(&d_a_decay_h);
706
707    // Test-only: keep d_dt_orange_bhnl alive for the oracle return.
708    #[cfg(test)]
709    let d_dt_orange_bhnl_save = d_dt_orange_bhnl.clone();
710    let d_dt_discretized_bhnl = d_dt_orange_bhnl + d_dt_discretized_k3_bhnl + d_dt_k1_bhnl;
711    san(&d_dt_discretized_bhnl);
712
713    let d_x_bnlhp = d_x_skip_bnlhp + d_x_k3_bnlhp + d_x_orange_bnlhp;
714    san(&d_x_bnlhp);
715
716    let d_b_bnlgr = d_b_k2_bnlgr + d_b_k3_bnlgr;
717    san(&d_b_bnlgr);
718    let d_c_bnlgr = d_c_k2_bnlgr + d_c_blue_bnlgr;
719    san(&d_c_bnlgr);
720
721    CombinedGrads {
722        d_a_decay_h,
723        d_dt_discretized_bhnl,
724        d_x_bnlhp,
725        d_b_bnlgr,
726        d_c_bnlgr,
727        d_d_h,
728        d_initial_state_bhpr,
729        #[cfg(test)]
730        d_da_local_bhnl,
731        #[cfg(test)]
732        d_dt_orange_bhnl: d_dt_orange_bhnl_save,
733    }
734}

752    ///     − d_dt_orange[b,h,n,l] · dt[b,h,n,l]
753    /// ```
754    /// where `out_x = y − D·x` (output before D-skip) and `d_dt_orange` is
755    /// the same-chunk d_dt contribution from ORANGE only.
756    ///
757    /// This is a stronger correctness check than the autodiff comparison:
758    /// both sides are computed via independent analytical paths and should
759    /// match to fp32 precision on small inputs. The identity is numerically
760    /// unstable for large inputs (catastrophic cancellation between einsum
761    /// and ddt·dt).
762    #[test]
763    fn oracle_da_local_matches_einsum_minus_ddt_dt() {
764        let device = Default::default();
765        let batch = 2;
766        let nchunks = 3;
767        let chunk_len = 4;
768        let nheads = 4;
769        let per_head_dim = 4;
770        let ngroups = 2;
771        let state_rank = 6;
772
773        // ─── Random inputs (small, gentle distributions) ─────────────────
774        let x_bnlhp = Tensor::<B, 5>::random(
775            [batch, nchunks, chunk_len, nheads, per_head_dim],
776            Distribution::Normal(0.0, 1.0),
777            &device,
778        );
779        let dt_bnlh = Tensor::<B, 4>::random(
780            [batch, nchunks, chunk_len, nheads],
781            Distribution::Uniform(0.05, 0.3),
782            &device,
783        );
784        let a_decay_h =
785            Tensor::<B, 1>::random([nheads], Distribution::Uniform(-1.0, -0.5), &device);
786        let b_bnlgr = Tensor::<B, 5>::random(
787            [batch, nchunks, chunk_len, ngroups, state_rank],
788            Distribution::Normal(0.0, 1.0),
789            &device,
790        );
791        let c_bnlgr = Tensor::<B, 5>::random(
792            [batch, nchunks, chunk_len, ngroups, state_rank],
793            Distribution::Normal(0.0, 1.0),
794            &device,
795        );
796        let d_h = Tensor::<B, 1>::random([nheads], Distribution::Normal(0.0, 0.1), &device);
797        let initial_state_bhpr = Tensor::<B, 4>::random(
798            [batch, nheads, per_head_dim, state_rank],
799            Distribution::Normal(0.0, 0.1),
800            &device,
801        );
802        let dt_discretized_bhnl = dt_bnlh.permute([0, 3, 1, 2]);
803
804        // ─── Forward (Serial path) ───────────────────────────────────────
805        let (da_cumsum_bhnl, da_chunk_end_bhn) =
806            serial::k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), a_decay_h.clone());
807        let cb_bngll = serial::k2_ssd_bmm(c_bnlgr.clone(), b_bnlgr.clone());
808        let intra_chunk_state_bnhpr = serial::k3_ssd_chunk_state(
809            x_bnlhp.clone(),
810            b_bnlgr.clone(),
811            da_cumsum_bhnl.clone(),
812            dt_discretized_bhnl.clone(),
813        );
814        let (chunk_input_state_bnhpr, _final_state_bhpr) = serial::k4_ssd_state_passing(
815            intra_chunk_state_bnhpr,
816            da_chunk_end_bhn,
817            initial_state_bhpr.clone(),
818        );
819        let y_bnlhp = serial::k5_ssd_chunk_scan(
820            da_cumsum_bhnl,
821            dt_discretized_bhnl.clone(),
822            x_bnlhp.clone(),
823            c_bnlgr.clone(),
824            cb_bngll,
825            chunk_input_state_bnhpr,
826            d_h.clone(),
827        );
828
829        // ─── out_x = y − D·x  (output before D-skip) ─────────────────────
830        let skip_bnlhp = d_h
831            .clone()
832            .unsqueeze_dims::<5>(&[0, 1, 2, 4]) // d_111h1
833            .expand([batch, nchunks, chunk_len, nheads, per_head_dim]) // d_bnlhp
834            * x_bnlhp.clone();
835        let out_x_bnlhp = y_bnlhp - skip_bnlhp;
836
837        // ─── Random upstream gradients ────────────────────────────────────
838        let d_y_bnlhp = Tensor::<B, 5>::random(
839            [batch, nchunks, chunk_len, nheads, per_head_dim],
840            Distribution::Normal(0.0, 1.0),
841            &device,
842        );
843        let d_final_bhpr = Tensor::<B, 4>::random(
844            [batch, nheads, per_head_dim, state_rank],
845            Distribution::Normal(0.0, 1.0),
846            &device,
847        );
848
849        // ─── Run combined_backward (gets d_da_local + d_dt_orange) ───────
850        let grads = combined_backward(
851            d_y_bnlhp.clone(),
852            d_final_bhpr,
853            x_bnlhp,
854            dt_discretized_bhnl.clone(),
855            b_bnlgr,
856            c_bnlgr,
857            d_h,
858            initial_state_bhpr,
859            a_decay_h,
860        );
861
862        // ─── Oracle: einsum(out_x, d_y, "bnlhp,bnlhp->bhnl") − d_dt_orange·dt
863        let einsum_bhnl: Tensor<B, 4> = (out_x_bnlhp * d_y_bnlhp)
864            .sum_dim(4) // einsum_bnlh1
865            .squeeze_dim::<4>(4) // einsum_bnlh
866            .permute([0, 3, 1, 2]); // einsum_bhnl
867        let oracle_bhnl = einsum_bhnl - grads.d_dt_orange_bhnl * dt_discretized_bhnl;
868
869        // ─── Compare ─────────────────────────────────────────────────────
870        let diff: f32 = (grads.d_da_local_bhnl - oracle_bhnl)
871            .abs()
872            .max()
873            .into_scalar()
874            .elem();
875        assert!(
876            diff < 1e-3,
877            "d_da_local oracle identity violated; max abs diff = {diff}",
878        );
879    }
