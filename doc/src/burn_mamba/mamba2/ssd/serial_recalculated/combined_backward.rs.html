15/// Same as [k3_ssd_chunk_state](serial::k3_ssd_chunk_state) but return some intermediaries
16/// that are useful to the custom backward operation.
17///
18/// Returns:
19/// - intra_chunk_state_bnhpr
20/// - b_bar_scale_bhnl
21/// - forward_decay_to_chunk_end_bhnl
22/// - b_scaled_bnhlr
23pub fn k3_ssd_chunk_state_extended<B: Backend>(
24    x_bnlhp: Tensor<B, 5>,
25    b_bnlgr: Tensor<B, 5>,
26    da_cumsum_bhnl: Tensor<B, 4>,
27    dt_discretized_bhnl: Tensor<B, 4>,
28) -> (Tensor<B, 5>, Tensor<B, 4>, Tensor<B, 4>, Tensor<B, 5>) {
29    use burn::tensor::s;
30
31    let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
32    let [.., ngroups, state_rank] = b_bnlgr.dims();
33
34    // permute b and x to prepare them for the mamtul
35    // - 1/15: permute: (x_bnlhp [in][*]) -> (x_bnhpl)
36    let x_bnhpl = x_bnlhp.clone().permute([0, 1, 3, 4, 2]);
37    assert_eq!(
38        [batch, nchunks, nheads, per_head_dim, chunk_len],
39        x_bnhpl.dims()
40    );
41    // - 2: permute: (b_bnlgr [in][*]) -> (b_bnglr)
42    let b_bnglr = b_bnlgr.permute([0, 1, 3, 2, 4]); // note: still in groups instead of heads
43    assert_eq!(
44        [batch, nchunks, ngroups, chunk_len, state_rank],
45        b_bnglr.dims()
46    );
47
48    // Expand B from ngroups to nheads by repeating each group's
49    // projection across all heads_per_group heads in that group.
50    let heads_per_group = nheads / ngroups;
51    let b_bnhlr = b_bnglr
52        // - 3: unsqueeze: (b_bnglr) -> (b_bng1lr)
53        .unsqueeze_dim::<6>(3) // b_bng1lr
54        // - 4: expand: (b_bng1lr) -> (b_bngHlr)
55        .expand([
56            batch,
57            nchunks,
58            ngroups,
59            heads_per_group,
60            chunk_len,
61            state_rank,
62        ]) // b_bngHlr
63        // - 5: reshape: (b_bngHlr) -> (b_bnhlr)
64        .reshape([batch, nchunks, nheads, chunk_len, state_rank]);
65
66    // scale b
67    let da_cumsum_last_in_chunk_bhn1 =
68        // - 6: slice: (da_cumsum_bhnl [in][*]) -> (da_cumsum_last_in_chunk_bhn1)
69        da_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
70    assert_eq!(
71        [batch, nheads, nchunks, 1],
72        da_cumsum_last_in_chunk_bhn1.dims()
73    );
74
75    // - 7: expand: (da_cumsum_last_in_chunk_bhn1) -> (da_cumsum_last_bhnl)
76    let da_cumsum_last_bhnl =
77        da_cumsum_last_in_chunk_bhn1.expand([batch, nheads, nchunks, chunk_len]);
78    // - 8: sub: (da_cumsum_last_bhnl, da_cumsum_bhnl [from K1][*]) -> (da_delta_bhnl)
79    let da_delta_bhnl = da_cumsum_last_bhnl - da_cumsum_bhnl.clone();
80    san(&da_delta_bhnl);
81    // - 9: exp: (da_delta_bhnl) -> (forward_decay_to_chunk_end_bhnl [+])
82    let forward_decay_to_chunk_end_bhnl = da_delta_bhnl.exp();
83    assert_eq!(
84        [batch, nheads, nchunks, chunk_len],
85        forward_decay_to_chunk_end_bhnl.dims()
86    );
87    san(&forward_decay_to_chunk_end_bhnl);
88
89    // - 10: mul: (forward_decay_to_chunk_end_bhnl [+], dt_discretized_bhnl [in][*]) -> (b_bar_scale_bhnl [+])
90    let b_bar_scale_bhnl = forward_decay_to_chunk_end_bhnl.clone() * dt_discretized_bhnl.clone();
91    assert_eq!([batch, nheads, nchunks, chunk_len], b_bar_scale_bhnl.dims());
92    san(&b_bar_scale_bhnl);
93
94    // - 11: permute: (b_bar_scale_bhnl [+]) -> (b_bar_scale_bnhl)
95    let b_bar_scale_bnhl = b_bar_scale_bhnl.clone().permute([0, 2, 1, 3]);
96    assert_eq!([batch, nchunks, nheads, chunk_len], b_bar_scale_bnhl.dims());
97    let b_bar_scale_bnhlr = b_bar_scale_bnhl
98        // - 12: unsqueeze: (b_bar_scale_bnhl) -> (b_bar_scale_bnhl1)
99        .unsqueeze_dim::<5>(4) // b_bar_scale_bnhl1
100        // - 13: expand: (b_bar_scale_bnhl1) -> (b_bar_scale_bnhlr)
101        .expand([batch, nchunks, nheads, chunk_len, state_rank]);
102    // - 14: mul: (b_bnhlr, b_bar_scale_bnhlr) -> (b_scaled_bnhlr [+])
103    let b_scaled_bnhlr = b_bnhlr * b_bar_scale_bnhlr;
104    assert_eq!(
105        [batch, nchunks, nheads, chunk_len, state_rank],
106        b_scaled_bnhlr.dims()
107    );
108    san(&b_scaled_bnhlr);
109
110    // - 15/15: matmul: (x_bnhpl, b_scaled_bnhlr [+]) -> (intra_chunk_state_bnhpr [out][!])
111    let intra_chunk_state_bnhpr: Tensor<B, 5> = x_bnhpl.matmul(b_scaled_bnhlr.clone());
112    assert_eq!(
113        [batch, nchunks, nheads, per_head_dim, state_rank],
114        intra_chunk_state_bnhpr.dims()
115    );
116    san(&intra_chunk_state_bnhpr);
117    (
118        intra_chunk_state_bnhpr,
119        b_bar_scale_bhnl,
120        forward_decay_to_chunk_end_bhnl,
121        b_scaled_bnhlr,
122    )
123}

125/// Core gradient computation.  All arguments use the shapes from the forward.
126///
127/// `d_y_bnlhp`         : upstream gradient of the scan output  [B,N,L,H,P]
128/// `d_final_bhpr`      : upstream gradient of the final state  [B,H,P,R]
129///
130/// Returns one `CombinedGrads` struct containing gradients for all 7 inputs.
131#[allow(clippy::too_many_arguments)]
132pub fn combined_backward<B: Backend>(
133    d_y_bnlhp: Tensor<B, 5>,
134    d_final_bhpr: Tensor<B, 4>,
135    // Saved forward inputs
136    x_bnlhp: Tensor<B, 5>,
137    dt_discretized_bhnl: Tensor<B, 4>,
138    b_bnlgr: Tensor<B, 5>,
139    c_bnlgr: Tensor<B, 5>,
140    d_h: Tensor<B, 1>,
141    initial_state_bhpr: Tensor<B, 4>,
142    da_cumsum_bhnl: Tensor<B, 4>, // from K1
143) -> CombinedGrads<B> {
144    let [batch, nheads, nchunks, chunk_len] = da_cumsum_bhnl.dims();
145    let [.., per_head_dim] = x_bnlhp.dims();
146    let [.., ngroups, state_rank] = b_bnlgr.dims();
147    let heads_per_group = nheads / ngroups;
148    let device = da_cumsum_bhnl.device();
149
150    san(&d_y_bnlhp);
151    san(&d_final_bhpr);
152    san(&x_bnlhp);
153    san(&dt_discretized_bhnl);
154    san(&b_bnlgr);
155    san(&c_bnlgr);
156    san(&d_h);
157    san(&initial_state_bhpr);
158    san(&da_cumsum_bhnl);
159
160    // ═══════════════════════════════════════════════════════════════════════
161    // RECOMPUTE FORWARD INTERMEDIATES (the memory-saving heart of this op)
162    // ═══════════════════════════════════════════════════════════════════════
163
164    // K1 (steps 5~6) ───────────────────────────────────────────────────────
165    let da_chunk_end_bhn = da_cumsum_bhnl
166        .clone()
167        .slice(s![.., .., .., -1]) // da_chunk_end_bhn1
168        .squeeze_dim::<3>(3);
169
170    // K2 ───────────────────────────────────────────────────────────────────
171    let cb_bngll = serial::k2_ssd_bmm(c_bnlgr.clone(), b_bnlgr.clone());
172    // let cb_bngll = k2_forward(&c_bnlgr, &b_bnlgr);          // [B,N,G,L,L]
173    san(&cb_bngll);
174
175    // K3 (with intermediates) ──────────────────────────────────────────────
176    let (
177        intra_chunk_state_bnhpr,
178        b_bar_scale_bhnl,
179        forward_decay_to_chunk_end_bhnl,
180        b_scaled_bnhlr,
181    ) = k3_ssd_chunk_state_extended(
182        x_bnlhp.clone(),
183        b_bnlgr.clone(),
184        da_cumsum_bhnl.clone(),
185        dt_discretized_bhnl.clone(),
186    );
187    san(&intra_chunk_state_bnhpr);
188    san(&b_bar_scale_bhnl);
189    san(&forward_decay_to_chunk_end_bhnl);
190    san(&b_scaled_bnhlr);
191
192    // K4 ───────────────────────────────────────────────────────────────────
193    let (chunk_input_state_bnhpr, _final_state_bhpr): (Tensor<B, 5>, Tensor<B, 4>) =
194        serial::k4_ssd_state_passing(
195            intra_chunk_state_bnhpr.clone(),
196            da_chunk_end_bhn.clone(),
197            initial_state_bhpr,
198        );
199    san(&chunk_input_state_bnhpr);
200    san(&_final_state_bhpr);
201
202    // ═══════════════════════════════════════════════════════════════════════
203    // K5 BACKWARD
204    // ═══════════════════════════════════════════════════════════════════════
205    // Expand CB for all heads
206    let cb_bnhll = cb_bngll
207        .clone()
208        .unsqueeze_dim::<6>(3) // cb_bng1ll
209        .expand([
210            batch,
211            nchunks,
212            ngroups,
213            heads_per_group,
214            chunk_len,
215            chunk_len,
216        ]) // cb_bngHll
217        .reshape([batch, nchunks, nheads, chunk_len, chunk_len]);
218
219    // Reshape inputs to [B,N,H,L,...] convention used inside K5
220    let da_cumsum_bnhl: Tensor<B, 4> = da_cumsum_bhnl.permute([0, 2, 1, 3]);
221    let dt_bnhl: Tensor<B, 4> = dt_discretized_bhnl.clone().permute([0, 2, 1, 3]);
222    let x_bnhlp: Tensor<B, 5> = x_bnlhp.clone().permute([0, 1, 3, 2, 4]);
223    let d_y_bnhlp: Tensor<B, 5> = d_y_bnlhp.clone().permute([0, 1, 3, 2, 4]);
224
225    // GQA-expand C: [B,N,L,G,R] → [B,N,H,L,R]
226    let c_bnhlr = c_bnlgr
227        .clone()
228        .unsqueeze_dim::<6>(4) // c_bnlg1r
229        .expand([
230            batch,
231            nchunks,
232            chunk_len,
233            ngroups,
234            heads_per_group,
235            state_rank,
236        ]) // c_bnlgHr
237        .reshape([batch, nchunks, chunk_len, nheads, state_rank]) // c_bnlhr
238        .permute([0, 1, 3, 2, 4]);
239
240    // ── SKIP backward ──────────────────────────────────────────────────────
241    // - | 36/36: add: (y_partial_bnlhp, skip_bnlhp) -> (y_bnlhp [out])
242    // - | (d_skip_bnlhp = d_y_bnlhp)
243    let d_skip_bnlhp = d_y_bnlhp.clone();
244    //
245    //
246    // For d_h:
247    // - - | 33: mul: (d_bnlhp, x_bnlhp[*]) -> (skip_bnlhp)
248    // - - | (d_d_bnlhp = d_skip_bnlhp * x_bnlhp)
249    // - - | 32: expand: (d_111h1) -> (d_bnlhp)
250    // - - | 31: unsqueeze-dims: (d_h [*]) -> (d_111h1)
251    //
252    // - - | d_d[h] = Σ_{b,n,l,p} dy * x   — use permute+reshape to avoid chained sum_dim
253    let d_d_h = {
254        // [B,N,L,H,P] → permute to [H,B,N,L,P] → reshape [H, rest] → sum → [H]
255        d_skip_bnlhp.clone()
256            .permute([3, 0, 1, 2, 4]) // d_y_hbnlp
257            .reshape([nheads, batch * nchunks * chunk_len * per_head_dim]) // d_y_hBNLP
258            * x_bnlhp
259                .clone()
260                .permute([3, 0, 1, 2, 4]) // x_hbnlp
261                .reshape([nheads, batch * nchunks * chunk_len * per_head_dim]) // x_hBNLP
262    }
263    .sum_dim(1) // d_d_h1
264    .reshape([nheads]);
265    san(&d_d_h);
266    //
267    // For d_x:
268    // - - | 33: mul: (d_bnlhp, x_bnlhp[*]) -> (skip_bnlhp)
269    // - - | (d_x_skip_bnlhp = d_skip_bnlhp * d_bnlhp)
270    let d_x_skip_bnlhp = d_skip_bnlhp
271        * d_h
272            .clone()
273            .unsqueeze_dims::<5>(&[0, 1, 2, 4]) // d_111h1
274            // d_bnlhp
275            .expand([batch, nchunks, chunk_len, nheads, per_head_dim]);
276    san(&d_x_skip_bnlhp);
277
278    // ── BLUE backward ──────────────────────────────────────────────────────
279    // - | 36/36: add: (y_partial_bnlhp, skip_bnlhp) -> (y_bnlhp [out])
280    // - | (d_y_partial_bnlhp = d_y_bnlhp)
281    let d_y_partial_bnhlp = d_y_bnhlp.clone();
282    //
283    // - | 35: permute: (y_partial_bnhlp) -> (y_partial_bnlhp)
284    // - | 34: add: (blue_scaled_bnhlp, orange_bnhlp) -> (y_partial_bnhlp)
285    // - | (d_blue_scaled_bnhlp = d_y_partial_bnhlp)
286    let d_blue_scaled_bnhlp = d_y_partial_bnhlp;
287    // - | 16: mul: (blue_bnhlp, exp_da_cumsum_bnhlp) -> (blue_scaled_bnhlp)
288    // - | (d_blue_bnhlp = d_blue_scaled_bnhlp * exp_da_cumsum_bnhlp)
289    //
290    // - | blue[b,n,h,l,p] = exp(da[b,n,h,l]) * Σ_r C[b,n,h,l,r] * state[b,n,h,p,r]
291    let exp_da_cumsum_bnhl: Tensor<B, 4> = da_cumsum_bnhl.clone().exp();
292    san(&exp_da_cumsum_bnhl);
293    let exp_da_cumsum_bnhlp = exp_da_cumsum_bnhl.clone().unsqueeze_dim::<5>(4).expand([
294        batch,
295        nchunks,
296        nheads,
297        chunk_len,
298        per_head_dim,
299    ]);
300    let d_blue_bnhlp: Tensor<B, 5> = d_blue_scaled_bnhlp.clone() * exp_da_cumsum_bnhlp.clone();
301    san(&d_blue_bnhlp);
302    //
303    // For d_chunk_input_state_bnhpr:
304    // - | 15: matmul: (c_bnhlr, chunk_input_state_bnhrp) -> (blue_bnhlp)
305    // - - | (d_chunk_input_state_bnhrp = c_bnhlr^T @ d_blue_bnhlp)
306    // - - | 14: permute: (chunk_input_state_bnhpr [!]) -> (chunk_input_state_bnhrp)
307    //
308    // - - | d_state[b,n,h,p,r] = Σ_l (scaled_dy[b,n,h,l,p] * C[b,n,h,l,r])
309    // - - |  = C^T[R,L] @ scaled_dy[L,P]  for fixed (b,n,h)
310    // - - |  [B,N,H,R,L] @ [B,N,H,L,P] → [B,N,H,R,P] → permute → [B,N,H,P,R]
311    let d_chunk_input_state_bnhpr = c_bnhlr
312        .clone()
313        .permute([0, 1, 2, 4, 3]) // c_bnhrl
314        .matmul(d_blue_bnhlp.clone()) // d_chunk_input_state_bnhrp
315        .permute([0, 1, 2, 4, 3]);
316    san(&d_chunk_input_state_bnhpr);
317    //
318    // For d_c from BLUE:
319    // - | 15: matmul: (c_bnhlr, chunk_input_state_bnhrp) -> (blue_bnhlp)
320    // - - | (d_c_bnhlr = d_blue_bnhlp @ chunk_input_state_bnhrp^T)
321    // - - | 7: permute: (c_bnlhr) -> (c_bnhlr)
322    // - - | 6: reshape: (c_bnlgHr) -> (c_bnlhr)
323    // - - | 5: expand: (c_bnlg1r) -> (c_bnlgHr)
324    // - - | 4: unsqueeze: (c_bnlgr [*]) -> (c_bnlg1r)
325    //
326    // - - | d_C[l,r] = Σ_p scaled_dy[l,p] * state[p,r]
327    // - - |  [B,N,H,L,P] @ [B,N,H,P,R] → [B,N,H,L,R]
328    let d_c_blue_bnhlr = d_blue_bnhlp.clone().matmul(chunk_input_state_bnhpr.clone());
329    san(&d_c_blue_bnhlr);
330    // - - | GQA reduce: [B,N,H,L,R] → [B,N,L,G,R]
331    let d_c_blue_bnlgr = d_c_blue_bnhlr
332        .reshape([
333            batch,
334            nchunks,
335            ngroups,
336            heads_per_group,
337            chunk_len,
338            state_rank,
339        ]) // d_c_blue_bngHlr
340        .sum_dim(3) // d_c_blue_bng1lr
341        .squeeze_dim::<5>(3) // d_c_blue_bnglr
342        .permute([0, 1, 3, 2, 4]);
343    san(&d_c_blue_bnlgr);
344    //
345    // For d_da_cumsum from BLUE:
346    // - | 16: mul: (blue_bnhlp, exp_da_cumsum_bnhlp) -> (blue_scaled_bnhlp)
347    // - | (d_exp_da_cumsum_bnhlp = d_blue_scaled_bnhlp * blue_bnhlp)
348    let blue_bnhlp = c_bnhlr
349        .clone()
350        .matmul(chunk_input_state_bnhpr.clone().permute([0, 1, 2, 4, 3])); // replay forward step 15
351    san(&blue_bnhlp);
352    let d_exp_da_cumsum_bnhlp = d_blue_scaled_bnhlp.clone() * blue_bnhlp;
353    san(&d_exp_da_cumsum_bnhlp);
354    //
355    // - | blue_no_scale = C @ state^T  [L,P]
356    // - - | 13: expand: (exp_da_cumsum_bnhl1) -> (exp_da_cumsum_bnhlp)
357    // - - | 12: unsqueeze: (exp_da_cumsum_bnhl) -> (exp_da_cumsum_bnhl1)
358    // - - | 11: exp: (da_cumsum_bnhl) -> (exp_da_cumsum_bnhl)
359    // - - | (d_da_cumsum_bnhl = d_exp_da_cumsum_bnhlp * exp(da_cumsum_bnhl))
360    // - - | 1/36: permute: (da_cumsum_bhnl [*]) -> (da_cumsum_bnhl)
361    //
362    // - - | d_da[l] = Σ_p dy[l,p] * exp_da[l] * blue_no_scale[l,p]
363    let d_da_blue_bnhl = (d_exp_da_cumsum_bnhlp * exp_da_cumsum_bnhlp)
364        .sum_dim(4) // d_da_blue_bnhl1
365        .squeeze_dim::<4>(4);
366    san(&d_da_blue_bnhl);
367    let d_da_blue_bhnl = d_da_blue_bnhl.permute([0, 2, 1, 3]);
368
369    // ── ORANGE backward ─────────────────────────────────────────────────────
370    //  y_orange[l,p] = Σ_{s≤l} CB[l,s] * exp(da[l]-da[s]) * dt[s] * x[s,p]
371    // Precompute weight matrix CB_w [B,N,H,L_tgt,L_src]
372    // replay forward steps 17-29
373    let da_cumsum_target_bnhll = da_cumsum_bnhl
374        .clone()
375        .unsqueeze_dim::<5>(4) // da_cumsum_bnhl1 // forward step 17
376        .expand([batch, nchunks, nheads, chunk_len, chunk_len]); // forward step 18
377    let da_cumsum_source_bnhll = da_cumsum_bnhl
378        .clone()
379        .unsqueeze_dim::<5>(3) // da_cumsum_bnh1l // forward step 19
380        .expand([batch, nchunks, nheads, chunk_len, chunk_len]); // forward step 20
381    let da_cumsum_diff_bnhll = da_cumsum_target_bnhll - da_cumsum_source_bnhll; // forward step 21
382    san(&da_cumsum_diff_bnhll);
383    let causal_mask_bnhll =
384        Tensor::tril_mask([batch, nchunks, nheads, chunk_len, chunk_len], 0, &device); // forward step 21.1
385    // forward step 21.2
386    // Causal mask and exp stabilizer (-inf above the main diagonal, 0 elsewhere).
387    let da_cumsum_diff_masked_bnhll =
388        da_cumsum_diff_bnhll.mask_fill(causal_mask_bnhll.clone(), f32::NEG_INFINITY);
389    let da_cumsum_diff_exp_bnhll = (da_cumsum_diff_masked_bnhll).exp(); // forward steps 22
390    san(&da_cumsum_diff_exp_bnhll);
391    let dt_source_bnhll = dt_bnhl
392        .clone()
393        .unsqueeze_dim::<5>(3) // dt_bnh1l // forward step 23
394        .expand([batch, nchunks, nheads, chunk_len, chunk_len]); // forward step 24
395    // // Causal mask (0 above the main diagonal, 1 elsewhere).
396    // let causal_mask_bnhll =
397    //     Tensor::ones([batch, nchunks, nheads, chunk_len, chunk_len], &device).tril(0); // forward steps 25-26
398    // CB_w[l,s] = CB[l,s] * decay[l,s] * dt[s] * mask[l,s]
399    let orange_lhs_partial1_bnhll: Tensor<B, 5> = // forward step 27
400        cb_bnhll.clone() * da_cumsum_diff_exp_bnhll.clone();
401    san(&orange_lhs_partial1_bnhll);
402    let orange_lhs_partial2_bnhll: Tensor<B, 5> = // forward step 28
403        orange_lhs_partial1_bnhll.clone() * dt_source_bnhll.clone();
404    san(&orange_lhs_partial2_bnhll);
405    // let orange_lhs_partial3_bnhll: Tensor<B, 5> = // forward step 29
406    //     orange_lhs_partial2_bnhll.clone() * causal_mask_bnhll.clone();
407    //
408    // Backwads:
409    // - | 36/36: add: (y_partial_bnlhp, skip_bnlhp) -> (y_bnlhp [out])
410    // - | (d_y_partial_bnlhp = d_y_bnlhp)
411    let d_y_partial_bnhlp = d_y_bnhlp.clone();
