16impl<B: Backend + Mamba2BackendExt, C: CheckpointStrategy> Mamba2BackendExt for Autodiff<B, C> {
17    /// Memory-efficient combined forward+backward.
18    ///
19    /// The two output tensors are concatenated into a single 1-dimensional tracked tensor
20    /// so that one `Backward<B, 7>` node covers both outputs.  The caller
21    /// receives split+reshaped slices of that combined tensor; burn's autodiff
22    /// accumulates their upstream gradients back into a single gradient vector
23    /// before firing this backward.
24    fn ssd_serial_recalculated(
25        x_bnlhp: FloatTensor<Self>,
26        dt_discretized_bhnl: FloatTensor<Self>,
27        b_bnlhr: FloatTensor<Self>,
28        c_bnlhr: FloatTensor<Self>,
29        d_h: FloatTensor<Self>,
30        initial_state_bhpr: FloatTensor<Self>,
31        a_decay_h: FloatTensor<Self>,
32    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
33        // ── Backward struct ──────────────────────────────────────────────────
34        #[derive(Debug)]
35        struct CombinedKernelsBackward;
36
37        #[derive(Clone, Debug)]
38        struct State<B: Backend> {
39            x_bnlhp: FloatTensor<B>,
40            dt_discretized_bhnl: FloatTensor<B>,
41            b_bnlhr: FloatTensor<B>,
42            c_bnlhr: FloatTensor<B>,
43            d_h: FloatTensor<B>,
44            initial_state_bhpr: FloatTensor<B>,
45            a_decay_h: FloatTensor<B>,
46            // flat byte-sizes for splitting the combined gradient vector
47            flat_len_y_BNLHP: usize,
48            flat_len_final_state_BHPR: usize,
49            // shapes needed to reconstruct tensors in the right ranks
50            shape_x_bnlhp: [usize; 5],
51            shape_dt_discretized_bhnl: [usize; 4],
52            shape_b_bnlhr: [usize; 5],
53            shape_c_bnlhr: [usize; 5],
54            shape_d_h: [usize; 1],
55            shape_initial_state_bhpr: [usize; 4],
56            shape_a_decay_h: [usize; 1],
57            shape_y_bnlhp: [usize; 5],          // (output 1)
58            shape_final_state_bhpr: [usize; 4], // (output 2)
59        }
60
61        /// State carried across the forward→backward boundary.
62        ///
63        /// Only the 7 original inputs are saved; all intermediates (cb, intra
64        /// state, chunk_input_state) are recomputed during `backward`.
65        #[allow(clippy::type_complexity)]
66        impl<B: Backend + Mamba2BackendExt> Backward<B, 7> for CombinedKernelsBackward {
67            type State = State<B>;
68
69            fn backward(
70                self,
71                ops: Ops<Self::State, 7>,
72                grads: &mut Gradients,
73                _checkpointer: &mut Checkpointer,
74            ) {
75                let [
76                    node_x_bnlhp,
77                    node_dt_discretized_bhnl,
78                    node_b_bnlhr,
79                    node_c_bnlhr,
80                    node_d_h,
81                    node_initial_state_bhpr,
82                    node_a_decay_h,
83                ] = ops.parents;
84
85                // Retrieve the gradient of the combined 1-dimensional output.
86                let d_combined: Tensor<B, 1> =
87                    Tensor::from_primitive(TensorPrimitive::Float(grads.consume::<B>(&ops.node)));
88
89                let State {
90                    x_bnlhp,
91                    dt_discretized_bhnl,
92                    b_bnlhr,
93                    c_bnlhr,
94                    d_h,
95                    initial_state_bhpr,
96                    a_decay_h,
97                    //
98                    flat_len_y_BNLHP,
99                    flat_len_final_state_BHPR,
100                    //
101                    shape_x_bnlhp,
102                    shape_dt_discretized_bhnl,
103                    shape_b_bnlhr,
104                    shape_c_bnlhr,
105                    shape_d_h,
106                    shape_initial_state_bhpr,
107                    shape_a_decay_h,
108                    //
109                    shape_y_bnlhp,
110                    shape_final_state_bhpr,
111                } = ops.state;
112
113                // ── Reconstruct saved tensors ──────────────────────────────
114                use crate::utils::primitive::mk;
115
116                let x_bnlhp = mk::<_, 5>(x_bnlhp).reshape(shape_x_bnlhp);
117                let dt_discretized_bhnl =
118                    mk::<_, 4>(dt_discretized_bhnl).reshape(shape_dt_discretized_bhnl);
119                let b_bnlhr = mk::<_, 5>(b_bnlhr).reshape(shape_b_bnlhr);
120                let c_bnlhr = mk::<_, 5>(c_bnlhr).reshape(shape_c_bnlhr);
121                let d_h = mk::<_, 1>(d_h).reshape(shape_d_h);
122                let initial_state_bhpr =
123                    mk::<_, 4>(initial_state_bhpr).reshape(shape_initial_state_bhpr);
124                let a_decay_h = mk::<_, 1>(a_decay_h).reshape(shape_a_decay_h);
125
126                // ── Split incoming combined gradient ───────────────────────
127                let (d_y_bnlhp, d_final_state_bhpr) = crate::utils::combined_grad::unflatten_pair(
128                    d_combined,
129                    flat_len_y_BNLHP,
130                    flat_len_final_state_BHPR,
131                    shape_y_bnlhp,
132                    shape_final_state_bhpr,
133                );
134
135                // ── Core gradient computation ──────────────────────────────
136                let CombinedGrads {
137                    d_x_bnlhp,
138                    d_dt_discretized_bhnl,
139                    d_b_bnlhr,
140                    d_c_bnlhr,
141                    d_d_h,
142                    d_initial_state_bhpr,
143                    d_a_decay_h,
144                    ..
145                } = combined_backward::combined_backward(
146                    d_y_bnlhp,
147                    d_final_state_bhpr,
148                    //
149                    x_bnlhp,
150                    dt_discretized_bhnl,
151                    b_bnlhr,
152                    c_bnlhr,
153                    d_h,
154                    initial_state_bhpr,
155                    a_decay_h,
156                );
157
158                // ── Register gradients ─────────────────────────────────────
159                // TODO: request Node to be re-exported.
160                //
161                // use burn::cubecl::stub::Arc;
162                // use burn::backend::autodiff::Node;
163                // let reg = |node: Option<Arc<_>>, grad: Tensor<B, _>| {
164                //     if let Some(n) = node {
165                //         grads.register::<B>(n.id, grad.into_primitive().tensor());
166                //     }
167                // };
168                // let () = reg(node_x_bnlhp, d_x_bnlhp);
169                // let () = reg(node_dt_discretized_bhnl, d_dt_discretized_bhnl);
170                // let () = reg(node_b_bnlhr, d_b_bnlhr);
171                // let () = reg(node_c_bnlhr, d_c_bnlhr);
172                // let () = reg(node_d_h, d_d_h);
173                // let () = reg(node_initial_state_bhpr, d_initial_state_bhpr);
174                // let () = reg(node_da_cumsum_bhnl, d_da_cumsum_bhnl);
175
176                if let Some(n) = node_x_bnlhp {
177                    grads.register::<B>(n.id, d_x_bnlhp.into_primitive().tensor());
178                }
179                if let Some(n) = node_dt_discretized_bhnl {
180                    grads.register::<B>(n.id, d_dt_discretized_bhnl.into_primitive().tensor());
181                }
182                if let Some(n) = node_b_bnlhr {
183                    grads.register::<B>(n.id, d_b_bnlhr.into_primitive().tensor());
184                }
185                if let Some(n) = node_c_bnlhr {
186                    grads.register::<B>(n.id, d_c_bnlhr.into_primitive().tensor());
187                }
188                if let Some(n) = node_d_h {
189                    grads.register::<B>(n.id, d_d_h.into_primitive().tensor());
190                }
191                if let Some(n) = node_initial_state_bhpr {
192                    grads.register::<B>(n.id, d_initial_state_bhpr.into_primitive().tensor());
193                }
194                if let Some(n) = node_a_decay_h {
195                    grads.register::<B>(n.id, d_a_decay_h.into_primitive().tensor());
196                }
197            }
198        } // end impl Backward
199
200        // ── Shape extraction helpers ───────────────────────────────────────
201        // Accessed via the AutodiffTensor wrappers (which own both .node
202        // and .primitive).
203        use burn::tensor::TensorMetadata;
204        let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.primitive.shape().dims();
205        let [_, _, _, _nheads_b, state_rank] = b_bnlhr.primitive.shape().dims();
206
207        let flat_len_y_BNLHP = batch * nchunks * chunk_len * nheads * per_head_dim;
208        let flat_len_final_state_BHPR = batch * nheads * per_head_dim * state_rank;
209
210        let shape_x_bnlhp: [usize; 5] = [batch, nchunks, chunk_len, nheads, per_head_dim];
211        let shape_dt_discretized_bhnl: [usize; 4] = [batch, nheads, nchunks, chunk_len];
212        let shape_b_bnlhr: [usize; 5] = [batch, nchunks, chunk_len, nheads, state_rank];
213        let shape_c_bnlhr: [usize; 5] = [batch, nchunks, chunk_len, nheads, state_rank];
214        let shape_d_h: [usize; 1] = [nheads];
215        let shape_initial_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
216        let shape_a_decay_h: [usize; 1] = [nheads];
217        let shape_y_bnlhp: [usize; 5] = [batch, nchunks, chunk_len, nheads, per_head_dim];
218        let shape_final_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
219
220        // ── Register backward / run forward ───────────────────────────────
221        match CombinedKernelsBackward
222            .prepare::<C>([
223                x_bnlhp.node.clone(),
224                dt_discretized_bhnl.node.clone(),
225                b_bnlhr.node.clone(),
226                c_bnlhr.node.clone(),
227                d_h.node.clone(),
228                initial_state_bhpr.node.clone(),
229                a_decay_h.node.clone(),
230            ])
231            .compute_bound()
232            .stateful() // requires compute_bound
233        {
234            OpsKind::Tracked(prep) => {
235                // Run the inner (non-autodiff) forward pass.
236                let (prim_y_bnlhp, prim_final_state_bhpr) = B::ssd_serial_recalculated(
237                    x_bnlhp.primitive.clone(),
238                    dt_discretized_bhnl.primitive.clone(),
239                    b_bnlhr.primitive.clone(),
240                    c_bnlhr.primitive.clone(),
241                    d_h.primitive.clone(),
242                    initial_state_bhpr.primitive.clone(),
243                    a_decay_h.primitive.clone(),
244                );
245
246                // prep.finish takes a single tensor, so pack both outputs into a
247                // single 1-D tensor; one Backward node then covers both.
248                let (combined, _, _) = crate::utils::combined_grad::flatten_pair(
249                    crate::utils::primitive::mk::<B, 5>(prim_y_bnlhp),
250                    crate::utils::primitive::mk::<B, 4>(prim_final_state_bhpr),
251                );
252
253                let state = State {
254                    x_bnlhp: x_bnlhp.primitive.clone(),
255                    dt_discretized_bhnl: dt_discretized_bhnl.primitive.clone(),
256                    b_bnlhr: b_bnlhr.primitive.clone(),
257                    c_bnlhr: c_bnlhr.primitive.clone(),
258                    d_h: d_h.primitive.clone(),
259                    initial_state_bhpr: initial_state_bhpr.primitive.clone(),
260                    a_decay_h: a_decay_h.primitive.clone(),
261                    //
262                    flat_len_y_BNLHP,
263                    flat_len_final_state_BHPR,
264                    //
265                    shape_x_bnlhp, shape_dt_discretized_bhnl, shape_b_bnlhr, shape_c_bnlhr, shape_d_h, shape_initial_state_bhpr, shape_a_decay_h,
266                    shape_y_bnlhp, shape_final_state_bhpr,
267                };
268                let tracked_combined: FloatTensor<Autodiff<B, C>> =
269                    prep.finish(state, combined.into_primitive().tensor());
270
271                // Split the tracked combined tensor back into the two outputs.
272                // The narrow/reshape ops are thin autodiff pass-throughs whose
273                // backwards accumulate into the combined gradient vector that
274                // `backward` above consumes.
275                let (tracked_y_bnlhp, tracked_final_state_bhpr) = crate::utils::combined_grad::unflatten_pair::<Autodiff<B, C>, 5, 4>(
276                    crate::utils::primitive::mk(tracked_combined),
277                    flat_len_y_BNLHP,
278                    flat_len_final_state_BHPR,
279                    shape_y_bnlhp,
280                    shape_final_state_bhpr,
281                );
282
283                (
284                    tracked_y_bnlhp.into_primitive().tensor(),
285                    tracked_final_state_bhpr.into_primitive().tensor(),
286                )
287            }
288
289            OpsKind::UnTracked(prep) => {
290                // No gradient tracking — just run the bare forward.
291                let (prim_y_bnlhp, prim_final_state_bhpr) = B::ssd_serial_recalculated(
292                    x_bnlhp.primitive,
293                    dt_discretized_bhnl.primitive,
294                    b_bnlhr.primitive,
295                    c_bnlhr.primitive,
296                    d_h.primitive,
297                    initial_state_bhpr.primitive,
298                    a_decay_h.primitive,
299                );
300
301                let (combined, _, _) = crate::utils::combined_grad::flatten_pair(
302                    crate::utils::primitive::mk::<B, 5>(prim_y_bnlhp),
303                    crate::utils::primitive::mk::<B, 4>(prim_final_state_bhpr),
304                );
305
306                let tracked_combined: FloatTensor<Autodiff<B, C>> =
307                    prep.finish(combined.into_primitive().tensor());
308
309                let (tracked_y_bnlhp, tracked_final_state_bhpr) = crate::utils::combined_grad::unflatten_pair::<Autodiff<B, C>, 5, 4>(
310                    crate::utils::primitive::mk(tracked_combined),
311                    flat_len_y_BNLHP,
312                    flat_len_final_state_BHPR,
313                    shape_y_bnlhp,
314                    shape_final_state_bhpr,
315                );
316
317                (
318                    tracked_y_bnlhp.into_primitive().tensor(),
319                    tracked_final_state_bhpr.into_primitive().tensor(),
320                )
321            }
322        } // end match
323    } // end fn ssd_serial_recalculated on Autodiff<B, C>
324} // end impl Mamba2BackendExt for Autodiff<B, C>
