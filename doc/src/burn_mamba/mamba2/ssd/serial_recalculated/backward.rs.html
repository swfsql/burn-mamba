16impl<B: Backend + Mamba2BackendExt, C: CheckpointStrategy> Mamba2BackendExt for Autodiff<B, C> {
17    /// Memory-efficient combined forward+backward.
18    ///
19    /// The two output tensors are concatenated into a single 1-D tracked tensor
20    /// so that one `Backward<B, 7>` node covers both outputs.  The caller
21    /// receives split+reshaped slices of that combined tensor; burn's autodiff
22    /// accumulates their upstream gradients back into a single gradient vector
23    /// before firing this backward.
24    fn ssd_serial_recalculated(
25        // AI init interface:
26        //
27        // da_cumsum_bhnl: FloatTensor<Self>,
28        // dt_discretized_bhnl: FloatTensor<Self>,
29        // x_bnlhp: FloatTensor<Self>,
30        // b_bnlgr: FloatTensor<Self>,
31        // c_bnlgr: FloatTensor<Self>,
32        // d_h: FloatTensor<Self>,
33        // initial_state_bhpr: FloatTensor<Self>,
34
35        // actual required interface:
36        //
37        x_bnlhp: FloatTensor<Self>,
38        dt_discretized_bhnl: FloatTensor<Self>,
39        b_bnlgr: FloatTensor<Self>,
40        c_bnlgr: FloatTensor<Self>,
41        d_h: FloatTensor<Self>,
42        initial_state_bhpr: FloatTensor<Self>,
43        da_cumsum_bhnl: FloatTensor<Self>,
44        // current inner interface:
45        //
46        // d_da_cumsum_bhnl: Tensor<B, 4>,
47        // d_dt_discretized_bhnl: Tensor<B, 4>,
48        // d_x_bnlhp: Tensor<B, 5>,
49        // d_b_bnlgr: Tensor<B, 5>,
50        // d_c_bnlgr: Tensor<B, 5>,
51        // d_d_h: Tensor<B, 1>,
52        // d_initial_state_bhpr: Tensor<B, 4>,
53    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
54        // ── Backward struct ──────────────────────────────────────────────────
55        #[derive(Debug)]
56        struct K2K3K4K5CombinedBackward;
57
58        #[derive(Clone, Debug)]
59        struct State<B: Backend> {
60            x_bnlhp: FloatTensor<B>,
61            dt_discretized_bhnl: FloatTensor<B>,
62            b_bnlgr: FloatTensor<B>,
63            c_bnlgr: FloatTensor<B>,
64            d_h: FloatTensor<B>,
65            initial_state_bhpr: FloatTensor<B>,
66            da_cumsum_bhnl: FloatTensor<B>, // (from K1)
67            // flat byte-sizes for splitting the combined gradient vector
68            flat_len_y_BNLHP: usize,
69            flat_len_final_state_BHPR: usize,
70            // shapes needed to reconstruct tensors in the right ranks
71            shape_x_bnlhp: [usize; 5],
72            shape_dt_discretized_bhnl: [usize; 4],
73            shape_b_bnlgr: [usize; 5],
74            shape_c_bnlgr: [usize; 5],
75            shape_d_h: [usize; 1],
76            shape_initial_state_bhpr: [usize; 4],
77            shape_da_cumsum_bhnl: [usize; 4],   // (from K1)
78            shape_y_bnlhp: [usize; 5],          // (output 1)
79            shape_final_state_bhpr: [usize; 4], // (output 2)
80        }
81
82        /// State carried across the forward→backward boundary.
83        ///
84        /// Only the 7 original inputs are saved; all intermediates (cb, intra
85        /// state, chunk_input_state) are recomputed during `backward`.
86        #[allow(clippy::type_complexity)]
87        impl<B: Backend + Mamba2BackendExt> Backward<B, 7> for K2K3K4K5CombinedBackward {
88            type State = State<B>;
89
90            fn backward(
91                self,
92                ops: Ops<Self::State, 7>,
93                grads: &mut Gradients,
94                _checkpointer: &mut Checkpointer,
95            ) {
96                let [
97                    node_x_bnlhp,
98                    node_dt_discretized_bhnl,
99                    node_b_bnlgr,
100                    node_c_bnlgr,
101                    node_d_h,
102                    node_initial_state_bhpr,
103                    node_da_cumsum_bhnl,
104                ] = ops.parents;
105
106                // Retrieve the gradient of the combined 1-D output.
107                let d_combined: Tensor<B, 1> =
108                    Tensor::from_primitive(TensorPrimitive::Float(grads.consume::<B>(&ops.node)));
109
110                let State {
111                    x_bnlhp,
112                    dt_discretized_bhnl,
113                    b_bnlgr,
114                    c_bnlgr,
115                    d_h,
116                    initial_state_bhpr,
117                    da_cumsum_bhnl,
118                    //
119                    flat_len_y_BNLHP,
120                    flat_len_final_state_BHPR,
121                    //
122                    shape_x_bnlhp,
123                    shape_dt_discretized_bhnl,
124                    shape_b_bnlgr,
125                    shape_c_bnlgr,
126                    shape_d_h,
127                    shape_initial_state_bhpr,
128                    shape_da_cumsum_bhnl,
129                    //
130                    shape_y_bnlhp,
131                    shape_final_state_bhpr,
132                } = ops.state;
133
134                // ── Reconstruct saved tensors ──────────────────────────────
135                use super::serial_recalculated::mk;
136
137                let x_bnlhp = mk::<_, 5>(x_bnlhp).reshape(shape_x_bnlhp);
138                let dt_discretized_bhnl =
139                    mk::<_, 4>(dt_discretized_bhnl).reshape(shape_dt_discretized_bhnl);
140                let b_bnlgr = mk::<_, 5>(b_bnlgr).reshape(shape_b_bnlgr);
141                let c_bnlgr = mk::<_, 5>(c_bnlgr).reshape(shape_c_bnlgr);
142                let d_h = mk::<_, 1>(d_h).reshape(shape_d_h);
143                let initial_state_bhpr =
144                    mk::<_, 4>(initial_state_bhpr).reshape(shape_initial_state_bhpr);
145                let da_cumsum_bhnl = mk::<_, 4>(da_cumsum_bhnl).reshape(shape_da_cumsum_bhnl);
146
147                // ── Split incoming combined gradient ───────────────────────
148                // d_combined : [y_flat_len_BNLHP + fs_flat_len_BHPR]
149                let flat_d_y_BNLHP = d_combined.clone().narrow(0, 0, flat_len_y_BNLHP);
150                let flat_d_final_state_BHPR =
151                    d_combined.narrow(0, flat_len_y_BNLHP, flat_len_final_state_BHPR);
152
153                let d_y_bnlhp: Tensor<B, 5> = flat_d_y_BNLHP.reshape(shape_y_bnlhp);
154                let d_final_state_bhpr: Tensor<B, 4> =
155                    flat_d_final_state_BHPR.reshape(shape_final_state_bhpr);
156
157                // ── Core gradient computation ──────────────────────────────
158                let CombinedGrads {
159                    d_x_bnlhp,
160                    d_dt_discretized_bhnl,
161                    d_b_bnlgr,
162                    d_c_bnlgr,
163                    d_d_h,
164                    d_initial_state_bhpr,
165                    d_da_cumsum_bhnl,
166                } = combined_backward::combined_backward(
167                    d_y_bnlhp,
168                    d_final_state_bhpr,
169                    //
170                    x_bnlhp,
171                    dt_discretized_bhnl,
172                    b_bnlgr,
173                    c_bnlgr,
174                    d_h,
175                    initial_state_bhpr,
176                    da_cumsum_bhnl,
177                );
178
179                // ── Register gradients ─────────────────────────────────────
180                // TODO: request Node to be re-exported.
181                //
182                // use burn::cubecl::stub::Arc;
183                // use burn::backend::autodiff::Node;
184                // let reg = |node: Option<Arc<_>>, grad: Tensor<B, _>| {
185                //     if let Some(n) = node {
186                //         grads.register::<B>(n.id, grad.into_primitive().tensor());
187                //     }
188                // };
189                // let () = reg(node_x_bnlhp, d_x_bnlhp);
190                // let () = reg(node_dt_discretized_bhnl, d_dt_discretized_bhnl);
191                // let () = reg(node_b_bnlgr, d_b_bnlgr);
192                // let () = reg(node_c_bnlgr, d_c_bnlgr);
193                // let () = reg(node_d_h, d_d_h);
194                // let () = reg(node_initial_state_bhpr, d_initial_state_bhpr);
195                // let () = reg(node_da_cumsum_bhnl, d_da_cumsum_bhnl);
196
197                if let Some(n) = node_x_bnlhp {
198                    grads.register::<B>(n.id, d_x_bnlhp.into_primitive().tensor());
199                }
200                if let Some(n) = node_dt_discretized_bhnl {
201                    grads.register::<B>(n.id, d_dt_discretized_bhnl.into_primitive().tensor());
202                }
203                if let Some(n) = node_b_bnlgr {
204                    grads.register::<B>(n.id, d_b_bnlgr.into_primitive().tensor());
205                }
206                if let Some(n) = node_c_bnlgr {
207                    grads.register::<B>(n.id, d_c_bnlgr.into_primitive().tensor());
208                }
209                if let Some(n) = node_d_h {
210                    grads.register::<B>(n.id, d_d_h.into_primitive().tensor());
211                }
212                if let Some(n) = node_initial_state_bhpr {
213                    grads.register::<B>(n.id, d_initial_state_bhpr.into_primitive().tensor());
214                }
215                if let Some(n) = node_da_cumsum_bhnl {
216                    grads.register::<B>(n.id, d_da_cumsum_bhnl.into_primitive().tensor());
217                }
218            }
219        } // end impl Backward
220
221        // ── Shape extraction helpers ───────────────────────────────────────
222        // Accessed via the AutodiffTensor wrappers (which own both .node
223        // and .primitive).
224        use burn::tensor::TensorMetadata;
225        let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.primitive.shape().dims();
226        let [_, _, _, ngroups, state_rank] = b_bnlgr.primitive.shape().dims();
227
228        let flat_len_y_BNLHP = batch * nchunks * chunk_len * nheads * per_head_dim;
229        let flat_len_final_state_BHPR = batch * nheads * per_head_dim * state_rank;
230
231        let shape_x_bnlhp: [usize; 5] = [batch, nchunks, chunk_len, nheads, per_head_dim];
232        let shape_dt_discretized_bhnl: [usize; 4] = [batch, nheads, nchunks, chunk_len];
233        let shape_b_bnlgr: [usize; 5] = [batch, nchunks, chunk_len, ngroups, state_rank];
234        let shape_c_bnlgr: [usize; 5] = [batch, nchunks, chunk_len, ngroups, state_rank];
235        let shape_d_h: [usize; 1] = [nheads];
236        let shape_initial_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
237        let shape_da_cumsum_bhnl: [usize; 4] = [batch, nheads, nchunks, chunk_len];
238        let shape_y_bnlhp: [usize; 5] = [batch, nchunks, chunk_len, nheads, per_head_dim];
239        let shape_final_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
240
241        // ── Register backward / run forward ───────────────────────────────
242        match K2K3K4K5CombinedBackward
243            .prepare::<C>([
244                x_bnlhp.node.clone(),
245                dt_discretized_bhnl.node.clone(),
246                b_bnlgr.node.clone(),
247                c_bnlgr.node.clone(),
248                d_h.node.clone(),
249                initial_state_bhpr.node.clone(),
250                da_cumsum_bhnl.node.clone(),
251            ])
252            .compute_bound()
253            .stateful() // requires compute_bound
254        {
255            OpsKind::Tracked(prep) => {
256                // Run the inner (non-autodiff) forward pass.
257                let (prim_y_bnlhp, prim_final_state_bhpr) = B::ssd_serial_recalculated(
258                    x_bnlhp.primitive.clone(),
259                    dt_discretized_bhnl.primitive.clone(),
260                    b_bnlgr.primitive.clone(),
261                    c_bnlgr.primitive.clone(),
262                    d_h.primitive.clone(),
263                    initial_state_bhpr.primitive.clone(),
264                    da_cumsum_bhnl.primitive.clone(),
265                );
266
267                // Note: prep.finish accepts only a single tensor.
268                // Flatten both outputs and cat into one 1-D tensor so that
269                // one Backward node covers both.
270                let flat_y_BNLHP: Tensor<B, 1> =
271                    Tensor::<B, 5>::from_primitive(TensorPrimitive::Float(prim_y_bnlhp))
272                        .reshape([flat_len_y_BNLHP]);
273                let flat_final_state_BHPR: Tensor<B, 1> =
274                    Tensor::<B, 4>::from_primitive(TensorPrimitive::Float(prim_final_state_bhpr))
275                        .reshape([flat_len_final_state_BHPR]);
276                let combined: Tensor<B, 1> =
277                    Tensor::cat(vec![flat_y_BNLHP, flat_final_state_BHPR], 0);
278
279                let state = State {
280                    x_bnlhp: x_bnlhp.primitive.clone(),
281                    dt_discretized_bhnl: dt_discretized_bhnl.primitive.clone(),
282                    b_bnlgr: b_bnlgr.primitive.clone(),
283                    c_bnlgr: c_bnlgr.primitive.clone(),
284                    d_h: d_h.primitive.clone(),
285                    initial_state_bhpr: initial_state_bhpr.primitive.clone(),
286                    da_cumsum_bhnl: da_cumsum_bhnl.primitive.clone(),
287                    //
288                    flat_len_y_BNLHP,
289                    flat_len_final_state_BHPR,
290                    //
291                    shape_x_bnlhp, shape_dt_discretized_bhnl, shape_b_bnlgr, shape_c_bnlgr, shape_d_h, shape_initial_state_bhpr, shape_da_cumsum_bhnl,
292                    shape_y_bnlhp, shape_final_state_bhpr,
293                };
294                let tracked_combined: FloatTensor<Autodiff<B, C>> =
295                    prep.finish(state, combined.into_primitive().tensor());
296
297                // Split the tracked 1-D tensor back into the two outputs.
298                // The narrow / reshape ops create thin pass-through autodiff
299                // nodes; their backward accumulates gradients into the
300                // combined gradient vector consumed above.
301                let tracked_combined: Tensor<Autodiff<B, C>, 1> =
302                    Tensor::from_primitive(TensorPrimitive::Float(tracked_combined));
303
304                let tracked_y_bnlhp: Tensor<Autodiff<B, C>, 5> = tracked_combined
305                    .clone()
306                    .narrow(0, 0, flat_len_y_BNLHP)
307                    .reshape(shape_y_bnlhp);
308                let tracked_final_state_bhpr: Tensor<Autodiff<B, C>, 4> = tracked_combined
309                    .narrow(0, flat_len_y_BNLHP, flat_len_final_state_BHPR)
310                    .reshape(shape_final_state_bhpr);
311
312                (
313                    tracked_y_bnlhp.into_primitive().tensor(),
314                    tracked_final_state_bhpr.into_primitive().tensor(),
315                )
316            }
317
318            OpsKind::UnTracked(prep) => {
319                // No gradient tracking needed — run bare forward and return.
320                let (prim_y_bnlhp, prim_final_state_bhpr) = B::ssd_serial_recalculated(
321                    x_bnlhp.primitive,
322                    dt_discretized_bhnl.primitive,
323                    b_bnlgr.primitive,
324                    c_bnlgr.primitive,
325                    d_h.primitive,
326                    initial_state_bhpr.primitive,
327                    da_cumsum_bhnl.primitive,
328                );
329
330                // Note: prep.finish accepts only a single tensor.
331                let flat_y_BNLHP: Tensor<B, 1> =
332                    Tensor::<B, 5>::from_primitive(TensorPrimitive::Float(prim_y_bnlhp))
333                        .reshape([flat_len_y_BNLHP]);
334                let flat_final_state_BHPR: Tensor<B, 1> =
335                    Tensor::<B, 4>::from_primitive(TensorPrimitive::Float(prim_final_state_bhpr))
336                        .reshape([flat_len_final_state_BHPR]);
337                let combined: Tensor<B, 1> = Tensor::cat(vec![flat_y_BNLHP, flat_final_state_BHPR], 0);
338
339                let tracked_combined: FloatTensor<Autodiff<B, C>> =
340                    prep.finish(combined.into_primitive().tensor());
341
342                let tracked_combined: Tensor<Autodiff<B, C>, 1> =
343                    Tensor::from_primitive(TensorPrimitive::Float(tracked_combined));
344                let tracked_y_bnlhp: Tensor<Autodiff<B, C>, 5> = tracked_combined
345                    .clone()
346                    .narrow(0, 0, flat_len_y_BNLHP)
347                    .reshape(shape_y_bnlhp);
348                let tracked_final_state_bhpr: Tensor<Autodiff<B, C>, 4> = tracked_combined
349                    .narrow(0, flat_len_y_BNLHP, flat_len_final_state_BHPR)
350                    .reshape(shape_final_state_bhpr);
351
352                (
353                    tracked_y_bnlhp.into_primitive().tensor(),
354                    tracked_final_state_bhpr.into_primitive().tensor(),
355                )
356            }
357        } // end match
358    } // end fn ssd_serial_recalculated on Autodiff<B, C>
359} // end impl Mamba2BackendExt for Autodiff<B, C>
