1//! # Custom autodiff node for the Mamba-2 recompute backward
2//!
3//! Implements [`Mamba2BackendExt`] for `Autodiff<B>` by registering a single
4//! Burn [`Backward`] node.  The forward stores only its (small) leaf inputs;
5//! during backprop those are replayed through the K1–K5 kernels and the
6//! analytic gradient math in [`combined_backward`], so the large intermediate
7//! tensors never have to be kept alive — the ~⅓ training-memory saving of the
8//! `SerialRecalculated` path.
9//!
10//! The two forward outputs (`y` and `final_state`) are flattened into one
11//! tracked 1-D tensor (via [`crate::utils::combined_grad`]) so that a single
12//! `Backward<B, 7>` node — one per the 7 differentiable inputs — covers both.
13
14#![allow(non_snake_case)]
15
16use crate::mamba2::ssd::serial_recalculated::{
17    Mamba2BackendExt,
18    combined_backward::{self, CombinedGrads},
19};
20use burn::backend::autodiff::{
21    Autodiff,
22    checkpoint::{base::Checkpointer, strategy::CheckpointStrategy},
23    grads::Gradients,
24    ops::{Backward, Ops, OpsKind},
25};
26use burn::backend::tensor::FloatTensor;
27use burn::backend::{Backend, BackendTypes};
28use burn::prelude::*;
29
30impl<B: Backend + Mamba2BackendExt, C: CheckpointStrategy> Mamba2BackendExt for Autodiff<B, C> {
31    /// Memory-efficient combined forward+backward.
32    ///
33    /// The two output tensors are concatenated into a single 1-dimensional tracked tensor
34    /// so that one `Backward<B, 7>` node covers both outputs.  The caller
35    /// receives split+reshaped slices of that combined tensor; burn's autodiff
36    /// accumulates their upstream gradients back into a single gradient vector
37    /// before firing this backward.
38    fn ssd_serial_recalculated(
39        x_bnlhp: FloatTensor<Self>,
40        dt_discretized_bhnl: FloatTensor<Self>,
41        b_bnlhr: FloatTensor<Self>,
42        c_bnlhr: FloatTensor<Self>,
43        d_h: FloatTensor<Self>,
44        initial_state_bhpr: FloatTensor<Self>,
45        a_decay_h: FloatTensor<Self>,
46    ) -> (FloatTensor<Self>, FloatTensor<Self>) {
47        // ── Backward struct ──────────────────────────────────────────────────
48        #[derive(Debug)]
49        struct CombinedKernelsBackward;
50
51        #[derive(Clone, Debug)]
52        struct State<B: Backend> {
53            x_bnlhp: <B as BackendTypes>::FloatTensorPrimitive,
54            dt_discretized_bhnl: <B as BackendTypes>::FloatTensorPrimitive,
55            b_bnlhr: <B as BackendTypes>::FloatTensorPrimitive,
56            c_bnlhr: <B as BackendTypes>::FloatTensorPrimitive,
57            d_h: <B as BackendTypes>::FloatTensorPrimitive,
58            initial_state_bhpr: <B as BackendTypes>::FloatTensorPrimitive,
59            a_decay_h: <B as BackendTypes>::FloatTensorPrimitive,
60            // flat byte-sizes for splitting the combined gradient vector
61            flat_len_y_BNLHP: usize,
62            flat_len_final_state_BHPR: usize,
63            // shapes needed to reconstruct tensors in the right ranks
64            shape_x_bnlhp: [usize; 5],
65            shape_dt_discretized_bhnl: [usize; 4],
66            shape_b_bnlhr: [usize; 5],
67            shape_c_bnlhr: [usize; 5],
68            shape_d_h: [usize; 1],
69            shape_initial_state_bhpr: [usize; 4],
70            shape_a_decay_h: [usize; 1],
71            shape_y_bnlhp: [usize; 5],          // (output 1)
72            shape_final_state_bhpr: [usize; 4], // (output 2)
73        }
74
75        /// State carried across the forward→backward boundary.
76        ///
77        /// Only the 7 original inputs are saved; all intermediates (cb, intra
78        /// state, chunk_input_state) are recomputed during `backward`.
79        #[allow(clippy::type_complexity)]
80        impl<B: Backend + Mamba2BackendExt> Backward<B, 7> for CombinedKernelsBackward {
81            type State = State<B>;
82
83            fn backward(
84                self,
85                ops: Ops<Self::State, 7>,
86                grads: &mut Gradients,
87                _checkpointer: &mut Checkpointer,
88            ) {
89                let [
90                    node_x_bnlhp,
91                    node_dt_discretized_bhnl,
92                    node_b_bnlhr,
93                    node_c_bnlhr,
94                    node_d_h,
95                    node_initial_state_bhpr,
96                    node_a_decay_h,
97                ] = ops.parents;
98
99                let d_combined: <B as BackendTypes>::FloatTensorPrimitive =
100                    grads.consume::<B>(&ops.node);
101
102                let State {
103                    x_bnlhp,
104                    dt_discretized_bhnl,
105                    b_bnlhr,
106                    c_bnlhr,
107                    d_h,
108                    initial_state_bhpr,
109                    a_decay_h,
110                    //
111                    flat_len_y_BNLHP,
112                    flat_len_final_state_BHPR,
113                    //
114                    shape_x_bnlhp,
115                    shape_dt_discretized_bhnl,
116                    shape_b_bnlhr,
117                    shape_c_bnlhr,
118                    shape_d_h,
119                    shape_initial_state_bhpr,
120                    shape_a_decay_h,
121                    //
122                    shape_y_bnlhp,
123                    shape_final_state_bhpr,
124                } = ops.state;
125
126                // ── Reconstruct saved tensors as rank-tagged primitives ────
127                use crate::utils::fprim::F;
128
129                let x_bnlhp = F::<B, 5>::new(x_bnlhp).reshape(shape_x_bnlhp);
130                let dt_discretized_bhnl =
131                    F::<B, 4>::new(dt_discretized_bhnl).reshape(shape_dt_discretized_bhnl);
132                let b_bnlhr = F::<B, 5>::new(b_bnlhr).reshape(shape_b_bnlhr);
133                let c_bnlhr = F::<B, 5>::new(c_bnlhr).reshape(shape_c_bnlhr);
134                let d_h = F::<B, 1>::new(d_h).reshape(shape_d_h);
135                let initial_state_bhpr =
136                    F::<B, 4>::new(initial_state_bhpr).reshape(shape_initial_state_bhpr);
137                let a_decay_h = F::<B, 1>::new(a_decay_h).reshape(shape_a_decay_h);
138
139                // ── Split incoming combined gradient ───────────────────────
140                let (d_y_bnlhp, d_final_state_bhpr) =
141                    crate::utils::combined_grad::unflatten_pair::<B, 5, 4>(
142                        d_combined,
143                        flat_len_y_BNLHP,
144                        flat_len_final_state_BHPR,
145                        shape_y_bnlhp,
146                        shape_final_state_bhpr,
147                    );
148
149                // ── Core gradient computation ──────────────────────────────
150                let CombinedGrads {
151                    d_x_bnlhp,
152                    d_dt_discretized_bhnl,
153                    d_b_bnlhr,
154                    d_c_bnlhr,
155                    d_d_h,
156                    d_initial_state_bhpr,
157                    d_a_decay_h,
158                    ..
159                } = combined_backward::combined_backward(
160                    F::<B, 5>::new(d_y_bnlhp),
161                    F::<B, 4>::new(d_final_state_bhpr),
162                    //
163                    x_bnlhp,
164                    dt_discretized_bhnl,
165                    b_bnlhr,
166                    c_bnlhr,
167                    d_h,
168                    initial_state_bhpr,
169                    a_decay_h,
170                );
171
172                // ── Register gradients ─────────────────────────────────────
173                if let Some(n) = node_x_bnlhp {
174                    grads.register::<B>(n.id, d_x_bnlhp.inner());
175                }
176                if let Some(n) = node_dt_discretized_bhnl {
177                    grads.register::<B>(n.id, d_dt_discretized_bhnl.inner());
178                }
179                if let Some(n) = node_b_bnlhr {
180                    grads.register::<B>(n.id, d_b_bnlhr.inner());
181                }
182                if let Some(n) = node_c_bnlhr {
183                    grads.register::<B>(n.id, d_c_bnlhr.inner());
184                }
185                if let Some(n) = node_d_h {
186                    grads.register::<B>(n.id, d_d_h.inner());
187                }
188                if let Some(n) = node_initial_state_bhpr {
189                    grads.register::<B>(n.id, d_initial_state_bhpr.inner());
190                }
191                if let Some(n) = node_a_decay_h {
192                    grads.register::<B>(n.id, d_a_decay_h.inner());
193                }
194            }
195        } // end impl Backward
196
197        // ── Shape extraction helpers ───────────────────────────────────────
198        // Accessed via the AutodiffTensor wrappers (which own both .node
199        // and .primitive).
200        use burn::backend::TensorMetadata;
201        let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.primitive.shape().dims();
202        let [_, _, _, _nheads_b, state_rank] = b_bnlhr.primitive.shape().dims();
203
204        let flat_len_y_BNLHP = batch * nchunks * chunk_len * nheads * per_head_dim;
205        let flat_len_final_state_BHPR = batch * nheads * per_head_dim * state_rank;
206
207        let shape_x_bnlhp: [usize; 5] = [batch, nchunks, chunk_len, nheads, per_head_dim];
208        let shape_dt_discretized_bhnl: [usize; 4] = [batch, nheads, nchunks, chunk_len];
209        let shape_b_bnlhr: [usize; 5] = [batch, nchunks, chunk_len, nheads, state_rank];
210        let shape_c_bnlhr: [usize; 5] = [batch, nchunks, chunk_len, nheads, state_rank];
211        let shape_d_h: [usize; 1] = [nheads];
212        let shape_initial_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
213        let shape_a_decay_h: [usize; 1] = [nheads];
214        let shape_y_bnlhp: [usize; 5] = [batch, nchunks, chunk_len, nheads, per_head_dim];
215        let shape_final_state_bhpr: [usize; 4] = [batch, nheads, per_head_dim, state_rank];
216
217        // ── Register backward / run forward ───────────────────────────────
218        match CombinedKernelsBackward
219            .prepare::<C>([
220                x_bnlhp.node.clone(),
221                dt_discretized_bhnl.node.clone(),
222                b_bnlhr.node.clone(),
223                c_bnlhr.node.clone(),
224                d_h.node.clone(),
225                initial_state_bhpr.node.clone(),
226                a_decay_h.node.clone(),
227            ])
228            .compute_bound()
229            .stateful() // requires compute_bound
230        {
231            OpsKind::Tracked(prep) => {
232                // Run the inner (non-autodiff) forward pass.
233                let (prim_y_bnlhp, prim_final_state_bhpr) = B::ssd_serial_recalculated(
234                    x_bnlhp.primitive.clone(),
235                    dt_discretized_bhnl.primitive.clone(),
236                    b_bnlhr.primitive.clone(),
237                    c_bnlhr.primitive.clone(),
238                    d_h.primitive.clone(),
239                    initial_state_bhpr.primitive.clone(),
240                    a_decay_h.primitive.clone(),
241                );
242
243                // prep.finish takes a single tensor, so pack both outputs into a
244                // single 1-D tensor; one Backward node then covers both.
245                let (prim_combined, _, _) = crate::utils::combined_grad::flatten_pair::<B>(
246                    prim_y_bnlhp,
247                    prim_final_state_bhpr,
248                );
249
250                let state = State {
251                    x_bnlhp: x_bnlhp.primitive.clone(),
252                    dt_discretized_bhnl: dt_discretized_bhnl.primitive.clone(),
253                    b_bnlhr: b_bnlhr.primitive.clone(),
254                    c_bnlhr: c_bnlhr.primitive.clone(),
255                    d_h: d_h.primitive.clone(),
256                    initial_state_bhpr: initial_state_bhpr.primitive.clone(),
257                    a_decay_h: a_decay_h.primitive.clone(),
258                    //
259                    flat_len_y_BNLHP,
260                    flat_len_final_state_BHPR,
261                    //
262                    shape_x_bnlhp, shape_dt_discretized_bhnl, shape_b_bnlhr, shape_c_bnlhr, shape_d_h, shape_initial_state_bhpr, shape_a_decay_h,
263                    shape_y_bnlhp, shape_final_state_bhpr,
264                };
265                let tracked_combined: FloatTensor<Autodiff<B, C>> =
266                    prep.finish(state, prim_combined);
267
268                // Split the tracked combined tensor back into the two outputs.
269                // The narrow/reshape ops are thin autodiff pass-throughs whose
270                // backwards accumulate into the combined gradient vector that
271                // `backward` above consumes.
272                let (tracked_y_bnlhp, tracked_final_state_bhpr) =
273                    crate::utils::combined_grad::autodiff_unflatten_pair::<B, C, 5, 4>(
274                    tracked_combined,
275                    flat_len_y_BNLHP,
276                    flat_len_final_state_BHPR,
277                    shape_y_bnlhp,
278                    shape_final_state_bhpr,
279                );
280
281                (
282                    tracked_y_bnlhp,
283                    tracked_final_state_bhpr,
284                )
285            }
286
287            OpsKind::UnTracked(prep) => {
288                // No gradient tracking — just run the bare forward.
289                let (prim_y_bnlhp, prim_final_state_bhpr) = B::ssd_serial_recalculated(
290                    x_bnlhp.primitive,
291                    dt_discretized_bhnl.primitive,
292                    b_bnlhr.primitive,
293                    c_bnlhr.primitive,
294                    d_h.primitive,
295                    initial_state_bhpr.primitive,
296                    a_decay_h.primitive,
297                );
298
299                let (combined, _, _) = crate::utils::combined_grad::flatten_pair::<B>(
300                    prim_y_bnlhp,
301                    prim_final_state_bhpr,
302                );
303
304                let tracked_combined: FloatTensor<Autodiff<B, C>> =
305                    prep.finish(combined);
306
307                let (tracked_y_bnlhp, tracked_final_state_bhpr) = crate::utils::combined_grad::autodiff_unflatten_pair::<B, C, 5, 4>(
308                    tracked_combined,
309                    flat_len_y_BNLHP,
310                    flat_len_final_state_BHPR,
311                    shape_y_bnlhp,
312                    shape_final_state_bhpr,
313                );
314
315                (
316                    tracked_y_bnlhp,
317                    tracked_final_state_bhpr,
318                )
319            }
320        } // end match
321    } // end fn ssd_serial_recalculated on Autodiff<B, C>
322} // end impl Mamba2BackendExt for Autodiff<B, C>
