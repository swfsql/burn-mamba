7impl<B: Backend> Mamba2SsdInput<B> {
8    /// Forward pass for the Mamba-2 SSD module.
9    ///
10    /// Returns:
11    /// - `y_bnlhp`.
12    /// - `final_state_bhpr`.
13    #[allow(non_snake_case)]
14    pub fn ssd_serial(self) -> (Tensor<B, 5>, Tensor<B, 4>) {
15        let input = self;
16        let [batch, nchunks, chunk_len, nheads, per_head_dim] = input.x_bnlhp.dims();
17        let [.., state_rank] = input.b_bnlhr.dims();
18        assert!(nchunks > 0, "sequence length must be at least 1");
19
20        san(&input.x_bnlhp);
21        san(&input.dt_bnlh);
22        san(&input.a_decay_h);
23        san(&input.b_bnlhr);
24        san(&input.c_bnlhr);
25        san(&input.d_h);
26        san(&input.initial_state_bhpr);
27
28        assert!(
29            input.init_state_hpr.is_none(),
30            "init_state_hpr not yet implemented"
31        );
32
33        // ── Permutes ──────────────────────────────────────────────────────────────────
34        // Note: dt_bnlh calculation (originally in Kernel 1) moved to Step 4 (before padding).
35        let dt_discretized_bhnl = input.dt_bnlh.permute([0, 3, 1, 2]);
36        assert_eq!(
37            [batch, nheads, nchunks, chunk_len],
38            dt_discretized_bhnl.dims()
39        );
40        san(&dt_discretized_bhnl);
41
42        // ── Kernel 1 ──────────────────────────────────────────────────────────────────
43        // IO: (..) -> (da_cumsum_bhnl [used in K3+K5][*], da_chunk_end_bhn [used in K4][omitted][*])
44        let (da_cumsum_bhnl, da_chunk_end_bhn): (Tensor<B, 4>, Tensor<B, 3>) =
45            k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), input.a_decay_h.clone());
46        assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
47        assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
48        san(&da_cumsum_bhnl);
49        san(&da_chunk_end_bhn);
50
51        // ── Kernel 2 ──────────────────────────────────────────────────────────────────
52        // IO: (..) -> (cb_bnhll [used in K5][!])
53        let cb_bnhll: Tensor<B, 5> = k2_ssd_bmm(input.c_bnlhr.clone(), input.b_bnlhr.clone());
54        assert_eq!(
55            [batch, nchunks, nheads, chunk_len, chunk_len],
56            cb_bnhll.dims()
57        );
58        san(&cb_bnhll);
59
60        // ── Kernel 3 ──────────────────────────────────────────────────────────────────
61        // IO: (..) -> (intra_chunk_state_bnhpr [used in K4][!])
62        let intra_chunk_state_bnhpr: Tensor<B, 5> = k3_ssd_chunk_state(
63            input.x_bnlhp.clone(),
64            input.b_bnlhr.clone(),
65            da_cumsum_bhnl.clone(),
66            dt_discretized_bhnl.clone(),
67        );
68        assert_eq!(
69            [batch, nchunks, nheads, per_head_dim, state_rank],
70            intra_chunk_state_bnhpr.dims()
71        );
72        san(&intra_chunk_state_bnhpr);
73
74        // ── Kernel 4 ──────────────────────────────────────────────────────────────────
75        // IO: (..) -> (chunk_input_state_bnhpr [used in K5][!], final_state_bhpr [final output])
76        let (chunk_input_state_bnhpr, final_state_bhpr): (Tensor<B, 5>, Tensor<B, 4>) =
77            k4_ssd_state_passing(
78                intra_chunk_state_bnhpr.clone(),
79                da_chunk_end_bhn.clone(),
80                input.initial_state_bhpr,
81            );
82        assert_eq!(
83            [batch, nchunks, nheads, per_head_dim, state_rank],
84            chunk_input_state_bnhpr.dims()
85        );
86        assert_eq!(
87            [batch, nheads, per_head_dim, state_rank],
88            final_state_bhpr.dims()
89        );
90        san(&chunk_input_state_bnhpr);
91        san(&final_state_bhpr);
92
93        // ── Kernel 5 ──────────────────────────────────────────────────────────────────
94        let y_bnlhp: Tensor<B, 5> = k5_ssd_chunk_scan(
95            da_cumsum_bhnl,
96            dt_discretized_bhnl,
97            input.x_bnlhp,
98            input.c_bnlhr,
99            cb_bnhll,
100            chunk_input_state_bnhpr,
101            input.d_h,
102        );
103        assert_eq!(
104            [batch, nchunks, chunk_len, nheads, per_head_dim],
105            y_bnlhp.dims()
106        );
107        san(&y_bnlhp);
108
109        (y_bnlhp, final_state_bhpr)
110    }
111}

113/// Based on the Kernel 1 Triton reference `_chunk_cumsum_fwd_kernel` (`ssd_chunk_state.py`).
114///
115/// Returns:
116/// - da_cumsum_bhnl `[used in K3+K5][*]` - intra-chunk cumsum.
117/// - da_chunk_end_bhn `[used in K4][omitted][*]` - last da_cumsum per chunk.
118pub fn k1_ssd_chunk_cumsum<B: Backend>(
119    dt_discretized_bhnl: Tensor<B, 4>,
120    a_decay_h: Tensor<B, 1>,
121) -> (Tensor<B, 4>, Tensor<B, 3>) {
122    let [batch, nheads, nchunks, chunk_len] = dt_discretized_bhnl.dims();
123    let da_cumsum_bhnl: Tensor<B, 4> = {
124        let a_decay_bhnl = a_decay_h
125            // - 1/6: unsqueeze-dims: (a_decay_h [*]) -> (a_decay_1h11)
126            .unsqueeze_dims::<4>(&[0, 2, 3]) // a_decay_1h11
127            // - 2: expand: (a_decay_1h11) -> (a_decay_bhnl)
128            .expand([batch, nheads, nchunks, chunk_len]);
129        // - 3: mul: (dt_discretized_bhnl [*], a_decay_bhnl) -> (da_bhnl)
130        // - 4: cumsum: (da_bhnl) -> (da_cumsum_bhnl [out][*])
131        (dt_discretized_bhnl * a_decay_bhnl).cumsum(3)
132    };
133    assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
134
135    let da_chunk_end_bhn = da_cumsum_bhnl
136        .clone()
137        // - 5: slice: (da_cumsum_bhnl [*]) -> (da_cumsum_bhn1)
138        .slice(s![.., .., .., -1]) // da_cumsum_bhn1
139        // - 6/6: squeeze: (da_cumsum_bhn1) -> (da_chunk_end_bhn [out])
140        .squeeze_dim::<3>(3);
141    assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
142
143    (da_cumsum_bhnl, da_chunk_end_bhn)
144}

146/// Based on the Kernel 2 Triton reference `_bmm_chunk_fwd_kernel` (`ssd_bmm.py`).
147///
148/// Returns:
149/// - cb_bnhll `[used in K5][!]`.
150pub fn k2_ssd_bmm<B: Backend>(c_bnlhr: Tensor<B, 5>, b_bnlhr: Tensor<B, 5>) -> Tensor<B, 5> {
151    let [batch, nchunks, chunk_len, nheads, _state_rank] = c_bnlhr.dims();
152
153    // - 1/3: permute: (c_bnlhr [in][*]) -> (c_bnhlr)
154    let c_bnhlr = c_bnlhr.permute([0, 1, 3, 2, 4]);
155    // - 2: permute: (b_bnlhr [in][*]) -> (b_bnhrl)
156    let b_bnhrl = b_bnlhr.permute([0, 1, 3, 4, 2]);
157    // - 3/3: matmul: (c_bnhlr, b_bnhrl) -> (cb_bnhll [out][!])
158    let cb_bnhll: Tensor<B, 5> = c_bnhlr.matmul(b_bnhrl);
159    assert_eq!(
160        [batch, nchunks, nheads, chunk_len, chunk_len],
161        cb_bnhll.dims()
162    );
163    // Note: cb_bnhll is then only used by Kernel 5.
164    cb_bnhll
165}

167/// Based on the Kernel 3 Triton reference `_chunk_state_fwd_kernel` (`ssd_chunk_state.py`).
168///
169/// Returns:
170/// - cb_bngll `[used in K5][!]` - state assuming zero initial state at each chunk boundary.
171/// - b_bar_scale_bhnl `[*]` - intermediary
172pub fn k3_ssd_chunk_state<B: Backend>(
173    x_bnlhp: Tensor<B, 5>,
174    b_bnlhr: Tensor<B, 5>,
175    da_cumsum_bhnl: Tensor<B, 4>,
176    dt_discretized_bhnl: Tensor<B, 4>,
177) -> Tensor<B, 5> {
178    use burn::tensor::s;
179
180    let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
181    let [.., state_rank] = b_bnlhr.dims();
182
183    // permute b and x to prepare them for the matmul
184    // - 1/15: permute: (x_bnlhp [in][*]) -> (x_bnhpl)
185    let x_bnhpl = x_bnlhp.clone().permute([0, 1, 3, 4, 2]);
186    assert_eq!(
187        [batch, nchunks, nheads, per_head_dim, chunk_len],
188        x_bnhpl.dims()
189    );
190    // - 2: permute: (b_bnlhr [in][*]) -> (b_bnhlr)
191    let b_bnhlr = b_bnlhr.permute([0, 1, 3, 2, 4]);
192    assert_eq!(
193        [batch, nchunks, nheads, chunk_len, state_rank],
194        b_bnhlr.dims()
195    );
196
197    // scale b
198    let b_scaled_bnhlr = {
199        let b_bar_scale_bhnl = {
200            let da_cumsum_last_in_chunk_bhn1 =
201                // - 6: slice: (da_cumsum_bhnl [in][*]) -> (da_cumsum_last_in_chunk_bhn1)
202                da_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
203            assert_eq!(
204                [batch, nheads, nchunks, 1],
205                da_cumsum_last_in_chunk_bhn1.dims()
206            );
207
208            // - 7: expand: (da_cumsum_last_in_chunk_bhn1) -> (da_cumsum_last_bhnl)
209            let da_cumsum_last_bhnl =
210                da_cumsum_last_in_chunk_bhn1.expand([batch, nheads, nchunks, chunk_len]);
211            // - 8: sub: (da_cumsum_last_bhnl, da_cumsum_bhnl [from K1][*]) -> (da_delta_bhnl)
212            let da_delta_bhnl = da_cumsum_last_bhnl - da_cumsum_bhnl.clone();
213            // - 9: exp: (da_delta_bhnl) -> (forward_decay_to_chunk_end_bhnl [+])
214            let forward_decay_to_chunk_end_bhnl = da_delta_bhnl.exp();
215            assert_eq!(
216                [batch, nheads, nchunks, chunk_len],
217                forward_decay_to_chunk_end_bhnl.dims()
218            );
219
220            // - 10: mul: (forward_decay_to_chunk_end_bhnl [+], dt_discretized_bhnl [in][*]) -> (b_bar_scale_bhnl [+])
221            forward_decay_to_chunk_end_bhnl * dt_discretized_bhnl.clone()
222        };
223        assert_eq!([batch, nheads, nchunks, chunk_len], b_bar_scale_bhnl.dims());
224
225        // - 11: permute: (b_bar_scale_bhnl [+]) -> (b_bar_scale_bnhl)
226        let b_bar_scale_bnhl = b_bar_scale_bhnl.permute([0, 2, 1, 3]);
227        assert_eq!([batch, nchunks, nheads, chunk_len], b_bar_scale_bnhl.dims());
228        let b_bar_scale_bnhlr = b_bar_scale_bnhl
229            // - 12: unsqueeze: (b_bar_scale_bnhl) -> (b_bar_scale_bnhl1)
230            .unsqueeze_dim::<5>(4) // b_bar_scale_bnhl1
231            // - 13: expand: (b_bar_scale_bnhl1) -> (b_bar_scale_bnhlr)
232            .expand([batch, nchunks, nheads, chunk_len, state_rank]);
233        // - 14: mul: (b_bnhlr, b_bar_scale_bnhlr) -> (b_scaled_bnhlr [+])
234        b_bnhlr * b_bar_scale_bnhlr
235    };
236    assert_eq!(
237        [batch, nchunks, nheads, chunk_len, state_rank],
238        b_scaled_bnhlr.dims()
239    );
240
241    // - 15/15: matmul: (x_bnhpl, b_scaled_bnhlr [+]) -> (intra_chunk_state_bnhpr [out][!])
242    let intra_chunk_state_bnhpr: Tensor<B, 5> = x_bnhpl.matmul(b_scaled_bnhlr);
243    assert_eq!(
244        [batch, nchunks, nheads, per_head_dim, state_rank],
245        intra_chunk_state_bnhpr.dims()
246    );
247    intra_chunk_state_bnhpr
248}

250/// Based on the Kernel 4 Triton reference `_state_passing_fwd_kernel` (`ssd_state_passing.py`).
251///
252/// Returns:
253/// - chunk_input_state_bnhpr `[used in K5][!]`.
254/// - final_state_bhpr `[final output]`.
255pub fn k4_ssd_state_passing<B: Backend>(
256    intra_chunk_state_bnhpr: Tensor<B, 5>,
257    da_chunk_end_bhn: Tensor<B, 3>,
258    initial_state_bhpr: Tensor<B, 4>,
259) -> (Tensor<B, 5>, Tensor<B, 4>) {
260    let [batch, nchunks, nheads, per_head_dim, state_rank] = intra_chunk_state_bnhpr.dims();
261    let flat_state_dim = per_head_dim * state_rank;
262
263    // - 1/5: init-mut: (initial_state_bhpr [in][*]) -> (running_state_bhpr)
264    let mut running_state_bhpr = initial_state_bhpr;
265    assert_eq!(
266        [batch, nheads, per_head_dim, state_rank],
267        running_state_bhpr.dims()
268    );
269
270    let mut chunk_input_state_vec_bhpr = Vec::with_capacity(nchunks + 1);
271    // - 2: vec-push: (running_state_bhpr [elem]) -> (chunk_input_state_vec_bhpr [vec][!])
272    chunk_input_state_vec_bhpr.push(running_state_bhpr.clone());
273
274    // - 3: serial-loop: (0..nchunks)
275    for i_chunk in 0..nchunks {
276        let intra_state_bhpr = intra_chunk_state_bnhpr
277            .clone()
278            //   - 3.1/3.9: slice: (intra_chunk_state_bnhpr [in][!]) -> (intra_chunk_state_b1hpr)
279            .slice(s![.., i_chunk, .., .., ..]) // intra_chunk_state_b1hpr
280            //   - 3.2: squeeze: (intra_chunk_state_b1hpr) -> (intra_state_bhpr)
281            .squeeze_dim::<4>(1);
282        assert_eq!(
283            [batch, nheads, per_head_dim, state_rank],
284            intra_state_bhpr.dims()
285        );
286
287        let decay_bhpr = da_chunk_end_bhn
288            .clone()
289            //   - 3.3: slice: (da_chunk_end_bhn [in][*]) -> (da_chunk_end_bh1)
290            .slice(s![.., .., i_chunk]) // da_chunk_end_bh1
291            //   - 3.4: exp: (da_chunk_end_bh1) -> (exp_da_chunk_end_bh1)
292            .exp() // exp_da_chunk_end_bh1
293            //   - 3.5: unsqueeze: (exp_da_chunk_end_bh1) -> (exp_da_chunk_end_bh11)
294            .unsqueeze_dim::<4>(3) // exp_da_chunk_end_bh11
295            //   - 3.6: expand: (exp_da_chunk_end_bh11) -> (decay_bhpr)
296            .expand([batch, nheads, per_head_dim, state_rank]);
297
298        // SSM recurrence: running_state = decay * running_state + intra_state
299        running_state_bhpr =
300        //   - 3.7: mul: (decay_bhpr, running_state_bhpr) -> (running_state_bhpr)
301            (decay_bhpr * running_state_bhpr) // running_state_bhpr
302        //   - 3.8: add: (running_state_bhpr, intra_state_bhpr) -> (running_state_bhpr)
303            + intra_state_bhpr;
304        //   - 3.9/3.9: vec-push: (running_state_bhpr [elem]) -> (chunk_input_state_vec_bhpr [vec][!])
305        chunk_input_state_vec_bhpr.push(running_state_bhpr.clone());
306    }
307
308    // - 4: vec-pop: (chunk_input_state_vec_bhpr [vec][!]) -> (final_state_bhpr [elem][out][!])
309    let final_state_bhpr = chunk_input_state_vec_bhpr.pop().unwrap();
310    assert_eq!(
311        [batch, nheads, per_head_dim, state_rank],
312        final_state_bhpr.dims()
313    );
314
315    // - 5/5: stack: (chunk_input_state_vec_bhpr [!]) -> (chunk_input_state_bnhpr [out][!])
316    let chunk_input_state_bnhpr = Tensor::stack(chunk_input_state_vec_bhpr, 1);
317    assert_eq!(
318        [batch, nchunks, nheads, per_head_dim, state_rank],
319        chunk_input_state_bnhpr.dims()
320    );
321
322    (chunk_input_state_bnhpr, final_state_bhpr)
323}

325/// Based on the Kernel 5 Triton reference `_chunk_scan_fwd_kernel` (`ssd_chunk_scan.py`).
326///
327/// Returns:
328/// - y_bnlhp `[final output]`
329pub fn k5_ssd_chunk_scan<B: Backend>(
330    da_cumsum_bhnl: Tensor<B, 4>,
331    dt_discretized_bhnl: Tensor<B, 4>,
332    x_bnlhp: Tensor<B, 5>,
333    c_bnlhr: Tensor<B, 5>,
334    cb_bnhll: Tensor<B, 5>,
335    chunk_input_state_bnhpr: Tensor<B, 5>,
336    d_h: Tensor<B, 1>,
337) -> Tensor<B, 5> {
338    let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
339    let device = x_bnlhp.device();
340
341    // Rearrange inputs to the common [batch, nchunks, nheads, ...] ordering used below.
342    // - 1/36: permute: (da_cumsum_bhnl [*]) -> (da_cumsum_bnhl)
343    let da_cumsum_bnhl = da_cumsum_bhnl.permute([0, 2, 1, 3]);
344    san(&da_cumsum_bnhl);
345    // - 2: permute: (dt_discretized_bhnl [*]) -> (dt_bnhl)
346    let dt_bnhl = dt_discretized_bhnl.permute([0, 2, 1, 3]);
347    san(&dt_bnhl);
348    // - 3: permute: (x_bnlhp [*]) -> (x_bnhlp)
349    let x_bnhlp = x_bnlhp.clone().permute([0, 1, 3, 2, 4]);
350    san(&x_bnhlp);
351
352    // B/C are already per-head — only a permute is needed.
353    let c_bnhlr = c_bnlhr.permute([0, 1, 3, 2, 4]);
354    san(&c_bnhlr);
355    san(&cb_bnhll);
356
357    // ── BLUE: exp(dA[l]) · C[l,:] @ state_in^T ─────────────────────────────
358    //
359    //   blue[b,n,h,l,p] = exp(da[b,n,h,l]) · Σ_r  c[b,n,h,l,r] · state[b,n,h,p,r]
360    //
361    //   [b,n,h,l,r] @ [b,n,h,r,p]  →  [b,n,h,l,p]
362    let exp_da_cumsum_bnhlp = da_cumsum_bnhl
363        .clone()
364        // - 11: exp: (da_cumsum_bnhl) -> (exp_da_cumsum_bnhl)
365        .exp()
366        // - 12: unsqueeze: (exp_da_cumsum_bnhl) -> (exp_da_cumsum_bnhl1)
367        .unsqueeze_dim::<5>(4) // exp_da_cumsum_bnhl1
368        // - 13: expand: (exp_da_cumsum_bnhl1) -> (exp_da_cumsum_bnhlp)
369        .expand([batch, nchunks, nheads, chunk_len, per_head_dim]);
370    san(&exp_da_cumsum_bnhlp);
371    // - 14: permute: (chunk_input_state_bnhpr [!]) -> (chunk_input_state_bnhrp)
372    let chunk_input_state_bnhrp = chunk_input_state_bnhpr.permute([0, 1, 2, 4, 3]);
373    // - 15: matmul: (c_bnhlr, chunk_input_state_bnhrp) -> (blue_bnhlp)
374    let blue_scaled_bnhlp = c_bnhlr
375        .matmul(chunk_input_state_bnhrp)  // blue_bnhlp
376        // - 16: mul: (blue_bnhlp, exp_da_cumsum_bnhlp) -> (blue_scaled_bnhlp)
377        * exp_da_cumsum_bnhlp;
378    san(&blue_scaled_bnhlp);
379
380    // ── ORANGE: causal CB_weighted @ X ──────────────────────────────────────
381    //
382    //   orange[b,n,h,l,p] = Σ_{s≤l} CB[l,s] · exp(da[l]-da[s]) · dt[s] · x[s,p]
383    //
384    // Precompute the full lower-triangular weight matrix, then do a single matmul.
385    //
386    let da_cumsum_target_bnhll = da_cumsum_bnhl
387        .clone()
388        // - 17: unsqueeze: (da_cumsum_bnhl) -> (da_cumsum_bnhl1)
389        .unsqueeze_dim::<5>(4) // da_cumsum_bnhl1
390        // - 18: expand: (da_cumsum_bnhl1) -> (da_cumsum_target_bnhll)
391        .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
392    // println!("{}", da_cumsum_target_bnhll);
393    san(&da_cumsum_target_bnhll);
394    let da_cumsum_source_bnhll = da_cumsum_bnhl
395        // - 19: unsqueeze: (da_cumsum_bnhl) -> (da_cumsum_bnh1l)
396        .unsqueeze_dim::<5>(3) // da_cumsum_bnh1l
397        // - 20: expand: (da_cumsum_bnh1l) -> (da_cumsum_source_bnhll)
398        .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
399    // println!("{}", da_cumsum_source_bnhll);
400    san(&da_cumsum_source_bnhll);
401    // - 21: sub: (da_cumsum_target_bnhll, da_cumsum_source_bnhll) -> (da_cumsum_diff_bnhll)
402    let da_cumsum_diff_bnhll = da_cumsum_target_bnhll - da_cumsum_source_bnhll;
403    san(&da_cumsum_diff_bnhll);
404
405    // note: overflow instability at step 22, a `minimal::segsum`-like upper triangle protection is necessary.
406    // - 21.1: tril-mask: (0) -> (causal_mask_ll), expanded as a view to causal_mask_bnhll.
407    // true above the main diagonal, false at diagonal and below.
408    // Built at [L,L] and broadcast — the mask values do not depend on (b,n,h).
409    let causal_mask_bnhll: Tensor<B, 5, burn::prelude::Bool> =
410        Tensor::<B, 2, burn::prelude::Bool>::tril_mask([chunk_len, chunk_len], 0, &device)
411            .reshape([1, 1, 1, chunk_len, chunk_len])
412            .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
413    // - 21.2: mask-fill: (da_cumsum_diff_bnhll, causal_mask_bnhll) -> (da_cumsum_diff_masked_bnhll)
414    // Causal mask and exp stabilizer: above upper diagonal set to -inf.
415    let da_cumsum_diff_masked_bnhll =
416        da_cumsum_diff_bnhll.mask_fill(causal_mask_bnhll, f32::NEG_INFINITY);
417
418    // - 22: exp: (da_cumsum_diff_masked_bnhll) -> (da_cumsum_diff_exp_bnhll)
