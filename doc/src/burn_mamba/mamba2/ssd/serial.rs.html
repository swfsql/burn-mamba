6impl<B: Backend> Mamba2<B> {
7    /// Forward pass for the Mamba-2 SSD module.
8    ///
9    /// # Tensor shapes flowing through the function
10    ///
11    /// ```text
12    /// x_bshp:                  [batch, sequence, nheads, per_head_dim]
13    /// dt_bsh:                  [batch, sequence, nheads]
14    /// a_decay_h:               [nheads]
15    /// b_bsgr:                  [batch, sequence, ngroups, state_rank]
16    /// c_bsgr:                  [batch, sequence, ngroups, state_rank]
17    /// d_h:                     [nheads]
18    /// ssm_initial_state_bhpr   [batch, nheads, per_head_dim, state_rank]
19    /// _init_states_hpr         [nheads, per_head_dim, state_rank]
20    /// ```
21    #[allow(non_snake_case)]
22    pub fn ssd_serial(
23        x_bshp: Tensor<B, 4>,
24        dt_bsh: Tensor<B, 3>,
25        a_decay_h: Tensor<B, 1>,
26        b_bsgr: Tensor<B, 4>,
27        c_bsgr: Tensor<B, 4>,
28        d_h: Tensor<B, 1>,
29        ssm_initial_state_bhpr: Tensor<B, 4>,
30        _init_states_hpr: Option<Tensor<B, 3>>,
31        ngroups: usize,
32        state_rank: usize,
33        // currently must be set by caller
34        chunk_len: usize,
35    ) -> (Tensor<B, 4>, Tensor<B, 4>) {
36        let [batch, sequence, nheads, per_head_dim] = x_bshp.dims();
37        let device = x_bshp.device();
38        assert_ne!(ngroups, 0);
39        assert_eq!(nheads % ngroups, 0);
40        assert!(sequence > 0, "sequence length must be at least 1");
41        // `heads_per_group` is called `nheads_ngroups_ratio` in every Triton kernel.
42        // It is the compile-time constant used by GQA (Grouped Query Attention) to map
43        // a head index to its B/C group: `group_idx = head_idx / heads_per_group`.
44        let heads_per_group = nheads / ngroups;
45
46        // We require seqlen % chunk_len == 0 for simplicity (the Triton kernel handles a
47        // partial last chunk via `chunk_len_limit = min(chunk_len, seqlen - pid_c * chunk_len)`).
48        debug_assert_eq!(sequence % chunk_len, 0);
49        let nchunks = sequence / chunk_len;
50
51        assert!(
52            _init_states_hpr.is_none(),
53            "init_states_hpr not yet implemented"
54        );
55
56        // ── Reshape into chunks ───────────────────────────────────────────────────────
57        let x_bnlhp = x_bshp.reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
58        let dt_bnlh = dt_bsh.reshape([batch, nchunks, chunk_len, nheads]);
59        let b_bnlgr = b_bsgr.reshape([batch, nchunks, chunk_len, ngroups, state_rank]);
60        let c_bnlgr = c_bsgr.reshape([batch, nchunks, chunk_len, ngroups, state_rank]);
61
62        // ── Permutes ──────────────────────────────────────────────────────────────────
63        // Note: dt_bnlh calculation (originally in Kernel 1) moved to Step 4 (before padding).
64        let dt_discretized_bhnl = dt_bnlh.permute([0, 3, 1, 2]);
65        assert_eq!(
66            [batch, nheads, nchunks, chunk_len],
67            dt_discretized_bhnl.dims()
68        );
69
70        // ── Kernel 1 ──────────────────────────────────────────────────────────────────
71        // IO: (..) -> (da_cumsum_bhnl [used in K3+K5][*], da_chunk_end_bhn [used in K4][omitted][*])
72        let (da_cumsum_bhnl, da_chunk_end_bhn): (Tensor<B, 4>, Tensor<B, 3>) =
73            k1_ssd_chunk_cumsum(dt_discretized_bhnl.clone(), a_decay_h.clone());
74        assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
75        assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
76
77        // ── Kernel 2 ──────────────────────────────────────────────────────────────────
78        // IO: (..) -> (cb_bngll [used in K5][!])
79        let cb_bngll: Tensor<B, 5> = k2_ssd_bmm(c_bnlgr.clone(), b_bnlgr.clone());
80        assert_eq!(
81            [batch, nchunks, ngroups, chunk_len, chunk_len],
82            cb_bngll.dims()
83        );
84        // Note: cb_bngll is then only used by Kernel 5.
85
86        // ── Kernel 3 ──────────────────────────────────────────────────────────────────
87        // IO: (..) -> (intra_chunk_states_bnhpr [used in K4][!])
88        let intra_chunk_states_bnhpr: Tensor<B, 5> = k3_ssd_chunk_state(
89            x_bnlhp.clone(),
90            b_bnlgr.clone(),
91            da_cumsum_bhnl.clone(),
92            dt_discretized_bhnl.clone(),
93        );
94        assert_eq!(
95            [batch, nchunks, nheads, per_head_dim, state_rank],
96            intra_chunk_states_bnhpr.dims()
97        );
98
99        // ── Kernel 4 ──────────────────────────────────────────────────────────────────
100        // IO: (..) -> (chunk_input_states_bnhpr [used in K5][!], final_ssm_state_bhpr [final output])
101        let flat_state_dim = per_head_dim * state_rank;
102        let (chunk_input_states_bnhpr, final_ssm_state_bhpr): (Tensor<B, 5>, Tensor<B, 4>) =
103            k4_ssd_state_passing(
104                intra_chunk_states_bnhpr.clone(),
105                da_chunk_end_bhn.clone(),
106                ssm_initial_state_bhpr,
107            );
108        assert_eq!(
109            [batch, nchunks, nheads, per_head_dim, state_rank],
110            chunk_input_states_bnhpr.dims()
111        );
112        assert_eq!(
113            [batch, nheads, per_head_dim, state_rank],
114            final_ssm_state_bhpr.dims()
115        );
116
117        // ── Kernel 5 ──────────────────────────────────────────────────────────────────
118        let y_bnlhp: Tensor<B, 5> = k5_ssd_chunk_scan(
119            da_cumsum_bhnl,
120            dt_discretized_bhnl,
121            x_bnlhp,
122            c_bnlgr,
123            cb_bngll,
124            chunk_input_states_bnhpr,
125            d_h,
126        );
127        assert_eq!(
128            [batch, nchunks, chunk_len, nheads, per_head_dim],
129            y_bnlhp.dims()
130        );
131
132        // ── Reshape into sequence ─────────────────────────────────────────────────────
133        let y_bshp = y_bnlhp.reshape([batch, sequence, nheads, per_head_dim]);
134
135        (y_bshp, final_ssm_state_bhpr)
136    }
137}

139/// Based on the Kernel 1 Triton reference `_chunk_cumsum_fwd_kernel` (`ssd_chunk_state.py`).
140///
141/// Returns:
142/// - da_cumsum_bhnl [used in K3+K5][*] - intra-chunk cumsum.
143/// - da_chunk_end_bhn [used in K4][omitted][*] - last da_cumsum per chunk.
144pub fn k1_ssd_chunk_cumsum<B: Backend>(
145    dt_discretized_bhnl: Tensor<B, 4>,
146    a_decay_h: Tensor<B, 1>,
147) -> (Tensor<B, 4>, Tensor<B, 3>) {
148    let [batch, nheads, nchunks, chunk_len] = dt_discretized_bhnl.dims();
149    let da_cumsum_bhnl: Tensor<B, 4> = {
150        let a_decay_bhnl = a_decay_h
151            // - 1/6: unsqueeze-dims: (a_decay_h [*]) -> (a_decay_1h11)
152            .unsqueeze_dims::<4>(&[0, 2, 3]) // a_decay_1h11
153            // - 2: expand: (a_decay_1h11) -> (a_decay_bhnl)
154            .expand([batch, nheads, nchunks, chunk_len]);
155        // - 3: mul: (dt_discretized_bhnl [*], a_decay_bhnl) -> (da_bhnl)
156        // - 4: cumsum: (da_bhnl) -> (da_cumsum_bhnl [out][*])
157        (dt_discretized_bhnl * a_decay_bhnl).cumsum(3)
158    };
159    assert_eq!([batch, nheads, nchunks, chunk_len], da_cumsum_bhnl.dims());
160
161    let da_chunk_end_bhn = da_cumsum_bhnl
162        .clone()
163        // - 5: slice: (da_cumsum_bhnl [*]) -> (da_cumsum_bhn1)
164        .slice(s![.., .., .., -1]) // da_cumsum_bhn1
165        // - 6/6: squeeze: (da_cumsum_bhn1) -> (da_chunk_end_bhn [out])
166        .squeeze_dim::<3>(3);
167    assert_eq!([batch, nheads, nchunks], da_chunk_end_bhn.dims());
168
169    (da_cumsum_bhnl, da_chunk_end_bhn)
170}

172/// Based on the Kernel 2 Triton reference `_bmm_chunk_fwd_kernel` (`ssd_bmm.py`).
173///
174/// Returns:
175/// - cb_bngll [used in K5][!].
176pub fn k2_ssd_bmm<B: Backend>(c_bnlgr: Tensor<B, 5>, b_bnlgr: Tensor<B, 5>) -> Tensor<B, 5> {
177    let [batch, nchunks, chunk_len, ngroups, state_rank] = c_bnlgr.dims();
178
179    // - 1/3: permute: (c_bnlgr [in][*]) -> (c_bnglr)
180    let c_bnglr = c_bnlgr.clone().permute([0, 1, 3, 2, 4]);
181    // - 2: permute: (b_bnlgr [in][*]) -> (b_bngrl)
182    let b_bngrl = b_bnlgr.clone().permute([0, 1, 3, 4, 2]);
183    // - 3/3: matmul: (c_bnglr, b_bngrl) -> (cb_bngll [out][!])
184    let cb_bngll: Tensor<B, 5> = c_bnglr.matmul(b_bngrl);
185    assert_eq!(
186        [batch, nchunks, ngroups, chunk_len, chunk_len],
187        cb_bngll.dims()
188    );
189    // Note: cb_bngll is then only used by Kernel 5.
190    cb_bngll
191}

193/// Based on the Kernel 3 Triton reference `_chunk_state_fwd_kernel` (`ssd_chunk_state.py`).
194///
195/// Returns:
196/// - cb_bngll [used in K5][!] - state assuming zero initial state at each chunk boundary.
197pub fn k3_ssd_chunk_state<B: Backend>(
198    x_bnlhp: Tensor<B, 5>,
199    b_bnlgr: Tensor<B, 5>,
200    da_cumsum_bhnl: Tensor<B, 4>,
201    dt_discretized_bhnl: Tensor<B, 4>,
202) -> Tensor<B, 5> {
203    use burn::tensor::s;
204
205    let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
206    let [_batch, _nchunks, _chunk_len, ngroups, state_rank] = b_bnlgr.dims();
207
208    // permute b and x to prepare them for the mamtul
209    // - 1/15: permute: (x_bnlhp [in][*]) -> (x_bnhpl)
210    let x_bnhpl = x_bnlhp.clone().permute([0, 1, 3, 4, 2]);
211    assert_eq!(
212        [batch, nchunks, nheads, per_head_dim, chunk_len],
213        x_bnhpl.dims()
214    );
215    // - 2: permute: (b_bnlgr [in][*]) -> (b_bnglr)
216    let b_bnglr = b_bnlgr.permute([0, 1, 3, 2, 4]); // note: still in groups instead of heads
217    assert_eq!(
218        [batch, nchunks, ngroups, chunk_len, state_rank],
219        b_bnglr.dims()
220    );
221
222    // Expand B from ngroups to nheads by repeating each group's
223    // projection across all heads_per_group heads in that group.
224    let heads_per_group = nheads / ngroups;
225    let b_bnhlr = b_bnglr
226        // - 3: unsqueeze: (b_bnglr) -> (b_bng1lr)
227        .unsqueeze_dim::<6>(3) // b_bng1lr
228        // - 4: expand: (b_bng1lr) -> (b_bngHlr)
229        .expand([
230            batch,
231            nchunks,
232            ngroups,
233            heads_per_group,
234            chunk_len,
235            state_rank,
236        ]) // b_bngHlr
237        // - 5: reshape: (b_bngHlr) -> (b_bnhlr)
238        .reshape([batch, nchunks, nheads, chunk_len, state_rank]);
239
240    // scale b
241    let b_scaled_bnhlr = {
242        let b_bar_scale_bhnl = {
243            let da_cumsum_last_in_chunk_bhn1 =
244                // - 6: slice: (da_cumsum_bhnl [in][*]) -> (da_cumsum_last_in_chunk_bhn1)
245                da_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
246            assert_eq!(
247                [batch, nheads, nchunks, 1],
248                da_cumsum_last_in_chunk_bhn1.dims()
249            );
250
251            let forward_decay_to_chunk_end_bhnl = (da_cumsum_last_in_chunk_bhn1
252            // - 7: expand: (da_cumsum_last_in_chunk_bhn1) -> (da_cumsum_last_bhnl)
253                .expand([batch, nheads, nchunks, chunk_len]) // da_cumsum_last_bhnl
254            // - 8: sub: (da_cumsum_last_bhnl, da_cumsum_bhnl [from K1][*]) -> (da_delta_bhnl)
255                - da_cumsum_bhnl.clone()) // da_delta_bhnl
256            // - 9: exp: (da_delta_bhnl) -> (forward_decay_to_chunk_end_bhnl)
257            .exp();
258            assert_eq!(
259                [batch, nheads, nchunks, chunk_len],
260                forward_decay_to_chunk_end_bhnl.dims()
261            );
262
263            // - 10: mul: (forward_decay_to_chunk_end_bhnl, dt_discretized_bhnl [in][*]) -> (b_bar_scale_bhnl)
264            forward_decay_to_chunk_end_bhnl * dt_discretized_bhnl.clone()
265        };
266        assert_eq!([batch, nheads, nchunks, chunk_len], b_bar_scale_bhnl.dims());
267
268        // - 11: permute: (b_bar_scale_bhnl) -> (b_bar_scale_bnhl)
269        let b_bar_scale_bnhl = b_bar_scale_bhnl.permute([0, 2, 1, 3]);
270        assert_eq!([batch, nchunks, nheads, chunk_len], b_bar_scale_bnhl.dims());
271        let b_bar_scale_bnhlr = b_bar_scale_bnhl
272            // - 12: unsqueeze: (b_bar_scale_bnhl) -> (b_bar_scale_bnhl1)
273            .unsqueeze_dim::<5>(4) // b_bar_scale_bnhl1
274            // - 13: expand: (b_bar_scale_bnhl1) -> (b_bar_scale_bnhlr)
275            .expand([batch, nchunks, nheads, chunk_len, state_rank]);
276        // - 14: mul: (b_bnhlr, b_bar_scale_bnhlr) -> (b_bar_scale_bnhlr)
277        b_bnhlr * b_bar_scale_bnhlr
278    };
279    assert_eq!(
280        [batch, nchunks, nheads, chunk_len, state_rank],
281        b_scaled_bnhlr.dims()
282    );
283
284    // - 15/15: matmul: (x_bnhpl, b_scaled_bnhlr) -> (intra_chunk_states_bnhpr [out][!])
285    let intra_chunk_states_bnhpr: Tensor<B, 5> = x_bnhpl.matmul(b_scaled_bnhlr);
286    assert_eq!(
287        [batch, nchunks, nheads, per_head_dim, state_rank],
288        intra_chunk_states_bnhpr.dims()
289    );
290    intra_chunk_states_bnhpr
291}

293/// Based on the Kernel 4 Triton reference `_state_passing_fwd_kernel` (`ssd_state_passing.py`).
294///
295/// Returns:
296/// - chunk_input_states_bnhpr [used in K5][!].
297/// - final_ssm_state_bhpr [final output].
298pub fn k4_ssd_state_passing<B: Backend>(
299    intra_chunk_states_bnhpr: Tensor<B, 5>,
300    da_chunk_end_bhn: Tensor<B, 3>,
301    ssm_initial_state_bhpr: Tensor<B, 4>,
302) -> (Tensor<B, 5>, Tensor<B, 4>) {
303    let [batch, nchunks, nheads, per_head_dim, state_rank] = intra_chunk_states_bnhpr.dims();
304    let flat_state_dim = per_head_dim * state_rank;
305
306    // - 1/5: init-mut: (ssm_initial_state_bhpr [in][*]) -> (running_state_bhpr)
307    let mut running_state_bhpr = ssm_initial_state_bhpr;
308    assert_eq!(
309        [batch, nheads, per_head_dim, state_rank],
310        running_state_bhpr.dims()
311    );
312
313    let mut chunk_input_states_vec_bhpr = Vec::with_capacity(nchunks + 1);
314    // - 2: vec-push: (running_state_bhpr [elem]) -> (chunk_input_states_vec_bhpr [vec][!])
315    chunk_input_states_vec_bhpr.push(running_state_bhpr.clone());
316
317    // - 3: serial-loop: (0..nchunks)
318    for i_chunk in 0..nchunks {
319        let intra_state_bhpr = intra_chunk_states_bnhpr
320            .clone()
321            //   - 3.1/3.9: slice: (intra_chunk_states_bnhpr [in][!]) -> (intra_chunk_states_b1hpr)
322            .slice(s![.., i_chunk, .., .., ..]) // intra_chunk_states_b1hpr
323            //   - 3.2: squeeze: (intra_chunk_states_b1hpr) -> (intra_state_bhpr)
324            .squeeze_dim::<4>(1);
325        assert_eq!(
326            [batch, nheads, per_head_dim, state_rank],
327            intra_state_bhpr.dims()
328        );
329
330        let decay_bhpr = da_chunk_end_bhn
331            .clone()
332            //   - 3.3: slice: (da_chunk_end_bhn [in][*]) -> (da_chunk_end_bh1)
333            .slice(s![.., .., i_chunk]) // da_chunk_end_bh1
334            //   - 3.4: exp: (da_chunk_end_bh1) -> (da_chunk_end_bh1)
335            .exp() // da_chunk_end_bh1
336            //   - 3.5: unsqueeze: (da_chunk_end_bh1) -> (da_chunk_end_bh11)
337            .unsqueeze_dim::<4>(3) // da_chunk_end_bh11
338            //   - 3.6: expand: (da_chunk_end_bh11) -> (decay_bhpr)
339            .expand([batch, nheads, per_head_dim, state_rank]);
340
341        // SSM recurrence: running_state = decay * running_state + intra_state
342        running_state_bhpr =
343        //   - 3.7: mul: (decay_bhpr, running_state_bhpr) -> (running_state_bhpr)
344            decay_bhpr * running_state_bhpr // running_state_bhpr
345        //   - 3.8: add: (running_state_bhpr, intra_state_bhpr) -> (running_state_bhpr)
346            + intra_state_bhpr;
347        //   - 3.9/3.9: vec-push: (running_state_bhpr [elem]) -> (chunk_input_states_vec_bhpr [vec][!])
348        chunk_input_states_vec_bhpr.push(running_state_bhpr.clone());
349    }
350
351    // - 4: vec-pop: (chunk_input_states_vec_bhpr [vec][!]) -> (final_ssm_state_bhpr [elem][out][!])
352    let final_ssm_state_bhpr = chunk_input_states_vec_bhpr.pop().unwrap();
353    assert_eq!(
354        [batch, nheads, per_head_dim, state_rank],
355        final_ssm_state_bhpr.dims()
356    );
357
358    // - 5/5: stack: (chunk_input_states_vec_bhpr [!]) -> (chunk_input_states_bnhpr [out][!])
359    let chunk_input_states_bnhpr = Tensor::stack(chunk_input_states_vec_bhpr, 1);
360    assert_eq!(
361        [batch, nchunks, nheads, per_head_dim, state_rank],
362        chunk_input_states_bnhpr.dims()
363    );
364
365    (chunk_input_states_bnhpr, final_ssm_state_bhpr)
366}

368/// Based on the Kernel 5 Triton reference `_chunk_scan_fwd_kernel` (`ssd_chunk_scan.py`).
369///
370/// Returns:
371/// - y_bnlhp [final output]
372pub fn k5_ssd_chunk_scan<B: Backend>(
373    da_cumsum_bhnl: Tensor<B, 4>,
374    dt_discretized_bhnl: Tensor<B, 4>,
375    x_bnlhp: Tensor<B, 5>,
376    c_bnlgr: Tensor<B, 5>,
377    cb_bngll: Tensor<B, 5>,
378    chunk_input_states_bnhpr: Tensor<B, 5>,
379    d_h: Tensor<B, 1>,
380) -> Tensor<B, 5> {
381    let [batch, nchunks, chunk_len, nheads, per_head_dim] = x_bnlhp.dims();
382    let [_b, _n, _l, ngroups, state_rank] = c_bnlgr.dims();
383    let heads_per_group = nheads / ngroups;
384    let device = x_bnlhp.device();
385
386    // Rearrange inputs to the common [batch, nchunks, nheads, ...] ordering used below.
387    // - 1/36: permute: (da_cumsum_bhnl [*]) -> (da_cumsum_bnhl)
388    let da_cumsum_bnhl = da_cumsum_bhnl.permute([0, 2, 1, 3]);
389    // - 2: permute: (dt_discretized_bhnl [*]) -> (dt_bnhl)
390    let dt_bnhl = dt_discretized_bhnl.permute([0, 2, 1, 3]);
391    // - 3: permute: (x_bnlhp [*]) -> (x_bnhlp)
392    let x_bnhlp = x_bnlhp.clone().permute([0, 1, 3, 2, 4]);
393
394    // GQA: expand C  [b,n,l,g,r] → [b,n,h,l,r].
395    let c_bnhlr = c_bnlgr
396        // - 4: unsqueeze: (c_bnlgr [*]) -> (c_bnlg1r)
397        .unsqueeze_dim::<6>(4) // c_bnlg1r
398        // - 5: expand: (c_bnlg1r) -> (c_bnlgHr)
399        .expand([
400            batch,
401            nchunks,
402            chunk_len,
403            ngroups,
404            heads_per_group,
405            state_rank,
406        ]) // c_bnlgHr
407        // - 6: reshape: (c_bnlgHr) -> (c_bnlhr)
408        .reshape([batch, nchunks, chunk_len, nheads, state_rank]) // c_bnlhr
409        // - 7: permute: (c_bnlhr) -> (c_bnhlr)
410        .permute([0, 1, 3, 2, 4]);
411
412    // GQA: expand CB [b,n,g,l,l] → [b,n,h,l,l].
413    let cb_bnhll = cb_bngll
414        // - 8: unsqueeze: (cb_bngll [!]) -> (cb_bng1ll)
415        .unsqueeze_dim::<6>(3) // cb_bng1ll
416        // - 9: expand: (cb_bng1ll) -> (cb_bngHll)
417        .expand([
418            batch,
419            nchunks,
420            ngroups,
421            heads_per_group,
422            chunk_len,
423            chunk_len,
424        ]) // cb_bngHll
425        // - 10: reshape: (cb_bngHll) -> (cb_bnhll)
426        .reshape([batch, nchunks, nheads, chunk_len, chunk_len]);
427
428    // ── BLUE: exp(dA[l]) · C[l,:] @ state_in^T ─────────────────────────────
429    //
430    //   blue[b,n,h,l,p] = exp(da[b,n,h,l]) · Σ_r  c[b,n,h,l,r] · state[b,n,h,p,r]
431    //
432    //   [b,n,h,l,r] @ [b,n,h,r,p]  →  [b,n,h,l,p]
433    let da_cumsum_bnhlp = da_cumsum_bnhl
434        .clone()
435        // - 11: exp: (da_cumsum_bnhl) -> (da_cumsum_bnhl)
436        .exp()
437        // - 12: unsqueeze: (da_cumsum_bnhl) -> (da_cumsum_bnhl1)
438        .unsqueeze_dim::<5>(4) // da_cumsum_bnhl1
439        // - 13: expand: (da_cumsum_bnhl1) -> (da_cumsum_bnhlp)
440        .expand([batch, nchunks, nheads, chunk_len, per_head_dim]);
441    // - 14: permute: (chunk_input_states_bnhpr [!]) -> (chunk_input_states_bnhrp)
442    let chunk_input_states_bnhrp = chunk_input_states_bnhpr.permute([0, 1, 2, 4, 3]);
443    // - 15: matmul: (c_bnhlr, chunk_input_states_bnhrp) -> (blue_bnhlp)
444    let blue_bnhlp = c_bnhlr
445        .matmul(chunk_input_states_bnhrp)  // blue_bnhlp
446        // - 16: mul: (blue_bnhlp, da_cumsum_bnhlp) -> (blue_bnhlp)
447        * da_cumsum_bnhlp;
448
449    // ── ORANGE: causal CB_weighted @ X ──────────────────────────────────────
450    //
451    //   orange[b,n,h,l,p] = Σ_{s≤l} CB[l,s] · exp(da[l]-da[s]) · dt[s] · x[s,p]
452    //
453    // Precompute the full lower-triangular weight matrix, then do a single matmul.
454    let da_cumsum_target_bnhll = da_cumsum_bnhl
455        .clone()
456        // - 17: unsqueeze: (da_cumsum_bnhl) -> (da_cumsum_bnhl1)
457        .unsqueeze_dim::<5>(4) // da_cumsum_bnhl1
458        // - 18: expand: (da_cumsum_bnhl1) -> (da_cumsum_target_bnhll)
459        .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
460    let da_cumsum_source_bnhll = da_cumsum_bnhl
461        // - 19: unsqueeze: (da_cumsum_bnhl) -> (da_cumsum_bnh1l)
462        .unsqueeze_dim::<5>(3) // da_cumsum_bnh1l
463        // - 20: expand: (da_cumsum_bnh1l) -> (da_cumsum_source_bnhll)
464        .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
465    let da_cumsum_diff_exp =
466        // - 21: sub: (da_cumsum_target_bnhll, da_cumsum_source_bnhll) -> (da_cumsum_diff)
467        (da_cumsum_target_bnhll - da_cumsum_source_bnhll) // da_cumsum_diff
468            // - 22: exp: (da_cumsum_diff) -> (da_cumsum_diff_exp)
469        .exp();
470    let dt_source_bnhll = dt_bnhl
471        // - 23: unsqueeze: (dt_bnhl) -> (dt_bnh1l)
472        .unsqueeze_dim::<5>(3) // dt_bnh1l
473        // - 24: expand: (dt_bnh1l) -> (dt_source_bnhll)
474        .expand([batch, nchunks, nheads, chunk_len, chunk_len]);
475
476    // Causal mask (0 above the main diagonal, 1 elsewhere).
477    // - 25: ones: (1) -> (causal_mask_bnhll)
478    let causal_mask_bnhll = Tensor::ones([batch, nchunks, nheads, chunk_len, chunk_len], &device)
479        // - 26: tril: (causal_mask_bnhll, 0) -> (causal_mask_bnhll)
480        .tril(0);
481
482    //   [b,n,h,l,l] @ [b,n,h,l,p]  →  [b,n,h,l,p]
483    let orange_bnhlp = (cb_bnhll
484        // - 27: mul: (cb_bnhll, da_cumsum_diff_exp) -> (orange_lhs_bnhll)
485        * da_cumsum_diff_exp // orange_lhs_bnhll
486        // - 28: mul: (orange_lhs_bnhll, dt_source_bnhll) -> (orange_lhs_bnhll)
487        * dt_source_bnhll
488        // - 29: mul: (orange_lhs_bnhll, causal_mask_bnhll) -> (orange_lhs_bnhll)
489        * causal_mask_bnhll)
490        // - 30: matmul: (orange_lhs_bnhll, x_bnhlp) -> (orange_bnhlp)
491        .matmul(x_bnhlp);
492
493    // ── SKIP: D[h] · x[l,p] ─────────────────────────────────────────────────
494    //
495    //   D_HAS_HDIM = False: D is a scalar per head, shape [nheads].
496    //   Triton: `acc += x_residual * D`
497    let skip_bnlhp = d_h
498        // - 31: unsqueeze-dims: (d_h [*]) -> (d_111h1)
499        .unsqueeze_dims::<5>(&[0, 1, 2, 4]) // d_111h1
500        // - 32: expand: (d_111h1) -> (d_bnlhp)
501        .expand([
502            batch,
503            nchunks,
504            chunk_len,
505            nheads,
506            per_head_dim,
507        ]) // d_bnlhp
508    // - 33: mul: (d_bnlhp, x_bnlhp[*]) -> (skip_bnlhp)
509    * x_bnlhp;
510
511    // Permute BLUE + ORANGE from [b,n,h,l,p] back to [b,n,l,h,p], then add SKIP.
512    let y_bnlhp: Tensor<B, 5> =
513        // - 34: add: (blue_bnhlp, orange_bnhlp) -> (y_partial_bnhlp)
514        (blue_bnhlp + orange_bnhlp) // y_partial_bnhlp
515            // - 35: permute: (y_partial_bnhlp) -> (y_partial_bnlhp)
516            .permute([0, 1, 3, 2, 4]) // y_partial_bnlhp
517        // - 36/36: add: (y_partial_bnlhp, skip_bnlhp) -> (y_bnlhp [out])
518        + skip_bnlhp; // y_bnlhp
519
520    assert_eq!(
521        [batch, nchunks, chunk_len, nheads, per_head_dim],
522        y_bnlhp.dims()
523    );
524    y_bnlhp
