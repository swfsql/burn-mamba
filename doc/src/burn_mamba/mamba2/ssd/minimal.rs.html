1//! ## The Chunkwise SSD Algorithm
2//!
3//! During training (and prefill), a naive sequential recurrence cannot
4//! exploit GPU tensor cores.  The **chunkwise SSD algorithm** (§4 of the
5//! paper) achieves this by splitting the sequence into chunks of length Q
6//! and decomposing the computation into four steps:
7//!
8//! ```text
9//!   Step 1  (intra-chunk, quadratic form)   →  Y_diag
10//!   Step 2  (input → chunk state)           →  states_bnhpr
11//!   Step 3  (inter-chunk state scan)        →  states_bnhpr, final_state
12//!   Step 4  (chunk state → output)          →  Y_off
13//!
14//!   Y = Y_diag + Y_off
15//! ```
16//!
17//! Steps 1, 2, 4 are fully parallel across chunks and use batched matrix
18//! multiplications (exploiting tensor cores).  Step 3 is a short sequential
19//! scan over `T/Q` elements rather than `T`.
20
21use crate::mamba2::prelude::*;
22use burn::prelude::*;

29    /// Minimal chunkwise SSD algorithm.
30    ///
31    /// Implements the four-step decomposition of the semiseparable matrix
32    /// multiplication described in §4 of the paper.  The sequence of length T
33    /// is split into `nchunks = ⌈T/Q⌉` chunks of length Q.
34    ///
35    /// ## The four steps
36    ///
37    /// ### Step 1 — Intra-chunk outputs (Y_diag)
38    ///
39    /// Within each chunk, compute the output assuming the initial hidden state
40    /// is zero.  This is the *quadratic attention form* of the SSD layer
41    /// restricted to a window of Q tokens (§4.1):
42    ///
43    /// ```text
44    ///   Y_diag[n] = (L[n] ∘ C[n] B[n]ᵀ) · X[n]
45    /// ```
46    ///
47    /// where `L[n]` is the Q×Q 1-semiseparable mask for chunk n.
48    /// This step is a batched GEMM (exploits tensor cores).
49    ///
50    /// ### Step 2 — Chunk states (states_bnhpr)
51    ///
52    /// Compute the final SSM state of each chunk assuming zero initial state
53    /// (§4.1, Eq. 20):
54    ///
55    /// ```text
56    ///   s[n] = Σ_{t ∈ chunk n}  exp(A_cum[end] - A_cum[t]) · B̄[t] · x[t]ᵀ
57    /// ```
58    ///
59    /// This is also a batched GEMM and is fully parallel across chunks.
60    ///
61    /// ### Step 3 — Inter-chunk state scan (state passing)
62    ///
63    /// Propagate the true hidden state across chunk boundaries using the
64    /// recurrence (§4.1, Eq. 22):
65    ///
66    /// ```text
67    ///   h[n] = Ā[n]_chunk · h[n-1] + s[n]
68    /// ```
69    ///
70    /// where `Ā[n]_chunk = exp(Σ_{t ∈ chunk n} Δₜ · A)` is the cumulative
71    /// decay over the whole chunk.  This step is implemented as a single
72    /// batched matrix multiplication using the 1-semiseparable structure of
73    /// the inter-chunk decay matrix (same `segsum` trick, now over chunks).
74    /// The scan has length `nchunks = T/Q` rather than T, so its cost is
75    /// negligible for typical chunk sizes.
76    ///
77    /// ### Step 4 — State-to-output (Y_off)
78    ///
79    /// For each chunk n, compute the contribution of the true initial state
80    /// `h[n-1]` to the outputs within that chunk (§4.1, Eq. 23):
81    ///
82    /// ```text
83    ///   Y_off[n, t] = C[n, t]ᵀ · exp(A_cum[t]) · h[n-1]
84    /// ```
85    ///
86    /// This is again a batched GEMM.
87    ///
88    /// ### Final output (with D skip-connection)
89    ///
90    /// ```text
91    ///   Y = Y_diag + Y_off + D · X
92    /// ```
93    ///
94    /// # Shapes
95    /// - `x_bshp`                  : `[batch, sequence, nheads, per_head_dim]`
96    /// - `dt_bsh`     (Δ)          : `[batch, sequence, nheads]`
97    /// - `a_head_decay_h` (A < 0)  : `[nheads]`
98    /// - `b_bsgr`     (B)          : `[batch, sequence, ngroups, state_rank]`
99    /// - `c_bsgr`     (C)          : `[batch, sequence, ngroups, state_rank]`
100    /// - `d_h`        (D)          : `[nheads]`
101    /// - `ssm_initial_state_bhpr`  : `[batch, nheads, per_head_dim, state_rank]`
102    /// - `init_states_hpr`         : `[nheads, per_head_dim, state_rank]`
103    /// - output.0 `y_bshp`         : `[batch, sequence, nheads, per_head_dim]`
104    /// - output.1 `final_state_bhpr`: `[batch, nheads, per_head_dim, state_rank]`
105    #[allow(non_snake_case)]
106    pub fn ssd_minimal(
107        x_bshp: Tensor<B, 4>,
108        dt_bsh: Tensor<B, 3>,
109        a_head_decay_h: Tensor<B, 1>,
110        b_bsgr: Tensor<B, 4>,
111        c_bsgr: Tensor<B, 4>,
112        d_h: Tensor<B, 1>,
113        ssm_initial_state_bhpr: Tensor<B, 4>,
114        init_states_hpr: Option<Tensor<B, 3>>,
115        ngroups: usize,
116        state_rank: usize,
117        chunk_len: usize,
118    ) -> (Tensor<B, 4>, Tensor<B, 4>) {
119        let [batch, sequence, nheads, per_head_dim] = x_bshp.dims();
120        let device = &x_bshp.device();
121
122        assert_eq!(nheads % ngroups, 0);
123        assert!(sequence >= 1, "sequence must be non-empty");
124        assert!(chunk_len > 0, "chunk_len must be positive");
125
126        assert_eq!(sequence % chunk_len, 0);
127        let nchunks = sequence / chunk_len;
128
129        // ── Compute discretised parameters ────────────────────────────────────
130        // Ā = exp(Δ · A)   stored in log-space as  a_bsh = Δ · A  (negative)
131        // B̄ = Δ · B        (Euler/ZOH approximation)
132
133        // Expand B and C from ngroups to nheads by repeating each group's
134        // projection across all heads_per_group heads in that group.
135        let heads_per_group = nheads / ngroups;
136
137        // b_bsgr → b_bshr  [batch, sequence, nheads, state_rank]
138        let b_bshr = b_bsgr
139            .clone()
140            .unsqueeze_dim::<5>(3) // [B, T, G, 1, N]
141            .expand([batch, sequence, ngroups, heads_per_group, state_rank])
142            .reshape([batch, sequence, nheads, state_rank]);
143
144        // c_bsgr → c_bshr  [batch, sequence, nheads, state_rank]
145        let c_bshr = c_bsgr
146            .clone()
147            .unsqueeze_dim::<5>(3) // [B, T, G, 1, N]
148            .expand([batch, sequence, ngroups, heads_per_group, state_rank])
149            .reshape([batch, sequence, nheads, state_rank]);
150
151        // B̄ₜ = Δₜ · Bₜ   [batch, sequence, nheads, state_rank]
152        let delta_b_bshr = dt_bsh.clone().unsqueeze_dim(3) * b_bshr;
153        assert_eq!([batch, sequence, nheads, state_rank], delta_b_bshr.dims());
154
155        // Ā in log-space: a_bsh = Δₜ · A  (negative scalar per [B, T, H])
156        let a_bsh = dt_bsh.clone()
157            * a_head_decay_h
158                .clone()
159                .unsqueeze_dims::<3>(&[0, 1]) // [H] → [1, 1, H]
160                .expand([batch, sequence, nheads]);
161
162        // ── Reshape into chunks ───────────────────────────────────────────────
163        // All tensors are reshaped from [B, T, ...] to [B, nchunks, chunk_len, ...]
164        // (denoted `n` and `l` in the suffix convention).
165
166        // x: [B, T, H, P] → [B, nchunks, chunk_len, H, P]
167        let x_bnlhp = x_bshp
168            .clone()
169            .reshape([batch, nchunks, chunk_len, nheads, per_head_dim]);
170        // a (log-decay): [B, T, H] → [B, nchunks, chunk_len, H] → [B, H, nchunks, chunk_len]
171        let a_bnlh = a_bsh.reshape([batch, nchunks, chunk_len, nheads]);
172        let a_bhnl = a_bnlh.permute([0, 3, 1, 2]);
173        assert_eq!([batch, nheads, nchunks, chunk_len], a_bhnl.dims());
174
175        // B̄: [B, T, H, N] → [B, nchunks, chunk_len, H, N]
176        let b_bnlhr = delta_b_bshr.reshape([batch, nchunks, chunk_len, nheads, state_rank]);
177        // C: [B, T, H, N] → [B, nchunks, chunk_len, H, N]
178        let c_bnlhr = c_bshr
179            .clone()
180            .reshape([batch, nchunks, chunk_len, nheads, state_rank]);
181
182        // Cumulative sum of log-decays within each chunk.
183        // a_cumsum_bhnl[b, h, n, t] = Σ_{k=0..t} Δ_{n,k} · A
184        // This is the log of the cumulative decay factor from the start of the
185        // chunk to position t (inclusive).
186        let a_cumsum_bhnl = a_bhnl.clone().cumsum(3);
187        assert_eq!([batch, nheads, nchunks, chunk_len], a_cumsum_bhnl.dims());
188
189        // =============================================================
190        // STEP 1: Intra-chunk outputs (diagonal blocks, Y_diag)
191        // =============================================================
192        //
193        // For each chunk n, compute Y_diag[n] = (L[n] ∘ C[n] B[n]ᵀ) · X[n]
194        // where L[n] ∈ ℝ^{Q×Q} is the 1-semiseparable mask for the chunk.
195        //
196        // L[n]_{i,j} = exp(Σ_{k=j+1..i} a_{n,k})  for i ≥ j
197        //            = exp(a_cumsum[n,i] - a_cumsum[n,j])   (using segsum trick)
198        //
199        // Implementation uses three batched matmuls:
200        //   (a) C[n] · B[n]ᵀ  (contract over state_rank N)  → temp1 [B, nchunks, H, Q, Q]
201        //   (b) temp1 ∘ L[n]                                 → temp2 [B, nchunks, H, Q, Q]
202        //   (c) temp2 · X[n]  (contract over Q)              → Y_diag [B, nchunks, Q, H, P]
203        let y_diag_bnlhp = {
204            // Permute to [B, nchunks, H, Q, N] for the matmul along Q and N.
205            let b_bnhlr = b_bnlhr.clone().permute([0, 1, 3, 2, 4]);
206            let c_bnhlr = c_bnlhr.clone().permute([0, 1, 3, 2, 4]);
207            assert_eq!(
208                [batch, nchunks, nheads, chunk_len, state_rank],
209                b_bnhlr.dims()
210            );
211            assert_eq!(
212                [batch, nchunks, nheads, chunk_len, state_rank],
213                c_bnhlr.dims()
214            );
215
216            // (a) C[n] · B[n]ᵀ → [B, nchunks, H, Q, Q]
217            //     Contracts over state_rank N.
218            let b_bnhrl = b_bnhlr.permute([0, 1, 2, 4, 3]); // [B, n, H, N, Q]
219            let cb_bnhll = c_bnhlr.matmul(b_bnhrl); // [B, n, H, Q, Q]
220            assert_eq!(
221                [batch, nchunks, nheads, chunk_len, chunk_len],
222                cb_bnhll.dims()
223            );
224
225            // (b) Element-wise multiply with the 1-SS mask L.
226            //     L = exp(segsum(a_bhnl))  [B, H, nchunks, Q, Q]
227            //     Lᵢⱼ = exp(a_cumsum[n,i] - a_cumsum[n,j])  (Eq. 4–5)
228            let l_bhnll = segsum(a_bhnl.clone()).exp();
229            assert_eq!(
230                [batch, nheads, nchunks, chunk_len, chunk_len],
231                l_bhnll.dims()
232            );
233
234            // Permute both to [B, n, Q, H, Q] for the broadcast multiply.
235            let cb_bnlhl = cb_bnhll.permute([0, 1, 3, 2, 4]);
236            assert_eq!(
237                [batch, nchunks, chunk_len, nheads, chunk_len],
238                cb_bnlhl.dims()
239            );
240            let l_bnlhl = l_bhnll.permute([0, 2, 3, 1, 4]);
241            assert_eq!(
242                [batch, nchunks, chunk_len, nheads, chunk_len],
243                l_bnlhl.dims()
244            );
245            let masked_cb_bnlhl = cb_bnlhl * l_bnlhl;
246
247            // (c) masked_CB · X → Y_diag.
248            //     Contract over the last Q dimension.
249            let masked_cb_bnhll = masked_cb_bnlhl.permute([0, 1, 3, 2, 4]);
250            assert_eq!(
251                [batch, nchunks, nheads, chunk_len, chunk_len],
252                masked_cb_bnhll.dims()
253            );
254
255            let x_bnhlp = x_bnlhp.clone().permute([0, 1, 3, 2, 4]); // [B, n, H, Q, P]
256            assert_eq!(
257                [batch, nchunks, nheads, chunk_len, per_head_dim],
258                x_bnhlp.dims()
259            );
260
261            let y_diag_bnhlp = masked_cb_bnhll.matmul(x_bnhlp);
262            assert_eq!(
263                [batch, nchunks, nheads, chunk_len, per_head_dim],
264                y_diag_bnhlp.dims()
265            );
266
267            y_diag_bnhlp.permute([0, 1, 3, 2, 4]) // → [B, n, Q, H, P]
268        };
269        assert_eq!(
270            [batch, nchunks, chunk_len, nheads, per_head_dim],
271            y_diag_bnlhp.dims()
272        );
273
274        // =============================================================
275        // STEP 2: Compute chunk states (input → state)
276        // =============================================================
277        //
278        // For each chunk n, compute the SSM state at the end of the chunk
279        // assuming the initial state is zero (Eq. 20):
280        //
281        //   s[n] = Σ_{t ∈ [0, Q)} exp(a_cumsum[n,-1] - a_cumsum[n,t]) · B̄[n,t] · x[n,t]ᵀ
282        //
283        // Equivalently:
284        //   decay_states[n, t] = exp(a_cum_last[n] - a_cum[n, t])
285        //   s[n] = Σ_t  decay_states[n, t] · x[n, t]ᵀ · B̄[n, t]     (outer product over P and N)
286        //
287        // This is a batched GEMM, fully parallel across n and b.
288        let states_bnhpr = {
289            // Decay from each position t to the end of the chunk:
290            //   decay_states[n, t] = exp(a_cum[n, Q-1] - a_cum[n, t])
291            let a_cumsum_last_bhn1 = a_cumsum_bhnl.clone().slice(s![.., .., .., -1]);
292            assert_eq!([batch, nheads, nchunks, 1], a_cumsum_last_bhn1.dims());
293
294            let decay_states_bhnl = (a_cumsum_last_bhn1 - a_cumsum_bhnl.clone()).exp();
295            assert_eq!(
296                [batch, nheads, nchunks, chunk_len],
297                decay_states_bhnl.dims()
298            );
299
300            // Multiply decay into x: decay[n, t] · x[n, t]  → [B, n, Q, H, P]
301            let decay_states_bnlh1 = decay_states_bhnl.permute([0, 2, 3, 1]).unsqueeze_dim(4);
302            assert_eq!(
303                [batch, nchunks, chunk_len, nheads, 1],
304                decay_states_bnlh1.dims()
305            );
306            let decayed_x_bnlhp = decay_states_bnlh1 * x_bnlhp.clone();
307            assert_eq!(
308                [batch, nchunks, chunk_len, nheads, per_head_dim],
309                decayed_x_bnlhp.dims()
310            );
311
312            // Contract over Q: (decayed_x[n, :, h, :])ᵀ · B̄[n, :, h, :]
313            //   [B, n, H, P, Q] × [B, n, H, Q, N] → [B, n, H, P, N]
314            let decayed_x_bnhpl = decayed_x_bnlhp.permute([0, 1, 3, 4, 2]);
315            assert_eq!(
316                [batch, nchunks, nheads, per_head_dim, chunk_len],
317                decayed_x_bnhpl.dims()
318            );
319            let b_bnhlr = b_bnlhr.clone().permute([0, 1, 3, 2, 4]);
320            assert_eq!(
321                [batch, nchunks, nheads, chunk_len, state_rank],
322                b_bnhlr.dims()
323            );
324
325            // TODO: issue for needing to ensure decayed_x_bnhpl is contiguous
326            let decayed_x_bnhpl_data = decayed_x_bnhpl.into_data();
327            let decayed_x_bnhpl: Tensor<B, 5> = Tensor::from_data(decayed_x_bnhpl_data, device);
328
329            decayed_x_bnhpl.matmul(b_bnhlr)
330        };
331        assert_eq!(
332            [batch, nchunks, nheads, per_head_dim, state_rank],
333            states_bnhpr.dims()
334        );
335
336        // =============================================================
337        // STEP 3: Inter-chunk state scan (state passing)
338        // =============================================================
339        //
340        // Propagate hidden states across chunk boundaries.  The recurrence is
341        //
342        //   h[n] = Ā_chunk[n] · h[n-1] + s[n]     (Eq. 22)
343        //
344        // where Ā_chunk[n] = exp(Σ_{t ∈ chunk n} Δₜ · A) = exp(a_cum[n, Q-1]).
345        //
346        // Unrolling the recurrence gives a matrix form identical to Step 2 but
347        // at the chunk level: each new state is a weighted sum of all previous
348        // chunk states.  We implement this with the same 1-SS segsum trick,
349        // now applied over the nchunks dimension.
350        //
351        // The result is `new_states[n]`, the true hidden state entering chunk n,
352        // for n ∈ {0, ..., nchunks-1}, plus the final state after all chunks.
353        let (states_bnhpr, final_state_bnpr) = {
354            // Prepend the initial state h₀ to the array of chunk states.
355            // Shape: [B, 1+nchunks, H, P, N]
356            let initial_states_b1hpr = ssm_initial_state_bhpr.unsqueeze_dim(1);
357            assert_eq!(
358                [batch, 1, nheads, per_head_dim, state_rank],
359                initial_states_b1hpr.dims()
360            );
361
362            // Optionally add learnable initial states (broadcast over batch).
363            let initial_states_b1hpr = if let Some(init_hpr) = init_states_hpr {
364                let init_b1hpr = init_hpr.unsqueeze_dim::<4>(0).expand([
365                    batch,
366                    1,
367                    nheads,
368                    per_head_dim,
369                    state_rank,
370                ]);
371                initial_states_b1hpr + init_b1hpr
372            } else {
373                initial_states_b1hpr
374            };
375
376            let states_bNhpr = Tensor::cat(vec![initial_states_b1hpr, states_bnhpr], 1);
377            assert_eq!(
378                [batch, 1 + nchunks, nheads, per_head_dim, state_rank],
379                states_bNhpr.dims()
380            );
381
382            // Build the inter-chunk decay matrix using segsum.
383            // a_cum_last[n] = Σ_{t ∈ chunk n} Δₜ · A   (the total log-decay of chunk n)
384            let a_cumsum_last_bhn = a_cumsum_bhnl
385                .clone()
386                .slice(s![.., .., .., -1])
387                .squeeze_dim(3); // [B, H, nchunks]
388            assert_eq!([batch, nheads, nchunks], a_cumsum_last_bhn.dims());
389
390            // Prepend a zero for the initial state (no decay before chunk 0).
391            let a_chunk_pad_bhN = Tensor::cat(
392                vec![
393                    Tensor::zeros(Shape::new([batch, nheads, 1]), device),
394                    a_cumsum_last_bhn,
395                ],
396                2,
397            ); // [B, H, 1+nchunks]
398            assert_eq!([batch, nheads, 1 + nchunks], a_chunk_pad_bhN.dims());
399
400            // 1-SS inter-chunk decay matrix.
401            //   decay_chunk[i, j] = exp(Σ_{k=j+1..i} a_cum_last[k])  (i ≥ j)
402            // Row i of this matrix, when multiplied by the state vector,
403            // gives the true hidden state entering chunk i.
404            let decay_chunk_bhNN = segsum(a_chunk_pad_bhN).exp();
405            assert_eq!(
406                [batch, nheads, 1 + nchunks, 1 + nchunks],
407                decay_chunk_bhNN.dims()
408            );
409
410            // Flatten the state's (P, N) dimensions for the matmul.
411            let flat_state_dim = per_head_dim * state_rank; // f = P·N
412            let states_bhNf = states_bNhpr
413                .clone()
414                .permute([0, 2, 1, 3, 4]) // [B, H, 1+n, P, N]
415                .reshape([batch, nheads, 1 + nchunks, flat_state_dim]);
416            assert_eq!(
417                [batch, nheads, 1 + nchunks, flat_state_dim],
418                states_bhNf.dims()
419            );
420
421            // Matmul: [B, H, 1+n, 1+n] × [B, H, 1+n, f] → [B, H, 1+n, f]
422            let new_states_bhNf = decay_chunk_bhNN.matmul(states_bhNf);
423            assert_eq!(
424                [batch, nheads, 1 + nchunks, flat_state_dim],
425                new_states_bhNf.dims()
426            );
427
428            let new_states_bhNpr =
429                new_states_bhNf.reshape([batch, nheads, 1 + nchunks, per_head_dim, state_rank]);
430
431            // Slice to get:
432            //   states[0..nchunks]  — the initial states entering each chunk
433            //   states[nchunks]     — the final state after the last real token
434            //
435            // For padded sequences the padding steps are identity operations
436            // (Δ=0 ⇒ Ā=1, B̄=0), so the state is carried unchanged through the
437            // pad region, and `states[nchunks]` is the correct final state.
438            let states_bhnpr = new_states_bhNpr
439                .clone()
440                .slice(s![.., .., 0..nchunks, .., ..]);
441            let final_state_bhpr = new_states_bhNpr
442                .slice(s![.., .., nchunks, .., ..])
443                .squeeze_dim(2);
444
445            (
446                states_bhnpr.permute([0, 2, 1, 3, 4]), // → [B, n, H, P, N]
447                final_state_bhpr,
448            )
449        };
450        assert_eq!(
451            [batch, nchunks, nheads, per_head_dim, state_rank],
452            states_bnhpr.dims()
453        );
454        assert_eq!(
455            [batch, nheads, per_head_dim, state_rank],
456            final_state_bnpr.dims()
457        );
458
459        // =============================================================
460        // STEP 4: State-to-output contribution (Y_off)
461        // =============================================================
462        //
463        // For each chunk n, compute the contribution of the true initial state
464        // h[n-1] to the outputs within that chunk (Eq. 23):
465        //
466        //   Y_off[n, t] = C[n, t]ᵀ · exp(a_cumsum[n, t]) · h[n-1]
467        //               = exp(a_cum[n,t]) · (C[n,t]ᵀ · h[n-1])
468        //
469        // where the scalar `exp(a_cum[n,t])` is the cumulative decay from the
470        // start of the chunk to position t.
471        //
472        // Implementation:
473        //   (a) C[n] · h[n-1]ᵀ  (contract over N)  → [B, n, H, Q, P]
474        //   (b) element-wise multiply with exp(a_cum)
475        let y_off_bnlhp = {
476            // exp(a_cumsum[n, t]): decay from start of chunk to position t.
477            let state_decay_out_bhnl = a_cumsum_bhnl.exp();
478            assert_eq!(
479                [batch, nheads, nchunks, chunk_len],
480                state_decay_out_bhnl.dims()
481            );
482
483            // (a) C[n] · h[n-1]ᵀ  → [B, n, H, Q, P]
484            //   C: [B, n, H, Q, N],  h: [B, n, H, N, P]  (transposed from [B,n,H,P,N])
485            let c_bnhlr = c_bnlhr.permute([0, 1, 3, 2, 4]); // [B, n, H, Q, N]
486            assert_eq!(
487                [batch, nchunks, nheads, chunk_len, state_rank],
488                c_bnhlr.dims()
489            );
490
491            let states_bnhrp = states_bnhpr.permute([0, 1, 2, 4, 3]); // [B, n, H, N, P]
492            assert_eq!(
493                [batch, nchunks, nheads, state_rank, per_head_dim],
494                states_bnhrp.dims()
495            );
496
497            let ch_bnhlp = c_bnhlr.matmul(states_bnhrp); // [B, n, H, Q, P]
498            assert_eq!(
499                [batch, nchunks, nheads, chunk_len, per_head_dim],
500                ch_bnhlp.dims()
501            );
502
503            // (b) Multiply by the intra-chunk cumulative decay.
504            let state_decay_out_bnhl1 = state_decay_out_bhnl.permute([0, 2, 1, 3]).unsqueeze_dim(4);
505            assert_eq!(
506                [batch, nchunks, nheads, chunk_len, 1],
507                state_decay_out_bnhl1.dims()
508            );
509
510            let y_off_bnhlp = ch_bnhlp * state_decay_out_bnhl1;
511            assert_eq!(
512                [batch, nchunks, nheads, chunk_len, per_head_dim],
513                y_off_bnhlp.dims()
514            );
515
516            y_off_bnhlp.permute([0, 1, 3, 2, 4]) // → [B, n, Q, H, P]
517        };
518        assert_eq!(
519            [batch, nchunks, chunk_len, nheads, per_head_dim],
520            y_off_bnlhp.dims()
521        );
522
523        // ── Combine Y_diag and Y_off, undo padding ────────────────────────────
524        let y_bnlhp = y_diag_bnlhp + y_off_bnlhp;
525
526        // Flatten chunks back into the sequence dimension.
527        let y_bshp = y_bnlhp.reshape([batch, sequence, nheads, per_head_dim]);
528
529        // ── D skip connection ─────────────────────────────────────────────────
530        // yₜ += D · xₜ
531        // D is a per-head scalar; broadcast over batch, sequence, and per_head_dim.
532        let d_bsh1 = d_h
533            .unsqueeze_dims::<4>(&[0, 1, 3]) // [H] → [1, 1, H, 1]
534            .expand([batch, sequence, nheads, 1]);
535        let y_bshp = y_bshp + d_bsh1 * x_bshp;
536
537        (y_bshp, final_state_bnpr)
538    }
539}

541// ---------------------------------------------------------------------------
542// segsum  (stable segment sum for the 1-SS mask)
543// ---------------------------------------------------------------------------
544
545/// Compute stable segment sums for constructing the 1-semiseparable mask.
546///
547/// Given a tensor `x` of shape `[..., T]`, returns a tensor of shape
548/// `[..., T, T]` where:
549///
550/// ```text
551///   out[..., i, j] = Σ_{k=j+1}^{i} x[..., k]     for i ≥ j   (lower triangle)
552///   out[..., i, j] = -∞                             for i < j   (upper triangle)
553/// ```
554///
555/// The 1-semiseparable mask is then obtained by exponentiating:
556///
557/// ```text
558///   L = exp(segsum(log_A))
559///   L[i, j] = exp(log_A[j+1] + ... + log_A[i])
560///            = A[j+1] · A[j+2] · ... · A[i]       (Eq. 4–5 in the paper)
561/// ```
562///
563/// ## Implementation
564///
565/// A naive computation of all pairwise products `A[j+1]·...·A[i]` would
566/// suffer from underflow for long sequences (e.g. `0.9^1000 ≈ 2.6×10⁻⁴⁶`).
567/// Working in log-space and computing differences of prefix sums avoids this:
568///
569/// ```text
570///   segsum(x)[i, j] = cumsum(x)[i] - cumsum(x)[j]
571/// ```
572///
573/// The upper triangle is masked to -∞ so that `exp(segsum(...))` gives 0
574/// for non-causal positions (the strict upper triangle of L must be zero).
575///
576/// ## Const-generic dimension handling
577///
578/// This function is generic over the input rank `D` and returns a tensor of
579/// rank `D + 1`.  Burn requires the output rank to be known at compile time,
580/// which is achieved through the const generic expression `{ D + 1 }`.
581fn segsum<B: Backend, const D: usize>(x: Tensor<B, D>) -> Tensor<B, { D + 1 }> {
582    assert!(D > 0);
583
584    // cumsum[..., t] = x[..., 0] + x[..., 1] + ... + x[..., t]
585    let x_cumsum = x.cumsum(D - 1);
586
587    // Broadcast along two different axes to compute all pairwise differences:
588    //   x_cumsum_row[..., i, j] = cumsum[..., i]   (i varies along axis D)
589    //   x_cumsum_col[..., i, j] = cumsum[..., j]   (j varies along axis D-1)
590    let x_cumsum_row = x_cumsum.clone().unsqueeze_dim(D); // [..., T, 1]
591    let x_cumsum_col = x_cumsum.unsqueeze_dim(D - 1); // [..., 1, T]
592
593    // diff[..., i, j] = cumsum[i] - cumsum[j]
594    //                 = x[j+1] + ... + x[i]    for i ≥ j
595    let diff = x_cumsum_row - x_cumsum_col; // [..., T, T]
596
597    // Mask the strict upper triangle (i < j) with -∞.
598    // triu(1) returns a tensor that is -∞ above the main diagonal and 0
599    // elsewhere; adding it to `diff` zeroes out the upper triangle of exp(diff).
600    let neg_inf_mask = Tensor::full_like(&diff, f32::NEG_INFINITY).triu(1);
601    diff + neg_inf_mask
602}
