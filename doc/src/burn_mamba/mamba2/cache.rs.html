1//! # Mamba-2 Inference Caches
2//!
3//! This module defines the state that must be preserved between calls during
4//! autoregressive (token-by-token) generation.  During *training* or *prefill*
5//! the full sequence is available at once and the chunked SSD algorithm is used
6//! (see [`crate::mamba2::Mamba2::forward`]).  During *decoding* the model
7//! processes one token per step and the SSM operates in its pure recurrent
8//! form (see [`crate::mamba2::Mamba2::step`]):
9//!
10//! ```text
11//!   hₜ = Āₜ hₜ₋₁ + B̄ₜ xₜ        (state update)
12//!   yₜ = Cₜᵀ hₜ + D xₜ            (output)
13//! ```
14//!
15//! Two pieces of state are required per layer:
16//!
17//! 1. **Convolution cache** — the last `conv_kernel` inputs to the depthwise
18//!    Conv1d, kept so that every decoding step can apply the causal filter
19//!    without re-processing previous tokens.
20//!
21//! 2. **SSM hidden state** — the matrix `hₜ ∈ ℝ^{P×N}` (per head), which
22//!    compresses the entire past context into a fixed-size representation
23//!    regardless of how many tokens have been generated.  This is the key
24//!    memory-efficiency advantage of SSMs over attention: the KV-cache of a
25//!    Transformer grows as O(T·N) with sequence length, whereas the SSM state
26//!    is always O(P·N).
27

32// ---------------------------------------------------------------------------
33// Mamba2Caches  (one cache entry per layer)
34// ---------------------------------------------------------------------------
35
36/// A collection of per-layer caches for a complete Mamba-2 network.
37///
38/// During autoregressive decoding, a [`Mamba2Caches`] instance is threaded
39/// through every call to [`crate::layer::Mamba2Layers::step`].  Each element
40/// of `caches` corresponds to one (virtual) layer in the network.
41#[derive(Module, Debug)]
42pub struct Mamba2Caches<B: Backend> {
43    /// Per-layer caches.
44    ///
45    /// Length: `n_real_caches` (the number of *virtual* layers, which may
46    /// exceed the number of *real* weight layers when weight-sharing / layer
47    /// scheduling is in use).
48    pub caches: Vec<Mamba2Cache<B>>,
49}
50
51/// Configuration / factory for [`Mamba2Caches`].
52#[derive(Config, Debug)]
53pub struct Mamba2CachesConfig {
54    /// Number of cache slots.  Equals the number of virtual layers in the
55    /// network (one cache per layer, even when layers share weights).
56    pub n_real_caches: usize,
57
58    /// Shared configuration that determines the shape of each individual
59    /// cache tensor.
60    pub cache: Mamba2CacheConfig,
61}
62
63impl Mamba2CachesConfig {
64    /// Convenience constructor that derives cache shapes directly from a
65    /// [`Mamba2Config`] block configuration.
66    pub fn new_from_block_config(
67        n_real_caches: usize,
68        batch: usize,
69        block_config: Mamba2Config,
70    ) -> Self {
71        Self {
72            n_real_caches,
73            cache: Mamba2CacheConfig::new_from_block_config(batch, block_config),
74        }
75    }
76
77    /// Allocate all cache tensors (zero-initialised) on `device`.
78    pub fn init<B: Backend>(&self, device: &B::Device) -> Mamba2Caches<B> {
79        let caches = (0..self.n_real_caches)
80            .map(|_| self.cache.clone().init(device))
81            .collect();
82        Mamba2Caches { caches }
83    }
84}

86// ---------------------------------------------------------------------------
87// Mamba2Cache  (state for a single layer)
88// ---------------------------------------------------------------------------
89
90/// The mutable state carried between decoding steps for a **single** Mamba-2
91/// layer.
92///
93/// Both tensors are updated in-place (via Burn's functional clone) at every
94/// call to [`crate::mamba2::Mamba2::step`].
95#[derive(Module, Debug)]
96pub struct Mamba2Cache<B: Backend> {
97    /// **Convolution rolling window.**
98    ///
99    /// Stores the last `conv_kernel` pre-activation feature vectors fed into
100    /// the depthwise Conv1d.  At each step, the oldest column is discarded and
101    /// the new token's projection is appended (a left-shift followed by an
102    /// insert into the rightmost column), maintaining strict causality.
103    ///
104    /// Shape: `[batch, conv_dim, conv_kernel]`
105    ///   - `conv_dim  = d_inner + 2 · ngroups · state_rank`
106    ///   - `conv_kernel` is typically 4
107    pub conv_bvk: Tensor<B, 3>,
108
109    /// **SSM hidden state** `hₜ`.
110    ///
111    /// This is the O(P·N) compressed summary of all tokens seen so far.
112    /// Updated via `hₜ = Āₜ hₜ₋₁ + B̄ₜ xₜ` at each decoding step.
113    ///
114    /// The tensor is indexed as `[batch, nheads, per_head_dim, state_rank]`
115    /// (i.e. `[B, H, P, N]` in the paper's notation), which is the transpose
116    /// of the mathematical `hₜ ∈ ℝ^{N×P}` but equivalent in content.
117    ///
118    /// Shape: `[batch, nheads, per_head_dim, state_rank]`
119    pub ssm_bhpr: Tensor<B, 4>,
120}
121
122/// Configuration / factory for a single [`Mamba2Cache`].
123#[derive(Config, Debug)]
124pub struct Mamba2CacheConfig {
125    /// Batch size.
126    pub batch: usize,
127
128    /// State rank `N` — the number of latent dimensions in the SSM hidden
129    /// state.  Corresponds to `state_rank` in [`Mamba2Config`].
130    #[config(default = 128)]
131    pub state_rank: usize,
132
133    /// Causal convolution window length.  Corresponds to `conv_kernel` in
134    /// [`Mamba2Config`].
135    #[config(default = 4)]
136    pub conv_kernel: usize,
137
138    /// Number of channels entering (and leaving) the depthwise convolution.
139    /// Equal to `d_inner + 2 · ngroups · state_rank`.
140    pub conv_dim: usize,
141
142    /// Head dimension `P`.  Corresponds to `per_head_dim` in [`Mamba2Config`].
143    #[config(default = 64)]
144    pub per_head_dim: usize,
145
146    /// Number of SSM heads `H`.
147    pub nheads: usize,
148}
149
150impl Mamba2CacheConfig {
151    /// Derive cache shapes from a Mamba-2 block configuration plus a batch
152    /// size.
153    pub fn new_from_block_config(batch: usize, block_config: Mamba2Config) -> Self {
154        Self {
155            batch,
156            state_rank: block_config.state_rank,
157            conv_kernel: block_config.conv_kernel,
158            conv_dim: block_config.conv_dim(),
159            per_head_dim: block_config.per_head_dim,
160            nheads: block_config.nheads(),
161        }
162    }
163
164    /// Allocate zero-initialised cache tensors on `device`.
165    ///
166    /// Zero initialisation is correct because:
167    /// - The convolution cache represents "no previous tokens" (identity padding).
168    /// - The SSM state represents `h₀ = 0` (zero initial condition), which is
169    ///   the standard default.  Learnable initial states (if configured) are
170    ///   added on top of this inside [`crate::mamba2::Mamba2::forward`] /
171    ///   [`crate::mamba2::Mamba2::step`].
172    pub fn init<B: Backend>(&self, device: &B::Device) -> Mamba2Cache<B> {
173        let conv_bvk = Tensor::zeros(
174            Shape::new([self.batch, self.conv_dim, self.conv_kernel]),
175            device,
176        );
177        let ssm_bhpr = Tensor::zeros(
178            Shape::new([self.batch, self.nheads, self.per_head_dim, self.state_rank]),
179            device,
180        );
181        Mamba2Cache { conv_bvk, ssm_bhpr }
182    }
183}
