1//! # Mamba-1 Layer and Layer Stack
2//!
3//! A **Mamba-1 layer** is the standard Pre-LN residual block used throughout
4//! the network.  It wraps a single [`Mamba1`] SSM block with an RMSNorm
5//! (applied to the input, *before* the block) and adds the input back as a
6//! residual connection:
7//!
8//! ```text
9//!   y = x + Mamba1( RMSNorm(x) )
10//! ```
11//!
12//! ## Virtual layers
13//!
14//! [`Mamba1Layers`] supports *virtual layers*: a larger logical depth achieved
15//! by cycling through a smaller set of *real* (weight-bearing) layers according
16//! to a [`Schedule`].  Each virtual layer keeps its **own cache** but shares
17//! the underlying parameters.  See [`crate::mamba2::layer`] for the structure
18//! this mirrors.
19

26// ---------------------------------------------------------------------------
27// Mamba1Layers  (the full layer stack)
28// ---------------------------------------------------------------------------
29
30/// A stack of Mamba-1 layers with optional virtual-layer scheduling.
31///
32/// The stack maintains `n_real_layers` distinct weight sets but can execute
33/// `n_virtual_layers` logical forward passes, cycling through weights
34/// according to the provided [`Schedule`].
35#[derive(Module, Debug)]
36pub struct Mamba1Layers {
37    /// Number of real (weight-bearing) layers.
38    pub n_real_layers: usize,
39
40    /// Optional `(n_virtual_layers, schedule)` for weight-sharing.
41    ///
42    /// When `None`, the virtual layer count falls back to `n_real_layers` (no
43    /// sharing).  Marked `module(skip)` so Burn does not treat it as a
44    /// trainable parameter.
45    #[module(skip)]
46    pub n_virtual_layers: Option<(usize, Schedule)>,
47
48    /// The actual weight-bearing layer instances.
49    ///
50    /// Length: `n_real_layers`.
51    pub real_layers: Vec<Mamba1Layer>,
52
53    /// When `true`, the residual connection of the **first** virtual layer is
54    /// scaled to zero.
55    pub ignore_first_residual: bool,
56
57    /// When `true`, the residual connection of the **last** virtual layer is
58    /// scaled to zero.
59    pub ignore_last_residual: bool,
60}
61
62/// Configuration / factory for [`Mamba1Layers`].
63#[derive(Config, Debug)]
64pub struct Mamba1LayersConfig {
65    /// Number of distinct weight sets to allocate.
66    pub n_real_layers: usize,
67
68    /// Optional virtual-layer scheduling.  See [`Mamba1Layers`] for details.
69    #[config(default = "None")]
70    pub n_virtual_layers: Option<(usize, Schedule)>,
71
72    /// Configuration shared by all Mamba-1 blocks in the stack.
73    pub mamba_block: Mamba1Config,
74
75    /// See [`Mamba1Layers::ignore_first_residual`].
76    #[config(default = false)]
77    pub ignore_first_residual: bool,
78
79    /// See [`Mamba1Layers::ignore_last_residual`].
80    #[config(default = false)]
81    pub ignore_last_residual: bool,
82}
83
84impl Mamba1LayersConfig {
85    /// Allocate and initialise all layers on `device`.
86    pub fn init(&self, device: &Device) -> Mamba1Layers {
87        let real_layers = (0..self.n_real_layers)
88            .map(|_| Mamba1LayerConfig::new(self.mamba_block.clone()).init(device))
89            .collect();
90
91        Mamba1Layers {
92            n_real_layers: self.n_real_layers,
93            n_virtual_layers: self.n_virtual_layers.clone(),
94            real_layers,
95            ignore_first_residual: self.ignore_first_residual,
96            ignore_last_residual: self.ignore_last_residual,
97        }
98    }
99}
100
101impl Mamba1Layers {
102    // -----------------------------------------------------------------------
103    // forward  (full sequence — used for training / prefill)
104    // -----------------------------------------------------------------------
105
106    /// Process a full sequence through every (virtual) layer.
107    ///
108    /// If `caches` is `None`, zero-initialised caches are created automatically.
109    ///
110    /// # Returns
111    /// `(output, updated_caches)` where `output` has shape
112    /// `[batch, sequence, d_model]`.
113    pub fn forward(
114        &self,
115        mut x: Tensor<3>,
116        caches: Option<Mamba1Caches>,
117    ) -> (Tensor<3>, Mamba1Caches) {
118        let n_virtual_layers = self.n_virtual_count();
119        let caches = caches.unwrap_or_else(|| self.make_zero_caches_3d(&x, n_virtual_layers));
120
121        assert_eq!(
122            caches.caches_len(),
123            n_virtual_layers,
124            "cache count must match the number of virtual layers; \
125             layers in forward() cannot share caches"
126        );
127
128        let mut caches: Vec<Option<Mamba1Cache>> = caches.into_options();
129
130        #[allow(clippy::needless_range_loop)]
131        for i in 0..n_virtual_layers {
132            let layer_idx = self.real_idx(i);
133            let layer = &self.real_layers[layer_idx];
134            let residual_scale = self.residual_scale(i, n_virtual_layers);
135
136            let cache = caches[i].take().unwrap();
137            let (x_, cache_) = layer.forward(x, Some(cache), residual_scale);
138            x = x_;
139            caches[i] = Some(cache_);
140        }
141
142        let caches = Mamba1Caches::from_options(caches);
143        (x, caches)
144    }
145
146    // -----------------------------------------------------------------------
147    // step  (single token — used for autoregressive decoding)
148    // -----------------------------------------------------------------------
149
150    /// Process a **single token** through every (virtual) layer.
151    ///
152    /// # Returns
153    /// `(output, updated_caches)` where `output` has shape `[batch, d_model]`.
154    pub fn step(
155        &self,
156        mut x: Tensor<2>,
157        caches: Option<Mamba1Caches>,
158    ) -> (Tensor<2>, Mamba1Caches) {
159        let n_virtual_layers = self.n_virtual_count();
160        let caches = caches.unwrap_or_else(|| self.make_zero_caches_2d(&x, n_virtual_layers));
161
162        assert_eq!(
163            caches.caches_len(),
164            n_virtual_layers,
165            "cache count must match the number of virtual layers; \
166             layers in step() cannot share caches"
167        );
168
169        let mut caches: Vec<Option<Mamba1Cache>> = caches.into_options();
170
171        #[allow(clippy::needless_range_loop)]
172        for i in 0..n_virtual_layers {
173            let layer_idx = self.real_idx(i);
174            let layer = &self.real_layers[layer_idx];
175            let residual_scale = self.residual_scale(i, n_virtual_layers);
176
177            let cache = caches[i].take().unwrap();
178            let (x_, cache_) = layer.step(x, cache, residual_scale);
179            x = x_;
180            caches[i] = Some(cache_);
181        }
182
183        let caches = Mamba1Caches::from_options(caches);
184        (x, caches)
185    }
186
187    // -----------------------------------------------------------------------
188    // Private helpers
189    // -----------------------------------------------------------------------
190
191    /// Effective number of forward passes (virtual layers).
192    fn n_virtual_count(&self) -> usize {
193        self.n_virtual_layers
194            .as_ref()
195            .map(|(l, _)| *l)
196            .unwrap_or(self.n_real_layers)
197    }
198
199    /// Map a virtual layer index to the corresponding real layer index using
200    /// the configured schedule (or identity when no schedule is set).
201    fn real_idx(&self, virtual_idx: usize) -> usize {
202        if let Some((n_virtual_layers, schedule)) = &self.n_virtual_layers {
203            schedule.real_idx(virtual_idx, *n_virtual_layers, self.n_real_layers)
204        } else {
205            virtual_idx
206        }
207    }
208
209    /// Returns 0.0 if this layer's residual should be suppressed, else 1.0.
210    fn residual_scale(&self, i: usize, n_virtual: usize) -> f32 {
211        let is_first = self.ignore_first_residual && i == 0;
212        let is_last = self.ignore_last_residual && i + 1 == n_virtual;
213        if is_first || is_last { 0.0 } else { 1.0 }
214    }
215
216    /// Build zero-initialised caches from a 3-dimensional input tensor
217    /// `[batch, sequence, d_model]`.
218    fn make_zero_caches_3d(&self, x: &Tensor<3>, n_virtual: usize) -> Mamba1Caches {
219        let [batch, _sequence, _d_model] = x.dims();
220        self.make_zero_caches(batch, n_virtual, &x.device())
221    }
222
223    /// Build zero-initialised caches from a 2-dimensional input tensor
224    /// `[batch, d_model]`.
225    fn make_zero_caches_2d(&self, x: &Tensor<2>, n_virtual: usize) -> Mamba1Caches {
226        let [batch, _d_model] = x.dims();
227        self.make_zero_caches(batch, n_virtual, &x.device())
228    }
229
230    /// Derive the cache shapes from the first real block's parameters and
231    /// allocate `n_virtual` zero caches.
232    fn make_zero_caches(
233        &self,
234        batch: usize,
235        n_virtual: usize,
236        device: &Device,
237    ) -> Mamba1Caches {
238        let block0 = &self.real_layers[0].mamba_block;
239        let [d_inner, state_rank] = block0.a_log.dims();
240        let [_, _, conv_kernel] = block0.conv1d.weight.dims();
241
242        Mamba1CachesConfig::new(
243            n_virtual,
244            Mamba1CacheConfig {
245                batch,
246                state_rank,
247                conv_kernel,
248                d_inner,
249            },
250        )
251        .init(device)
252    }
253}

255// ---------------------------------------------------------------------------
256// Mamba1Layer  (single Pre-LN residual block)
257// ---------------------------------------------------------------------------
258
259/// A single Mamba-1 residual block:
260///
261/// ```text
262///   output = x·scale + Mamba1( RMSNorm(x) )
263/// ```
264///
265/// where `scale` is 1.0 normally and 0.0 when the residual connection is
266/// intentionally suppressed by the layer stack configuration.
267#[derive(Module, Debug)]
268pub struct Mamba1Layer {
269    /// Pre-norm applied to the input before the SSM block.
270    pub norm: RmsNorm,
271    /// The Mamba-1 SSM block (see [`Mamba1`]).
272    pub mamba_block: Mamba1,
273}
274
275/// Configuration / factory for [`Mamba1Layer`].
276#[derive(Config, Debug)]
277pub struct Mamba1LayerConfig {
278    /// Configuration for the inner Mamba-1 block.
279    pub mamba_block: Mamba1Config,
280}
281
282impl Mamba1LayerConfig {
283    /// Allocate and initialise the layer on `device`.
284    pub fn init(&self, device: &Device) -> Mamba1Layer {
285        Mamba1Layer {
286            norm: RmsNormConfig::new(self.mamba_block.d_model).init(device),
287            mamba_block: self.mamba_block.init(device),
288        }
289    }
290}
291
292impl Mamba1Layer {
293    /// Run the Pre-LN residual block over a full sequence.
294    ///
295    /// Computes `output = x · residual_scale + Mamba1( RMSNorm(x) )`.
296    ///
297    /// # Shapes
298    ///   - `x`    : `[batch, sequence, d_model]`
299    ///   - output : `[batch, sequence, d_model]`
300    pub fn forward(
301        &self,
302        x: Tensor<3>,
303        cache: Option<Mamba1Cache>,
304        residual_scale: f32,
305    ) -> (Tensor<3>, Mamba1Cache) {
306        let [batch, sequence, d_model] = x.dims();
307
308        let res = x.clone() * residual_scale;
309        let normed = self.norm.forward(x);
310
311        let (x, cache) = self.mamba_block.forward(normed, cache);
312        assert_eq!([batch, sequence, d_model], x.dims());
313
314        (x + res, cache)
315    }
316
317    /// Run the Pre-LN residual block for a **single** decoding step.
318    ///
319    /// Computes `output = x · residual_scale + Mamba1.step( RMSNorm(x) )`.
320    ///
321    /// # Shapes
322    ///   - `x`    : `[batch, d_model]`
323    ///   - output : `[batch, d_model]`
324    pub fn step(
325        &self,
326        x: Tensor<2>,
327        cache: Mamba1Cache,
328        residual_scale: f32,
329    ) -> (Tensor<2>, Mamba1Cache) {
330        let [batch, d_model] = x.dims();
331
332        let res = x.clone() * residual_scale;
333        let normed = self.norm.forward(x);
334
335        let (x, cache) = self.mamba_block.step(normed, cache);
336        assert_eq!([batch, d_model], x.dims());
337
338        (x + res, cache)
339    }
340}
