1//! # Mamba-1 Language Model Network
2//!
3//! Assembles a complete autoregressive language model from the Mamba-1
4//! components:
5//!
6//! ```text
7//!   tokens [batch, sequence]
8//!       │  Embedding (vocab_size → d_model)
9//!       ▼  (×n_virtual_layers)
10//!   Mamba1Layer [Pre-LN residual block]
11//!       │  RMSNorm (final)
12//!       ▼  LM head (d_model → vocab_size)
13//!   logits [batch, sequence, vocab_size]
14//! ```
15//!
16//! Mirrors [`crate::mamba2::network`]: vocabulary is padded up to
17//! `pad_vocab_size_multiple`, the LM head can be weight-tied to the embedding
18//! (`missing_lm_head = true`), and both [`Mamba1Network::forward`] and
19//! [`Mamba1Network::step`] thread a [`Mamba1Caches`] so prefill can be followed
20//! by decoding.
21//!
22//! References:
23//! - [huggingface/candle](https://github.com/huggingface/candle/blob/fd7c8565646039e35925b8730d27ddad195d7e73/candle-examples/examples/mamba-minimal/)
24//! - [johnma2006/mamba-minimal](https://github.com/johnma2006/mamba-minimal/blob/61f01953ca153f8c4a850d7111beecbf4be9cee1/)
25

32/// A complete Mamba-1 language model.
33#[derive(Module, Debug)]
34pub struct Mamba1Network<B: Backend> {
35    /// Token embedding table (`padded_vocab_size → d_model`).
36    pub embedding: Embedding<B>,
37    /// The stack of Mamba-1 residual layers.
38    pub layers: Mamba1Layers<B>,
39    /// Final RMSNorm applied before the LM head.
40    pub norm_f: RmsNorm<B>,
41    /// If missing, re-utilizes a transposed `embedding` weight.
42    pub lm_head: Option<Linear<B>>,
43}
44
45/// Configuration / factory for [`Mamba1Network`].
46#[derive(Config, Debug)]
47pub struct Mamba1NetworkConfig {
48    /// Number of real (weight-bearing) Mamba-1 layers.
49    pub n_real_layers: usize,
50
51    /// Optional virtual-layer scheduling.  See [`Mamba1Layers`] for details.
52    #[config(default = "None")]
53    pub n_virtual_layers: Option<(usize, Schedule)>,
54
55    /// If vocab_size is divisible by pad_vocab_size_multiple, this should be considered the unpadded vocab size.
56    /// Otherwise, this is padded into `((vocab_size / self.pad_vocab_size_multiple) + 1) * pad_vocab_size_multiple`.
57    pub vocab_size: usize,
58
59    /// If no pad is required, vocab_size must be divisible by pad_vocab_size_multiple.
60    /// If pad is required, vocab_size increases until it's divisible by pad_vocab_size_multiple.
61    ///
62    /// To disable vocab padding, you can set this to `1`.
63    pub pad_vocab_size_multiple: usize,
64
65    /// Configuration shared by all Mamba-1 blocks.
66    pub mamba_block: Mamba1Config,
67
68    /// If set to true, `lm_head` is set to `None` and it re-utilizes the transposed `embedding` weights.
69    pub missing_lm_head: bool,
70}
71
72impl Mamba1NetworkConfig {
73    /// Returns the initialized model.
74    pub fn init<B: Backend>(&self, device: &B::Device) -> Mamba1Network<B> {
75        let layers = Mamba1LayersConfig {
76            n_real_layers: self.n_real_layers,
77            n_virtual_layers: self.n_virtual_layers.clone(),
78            mamba_block: self.mamba_block.clone(),
79            ignore_first_residual: false,
80            ignore_last_residual: false,
81        }
82        .init(device);
83
84        let padded_vocab_size = {
85            if self.vocab_size.is_multiple_of(self.pad_vocab_size_multiple) {
86                self.vocab_size
87            } else {
88                ((self.vocab_size / self.pad_vocab_size_multiple) + 1)
89                    * self.pad_vocab_size_multiple
90            }
91        };
92
93        Mamba1Network {
94            embedding: EmbeddingConfig::new(padded_vocab_size, self.mamba_block.d_model)
95                .init(device),
96            layers,
97            norm_f: RmsNormConfig::new(self.mamba_block.d_model).init(device),
98            lm_head: if self.missing_lm_head {
99                None
100            } else {
101                Some(
102                    LinearConfig::new(self.mamba_block.d_model, padded_vocab_size)
103                        .with_bias(false)
104                        .init(device),
105                )
106            },
107        }
108    }
109}

111impl<B: Backend> Mamba1Network<B> {
112    /// See also [`Self::step`].
113    ///
114    /// Processes a full token sequence and returns next-token logits along with
115    /// the updated caches (ready for a subsequent [`Self::step`] call).
116    ///
117    /// # Shapes
118    ///   - Input `[batch, sequence]`
119    ///   - Output `[batch, sequence, padded_vocab_size]`
120    pub fn forward(
121        &self,
122        x: Tensor<B, 2, Int>,
123        caches: Option<Mamba1Caches<B>>,
124    ) -> (Tensor<B, 3>, Mamba1Caches<B>) {
125        let [batch, sequence] = x.dims();
126        let [padded_vocab, d_model] = self.embedding.weight.dims();
127
128        let x = self.embedding.forward(x);
129        assert_eq!([batch, sequence, d_model], x.dims());
130
131        let (mut x, caches) = self.layers.forward(x, caches);
132        assert_eq!([batch, sequence, d_model], x.dims());
133
134        x = self.norm_f.forward(x);
135        x = self.apply_lm_head(x, d_model, padded_vocab);
136        assert_eq!([batch, sequence, padded_vocab], x.dims());
137
138        (x, caches)
139    }
140
141    /// See also [`Self::forward`].
142    ///
143    /// Processes a **single** token and returns next-token logits along with
144    /// the caches updated for the next step.
145    ///
146    /// # Shapes
147    ///   - Input `[batch]`
148    ///   - Output `[batch, padded_vocab_size]`
149    pub fn step(
150        &self,
151        x: Tensor<B, 1, Int>,
152        caches: Option<Mamba1Caches<B>>,
153    ) -> (Tensor<B, 2>, Mamba1Caches<B>) {
154        let [batch] = x.dims();
155        let [padded_vocab, d_model] = self.embedding.weight.dims();
156
157        let x = x.unsqueeze_dim::<2>(1);
158        assert_eq!([batch, 1], x.dims());
159
160        let x = self.embedding.forward(x);
161        assert_eq!([batch, 1, d_model], x.dims());
162        let x = x.squeeze_dim(1);
163        assert_eq!([batch, d_model], x.dims());
164
165        let (mut x, caches) = self.layers.step(x, caches);
166        assert_eq!([batch, d_model], x.dims());
167
168        x = self.norm_f.forward(x);
169
170        // Re-use the `apply_lm_head` helper by temporarily unsqueezing the
171        // sequence dimension then squeezing it back out.
172        let x = x.unsqueeze_dim(1);
173        let logits = self.apply_lm_head(x, d_model, padded_vocab);
174        assert_eq!([batch, 1, padded_vocab], logits.dims());
175        let logits = logits.squeeze_dim(1);
176        assert_eq!([batch, padded_vocab], logits.dims());
177
178        (logits, caches)
179    }
180
181    /// Apply the LM head projection to a `[batch, sequence, d_model]` tensor,
182    /// returning `[batch, sequence, padded_vocab]`.
183    ///
184    /// Uses the dedicated `lm_head` linear layer when available, or the
185    /// transposed embedding weight matrix otherwise (weight tying).
186    fn apply_lm_head(&self, x: Tensor<B, 3>, d_model: usize, padded_vocab: usize) -> Tensor<B, 3> {
187        if let Some(lm_head) = &self.lm_head {
188            lm_head.forward(x)
189        } else {
190            let weight = self.embedding.weight.clone().map(|w| w.permute([1, 0]));
191            assert_eq!([d_model, padded_vocab], weight.dims());
192            let linear = Linear { weight, bias: None };
193            linear.forward(x)
194        }
195    }
196}
