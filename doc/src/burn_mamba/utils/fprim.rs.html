1//! # Rank-tagged primitive tensor wrapper for the custom backward math
2//!
3//! [`F`] is a thin newtype over a backend's [`FloatTensor`] primitive that
4//! mirrors the subset of the high-level [`Tensor`](burn::tensor::Tensor) method
5//! API used by the recompute-backward gradient math
6//! (`*/serial_recalculated/combined_backward.rs`).
7//!
8//! Why it exists: in Burn 0.22 the high-level `Tensor` is pinned to the global
9//! `Dispatch` backend, so it cannot be built from an arbitrary backend `B`'s
10//! primitive.  A custom [`Backward`](burn::backend::autodiff::ops::Backward)
11//! node runs with a *generic* `B`, so its gradient math must operate directly on
12//! `B`'s primitives via the `B::float_*` ops.  This wrapper keeps that
13//! primitive-level math reading like the original `Tensor` code (method
14//! chaining, shape-suffixed names) instead of deeply nested free-function calls.
15//!
16//! The rank `D` is a compile-time tag for parity with the ported code and to
17//! catch rank mistakes; every operation ultimately defers to `B`'s
18//! runtime-shaped primitive ops.
19
20use burn::backend::Backend;
21use burn::backend::get_device_settings;
22use burn::backend::ops::{BoolTensorOps, FloatTensorOps, IntTensorOps};
23use burn::backend::tensor::{BoolTensor, Device, FloatTensor, IntTensor};
24use burn::backend::{FloatDType, Scalar, Shape, Slice, SliceArg, TensorMetadata};

26/// A backend float-tensor primitive tagged with a compile-time rank `D`.
27///
28/// Mirrors the slice of [`Tensor`](burn::tensor::Tensor)'s method API needed by
29/// the custom backward gradient math, operating directly on `B`'s primitives.
30pub(crate) struct F<B: Backend, const D: usize>(pub FloatTensor<B>);
31
32impl<B: Backend, const D: usize> Clone for F<B, D> {
33    fn clone(&self) -> Self {
34        F(self.0.clone())
35    }
36}
37
38impl<B: Backend, const D: usize> F<B, D> {
39    /// Wrap a raw primitive.
40    pub fn new(p: FloatTensor<B>) -> Self {
41        F(p)
42    }
43
44    /// Unwrap to the raw primitive.
45    pub fn inner(self) -> FloatTensor<B> {
46        self.0
47    }
48
49    /// Runtime shape as a `[usize; D]` array.
50    pub fn dims(&self) -> [usize; D] {
51        self.0.shape().dims()
52    }
53
54    /// Device the tensor lives on.
55    pub fn device(&self) -> Device<B> {
56        B::float_device(&self.0)
57    }
58
59    /// Float dtype of the tensor.
60    pub fn dtype(&self) -> FloatDType {
61        self.0.dtype().into()
62    }
63
64    /// Batched matrix multiplication over the last two dims.
65    pub fn matmul(self, rhs: Self) -> Self {
66        F(B::float_matmul(self.0, rhs.0))
67    }
68
69    /// Permute the axes (rank-preserving).
70    pub fn permute(self, axes: [usize; D]) -> Self {
71        F(B::float_permute(self.0, &axes))
72    }
73
74    /// Swap two axes.
75    pub fn swap_dims(self, dim1: usize, dim2: usize) -> Self {
76        F(B::float_swap_dims(self.0, dim1, dim2))
77    }
78
79    /// Element-wise `exp`.
80    pub fn exp(self) -> Self {
81        F(B::float_exp(self.0))
82    }
83
84    /// Sum along `dim`, keeping it as a size-1 axis (rank-preserving).
85    pub fn sum_dim(self, dim: usize) -> Self {
86        F(B::float_sum_dim(self.0, dim))
87    }
88
89    /// Cumulative sum along `dim`.
90    pub fn cumsum(self, dim: usize) -> Self {
91        F(B::float_cumsum(self.0, dim))
92    }
93
94    /// Slice the tensor (rank-preserving), accepting the same `s![..]` args as
95    /// the high-level API.
96    pub fn slice<S: SliceArg>(self, slices: S) -> Self {
97        let shape = self.0.shape();
98        let slices = slices.into_slices(&shape);
99        F(B::float_slice(self.0, &slices))
100    }
101
102    /// Narrow `dim` to `[start, start + length)` (rank-preserving).
103    pub fn narrow(self, dim: usize, start: usize, length: usize) -> Self {
104        let mut slices: Vec<Slice> = (0..dim).map(|_| Slice::from(..)).collect();
105        slices.push(Slice::from(start..start + length));
106        let shape = self.0.shape();
107        let slices = (&slices[..]).into_slices(&shape);
108        F(B::float_slice(self.0, &slices))
109    }
110
111    /// Reshape to a new rank `D2`.
112    pub fn reshape<const D2: usize>(self, shape: [usize; D2]) -> F<B, D2> {
113        F(B::float_reshape(self.0, Shape::new(shape)))
114    }
115
116    /// Broadcast-expand to `shape` (rank-preserving).
117    pub fn expand(self, shape: [usize; D]) -> Self {
118        F(B::float_expand(self.0, Shape::new(shape)))
119    }
120
121    /// Remove the size-1 axis at `dim`, yielding rank `D2 = D - 1`.
122    pub fn squeeze_dim<const D2: usize>(self, dim: usize) -> F<B, D2> {
123        let current = self.0.shape().dims::<D>();
124        let mut new_dims = [0usize; D2];
125        new_dims[..dim].copy_from_slice(&current[..dim]);
126        new_dims[dim..].copy_from_slice(&current[dim + 1..]);
127        F(B::float_reshape(self.0, Shape::new(new_dims)))
128    }
129
130    /// Insert a size-1 axis at `dim`, yielding rank `D2 = D + 1`.
131    pub fn unsqueeze_dim<const D2: usize>(self, dim: usize) -> F<B, D2> {
132        let shape = self.0.shape().dims::<D>();
133        let mut dims = [1usize; D2];
134        dims[0..dim].copy_from_slice(&shape[0..dim]);
135        if dim < D {
136            dims[dim] = 1;
137            dims[(dim + 1)..].copy_from_slice(&shape[dim..]);
138        } else {
139            dims[dim] = 1;
140        }
141        F(B::float_reshape(self.0, Shape::new(dims)))
142    }
143
144    /// Insert size-1 axes at the given output positions, yielding rank `D2`.
145    ///
146    /// Mirrors [`Tensor::unsqueeze_dims`](burn::tensor::Tensor::unsqueeze_dims):
147    /// negative indices count from the back and duplicates insert multiple axes.
148    pub fn unsqueeze_dims<const D2: usize>(self, axes: &[isize]) -> F<B, D2> {
149        let old_dims = self.0.shape().dims::<D>();
150        let mut new_dims = [1usize; D2];
151
152        // Resolve negative indices (counting from the back, in reverse order).
153        let mut neg_offset = D2;
154        let mut dim_indices = axes
155            .iter()
156            .map(|&d| {
157                (if d < 0 {
158                    neg_offset -= 1;
159                    d + neg_offset as isize + 1
160                } else {
161                    d
162                }) as usize
163            })
164            .collect::<Vec<usize>>();
165        dim_indices.sort_unstable();
166        // Duplicate axes mean "insert N dims at that index": bump duplicates.
167        for i in 1..dim_indices.len() {
168            if dim_indices[i] <= dim_indices[i - 1] {
169                dim_indices[i] = dim_indices[i - 1] + 1;
170            }
171        }
172
173        let mut dim_indices_curr = 0usize;
174        let mut old_dims_curr = 0usize;
175        for new_dims_curr in 0..D2 {
176            if dim_indices_curr == dim_indices.len() {
177                new_dims[new_dims_curr..].copy_from_slice(&old_dims[old_dims_curr..]);
178                break;
179            }
180            if new_dims_curr == dim_indices[dim_indices_curr] {
181                dim_indices_curr += 1;
182            } else {
183                new_dims[new_dims_curr] = old_dims[old_dims_curr];
184                old_dims_curr += 1;
185            }
186        }
187
188        F(B::float_reshape(self.0, Shape::new(new_dims)))
189    }
190
191    /// Zero everything strictly below the `diagonal` (keeps the upper triangle).
192    ///
193    /// Equivalent to [`Tensor::triu`](burn::tensor::Tensor::triu): builds the
194    /// triangular bool mask over the last two dims and fills the masked region
195    /// with `0`.
196    pub fn triu(self, diagonal: i64) -> Self {
197        let dims = self.0.shape().dims::<D>();
198        let rows = dims[D - 2];
199        let cols = dims[D - 1];
200        let device = B::float_device(&self.0);
201
202        let mask2 = tri_bool::<B>(rows, cols, diagonal, false, &device);
203        let mut lead = [1usize; D];
204        lead[D - 2] = rows;
205        lead[D - 1] = cols;
206        let mask = B::bool_reshape(mask2, Shape::new(lead));
207        let mask = B::bool_expand(mask, Shape::new(dims));
208        F(B::float_mask_fill(self.0, mask, Scalar::from(0.0f32)))
209    }
210
211    /// Fill the positions where `mask` is `true` with `value`.
212    pub fn mask_fill(self, mask: Mask<B>, value: f32) -> Self {
213        F(B::float_mask_fill(self.0, mask.0, Scalar::from(value)))
214    }

216    /// Concatenate same-rank tensors along `dim` (rank-preserving).
217    pub fn cat(tensors: Vec<F<B, D>>, dim: usize) -> Self {
218        F(B::float_cat(
219            tensors.into_iter().map(|t| t.0).collect(),
220            dim,
221        ))
222    }
223
224    /// Stack same-rank tensors along a fresh axis `dim`, yielding rank `D2 = D + 1`.
225    pub fn stack<const D2: usize>(tensors: Vec<F<B, D>>, dim: usize) -> F<B, D2> {
226        let unsqueezed = tensors
227            .into_iter()
228            .map(|t| {
229                let current = t.0.shape().dims::<D>();
230                let mut new_dims = [1usize; D2];
231                new_dims[0..dim].copy_from_slice(&current[0..dim]);
232                new_dims[dim] = 1;
233                new_dims[(dim + 1)..].copy_from_slice(&current[dim..]);
234                B::float_reshape(t.0, Shape::new(new_dims))
235            })
236            .collect::<Vec<_>>();
237        F(B::float_cat(unsqueezed, dim))
238    }

240    /// All-zeros tensor of the given shape, dtype and device.
241    pub fn zeros(shape: [usize; D], device: &Device<B>, dtype: FloatDType) -> Self {
242        F(B::float_zeros(Shape::new(shape), device, dtype))
243    }

245    /// Constant-filled tensor of the given shape, dtype and device.

247        F(B::float_full(
248            Shape::new(shape),
249            Scalar::from(value),
250            device,
251            dtype,
252        ))
253    }

284/// A boolean mask primitive used with [`F::mask_fill`].
285///
286/// Mirrors the slice of the bool-tensor API needed to build and broadcast the
287/// causal masks in the custom backward (construct → reshape → expand).
288pub(crate) struct Mask<B: Backend>(pub BoolTensor<B>);
289
290impl<B: Backend> Clone for Mask<B> {
291    fn clone(&self) -> Self {
292        Mask(self.0.clone())
293    }
294}
295
296impl<B: Backend> Mask<B> {
297    /// `[rows, cols]` mask that is `true` strictly above the `offset` diagonal.
298    ///
299    /// Matches [`Tensor::tril_mask`](burn::tensor::Tensor::tril_mask): the
300    /// `true` entries are the region a `tril` would fill.
301    pub fn tril_mask(rows: usize, cols: usize, offset: i64, device: &Device<B>) -> Self {
302        Mask(tri_bool::<B>(rows, cols, offset, true, device))
303    }
304
305    /// Reshape the mask to a new rank.
306    pub fn reshape<const N: usize>(self, shape: [usize; N]) -> Self {
307        Mask(B::bool_reshape(self.0, Shape::new(shape)))
308    }
309
310    /// Broadcast-expand the mask.
311    pub fn expand<const N: usize>(self, shape: [usize; N]) -> Self {
312        Mask(B::bool_expand(self.0, Shape::new(shape)))
313    }
314}

316/// Build a `[rows, cols]` triangular boolean mask on-device.
317///
318/// Mirrors `tri_mask` in Burn: with `matrix = row - col`, the result is
319/// `matrix < -offset` when `lower` (the `tril_mask`/lower-triangle region) and
320/// `matrix > -offset` otherwise (the `triu_mask`/upper-triangle region) — i.e.
321/// the original `row - col + offset ≷ 0` test with `offset` folded into the
322/// comparison threshold.
323///
324/// Built from on-device `arange`/comparison rather than uploaded host `bool`
325/// data: cubecl backends reject a `Bool(Native)` `bool_from_data`, and the
326/// comparison output here adopts the device's configured bool store.
327fn tri_bool<B: Backend>(
328    rows: usize,
329    cols: usize,
330    offset: i64,
331    lower: bool,
332    device: &Device<B>,
333) -> BoolTensor<B> {
334    let settings = get_device_settings::<B>(device);
335    let shape = Shape::new([rows, cols]);
336
337    let rows_i: IntTensor<B> = B::int_reshape(
338        B::int_arange(0..rows as i64, device, settings.int_dtype),
339        Shape::new([rows, 1]),
340    );
341    let cols_i: IntTensor<B> = B::int_reshape(
342        B::int_arange(0..cols as i64, device, settings.int_dtype),
343        Shape::new([1, cols]),
344    );
345    // matrix_rc = row - col (broadcast to [rows, cols]).
346    let matrix = B::int_sub(
347        B::int_expand(rows_i, shape.clone()),
348        B::int_expand(cols_i, shape),
349    );
350    let threshold = Scalar::from(-offset);
351    if lower {
352        B::int_lower_elem(matrix, threshold, settings.bool_dtype)
353    } else {
354        B::int_greater_elem(matrix, threshold, settings.bool_dtype)
355    }
356}

358/// Primitive analogue of [`crate::utils::sanity::sanity`] for [`F`].
359///
360/// Panics if `t` contains a `NaN` (when [`crate::DENY_NAN`] is set) or an `Inf`
361/// (when [`crate::DENY_INF`] is set).  A no-op — with no device read — when both
362/// flags are `false` (the default), so it can be sprinkled through the backward
363/// math at no release-build cost.
364pub(crate) fn san<B: Backend, const D: usize>(t: &F<B, D>) {
365    if !crate::DENY_NAN && !crate::DENY_INF {
366        return;
367    }
368    let data = burn::tensor::read_sync(B::float_into_data(t.0.clone()))
369        .expect("sanity check: failed to read tensor data");
370    let mut has_nan = false;
371    let mut has_inf = false;
372    for v in data.iter::<f64>() {
373        if crate::DENY_NAN && v.is_nan() {
374            has_nan = true;
375        }
376        if crate::DENY_INF && v.is_infinite() {
377            has_inf = true;
378        }
379    }
380    if has_nan {
381        eprintln!("got a NaN");
382    }
383    if has_inf {
384        eprintln!("got a INF");
385    }
386    if has_nan || has_inf {
387        panic!("sanity check failed");
388    }
