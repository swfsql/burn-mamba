1use burn::prelude::*;
2use burn::tensor::DType;
3use burn::tensor::Element;

5/// Applies the log sigmoid function element-wise.
6///
7/// `log_sigmoid(x) = log(1 / (1 + exp(-x)))`
8pub fn log_sigmoid<const D: usize, B: Backend>(x: Tensor<B, D>) -> Tensor<B, D> {
9    match <B::FloatElem as Element>::dtype() {
10        DType::F64 | DType::F32 | DType::Flex32 | DType::BF16 => {
11            // log_sigmoid(x) = log(1 / (1 + exp(-x)))
12            (x.neg().exp() + 1.).recip().log()
13        }
14        DType::F16 => {
15            // log_sigmoid(x) = -softplus(-x)
16            -crate::utils::softplus::softplus(x.neg())
17        }
18        DType::I64
19        | DType::I32
20        | DType::I16
21        | DType::I8
22        | DType::U64
23        | DType::U32
24        | DType::U16
25        | DType::U8 => {
26            unreachable!()
27        }
28        DType::Bool => {
29            unreachable!()
30        }
31        DType::QFloat(_) => {
32            unimplemented!()
33        }
34    }
35}
